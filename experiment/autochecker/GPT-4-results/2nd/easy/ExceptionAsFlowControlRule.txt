public class ExceptionAsFlowControlRule extends AbstractJavaRulechainRule {
    public ExceptionAsFlowControlRule() {
        super(ASTTryStatement.class);
    }
    
    @Override
    public Object visit(ASTTryStatement node, Object data) {
        
        // Iterate over catch clauses of the try statement
        node.getCatchClauses().forEach(catchClause -> {
            // Check if the catch block contains a throw statement
            catchClause.getBlock().findFirstDescendant(ASTThrowStatement.class).ifPresent(throwStatement -> {
                // If the thrown exception in the nested catch block is the same as caught in the outer catch block,
                // report a violation
                if (throwStatement.jjtGetChild(0).getImage().equals(node.getCatchClauses().get(0).getParameter().getTypeImage())) {
                    addViolation(data, throwStatement);
                }
            });
        });
        
        return super.visit(node, data);
    }
}

=========================logics=========================
1. Get the name of method.

3. Get the catch branches of the try statement.
4. For each catch branch, check whether it has a throw statement.
5. Get the exception type thrown in throw statement.
6. Check whether the thrown exception type is a subclass of the catch parameter type.

        super(ASTTryStatement.class);
    }

    @Override

        ASTCatchClause catchClause = node.getFirstChildOfType(ASTCatchClause.class);

package net.sourceforge.pmd.lang.java.rule.design;


import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ExceptionAsFlowControlRule extends AbstractJavaRulechainRule {

    public ExceptionAsFlowControlRule() {
        super(ASTTryStatement.class);
    }

    @Override
    public Object visit(ASTTryStatement node, Object data) {
        JTypeMirror thrownType = getThrownExceptionType(node);
        if (node.getCatchClauses().anyMatch(catchClause -> isCaughtExceptionTypeSubtype(catchClause, thrownType))) {
            addViolation(data, node);
        }
        return super.visit(node, data);
    }

    private JTypeMirror getThrownExceptionType(ASTTryStatement node) {
        ASTThrowStatement throwStatement = node.getFirstDescendantOfType(ASTThrowStatement.class);
        return throwStatement.getExpr().getTypeMirror();
    }

    private boolean isCaughtExceptionTypeSubtype(ASTCatchClause catchClause, JTypeMirror thrownType) {
        ASTCatchParameter catchParameter = catchClause.getParameter();
        return catchParameter.getAllExceptionTypes().anyMatch(type -> TypeTestUtil.isA(thrownType, type.getTypeMirror()));
    }
}
