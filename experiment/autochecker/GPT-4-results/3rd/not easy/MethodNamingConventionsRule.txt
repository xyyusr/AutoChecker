package net.sourceforge.pmd.lang.java.rule.codestyle;


import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class MethodNamingConventionsRule extends AbstractJavaRulechainRule {

    private static final String INSTANCE_METHOD_REGEX = "[a-z][a-zA-Z0-9]*";
    private static final String STATIC_METHOD_REGEX = "[a-z][a-zA-Z0-9]*";
    private static final String NATIVE_METHOD_REGEX = "[a-z][a-zA-Z0-9]*";
    private static final String JUNIT3_METHOD_REGEX = "test[A-Z0-9][a-zA-Z0-9]*";
    private static final String JUNIT4_5_METHOD_REGEX = "[a-z][a-zA-Z0-9]*";

    public MethodNamingConventionsRule() {
        super(ASTMethodDeclaration.class);
    }

    @Override
    public Object visit(ASTMethodDeclaration node, Object data) {
        String methodName = node.getName();

        if (TestFrameworksUtil.isJUnitMethod(node)) {
            if (methodName.matches(JUNIT3_METHOD_REGEX) || methodName.matches(JUNIT4_5_METHOD_REGEX)) {
                return super.visit(node, data);
            }
        } else if (node.isStatic() || node.isSyntacticallyStatic()) {
            if (methodName.matches(STATIC_METHOD_REGEX)) {
                return super.visit(node, data);
            }
        } else if (node.isNative()) {
            if (methodName.matches(NATIVE_METHOD_REGEX)) {
                return super.visit(node, data);
            }
        } else if (methodName.matches(INSTANCE_METHOD_REGEX)) {
            return super.visit(node, data);
        }

        addViolation(data, (Node)node);
        return super.visit(node, data);
    }
}

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class MethodNamingConventionsRule extends AbstractJavaRulechainRule {

    private static final String INSTANCE_METHOD_REGEX = "[a-z][a-zA-Z0-9]*";
    private static final String STATIC_METHOD_REGEX = "[a-z][a-zA-Z0-9]*";
    private static final String NATIVE_METHOD_REGEX = "[a-z][a-zA-Z0-9]*";
    private static final String JUNIT3_TEST_METHOD_REGEX = "test[A-Z0-9][a-zA-Z0-9]*";
    private static final String JUNIT4_TEST_METHOD_REGEX = "[a-z][a-zA-Z0-9]*";
    private static final String JUNIT5_TEST_METHOD_REGEX = "[a-z][a-zA-Z0-9]*";

    public MethodNamingConventionsRule() {
        super(ASTMethodDeclaration.class);
    }

    @Override
    public Object visit(ASTMethodDeclaration node, Object data) {
        String methodName = node.getName();

        if (node.isNative()) {
            if (!methodName.matches(NATIVE_METHOD_REGEX)) {
                addViolation(data, node);
            }
        } else if (node.isStatic()) {
            if (!methodName.matches(STATIC_METHOD_REGEX)) {
                addViolation(data, node);
            }
        } else if (node.getDeclaredAnnotations().anyMatch(a -> a.getName().equals("Test"))) {
            if (!methodName.matches(JUNIT4_TEST_METHOD_REGEX) && !methodName.matches(JUNIT5_TEST_METHOD_REGEX)) {
                addViolation(data, node);
            }
        } else if (methodName.startsWith("test")) {
            if (!methodName.matches(JUNIT3_TEST_METHOD_REGEX)) {
                addViolation(data, node);
            }
        } else {
            if (!methodName.matches(INSTANCE_METHOD_REGEX)) {
                addViolation(data, node);
            }
        }

        return super.visit(node, data);
    }
}

import java.util.regex.Pattern;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;

public class MethodNamingConventionsRule extends AbstractJavaRulechainRule {

    private static final Pattern METHOD_NAME_PATTERN = Pattern.compile("[a-z][a-zA-Z0-9]*");
    private static final Pattern TEST_METHOD_NAME_PATTERN = Pattern.compile("test[A-Z0-9][a-zA-Z0-9]*");

    public MethodNamingConventionsRule() {
        super(ASTMethodDeclaration.class);
    }

    @Override
    public Object visit(ASTMethodDeclaration node, Object data) {
        String methodName = node.getName();

        // Check if it's a JUnit method
        if (TestFrameworksUtil.isJUnitMethod(node)) {
            if (!TEST_METHOD_NAME_PATTERN.matcher(methodName).matches()) {
                addViolation(data, node);
            }
        } else {
            // Check if it's a regular, static or native method
            if (!METHOD_NAME_PATTERN.matcher(methodName).matches()) {
                addViolation(data, node);
            }
        }

        return super.visit(node, data);
    }
}
