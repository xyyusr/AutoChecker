import java.util.*;
import java.lang.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;

import net.sourceforge.pmd.lang.ast.NodeStream;
public class AssignmentToNonFinalStaticRule extends AbstractJavaRulechainRule {
    public AssignmentToNonFinalStaticRule() {
        super(ASTConstructorDeclaration.class, ASTAssignmentExpression.class);
    }

    @Override
    public Object visit(ASTConstructorDeclaration node, Object data) {
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTAssignmentExpression node, Object data) {
        ASTVariableAccess leftHandSide = node.getFirstChildOfType(ASTVariableAccess.class);
        if (leftHandSide != null) {
            String variableName = leftHandSide.getName();
            ASTClassOrInterfaceDeclaration classNode = node.getFirstParentOfType(ASTClassOrInterfaceDeclaration.class);
            if (classNode != null) {
                NodeStream<ASTFieldDeclaration> fields = classNode.getDeclarations(ASTFieldDeclaration.class);
                for (ASTFieldDeclaration field : fields) {
                    ASTVariableDeclaratorId fieldId = field.getFirstDescendantOfType(ASTVariableDeclaratorId.class);
                    if (fieldId != null && fieldId.getName().equals(variableName)) {
                        ASTModifierList modifiers = field.getFirstChildOfType(ASTModifierList.class);
                        if (modifiers != null && !modifiers.hasModifiers(JModifier.FINAL) && modifiers.hasModifiers(JModifier.STATIC)) {
                            addViolation(data, node);
                        }
                    }
                }
            }
        }
        return super.visit(node, data);
    }
}
