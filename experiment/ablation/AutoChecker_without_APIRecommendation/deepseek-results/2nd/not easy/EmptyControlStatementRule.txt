public class EmptyControlStatementRule extends AbstractJavaRulechainRule {
    public EmptyControlStatementRule() {
        super(ASTTryStatement.class, ASTIfStatement.class, ASTWhileStatement.class, ASTForStatement.class, ASTDoStatement.class, ASTSwitchStatement.class, ASTSynchronizedStatement.class, ASTBlock.class);
    }

    @Override
    public Object visit(ASTTryStatement node, Object data) {
        ASTBlock tryBlock = node.getFirstChildOfType(ASTBlock.class);
        if (tryBlock != null && tryBlock.getNumChildren() == 0 && node.getResources().isEmpty()) {
            addViolation(data, tryBlock);
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTIfStatement node, Object data) {
        ASTStatement thenStmt = node.getThenBranch();
        if (thenStmt instanceof ASTBlock && ((ASTBlock) thenStmt).getNumChildren() == 0) {
            addViolation(data, thenStmt);
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTWhileStatement node, Object data) {
        ASTStatement body = node.getBody();
        if (body instanceof ASTBlock && ((ASTBlock) body).getNumChildren() == 0) {
            addViolation(data, body);
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTForStatement node, Object data) {
        ASTStatement body = node.getBody();
        if (body instanceof ASTBlock && ((ASTBlock) body).getNumChildren() == 0) {
            addViolation(data, body);
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTDoStatement node, Object data) {
        ASTStatement body = node.getBody();
        if (body instanceof ASTBlock && ((ASTBlock) body).getNumChildren() == 0) {
            addViolation(data, body);
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTSwitchStatement node, Object data) {
        if (node.getNumChildren() == 0) {
            addViolation(data, node);
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTSynchronizedStatement node, Object data) {
        ASTBlock body = node.getBody();
        if (body != null && body.getNumChildren() == 0) {
            addViolation(data, body);
        }
        return super.visit(node, data);
    }

    @Override

        if (node.getNumChildren() == 0 && node.getParent() instanceof ASTInitializer) {
            addViolation(data, node);
