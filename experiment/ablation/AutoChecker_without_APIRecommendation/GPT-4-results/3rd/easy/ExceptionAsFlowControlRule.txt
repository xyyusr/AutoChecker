import java.util.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;

public class ExceptionAsFlowControlRule extends AbstractJavaRulechainRule {
    private final Map<String, Class> classMap = new HashMap<>();

    public ExceptionAsFlowControlRule() {
        addRuleChainVisit(ASTCatchStatement.class);
    }

    @Override
    public Object visit(ASTCatchStatement catchStatement, Object data) {
        ASTClassOrInterfaceType catchTypeNode = catchStatement.getFirstDescendantOfType(ASTCatchType.class).getFirstDescendantOfType(ASTClassOrInterfaceType.class);
        ASTTryStatement tryStatement = catchStatement.getFirstParentOfType(ASTTryStatement.class);

        if (tryStatement != null) {
            List<ASTThrowStatement> throwStatements = tryStatement.findDescendantsOfType(ASTThrowStatement.class);
            for (ASTThrowStatement throwStatement : throwStatements) {
                ASTClassOrInterfaceType throwTypeNode = throwStatement.getFirstDescendantOfType(ASTClassOrInterfaceType.class);
                if (throwTypeNode != null && isSubclass(throwTypeNode.getImage(), catchTypeNode.getImage())) {
                    addViolation(data, catchStatement);
                }
            }
        }

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ExceptionAsFlowControlRule extends AbstractJavaRulechainRule {

    public ExceptionAsFlowControlRule() {
        super(ASTCatchClause.class, ASTThrowStatement.class, ASTConstructorCall.class);
    }

    @Override
    public Object visit(ASTCatchClause catchClauseNode, Object data) {
        String catchExceptionType = catchClauseNode.getFirstDescendantOfType(ASTCatchParameter.class).getTypeNode().getTypeImage();
        ASTTryStatement innerTryStatement = catchClauseNode.getFirstParentOfType(ASTTryStatement.class);

        if (innerTryStatement != null) {
            ASTThrowStatement throwStatement = innerTryStatement.getFirstDescendantOfType(ASTThrowStatement.class);

            if (throwStatement != null) {
                String throwExceptionType = throwStatement.getFirstDescendantOfType(ASTConstructorCall.class).getTypeNode().getTypeImage();

                // Check if the thrown exception is not inside a conditional control structure
                if (catchExceptionType.equals(throwExceptionType) && throwStatement.getFirstParentOfType(ASTIfStatement.class) == null) {
                    addViolation(data, catchClauseNode);
                }
            }
        }

        return super.visit(catchClauseNode, data);
    }

    @Override
    public Object visit(ASTThrowStatement throwStatementNode, Object data) {
        return super.visit(throwStatementNode, data);
    }

    @Override
    public Object visit(ASTConstructorCall constructorCallNode, Object data) {
        return super.visit(constructorCallNode, data);
    }
}
