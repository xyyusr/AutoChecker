package net.sourceforge.pmd.lang.java.rule.codestyle;


import java.util.HashMap;
import java.util.Map;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class IdenticalCatchBranchesRule extends AbstractJavaRulechainRule {

    public IdenticalCatchBranchesRule() {
        super(ASTCatchClause.class);
    }

    @Override
    public Object visit(ASTCatchClause node, Object data) {
        if (isIdenticalCatchBranch(node)) {
            addViolation(data, node);
        }
        return super.visit(node, data);
    }

    private boolean isIdenticalCatchBranch(ASTCatchClause node) {
        Node parent = node.jjtGetParent();
        Map<String, String> catchClauses = new HashMap<>();

        for (int i = 0; i < parent.jjtGetNumChildren(); i++) {
            Node child = parent.jjtGetChild(i);
            if (child instanceof ASTCatchClause) {
                ASTBlock block = child.getFirstChildOfType(ASTBlock.class);
                ASTThrowStatement throwStatement = block.getFirstChildOfType(ASTThrowStatement.class);
                if (throwStatement != null) {
                    ASTCatchParameter catchParameter = child.getFirstChildOfType(ASTCatchParameter.class);
                    String exceptionType = catchParameter.getFirstChildOfType(ASTCatchParameter.class).getImage();
                    String throwStatementImage = throwStatement.jjtGetChild(0).getImage();
                    if (catchClauses.containsKey(exceptionType) && catchClauses.get(exceptionType).equals(throwStatementImage)) {
                        return true;
                    } else {
                        catchClauses.put(exceptionType, throwStatementImage);
                    }
                }
            }
        }

        return false;
    }
}

import java.util.List;
import java.util.stream.Collectors;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class IdenticalCatchBranchesRule extends AbstractJavaRulechainRule {

    public IdenticalCatchBranchesRule() {
        super(ASTTryStatement.class);
    }

    @Override
    public Object visit(ASTTryStatement node, Object data) {
        List<String> catchBlocks = node.findChildrenOfType(ASTCatchClause.class)
                .stream()
                .map(ASTCatchClause::getImage)
                .collect(Collectors.toList());

        for (int i = 0; i < catchBlocks.size() - 1; i++) {
            for (int j = i + 1; j < catchBlocks.size(); j++) {
                if (catchBlocks.get(i).equals(catchBlocks.get(j))) {
                    addViolation(data, node);
                }
            }
        }

        return super.visit(node, data);
    }
}

import java.util.HashMap;
import java.util.Map;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class IdenticalCatchBranchesRule extends AbstractJavaRulechainRule {
    private Map<String, Node> previousNodes = new HashMap<>();

    public IdenticalCatchBranchesRule() {
        super(ASTCatchClause.class);
    }

    @Override
    public Object visit(ASTCatchClause node, Object data) {
        AbstractJavaNode grandparent = (AbstractJavaNode) node.jjtGetParent().jjtGetParent();
        if (grandparent != null && grandparent.hasDescendantOfType(ASTThrowStatement.class)) {
            String codeBlock = getFirstChildOfType(node, ASTThrowStatement.class).getImage();
            if (previousNodes.containsKey(codeBlock)) {
                addViolation(data, node);
            } else {
                previousNodes.put(codeBlock, node);
            }
        }
        return super.visit(node, data);
    }

    private <T extends Node> T getFirstChildOfType(Node node, Class<T> klass) {
        for (int i = 0; i < node.jjtGetNumChildren(); i++) {
            Node child = node.jjtGetChild(i);
            if (klass.isInstance(child)) {
                return klass.cast(child);
            }
        }
        return null;
    }
}
