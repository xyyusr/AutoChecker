package net.sourceforge.pmd.lang.java.rule.errorprone;

import java.util.HashMap;
import java.util.Map;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class AvoidDuplicateLiteralsRule extends AbstractJavaRulechainRule {

    private static final int THRESHOLD = 4;
    private static final int MINIMUM_LENGTH = 3;
    private Map<String, Integer> stringOccurrences;

    public AvoidDuplicateLiteralsRule() {
        super(ASTCompilationUnit.class, ASTStringLiteral.class);
    }

    @Override
    public Object visit(ASTCompilationUnit node, Object data) {
        stringOccurrences = new HashMap<>();
        super.visit(node, data);
        for (Map.Entry<String, Integer> entry : stringOccurrences.entrySet()) {
            if (entry.getValue() >= THRESHOLD) {
                addViolation(data, node);
                break;
            }
        }
        return data;
    }

    @Override
    public Object visit(ASTStringLiteral node, Object data) {
        String literal = node.getImage();
        if (literal.length() >= MINIMUM_LENGTH) {
            stringOccurrences.put(literal, stringOccurrences.getOrDefault(literal, 0) + 1);
        }
        return super.visit(node, data);
    }
}

package net.sourceforge.pmd.lang.java.rule.errorprone;

import java.util.HashMap;
import java.util.Map;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class AvoidDuplicateLiteralsRule extends AbstractJavaRulechainRule {

    private Map<String, Integer> stringOccurrences;

    public AvoidDuplicateLiteralsRule() {
        super(ASTLiteral.class);
        stringOccurrences = new HashMap<>();
    }

    @Override
    public Object visit(ASTLiteral node, Object data) {
        String literalValue = node.getImage();

        // Ignore string literals length less than 3
        if (literalValue.length() < 3) {
            return super.visit(node, data);
        }

        // Count occurrences of each string literal
        int count = stringOccurrences.containsKey(literalValue) ? stringOccurrences.get(literalValue) : 0;
        stringOccurrences.put(literalValue, count + 1);

        // If a string literal occurs more than 4 times, add a violation
        if (stringOccurrences.get(literalValue) >= 4) {
            addViolation(data, node);
        }

        return super.visit(node, data);
    }
}

import java.util.HashMap;
import java.util.Map;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class AvoidDuplicateLiteralsRule extends AbstractJavaRulechainRule {

    private static final int THRESHOLD = 4;
    private static final int MINIMUM_LENGTH = 3;

    private Map<String, Integer> literalCount = new HashMap<>();

    public AvoidDuplicateLiteralsRule() {
        super(ASTCompilationUnit.class);
    }

    @Override
    public Object visit(ASTCompilationUnit node, Object data) {
        literalCount.clear();
        super.visit(node, data);
        checkForDuplicateLiterals(data);
        return data;
    }

    @Override
    public Object visit(ASTStringLiteral node, Object data) {
        String literal = node.getImage();
        if (literal.length() >= MINIMUM_LENGTH) {
            literalCount.merge(literal, 1, Integer::sum);
        }
        return super.visit(node, data);
    }

    private void checkForDuplicateLiterals(Object data) {
        for (Map.Entry<String, Integer> entry : literalCount.entrySet()) {
            if (entry.getValue() >= THRESHOLD) {
                addViolation(data, (Node) data);
            }
        }
    }
}
