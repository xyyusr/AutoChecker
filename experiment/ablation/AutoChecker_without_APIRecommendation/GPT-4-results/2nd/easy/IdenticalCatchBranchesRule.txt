import java.util.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class IdenticalCatchBranchesRule extends AbstractJavaRulechainRule {

    public IdenticalCatchBranchesRule() {
        super(ASTTryStatement.class);
    }

    @Override
    public Object visit(ASTTryStatement node, Object data) {
        List<ASTCatchClause> catchClauses = node.findChildNodesWithXPath(".//CatchClause");
        for (int i = 0; i < catchClauses.size() - 1; i++) {
            ASTCatchClause firstClause = catchClauses.get(i);
            ASTCatchClause secondClause = catchClauses.get(i + 1);
            if (isIdenticalCatchBranches(firstClause, secondClause)) {
                addViolation(data, secondClause);
            }
        }
        return super.visit(node, data);
    }

    private boolean isIdenticalCatchBranches(ASTCatchClause firstClause, ASTCatchClause secondClause) {
        String firstBlock = firstClause.getFirstDescendantOfType(ASTBlock.class).getImage();
        String secondBlock = secondClause.getFirstDescendantOfType(ASTBlock.class).getImage();
        return firstBlock.equals(secondBlock);
    }
}

package net.sourceforge.pmd.lang.java.rule.codestyle;
import java.util.Map;
import java.util.HashMap;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class IdenticalCatchBranchesRule extends AbstractJavaRulechainRule {
    public IdenticalCatchBranchesRule() {
        super(ASTCatchClause.class);
    }

    @Override
    public Object visit(ASTCatchClause node, Object data) {
        Node parent = node.jjtGetParent();
        Map<String, Integer> catchBlocks = new HashMap<>();

        for (int i = 0; i < parent.jjtGetNumChildren(); i++) {
            Node child = parent.jjtGetChild(i);
            if (child instanceof ASTCatchClause) {
                Node block = child.jjtGetChild(1); // get ASTBlock
                if (block instanceof ASTBlock) {
                    Node throwStatement = block.jjtGetChild(0); // get ASTThrowStatement
                    if (throwStatement instanceof ASTThrowStatement) {
                        String throwStatementCode = throwStatement.jjtGetFirstToken().toString();
                        if (catchBlocks.containsKey(throwStatementCode)) {
                            addViolation(data, node);
                        } else {
                            catchBlocks.put(throwStatementCode, 1);
                        }
                    }
                }
            }
        }

        return super.visit(node, data);
    }
}
