    public AssignmentToNonFinalStaticRule() {
        addRuleChainVisit(ASTConstructorDeclaration.class);
    }

    @Override

        for (int i = 0; i < node.jjtGetNumChildren(); i++) {
            Node childNode = node.jjtGetChild(i);

                    }
                }

public class AssignmentToNonFinalStaticRule extends AbstractJavaRulechainRule {

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class AssignmentToNonFinalStaticRule extends AbstractJavaRulechainRule {

    public AssignmentToNonFinalStaticRule() {
        addRuleChainVisit(ASTConstructorDeclaration.class);
    }

    @Override
    public Object visit(ASTConstructorDeclaration node, Object data) {
        for (int i = 0; i < node.jjtGetNumChildren(); i++) {
            Node childNode = node.jjtGetChild(i);
            if (childNode instanceof ASTAssignmentExpression) {
                checkAssignmentExpression((ASTAssignmentExpression) childNode, data);
            }
        }
        return super.visit(node, data);
    }

    private void checkAssignmentExpression(ASTAssignmentExpression node, Object data) {
        for (int i = 0; i < node.jjtGetNumChildren(); i++) {
            Node childNode = node.jjtGetChild(i);
            if (childNode instanceof ASTVariableAccess) {
                ASTVariableAccess variableAccess = (ASTVariableAccess) childNode;
                if (variableAccess.getImage() != null && isNonFinalStaticField(variableAccess.getImage())) {
                    addViolation(data, node);
                }
            }
        }
    }

    private boolean isNonFinalStaticField(String fieldName) {
        ASTFieldDeclaration fieldDeclaration = findFieldDeclaration(fieldName);
        return fieldDeclaration != null && fieldDeclaration.isStatic() && !fieldDeclaration.isFinal();
    }

    private ASTFieldDeclaration findFieldDeclaration(String fieldName) {
        for (ASTFieldDeclaration fieldDeclaration : getNodesOfType(ASTFieldDeclaration.class)) {
            if (fieldDeclaration.getVariableName().equals(fieldName)) {
                return fieldDeclaration;
            }
        }
        return null;
    }
}
