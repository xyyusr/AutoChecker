import net.sourceforge.pmd.lang.ast.NodeStream;
public class MethodNamingConventionsRule extends AbstractJavaRulechainRule {

    private static final Pattern METHOD_PATTERN = Pattern.compile("[a-z][a-zA-Z0-9]*");
    private static final Pattern TEST_METHOD_PATTERN = Pattern.compile("test[A-Z0-9][a-zA-Z0-9]*");
    private static final Pattern JUNIT4_METHOD_PATTERN = Pattern.compile("[a-z][a-zA-Z0-9]*");
    private static final Pattern JUNIT5_METHOD_PATTERN = Pattern.compile("[a-z][a-zA-Z0-9]*");

    public MethodNamingConventionsRule() {
        super(ASTClassOrInterfaceDeclaration.class, ASTMethodDeclaration.class);
    }

    @Override
    public Object visit(ASTClassOrInterfaceDeclaration node, Object data) {
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTMethodDeclaration node, Object data) {
        ASTMethodDeclarator methodDeclarator = node.getFirstDescendantOfType(ASTMethodDeclarator.class);
        if (methodDeclarator != null) {
            String methodName = methodDeclarator.getImage();

            Pattern patternToUse = METHOD_PATTERN;

            if (isJUnit3TestMethod(node)) {
                patternToUse = TEST_METHOD_PATTERN;
            } else if (isJUnit4TestMethod(node)) {
                patternToUse = JUNIT4_METHOD_PATTERN;
            } else if (isJUnit5TestMethod(node)) {
                patternToUse = JUNIT5_METHOD_PATTERN;
            }

            if (!patternToUse.matcher(methodName).matches()) {
                addViolation(data, methodDeclarator);
            }
        }

        return super.visit(node, data);
    }

    private boolean isJUnit3TestMethod(ASTMethodDeclaration node) {
        return node.getName().startsWith("test") && node.getResultType().isVoid() && node.getArgumentCount() == 0;
    }

    private boolean isJUnit4TestMethod(ASTMethodDeclaration node) {
        for (ASTAnnotation annotation : node.findDescendantsOfType(ASTAnnotation.class)) {
            if ("Test".equals(annotation.getAnnotationName())) {
                return true;
            }
        }
        return false;
    }

    private boolean isJUnit5TestMethod(ASTMethodDeclaration node) {
        for (ASTAnnotation annotation : node.findDescendantsOfType(ASTAnnotation.class)) {
            if ("Test".equals(annotation.getAnnotationName())) {
                return true;
            }
        }
        return false;
    }

package net.sourceforge.pmd.lang.java.rule.codestyle;

import java.util.regex.Pattern;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class MethodNamingConventionsRule extends AbstractJavaRulechainRule {

    private static final Pattern METHOD_PATTERN = Pattern.compile("[a-z][a-zA-Z0-9]*");
    private static final Pattern STATIC_METHOD_PATTERN = Pattern.compile("[a-z][a-zA-Z0-9]*");
    private static final Pattern NATIVE_METHOD_PATTERN = Pattern.compile("[a-z][a-zA-Z0-9]*");
    private static final Pattern JUNIT3_METHOD_PATTERN = Pattern.compile("test[A-Z0-9][a-zA-Z0-9]*");
    private static final Pattern JUNIT4_METHOD_PATTERN = Pattern.compile("[a-z][a-zA-Z0-9]*");
    private static final Pattern JUNIT5_METHOD_PATTERN = Pattern.compile("[a-z][a-zA-Z0-9]*");

    public MethodNamingConventionsRule() {
        super(ASTMethodDeclaration.class);
    }

    @Override
    public Object visit(ASTMethodDeclaration node, Object data) {
        String methodName = node.getName();

        if (node.isStatic() && !STATIC_METHOD_PATTERN.matcher(methodName).matches()) {
            addViolation(data, node);
            return data;
        }

        if (node.isNative() && !NATIVE_METHOD_PATTERN.matcher(methodName).matches()) {
            addViolation(data, node);
            return data;
        }

        if (TestFrameworksUtil.isJUnitMethod(node)) {
            ASTClassOrInterfaceDeclaration parentClassOrInterfaceDeclaration =
                    (ASTClassOrInterfaceDeclaration) node.getFirstParentOfType(ASTClassOrInterfaceDeclaration.class);

            if (parentClassOrInterfaceDeclaration != null) {
                if (parentClassOrInterfaceDeclaration.getSimpleName().startsWith("Test") && 
                    !JUNIT3_METHOD_PATTERN.matcher(methodName).matches()) {
                    addViolation(data, node);
                    return data;
                }

                if (parentClassOrInterfaceDeclaration.getSimpleName().startsWith("TestCase") && 
                    !JUNIT4_METHOD_PATTERN.matcher(methodName).matches()) {
                    addViolation(data, node);
                    return data;
                }

                if (parentClassOrInterfaceDeclaration.getSimpleName().startsWith("TournamentTest") && 
                    !JUNIT5_METHOD_PATTERN.matcher(methodName).matches()) {
                    addViolation(data, node);
                    return data;
                }
            }
        }

        if (!METHOD_PATTERN.matcher(methodName).matches()) {
            addViolation(data, node);
            return data;
        }

        return data;
    }
}

package net.sourceforge.pmd.lang.java.rule.codestyle;


import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class MethodNamingConventionsRule extends AbstractJavaRulechainRule {
    private static final String REGEX_INSTANCE_METHOD = "[a-z][a-zA-Z0-9]*";
    private static final String REGEX_STATIC_METHOD = "[a-z][a-zA-Z0-9]*";
    private static final String REGEX_NATIVE_METHOD = "[a-z][a-zA-Z0-9]*";
    private static final String REGEX_JUNIT3 = "test[A-Z0-9][a-zA-Z0-9]*";
    private static final String REGEX_JUNIT4 = "[a-z][a-zA-Z0-9]*";
    private static final String REGEX_JUNIT5 = "[a-z][a-zA-Z0-9]*";

    public MethodNamingConventionsRule() {
        super(ASTMethodDeclaration.class);
    }

    @Override
    public Object visit(ASTMethodDeclaration node, Object data) {
        String methodName = node.getName();

        if (node.isStatic() && !methodName.matches(REGEX_STATIC_METHOD)) {
            addViolation(data, node);
        } else if (node.isNative() && !methodName.matches(REGEX_NATIVE_METHOD)) {
            addViolation(data, node);
        } else if (TestFrameworksUtil.isJUnitMethod(node)) {
            if (TestFrameworksUtil.isJunit4Test(node) && !methodName.matches(REGEX_JUNIT4)) {
                addViolation(data, node);
            } else if (TestFrameworksUtil.isJunit5Test(node) && !methodName.matches(REGEX_JUNIT5)) {
                addViolation(data, node);
            } else if (!methodName.matches(REGEX_JUNIT3)) {
                addViolation(data, node);
            }
        } else if (!methodName.matches(REGEX_INSTANCE_METHOD)) {
            addViolation(data, node);
        }

        return super.visit(node, data);
    }
}
