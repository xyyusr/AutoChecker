package net.sourceforge.pmd.lang.java.rule.performance;


import java.util.List;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class StringInstantiationRule extends AbstractJavaRulechainRule {

    public StringInstantiationRule() {
        super(ASTPrimaryExpression.class);
    }

    @Override
    public Object visit(ASTPrimaryExpression node, Object data) {
        ASTPrimaryPrefix prefix = node.getFirstDescendantOfType(ASTPrimaryPrefix.class);
        if (prefix != null && prefix.usesThisModifier()) {
            return super.visit(node, data);
        }

        List<ASTPrimarySuffix> suffixes = node.findDescendantsOfType(ASTPrimarySuffix.class);
        for (ASTPrimarySuffix suffix : suffixes) {
            if (suffix.isArguments()) {
                return super.visit(node, data);
            }
        }

        if (isStringConstructor(node) && !isInImportDeclaration(node) && !isInMethodReturningString(node)) {
            addViolation(data, node);
        }

        return super.visit(node, data);
    }

    private boolean isStringConstructor(ASTPrimaryExpression node) {
        ASTConstructorDeclaration constructorDeclaration = node.getFirstParentOfType(ASTConstructorDeclaration.class);
        return constructorDeclaration != null && "String".equals(constructorDeclaration.getName());
    }

    private boolean isInImportDeclaration(ASTPrimaryExpression node) {
        return node.getFirstParentOfType(ASTImportDeclaration.class) != null;
    }

    private boolean isInMethodReturningString(ASTPrimaryExpression node) {
        ASTMethodDeclaration methodDeclaration = node.getFirstParentOfType(ASTMethodDeclaration.class);
        if (methodDeclaration == null) {
            return false;
        }

        ASTClassOrInterfaceDeclaration classOrInterfaceDeclaration = methodDeclaration.getFirstParentOfType(ASTClassOrInterfaceDeclaration.class);
        if (classOrInterfaceDeclaration == null) {
            return false;
        }

        String className = classOrInterfaceDeclaration.getName();
        ClassScope classScope = (ClassScope) classOrInterfaceDeclaration.getScope();

        return "String".equals(methodDeclaration.getResultType().getTypeImage())
                && className.equals(classScope.getEnclosingClassScope().getClassName());
    }
}

import java.lang.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;

public class StringInstantiationRule extends AbstractJavaRulechainRule {
    public StringInstantiationRule() {
        super(ASTConstructorCall.class);
    }
    
    @Override
    public Object visit(ASTConstructorCall node, Object data) {
        if (isStringConstructor(node)) {
            addViolation(data, node);
        }
        return super.visit(node, data);
    }
    
    private boolean isStringConstructor(ASTConstructorCall node) {
        ASTClassOrInterfaceType typeNode = node.getTypeNode();
        if (typeNode.getType() == String.class) {
            if (node.getArgumentCount() == 0) {
                return true;
            } else if (node.getArgumentCount() == 1) {
                ASTExpression firstArg = node.getArgument(0);
                if (firstArg.getType() == String.class || firstArg.getType() == byte[].class) {
                    return true;
                }
            }
        }
        return false;
    }
}
