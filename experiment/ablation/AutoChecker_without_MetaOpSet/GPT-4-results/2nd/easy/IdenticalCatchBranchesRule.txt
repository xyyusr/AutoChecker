    public IdenticalCatchBranchesRule() {
        super(ASTCatchClause.class);
    }
    @Override
    public Object visit(ASTCatchClause node, Object data) {
        ASTBlockStatement block = node.getFirstChildOfType(ASTBlockStatement.class);
        if (block != null) {
            List<ASTThrowStatement> throwStatements = block.findDescendantsOfType(ASTThrowStatement.class);
            if (throwStatements.size() == 1) {
                ASTCatchParameter catchParameter = node.getFirstChildOfType(ASTCatchParameter.class);
                if (catchParameter != null) {
                    List<ASTCatchClause> allCatchClauses = node.getFirstParentOfType(ASTTryStatement.class).findChildrenOfType(ASTCatchClause.class);
                    for (ASTCatchClause catchClause : allCatchClauses) {
                        if (catchClause != node && catchClause.getCatchParameter().getAllExceptionTypes().equals(catchParameter.getAllExceptionTypes())) {
                            addViolation(data, node);
                            break;
                        }
                    }
                }
            }
        }
        return super.visit(node, data);
    }
}

package net.sourceforge.pmd.lang.java.rule.codestyle;


import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class IdenticalCatchBranchesRule extends AbstractJavaRulechainRule {

    public IdenticalCatchBranchesRule() {
        super(ASTCompilationUnit.class);
    }

    @Override
    public Object visit(ASTCompilationUnit node, Object data) {
        super.visit(node, data);
        node.findDescendantsOfType(ASTClassOrInterfaceDeclaration.class)
            .forEach(classDeclaration -> classDeclaration.findDescendantsOfType(ASTInitializer.class)
                .forEach(initializer -> initializer.findDescendantsOfType(ASTTryStatement.class)
                    .forEach(tryStatement -> checkIdenticalCatchBranches(tryStatement))));
        return data;
    }

    private void checkIdenticalCatchBranches(ASTTryStatement tryStatement) {
        ASTBlock tryBlock = tryStatement.getFirstChildOfType(ASTBlock.class);
        tryStatement.findDescendantsOfType(ASTCatchClause.class)
            .stream()
            .reduce((first, second) -> {
                if (first.equals(second)) {
                    tryBlock.jjtAddChild(second.getFirstChild(), tryBlock.jjtGetNumChildren());
                    second.jjtGetParent().jjtRemoveChild(second);
                }
                return second;
            });
    }
}

package net.sourceforge.pmd.lang.java.rule.codestyle;
import java.util.*;
import java.lang.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class IdenticalCatchBranchesRule extends AbstractJavaRulechainRule {
    public IdenticalCatchBranchesRule() {
        super(ASTTryStatement.class);
    }

    @Override
    public Object visit(ASTTryStatement node, Object data) {
        NodeStream<ASTCatchClause> catchClauses = node.getCatchClauses();
        Set<String> exceptionTypes = new HashSet<>();
        
        for (ASTCatchClause catchClause : catchClauses) {
            NodeStream<ASTClassOrInterfaceType> exceptionTypeStream = catchClause.getAllExceptionTypes();
            StringBuilder exceptionTypeString = new StringBuilder();
            for (ASTClassOrInterfaceType exceptionType : exceptionTypeStream) {
                exceptionTypeString.append(exceptionType.getTypeImage()).append(" ");
            }
            if (!exceptionTypes.add(exceptionTypeString.toString())) {
                addViolation(data, catchClause);
            }
        }
        
        return super.visit(node, data);
    }
}
