import java.lang.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;

import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ExcessivePublicCountRule extends AbstractJavaRulechainRule {
    public ExcessivePublicCountRule() {
        super(ASTFieldDeclaration.class);
    }

    @Override
    public Object visit(ASTFieldDeclaration node, Object data) {
        int publicCount = 0;
        ASTModifierList modifierList = node.getFirstChildOfType(ASTModifierList.class);
        if (modifierList != null && modifierList.jjtGetNumChildren() > 0) {
            for (int i = 0; i < modifierList.jjtGetNumChildren(); i++) {
                JavaNode child = (JavaNode) modifierList.jjtGetChild(i);
                if (child instanceof ASTPublicModifier) {
                    publicCount++;
                }
            }
        }
        if (publicCount > 2) {
            addViolation(data, node);
        }
        return super.visit(node, data);
    }
}

    public Object visit(ASTClassOrInterfaceDeclaration node, Object data) {
        int publicMethodCount = 0;
        
        for (ASTMethodDeclaration method : node.findDescendantsOfType(ASTMethodDeclaration.class)) {

        }
