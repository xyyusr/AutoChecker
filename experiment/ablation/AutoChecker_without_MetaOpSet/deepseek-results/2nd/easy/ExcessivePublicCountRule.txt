import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ExcessivePublicCountRule extends AbstractJavaRulechainRule {
    private static final int THRESHOLD = 2;

    public ExcessivePublicCountRule() {
        super(ASTClassOrInterfaceDeclaration.class);
    }

    @Override
    public Object visit(ASTClassOrInterfaceDeclaration node, Object data) {
        int publicFieldCount = node.descendants(ASTFieldDeclaration.class)
            .filter(field -> field.getModifiers().isPublic())
            .count();

        if (publicFieldCount > THRESHOLD) {
            addViolation(data, node);
        }

        return super.visit(node, data);
    }
}

public class ExcessivePublicCountRule extends AbstractJavaRulechainRule {
    private static final int DEFAULT_THRESHOLD = 2;

    public ExcessivePublicCountRule() {
        super(ASTClassOrInterfaceDeclaration.class);
    }

    @Override
    public Object visit(ASTClassOrInterfaceDeclaration node, Object data) {
        int publicCount = 0;

        NodeStream<ASTMethodDeclaration> methods = node.getFirstChildOfType(ASTClassOrInterfaceBody.class)
                                                      .children(ASTMethodDeclaration.class);
        for (ASTMethodDeclaration method : methods) {
            if (method.getModifiers().isPublic()) {
                publicCount++;
            }
        }

        NodeStream<ASTFieldDeclaration> fields = node.getFirstChildOfType(ASTClassOrInterfaceBody.class)
                                                     .children(ASTFieldDeclaration.class);
        for (ASTFieldDeclaration field : fields) {
            if (field.getModifiers().isPublic()) {
                publicCount++;
            }
        }

        if (publicCount > DEFAULT_THRESHOLD) {
            addViolation(data, node);
        }

        return super.visit(node, data);
    }
}
