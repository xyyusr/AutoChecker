package net.sourceforge.pmd.lang.java.rule.errorprone;

import java.util.*;
import java.lang.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class BrokenNullCheckRule extends AbstractJavaRulechainRule {
    public BrokenNullCheckRule() {
        super(ASTIfStatement.class);
    }

    @Override
    public Object visit(ASTIfStatement node, Object data) {
        ASTExpression condition = node.getCondition();
        if (condition instanceof ASTBinaryExpression) {
            ASTBinaryExpression binaryExpression = (ASTBinaryExpression) condition;
            if (binaryExpression.getOperator() == JavaTokenKind.AND || binaryExpression.getOperator() == JavaTokenKind.OR) {
                checkBinaryExpression(binaryExpression, data);
            }
        }
        return super.visit(node, data);
    }

    private void checkBinaryExpression(ASTBinaryExpression binaryExpression, Object data) {
        ASTExpression left = binaryExpression.getLeftOperand();
        ASTExpression right = binaryExpression.getRightOperand();

        if (isNullCheck(left) && isPotentialNPE(right)) {
            if (binaryExpression.getOperator() == JavaTokenKind.AND) {
                addViolation(data, binaryExpression);
            }
        } else if (isNullCheck(right) && isPotentialNPE(left)) {
            if (binaryExpression.getOperator() == JavaTokenKind.AND) {
                addViolation(data, binaryExpression);
            }
        } else if (isNotNullCheck(left) && isPotentialNPE(right)) {
            if (binaryExpression.getOperator() == JavaTokenKind.OR) {
                addViolation(data, binaryExpression);
            }
        } else if (isNotNullCheck(right) && isPotentialNPE(left)) {
            if (binaryExpression.getOperator() == JavaTokenKind.OR) {
                addViolation(data, binaryExpression);
            }
        }
    }

    private boolean isNullCheck(ASTExpression expression) {
        if (expression instanceof ASTEqualityExpression) {
            ASTEqualityExpression equalityExpression = (ASTEqualityExpression) expression;
            return equalityExpression.getOperator() == JavaTokenKind.EQ && (isNullLiteral(equalityExpression.getLeftOperand()) || isNullLiteral(equalityExpression.getRightOperand()));
        }
        return false;
    }

    private boolean isNotNullCheck(ASTExpression expression) {
        if (expression instanceof ASTEqualityExpression) {
            ASTEqualityExpression equalityExpression = (ASTEqualityExpression) expression;
            return equalityExpression.getOperator() == JavaTokenKind.NE && (isNullLiteral(equalityExpression.getLeftOperand()) || isNullLiteral(equalityExpression.getRightOperand()));
        }
        return false;
    }

    private boolean isNullLiteral(ASTExpression expression) {
        return expression instanceof ASTNullLiteral;
    }

    private boolean isPotentialNPE(ASTExpression expression) {
        if (expression instanceof ASTMethodCall) {
            return true;
        } else if (expression instanceof ASTFieldAccess) {
            return true;
        } else if (expression instanceof ASTArrayAccess) {
            return true;
        } else if (expression instanceof ASTUnaryExpression) {
            ASTUnaryExpression unaryExpression = (ASTUnaryExpression) expression;
            return unaryExpression.getOperator() == JavaTokenKind.BANG && isPotentialNPE(unaryExpression.getOperand());
        } else if (expression instanceof ASTBinaryExpression) {
            ASTBinaryExpression binaryExpression = (ASTBinaryExpression) expression;
            return isPotentialNPE(binaryExpression.getLeftOperand()) || isPotentialNPE(binaryExpression.getRightOperand());
        } else if (expression instanceof ASTConditionalExpression) {
            ASTConditionalExpression conditionalExpression = (ASTConditionalExpression) expression;
            return isPotentialNPE(conditionalExpression.getTrueExpression()) || isPotentialNPE(conditionalExpression.getFalseExpression());
        } else if (expression instanceof ASTCastExpression) {
            ASTCastExpression castExpression = (ASTCastExpression) expression;
            return isPotentialNPE(castExpression.getOperand());
        } else if (expression instanceof ASTInstanceOfExpression) {
            ASTInstanceOfExpression instanceOfExpression = (ASTInstanceOfExpression) expression;
            return isPotentialNPE(instanceOfExpression.getLeftOperand());
        } else if (expression instanceof ASTAssignmentExpression) {
            ASTAssignmentExpression assignmentExpression = (ASTAssignmentExpression) expression;

public class BrokenNullCheckRule extends AbstractJavaRulechainRule {
    public BrokenNullCheckRule() {
        super(ASTIfStatement.class);
    }

    @Override
    public Object visit(ASTIfStatement node, Object data) {
        ASTExpression condition = node.getCondition();
        if (condition instanceof ASTInfixExpression) {
            ASTInfixExpression infix = (ASTInfixExpression) condition;
            if (infix.getOperator() == JavaTokenKind.AND || infix.getOperator() == JavaTokenKind.OR) {
                ASTExpression left = infix.getLeftOperand();
                ASTExpression right = infix.getRightOperand();
                if (isNullCheck(left) && isPotentialNPE(right)) {
                    if (infix.getOperator() == JavaTokenKind.AND && isNullCheck(left) && isPotentialNPE(right)) {
                        addViolation(data, infix);
                    } else if (infix.getOperator() == JavaTokenKind.OR && isNotNullCheck(left) && isPotentialNPE(right)) {
                        addViolation(data, infix);
                    }
                }
            }
        }
        return super.visit(node, data);
    }

    private boolean isNullCheck(ASTExpression expr) {
        if (expr instanceof ASTEqualityExpression) {
            ASTEqualityExpression equality = (ASTEqualityExpression) expr;
            return (equality.getOperator() == JavaTokenKind.EQ && equality.getRightOperand() instanceof ASTNullLiteral) ||
                   (equality.getOperator() == JavaTokenKind.EQ && equality.getLeftOperand() instanceof ASTNullLiteral);
        }
        return false;
    }

    private boolean isNotNullCheck(ASTExpression expr) {
        if (expr instanceof ASTEqualityExpression) {
            ASTEqualityExpression equality = (ASTEqualityExpression) expr;
            return (equality.getOperator() == JavaTokenKind.NE && equality.getRightOperand() instanceof ASTNullLiteral) ||
                   (equality.getOperator() == JavaTokenKind.NE && equality.getLeftOperand() instanceof ASTNullLiteral);
        }
        return false;
    }

    private boolean isPotentialNPE(ASTExpression expr) {
        if (expr instanceof ASTMethodCall || expr instanceof ASTFieldAccess || expr instanceof ASTArrayAccess || expr instanceof ASTUnaryExpression) {
            return true;
        }
        if (expr instanceof ASTInfixExpression) {
            ASTInfixExpression infix = (ASTInfixExpression) expr;
            return isPotentialNPE(infix.getLeftOperand()) || isPotentialNPE(infix.getRightOperand());
        }
        return false;
    }
}

public class BrokenNullCheckRule extends AbstractJavaRulechainRule {
    public BrokenNullCheckRule() {
        super(ASTIfStatement.class);
    }

    @Override
    public Object visit(ASTIfStatement node, Object data) {
        ASTExpression condition = node.getCondition();
        if (condition instanceof ASTInfixExpression) {
            ASTInfixExpression infix = (ASTInfixExpression) condition;
            if (infix.getOperator() == JavaTokenKind.AND || infix.getOperator() == JavaTokenKind.OR) {
                checkNullCheck(infix, data);
            }
        }
        return super.visit(node, data);
    }

    private void checkNullCheck(ASTInfixExpression infix, Object data) {
        ASTExpression left = infix.getLeftOperand();
        ASTExpression right = infix.getRightOperand();

        if (isNullCheck(left) && isPotentialNPE(right)) {
            if (infix.getOperator() == JavaTokenKind.AND) {
                addViolation(data, infix);
            }
        } else if (isNullCheck(right) && isPotentialNPE(left)) {
            if (infix.getOperator() == JavaTokenKind.AND) {
                addViolation(data, infix);
            }
        } else if (isNotNullCheck(left) && isPotentialNPE(right)) {
            if (infix.getOperator() == JavaTokenKind.OR) {
                addViolation(data, infix);
            }
        } else if (isNotNullCheck(right) && isPotentialNPE(left)) {
            if (infix.getOperator() == JavaTokenKind.OR) {
                addViolation(data, infix);
            }
        }
    }

    private boolean isNullCheck(ASTExpression expr) {
        if (expr instanceof ASTEqualityExpression) {
            ASTEqualityExpression equality = (ASTEqualityExpression) expr;
            return equality.getOperator() == JavaTokenKind.EQ && (isNullLiteral(equality.getLeftOperand()) || isNullLiteral(equality.getRightOperand()));
        }
        return false;
    }

    private boolean isNotNullCheck(ASTExpression expr) {
        if (expr instanceof ASTEqualityExpression) {
            ASTEqualityExpression equality = (ASTEqualityExpression) expr;
            return equality.getOperator() == JavaTokenKind.NE && (isNullLiteral(equality.getLeftOperand()) || isNullLiteral(equality.getRightOperand()));
        }
        return false;
    }

    private boolean isNullLiteral(ASTExpression expr) {
        return expr instanceof ASTLiteral && ((ASTLiteral) expr).isNull();
    }

    private boolean isPotentialNPE(ASTExpression expr) {
        if (expr instanceof ASTMethodCall || expr instanceof ASTFieldAccess || expr instanceof ASTArrayAccess || expr instanceof ASTConstructorCall || expr instanceof ASTMethodReference) {
            return true;
        }
        if (expr instanceof ASTInfixExpression) {
            ASTInfixExpression infix = (ASTInfixExpression) expr;
            return isPotentialNPE(infix.getLeftOperand()) || isPotentialNPE(infix.getRightOperand());
        }
        if (expr instanceof ASTUnaryExpression) {
            ASTUnaryExpression unary = (ASTUnaryExpression) expr;
            return isPotentialNPE(unary.getOperand());
        }
        return false;
    }

package net.sourceforge.pmd.lang.java.rule.errorprone;

import java.util.*;
import java.lang.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class BrokenNullCheckRule extends AbstractJavaRulechainRule {
    public BrokenNullCheckRule() {
        super(ASTIfStatement.class);
    }

    @Override
    public Object visit(ASTIfStatement node, Object data) {
        ASTExpression condition = node.getCondition();
        if (condition instanceof ASTBinaryExpression) {
            ASTBinaryExpression binaryExpr = (ASTBinaryExpression) condition;
            if (binaryExpr.getOperator() == JavaTokenKind.AND || binaryExpr.getOperator() == JavaTokenKind.OR) {
                checkBrokenNullCheck(binaryExpr, data);
            }
        }
        return super.visit(node, data);
    }

    private void checkBrokenNullCheck(ASTBinaryExpression binaryExpr, Object data) {
        ASTExpression left = binaryExpr.getLeftOperand();
        ASTExpression right = binaryExpr.getRightOperand();

        if (isNullCheck(left) && isPotentialNPE(right)) {
            if (binaryExpr.getOperator() == JavaTokenKind.AND) {
                addViolation(data, binaryExpr);
            }
        } else if (isNullCheck(right) && isPotentialNPE(left)) {
            if (binaryExpr.getOperator() == JavaTokenKind.AND) {
                addViolation(data, binaryExpr);
            }
        } else if (isNotNullCheck(left) && isPotentialNPE(right)) {
            if (binaryExpr.getOperator() == JavaTokenKind.OR) {
                addViolation(data, binaryExpr);
            }
        } else if (isNotNullCheck(right) && isPotentialNPE(left)) {
            if (binaryExpr.getOperator() == JavaTokenKind.OR) {
                addViolation(data, binaryExpr);
            }
        }
    }

    private boolean isNullCheck(ASTExpression expr) {
        if (expr instanceof ASTEqualityExpression) {
            ASTEqualityExpression equalityExpr = (ASTEqualityExpression) expr;
            return (equalityExpr.getOperator() == JavaTokenKind.EQ && equalityExpr.getRightOperand() instanceof ASTNullLiteral) ||
                   (equalityExpr.getOperator() == JavaTokenKind.EQ && equalityExpr.getLeftOperand() instanceof ASTNullLiteral);
        }
        return false;
    }

    private boolean isNotNullCheck(ASTExpression expr) {
        if (expr instanceof ASTEqualityExpression) {
            ASTEqualityExpression equalityExpr = (ASTEqualityExpression) expr;
            return (equalityExpr.getOperator() == JavaTokenKind.NE && equalityExpr.getRightOperand() instanceof ASTNullLiteral) ||
                   (equalityExpr.getOperator() == JavaTokenKind.NE && equalityExpr.getLeftOperand() instanceof ASTNullLiteral);
        }
        return false;
    }

    private boolean isPotentialNPE(ASTExpression expr) {
        if (expr instanceof ASTMethodCall || expr instanceof ASTFieldAccess || expr instanceof ASTArrayAccess || expr instanceof ASTConstructorCall) {
            return true;
        }
        if (expr instanceof ASTUnaryExpression) {
            ASTUnaryExpression unaryExpr = (ASTUnaryExpression) expr;
            return unaryExpr.getOperator() == JavaTokenKind.BANG && isPotentialNPE(unaryExpr.getOperand());
        }
        if (expr instanceof ASTBinaryExpression) {
            ASTBinaryExpression binaryExpr = (ASTBinaryExpression) expr;
            return isPotentialNPE(binaryExpr.getLeftOperand()) || isPotentialNPE(binaryExpr.getRightOperand());
        }
        return false;
    }
