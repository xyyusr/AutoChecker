package net.sourceforge.pmd.lang.java.rule.performance;

import java.util.*;
import java.lang.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class AvoidInstantiatingObjectsInLoopsRule extends AbstractJavaRulechainRule {
    public AvoidInstantiatingObjectsInLoopsRule() {
        super(ASTForStatement.class, ASTWhileStatement.class, ASTDoStatement.class);
    }

    @Override
    public Object visit(ASTForStatement node, Object data) {
        checkForNewObjects(node.getBody(), data);
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTWhileStatement node, Object data) {
        checkForNewObjects(node.getBody(), data);
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTDoStatement node, Object data) {
        checkForNewObjects(node.getBody(), data);
        return super.visit(node, data);
    }

    private void checkForNewObjects(ASTBlock node, Object data) {
        if (node == null) {
            return;
        }

        NodeStream<ASTAllocationExpression> allocations = node.descendants(ASTAllocationExpression.class);
        for (ASTAllocationExpression allocation : allocations) {
            if (!isException(allocation)) {
                addViolation(data, allocation);
            }
        }
    }

    private boolean isException(ASTAllocationExpression allocation) {
        JavaNode parent = allocation.getParent();
        if (parent instanceof ASTThrowStatement || parent instanceof ASTReturnStatement) {
            return true;
        }

        if (parent instanceof ASTAssignmentExpression) {
            JavaNode grandParent = parent.getParent();
            if (grandParent instanceof ASTArrayDimsAndInits || grandParent instanceof ASTArrayInitializer) {
                return true;
            }
        }

        if (parent instanceof ASTArgumentList) {
            return true;
        }

        if (parent instanceof ASTArrayDimsAndInits) {
            return true;
        }

        if (parent instanceof ASTArrayInitializer) {
            return true;
        }

        if (parent instanceof ASTFieldDeclaration) {
            return true;
        }

        if (parent instanceof ASTVariableDeclarator) {
            JavaNode grandParent = parent.getParent();
            if (grandParent instanceof ASTFieldDeclaration) {
                return true;
            }
        }

        return false;
    }
}

            return;

                addViolation(data, allocation);
            }

        if (parent instanceof ASTThrowStatement || parent instanceof ASTReturnStatement) {
            return true;
        }

        if (parent instanceof ASTAssignmentExpression) {
            JavaNode grandParent = parent.getParent();
            if (grandParent instanceof ASTArrayDimsAndInits || grandParent instanceof ASTArrayInitializer) {
                return true;

            return true;

                return true;
            }

public class AvoidInstantiatingObjectsInLoopsRule extends AbstractJavaRulechainRule {
    public AvoidInstantiatingObjectsInLoopsRule() {
        super(ASTForStatement.class, ASTWhileStatement.class, ASTDoStatement.class);
    }

    @Override

        checkForNewObjects(node.getBody(), data);
        return super.visit(node, data);

    @Override
    public Object visit(ASTWhileStatement node, Object data) {
        checkForNewObjects(node.getBody(), data);
        return super.visit(node, data);

    @Override
    public Object visit(ASTDoStatement node, Object data) {
        checkForNewObjects(node.getBody(), data);
        return super.visit(node, data);

    private void checkForNewObjects(ASTBlock node, Object data) {
        if (node == null) {
            return;
        }

        NodeStream<ASTAllocationExpression> allocations = node.descendants(ASTAllocationExpression.class);
        for (ASTAllocationExpression allocation : allocations) {
            if (!isException(allocation)) {
                addViolation(data, allocation);
            }
        }
    }

    private boolean isException(ASTAllocationExpression allocation) {
        JavaNode parent = allocation.getParent();
        if (parent instanceof ASTThrowStatement || parent instanceof ASTReturnStatement) {
            return true;
        }

        if (parent instanceof ASTAssignmentExpression) {
            ASTAssignmentExpression assignment = (ASTAssignmentExpression) parent;
            if (assignment.getLeftOperand() instanceof ASTArrayAccess || assignment.getLeftOperand() instanceof ASTFieldAccess) {
                return true;
            }

            ASTMethodCall methodCall = (ASTMethodCall) parent;
            if (methodCall.getMethodName().equals("add") || methodCall.getMethodName().equals("put")) {
                return true;
            }
        }

        if (parent instanceof ASTArrayInitializer) {
            return true;
        }

        if (parent instanceof ASTArrayAllocation) {
            return true;
        }

        if (parent instanceof ASTMethodReference) {
            return true;
        }

        return false;
    }
