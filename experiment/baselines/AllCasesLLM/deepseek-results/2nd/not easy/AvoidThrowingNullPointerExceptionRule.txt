import java.lang.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class AvoidThrowingNullPointerExceptionRule extends AbstractJavaRulechainRule {
    public AvoidThrowingNullPointerExceptionRule() {
        super(ASTThrowStatement.class, ASTVariableDeclarator.class, ASTAssignmentExpression.class);
    }

    @Override
    public Object visit(ASTThrowStatement node, Object data) {
        ASTExpression thrownExpression = node.getExpr();
        if (thrownExpression instanceof ASTVariableAccess) {
            ASTVariableAccess variableAccess = (ASTVariableAccess) thrownExpression;
            String variableName = variableAccess.getName();
            ASTVariableDeclarator variableDeclarator = findVariableDeclarator(variableName, node);
            if (variableDeclarator != null) {
                ASTExpression initializer = variableDeclarator.getInitializer();
                if (initializer != null && initializer instanceof ASTConstructorCall) {
                    ASTConstructorCall constructorCall = (ASTConstructorCall) initializer;
                    if (constructorCall.getTypeMirror().isSubtypeOf(NullPointerException.class)) {
                        addViolation(data, node);
                    }
                }
            }
        } else if (thrownExpression instanceof ASTConstructorCall) {
            ASTConstructorCall constructorCall = (ASTConstructorCall) thrownExpression;
            if (constructorCall.getTypeMirror().isSubtypeOf(NullPointerException.class)) {
                addViolation(data, node);
            }
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTAssignmentExpression node, Object data) {
        ASTVariableAccess variableAccess = node.getLeftOperand();
        if (variableAccess != null) {
            String variableName = variableAccess.getName();
            ASTVariableDeclarator variableDeclarator = findVariableDeclarator(variableName, node);
            if (variableDeclarator != null) {
                ASTExpression initializer = variableDeclarator.getInitializer();
                if (initializer != null && initializer instanceof ASTConstructorCall) {
                    ASTConstructorCall constructorCall = (ASTConstructorCall) initializer;
                    if (constructorCall.getTypeMirror().isSubtypeOf(NullPointerException.class)) {
                        addViolation(data, node);
                    }
                }
            }
        }
        return super.visit(node, data);
    }

    private ASTVariableDeclarator findVariableDeclarator(String variableName, JavaNode context) {
        NodeStream<ASTVariableDeclarator> variableDeclarators = context.ancestors().filter(ASTVariableDeclarator.class);
        for (ASTVariableDeclarator declarator : variableDeclarators) {
            if (declarator.getName().equals(variableName)) {
                return declarator;
            }
        }
        return null;
    }
}

package net.sourceforge.pmd.lang.java.rule.design;

import java.util.*;
import java.lang.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class AvoidThrowingNullPointerExceptionRule extends AbstractJavaRulechainRule {
    public AvoidThrowingNullPointerExceptionRule() {
        super(ASTThrowStatement.class, ASTVariableDeclarator.class, ASTAssignmentExpression.class);
    }

    @Override
    public Object visit(ASTThrowStatement node, Object data) {
        ASTExpression thrownExpression = node.getExpr();
        if (thrownExpression instanceof ASTVariableAccess) {
            ASTVariableAccess variableAccess = (ASTVariableAccess) thrownExpression;
            ASTVariableDeclarator declarator = variableAccess.getLocalVariable();
            if (declarator != null && isNullPointerException(declarator.getInitializer())) {
                addViolation(data, node);
            }
        } else if (thrownExpression instanceof ASTAllocationExpression) {
            ASTAllocationExpression allocation = (ASTAllocationExpression) thrownExpression;
            if (isNullPointerException(allocation)) {
                addViolation(data, node);
            }
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTVariableDeclarator node, Object data) {
        ASTExpression initializer = node.getInitializer();
        if (initializer != null && isNullPointerException(initializer)) {
            node.setUserData("isNPE", true);
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTAssignmentExpression node, Object data) {
        ASTExpression rightHandSide = node.getRightHandSide();
        if (rightHandSide != null && isNullPointerException(rightHandSide)) {
            node.setUserData("isNPE", true);
        } else if (rightHandSide != null && !isNullPointerException(rightHandSide)) {
            node.setUserData("isNPE", false);
        }
        return super.visit(node, data);
    }

    private boolean isNullPointerException(ASTExpression expression) {
        if (expression instanceof ASTAllocationExpression) {
            ASTAllocationExpression allocation = (ASTAllocationExpression) expression;
            ASTClassOrInterfaceType type = allocation.getTypeNode();
            return type != null && "NullPointerException".equals(type.getSimpleName());
        }
        return false;
    }
}

package net.sourceforge.pmd.lang.java.rule.design;

import java.util.*;
import java.lang.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class AvoidThrowingNullPointerExceptionRule extends AbstractJavaRulechainRule {

    public AvoidThrowingNullPointerExceptionRule() {
        super(ASTThrowStatement.class, ASTVariableDeclarator.class, ASTAssignmentExpression.class);
    }

    @Override
    public Object visit(ASTThrowStatement node, Object data) {
        ASTExpression thrownExpression = node.getExpr();
        if (thrownExpression instanceof ASTVariableAccess) {
            ASTVariableAccess variableAccess = (ASTVariableAccess) thrownExpression;
            String variableName = variableAccess.getName();
            List<ASTVariableDeclarator> declarations = node.getEnclosingType().descendants(ASTVariableDeclarator.class)
                    .filter(decl -> decl.getName().equals(variableName))
                    .toList();
            for (ASTVariableDeclarator decl : declarations) {
                ASTExpression initializer = decl.getInitializer();
                if (initializer != null && isNullPointerException(initializer)) {
                    addViolation(data, node);
                    break;
                }
            }
        } else if (thrownExpression instanceof ASTAllocationExpression) {
            ASTAllocationExpression allocation = (ASTAllocationExpression) thrownExpression;
            if (isNullPointerException(allocation)) {
                addViolation(data, node);
            }
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTAssignmentExpression node, Object data) {
        ASTExpression rightHandSide = node.getRightHandSide();
        if (isNullPointerException(rightHandSide)) {
            String variableName = node.getLeftHandSide().getVarId().getName();
            NodeStream<ASTThrowStatement> throwStatements = node.getEnclosingType().descendants(ASTThrowStatement.class)
                    .filter(throwStmt -> {
                        ASTExpression expr = throwStmt.getExpr();
                        return expr instanceof ASTVariableAccess && ((ASTVariableAccess) expr).getName().equals(variableName);
                    });
            if (!throwStatements.isEmpty()) {
                addViolation(data, throwStatements.first());
            }
        }
        return super.visit(node, data);
    }

    private boolean isNullPointerException(ASTExpression expression) {
        if (expression instanceof ASTAllocationExpression) {
            ASTAllocationExpression allocation = (ASTAllocationExpression) expression;
            ASTClassOrInterfaceType type = allocation.getTypeNode();
            return type != null && "NullPointerException".equals(type.getSimpleName());
        }
        return false;
    }
}

package net.sourceforge.pmd.lang.java.rule.design;

import java.util.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class AvoidThrowingNullPointerExceptionRule extends AbstractJavaRulechainRule {
    public AvoidThrowingNullPointerExceptionRule() {
        super(ASTThrowStatement.class, ASTVariableDeclarator.class, ASTAssignmentExpression.class);
    }

    @Override
    public Object visit(ASTThrowStatement node, Object data) {
        ASTExpression thrownExpression = node.getExpr();
        if (thrownExpression instanceof ASTVariableAccess) {
            ASTVariableAccess varAccess = (ASTVariableAccess) thrownExpression;
            String varName = varAccess.getName();
            List<ASTVariableDeclarator> varDeclarators = node.getParents().toStream()
                    .filter(ASTVariableDeclarator.class::isInstance)
                    .map(ASTVariableDeclarator.class::cast)
                    .filter(v -> v.getName().equals(varName))
                    .toList();

            for (ASTVariableDeclarator varDecl : varDeclarators) {
                ASTExpression initializer = varDecl.getInitializer();
                if (initializer != null && isNullPointerException(initializer)) {
                    addViolation(data, node);
                    break;
                }
            }

            List<ASTAssignmentExpression> assignments = node.getParents().toStream()
                    .filter(ASTAssignmentExpression.class::isInstance)
                    .map(ASTAssignmentExpression.class::cast)
                    .filter(a -> a.getLeftOperand() instanceof ASTVariableAccess)
                    .filter(a -> ((ASTVariableAccess) a.getLeftOperand()).getName().equals(varName))
                    .toList();

            for (ASTAssignmentExpression assignment : assignments) {
                if (isNullPointerException(assignment.getRightOperand())) {
                    addViolation(data, node);
                    break;
                }
            }
        } else if (thrownExpression instanceof ASTConstructorCall) {
            ASTConstructorCall constructorCall = (ASTConstructorCall) thrownExpression;
            if (isNullPointerException(constructorCall)) {
                addViolation(data, node);
            }
        }
        return super.visit(node, data);
    }

    private boolean isNullPointerException(ASTExpression expression) {
        if (expression instanceof ASTConstructorCall) {
            ASTConstructorCall constructorCall = (ASTConstructorCall) expression;
            return constructorCall.getTypeMirror().getSymbol().getSimpleName().equals("NullPointerException");
        }
        return false;
    }
}
