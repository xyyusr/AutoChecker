package net.sourceforge.pmd.lang.java.rule.design;

import java.util.*;
import java.lang.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ClassWithOnlyPrivateConstructorsShouldBeFinalRule extends AbstractJavaRulechainRule {
    public ClassWithOnlyPrivateConstructorsShouldBeFinalRule() {
        super(ASTClassOrInterfaceDeclaration.class);
    }

    @Override
    public Object visit(ASTClassOrInterfaceDeclaration node, Object data) {
        if (node.isFinal() || node.isInterface() || node.isAbstract()) {
            return super.visit(node, data);
        }

        NodeStream<ASTConstructorDeclaration> constructors = node.getDeclarations(ASTConstructorDeclaration.class);
        if (constructors.isEmpty()) {
            return super.visit(node, data);
        }

        boolean allPrivate = constructors.allMatch(constructor -> constructor.isPrivate());
        if (!allPrivate) {
            return super.visit(node, data);
        }

        NodeStream<ASTClassOrInterfaceDeclaration> subclasses = node.getRoot().descendants(ASTClassOrInterfaceDeclaration.class)
            .filter(subclass -> subclass.getSuperClassTypeNode() != null && subclass.getSuperClassTypeNode().getType() == node.getType());

        if (subclasses.isEmpty()) {
            addViolation(data, node);
        }

        return super.visit(node, data);
    }
}

package net.sourceforge.pmd.lang.java.rule.design;

import java.util.List;
import java.util.stream.Collectors;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ClassWithOnlyPrivateConstructorsShouldBeFinalRule extends AbstractJavaRulechainRule {
    public ClassWithOnlyPrivateConstructorsShouldBeFinalRule() {
        super(ASTClassOrInterfaceDeclaration.class);
    }

    @Override
    public Object visit(ASTClassOrInterfaceDeclaration node, Object data) {
        if (node.isFinal() || node.isInterface() || node.isAnnotation()) {
            return super.visit(node, data);
        }

        if (node.isAbstract()) {
            if (node.getDeclaredMethods().any(it -> it.isAbstract())) {
                return super.visit(node, data);
            }
            if (node.getDeclaredMethods().isEmpty() && node.getSuperClassTypeNode() != null) {
                return super.visit(node, data);
            }
        }

        List<ASTConstructorDeclaration> constructors = node.getDeclaredConstructors().toList();
        if (constructors.isEmpty()) {

public class ClassWithOnlyPrivateConstructorsShouldBeFinalRule extends AbstractJavaRulechainRule {
    public ClassWithOnlyPrivateConstructorsShouldBeFinalRule() {
        super(ASTClassOrInterfaceDeclaration.class);
    }

    @Override
    public Object visit(ASTClassOrInterfaceDeclaration node, Object data) {
        if (node.isFinal() || node.isInterface() || node.isAbstract()) {
            return super.visit(node, data);
        }

        NodeStream<ASTConstructorDeclaration> constructors = node.getDeclarations(ASTConstructorDeclaration.class);
        if (constructors.isEmpty()) {
            return super.visit(node, data);
        }

        boolean allConstructorsPrivate = constructors.allMatch(constructor -> constructor.isPrivate());
        if (!allConstructorsPrivate) {
            return super.visit(node, data);
        }

        NodeStream<ASTClassOrInterfaceDeclaration> subclasses = node.getRoot().descendants(ASTClassOrInterfaceDeclaration.class)
            .filter(subclass -> subclass.getSuperClassTypeNode() != null && subclass.getSuperClassTypeNode().getType() == node.getType());

        boolean hasSubclassInSameCompilationUnit = subclasses.anyMatch(subclass -> subclass.getEnclosingType() == node.getEnclosingType());

        if (!hasSubclassInSameCompilationUnit) {
            addViolation(data, node);
        }

        return super.visit(node, data);
    }
