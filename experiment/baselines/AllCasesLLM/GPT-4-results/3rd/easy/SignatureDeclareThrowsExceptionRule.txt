package net.sourceforge.pmd.lang.java.rule.design;

import java.util.List;
import java.util.stream.Collectors;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class SignatureDeclareThrowsExceptionRule extends AbstractJavaRulechainRule {

    public SignatureDeclareThrowsExceptionRule() {
        super(ASTMethodDeclaration.class, ASTConstructorDeclaration.class);
    }

    @Override
    public Object visit(ASTMethodDeclaration node, Object data) {
        if (isJUnitMethod(node) || isOverridingMethod(node)) {
            return super.visit(node, data);
        }

        List<JTypeMirror> exceptionTypes = node.getThrownExceptionTypes();
        for (JTypeMirror exceptionType : exceptionTypes) {
            if (isGenericException(exceptionType)) {
                addViolation(data, node, "Method should not explicitly throw java.lang.Exception");
                break;
            }
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTConstructorDeclaration node, Object data) {
        List<JTypeMirror> exceptionTypes = node.getThrownExceptionTypes();
        for (JTypeMirror exceptionType : exceptionTypes) {
            if (isGenericException(exceptionType)) {
                addViolation(data, node, "Constructor should not explicitly throw java.lang.Exception");
                break;
            }
        }
        return super.visit(node, data);
    }

    private boolean isGenericException(JTypeMirror exceptionType) {
        return exceptionType.isClass() && "java.lang.Exception".equals(exceptionType.getSymbol().getBinaryName());
    }

    private boolean isJUnitMethod(ASTMethodDeclaration node) {
        String methodName = node.getName();
        ASTClassOrInterfaceDeclaration classNode = node.getEnclosingType();
        if (classNode == null) {
            return false;
        }

        if (methodName.equals("setUp") || methodName.equals("tearDown") || methodName.startsWith("test")) {
            JClassSymbol superClass = classNode.getSuperClassSymbol();
            if (superClass != null && "junit.framework.TestCase".equals(superClass.getBinaryName())) {
                return true;
            }
        }

        return false;
    }

    private boolean isOverridingMethod(ASTMethodDeclaration node) {
        return node.isOverridden();
    }
}
