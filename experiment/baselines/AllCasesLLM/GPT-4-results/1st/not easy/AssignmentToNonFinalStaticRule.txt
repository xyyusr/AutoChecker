    @Override
    public Object visit(ASTConstructorDeclaration node, Object data) {
        List<ASTAssignmentOperator> assignments = node.findDescendantsOfType(ASTAssignmentOperator.class);

        for (ASTAssignmentOperator assignmentOperator : assignments) {
            ASTName name = assignmentOperator.getFirstChildOfType(ASTName.class);

            if (name != null) {
                String varName = name.getImage();

                if (varName.contains(".")) {
                    varName = varName.substring(varName.indexOf('.') + 1);

                            addViolation(data, assignmentOperator);
                        }
                    }
                }
