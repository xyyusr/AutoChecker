package net.sourceforge.pmd.lang.java.rule.errorprone;


import java.util.Objects;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class BrokenNullCheckRule extends AbstractJavaRulechainRule {

    public BrokenNullCheckRule() {
        addRuleChainVisit(ASTStatementExpression.class);
    }

    @Override
    public Object visit(ASTStatementExpression node, Object data) {
        for (ASTPrimaryExpression expr : node.findDescendantsOfType(ASTPrimaryExpression.class)) {
            ASTPrimaryPrefix prefix = expr.getFirstDescendantOfType(ASTPrimaryPrefix.class);
            if (prefix != null) {
                ASTName name = prefix.getFirstDescendantOfType(ASTName.class);
                if (name != null) {
                    checkForBrokenNullCheck(name, data);
                }
            }
        }
        return super.visit(node, data);
    }

    private void checkForBrokenNullCheck(ASTName name, Object data) {
        ASTPrimarySuffix suffix = name.getFirstParentOfType(ASTPrimarySuffix.class);
        if (suffix != null && suffix.isArguments()) {
            return;
        }

        Node parent = name.jjtGetParent().jjtGetParent();
        if (!(parent instanceof ASTEqualityExpression)) {
            return;
        }

        ASTEqualityExpression equality = (ASTEqualityExpression) parent;
        if (equality.hasDescendantOfType(ASTConditionalAndExpression.class) ||
                equality.hasDescendantOfType(ASTConditionalOrExpression.class)) {
            return;
        }

        ASTExpression expr = equality.getFirstParentOfType(ASTExpression.class);
        if (expr != null && Objects.equals(name.getImage(), getFirstVariableName(expr))) {
            addViolation(data, name);
        }
    }

    private String getFirstVariableName(ASTExpression expr) {
        ASTPrimaryExpression primaryExpression = expr.getFirstDescendantOfType(ASTPrimaryExpression.class);
        if (primaryExpression == null) {
            return null;
        }

        ASTPrimaryPrefix prefix = primaryExpression.getFirstDescendantOfType(ASTPrimaryPrefix.class);
        if (prefix == null) {
            return null;
        }

        ASTName name = prefix.getFirstDescendantOfType(ASTName.class);
        if (name == null) {
            return null;
        }

        return name.getImage();
    }
}
