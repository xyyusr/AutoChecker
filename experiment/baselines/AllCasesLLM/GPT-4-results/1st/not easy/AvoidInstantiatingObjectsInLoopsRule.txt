import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;

import net.sourceforge.pmd.lang.ast.NodeStream;
public class AvoidInstantiatingObjectsInLoopsRule extends AbstractJavaRulechainRule {

    public AvoidInstantiatingObjectsInLoopsRule() {
        addRuleChainVisit(ASTAllocationExpression.class);
    }

    @Override
    public Object visit(ASTAllocationExpression node, Object data) {
        if (isWithinLoop(node) && !isReturned(node) && !isThrown(node) && !isInTryCatch(node)) {
            addViolation(data, node);
        }
        return super.visit(node, data);
    }

    private boolean isWithinLoop(ASTAllocationExpression node) {
        return node.getFirstParentOfType(ASTWhileStatement.class) != null
                || node.getFirstParentOfType(ASTForStatement.class) != null
                || node.getFirstParentOfType(ASTDoStatement.class) != null;
    }

    private boolean isReturned(ASTAllocationExpression node) {
        return node.getFirstParentOfType(ASTReturnStatement.class) != null;
    }

    private boolean isThrown(ASTAllocationExpression node) {
        return node.getFirstParentOfType(ASTThrowStatement.class) != null;
    }

    private boolean isInTryCatch(ASTAllocationExpression node) {
        return node.getFirstParentOfType(ASTBreakStatement.class) != null
                || node.getFirstParentOfType(ASTContinueStatement.class) != null
                || node.getFirstParentOfType(ASTLocalVariableDeclaration.class) != null;
    }
