import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class InefficientEmptyStringCheckRule extends AbstractJavaRulechainRule {
    @Override
    public Object visit(ASTCompilationUnit node, Object data) {
        return super.visit(node, data);
    }

    private static Set<String> getMethods() {
        Set<String> methods = new HashSet<>();
        methods.add("length");
        methods.add("isEmpty");
        return methods;
    }

    @Override
    public Object visit(ASTMethodDeclaration node, Object data) {
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTMethodInvocation node, Object data) {
        Object result = super.visit(node, data);
        if (result != null) {
            return result;
        }
        if (isMethodInvocation(node, "length", data)) {
            ASTExpression parent = node.jjtGetParent();
            if (parent instanceof ASTExpressionUnary || parent instanceof ASTExpressionPrefix) {
                if (isMethodInvocation(parent, "isEmpty", data)) {
                    addViolation(this, node);
                }
            }
            return null;
        }
        if (isMethodInvocation(node, "isEmpty", data)) {
            ASTExpression parent = node.jjtGetParent();
            if (parent instanceof ASTExpressionUnary || parent instanceof ASTExpressionPrefix) {
                if (isMethodInvocation(parent, "length", data)) {
                    addViolation(this, node);
                }
            }
            return null;
        }
        return result;
    }

    private boolean isMethodInvocation(ASTExpression node, String methodName, Object data) {
        if (!(node instanceof ASTExpressionMethodInvocation)) {
            return false;
        }
        ASTExpression methodInvocation = (ASTExpression) node;
        if (!methodInvocation.getImage().equals(methodName)) {
            return false;
        }
        if (methodInvocation.jjtGetChild(0) instanceof ASTIdentifier) {
            return true;
        }
        return false;
    }

    @Override
    public Object visit(ASTExpressionMethodInvocation node, Object data) {
        Object result = super.visit(node, data);
        if (result != null) {
            return result;
        }
        if (node.jjtGetChild(0) instanceof ASTIdentifier) {
            ASTIdentifier identifier = (ASTIdentifier) node.jjtGetChild(0);
            ASTExpression methodInvocation = (ASTExpression) node.jjtGetChild(1);
            if (identifier.getImage().equals("length") || identifier.getImage().equals("isEmpty")) {
                if (methodInvocation instanceof ASTMethodInvocation) {
                    ASTIdentifier methodIdentifier = (ASTIdentifier) methodInvocation.jjtGetChild(0);
                    if (methodIdentifier.getImage().equals("length") || methodIdentifier.getImage().equals("isEmpty")) {
                        addViolation(this, node);
                    }
                } else if (methodInvocation instanceof ASTExpressionUnary) {
                    ASTExpression unaryOperator = (ASTExpression) methodInvocation.jjtGetChild(0);
                    if (unaryOperator instanceof ASTExpressionUnaryPostfix) {
                        if (identifier.getImage().equals("length")) {
                            addViolation(this, node);
                        }
                    }
                }
            }
        }
        return result;
    }

    @Override
    public Object visit(ASTExpressionUnary node, Object data) {
        Object result = super.visit(node, data);
        if (result != null) {
            return result;
        }
        if (node.jjtGetChild(0) instanceof ASTExpressionUnaryPostfix) {
            ASTExpression unaryOperator = (ASTExpression) node.jjtGetChild(0);
            if (unaryOperator instanceof ASTExpressionMethodInvocation) {
                ASTExpression methodInvocation = (ASTExpression) unaryOperator;
                if (methodInvocation instanceof ASTMethodDeclaration) {
                    ASTIdentifier identifier = (ASTIdentifier) methodInvocation.jjtGetChild(0);
                    if (identifier.getImage().equals("length") || identifier.getImage().equals("isEmpty")) {
                        addViolation(this, node);
                    }
                }
            }
        }
        return result;
    }

    @Override
    public Object visit(ASTPrimaryExpression node, Object data) {
        Object result = super.visit(node, data);
        if (result != null) {
            return result;
        }
        if (node instanceof ASTStringLiteral) {
            ASTStringLiteral stringLiteral = (ASTStringLiteral) node;
            if (stringLiteral.getImage().isEmpty()) {
                addViolation(this, node);
            }
        }
        return result;
    }

import java.util.*;
import java.lang.*;
import java.util.regex.Pattern;
import java.util.regex.Matcher;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class InefficientEmptyStringCheckRule extends AbstractJavaRulechainRule {

    public InefficientEmptyStringCheckRule() {
        super(ASTCallExpression.class, ASTMethodCallExpression.class, ASTMemberMethodCallExpression.class);
    }

    @Override
    public Object visit(ASTCallExpression node, Object data) {
        if (node.getImage().equals("isEmpty")) {
            if (isStringTrimCalledBefore(node)) {
                addViolation(data, node);
            }
        } else if (node.getImage().equals("length")) {
            if (isStringTrimCalledBefore(node)) {
                addViolation(data, node);
            } else if (isLengthCalledOnTrimResult(node)) {
                addViolation(data, node);
            }
        }
        return super.visit(node, data);
    }

    private boolean isStringTrimCalledBefore(ASTCallExpression node) {
        for (ASTExpression expr : node.getChildren()) {
            if (expr.getImage().equals("trim")) {
                return true;
            }
        }
        return false;
    }

    private boolean isLengthCalledOnTrimResult(ASTCallExpression node) {
        Pattern pattern = Pattern.compile(".*\\.length\\(\\)");
        Matcher matcher = pattern.matcher(node.getImage());
        if (matcher.matches()) {
            return true;
        }
        return false;
    }

    @Override
    public Object visit(ASTMethodCallExpression node, Object data) {
        if (node.getImage().equals("isEmpty") && isStringTrimCalledBefore(node)) {
            addViolation(data, node);
        } else if (node.getImage().equals("length") && isStringTrimCalledBefore(node)) {
            addViolation(data, node);
        } else if (node.getImage().equals("length") && isLengthCalledOnTrimResult(node)) {
            addViolation(data, node);
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTMemberMethodCallExpression node, Object data) {
        if (node.getImage().equals("isEmpty") && isStringTrimCalledBefore(node)) {
            addViolation(data, node);
        } else if (node.getImage().equals("length") && isStringTrimCalledBefore(node)) {
            addViolation(data, node);
        } else if (node.getImage().equals("length") && isLengthCalledOnTrimResult(node)) {
            addViolation(data, node);
        }
