package net.sourceforge.pmd.lang.java.rule.codestyle;

import java.util.*;
import java.lang.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class IdenticalCatchBranchesRule extends AbstractJavaRulechainRule {

    public IdenticalCatchBranchesRule() {
        super(JavaBlock.class);
    }

    @Override
    public Object visit(JavaBlock node, Object data) {
        List<JavaCatchClause> catches = new ArrayList<>();
        List<JavaExpression> expressions = new ArrayList<>();
        for (JavaStatement statement : (List<JavaStatement>) node.getStatements()) {
            if (statement instanceof JavaCatchClause) {
                catches.add((JavaCatchClause) statement);
                JavaExpression expression = ((JavaCatchClause) statement).getException();
                expressions.add(expression);
            }
        }
        if (catches.size() > 1 && expressions.size() > 1) {
            for (int i = 0; i < catches.size(); i++) {
                if (areCatchesIdentical(catches.get(i), catches.get(0), expressions.get(0))) {
                    addViolation(this, data, node, node.getImage() + " identical catch branches");
                }
            }
        }
        return super.visit(node, data);
    }

    private boolean areCatchesIdentical(JavaCatchClause catchClause, JavaCatchClause referenceClause, JavaExpression referenceExpression) {
        return catchClause.getException().getImage().equals(referenceClause.getException().getImage())
                && areExpressionsIdentical(catchClause.getException(), referenceExpression);
    }

    private boolean areExpressionsIdentical(JavaExpression expression, JavaExpression referenceExpression) {
        if (expression instanceof JavaName) {
            return ((JavaName) expression).getImage().equals(((JavaName) referenceExpression).getImage());
        } else if (expression instanceof JavaParenthesizedExpression) {
            return ((JavaParenthesizedExpression) referenceExpression).getExpression().getImage().equals(((JavaParenthesizedExpression) expression).getExpression().getImage());
        } else {
            return false;
        }
    }
}

package net.sourceforge.pmd.lang.java.rule.codestyle;

import java.util.List;
import java.util.Map;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class IdenticalCatchBranchesRule extends AbstractJavaRulechainRule {

    @Override
    public Object visit(ASTTryStatement node, Object data) {
        if (node.jjtGetNumChildren() > 0 && node.jjtGetChild(0) instanceof ASTCatches) {
            ASTCatches catches = (ASTCatches) node.jjtGetChild(0);
            for (ASTCatchClause catchClause : catches.getCatchClauses()) {
                checkCatchBranch(catchClause);
            }
        }
        return super.visit(node, data);
    }

    private void checkCatchBranch(ASTCatchClause catchClause) {
        List<String> catchVariableNames = new ArrayList<>();
        Map<String, Type> catchVariableTypes = new HashMap<>();
        for (JavaNode child : catchClause.jjtGetChildren()) {
            if (child instanceof ASTCatchVariable) {
                ASTCatchVariable catchVariable = (ASTCatchVariable) child;
                catchVariableNames.add(catchVariable.getName().getImage());
                catchVariableTypes.put(catchVariable.getName().getImage(), catchVariable.getType());
            }
        }

        if (catchClause.jjtGetNumChildren() > 0 && catchClause.jjtGetChild(0) instanceof ASTExpressionList) {
            ASTExpressionList expressions = (ASTExpressionList) catchClause.jjtGetChild(0);
            for (JavaNode expression : expressions.getExpressions()) {
                if (expression instanceof ASTAssignment) {
                    ASTAssignment assignment = (ASTAssignment) expression;
                    if (catchVariableNames.contains(assignment.getTarget().getImage())) {
                        VariableDeclarationSegmentSymbol symbol = new VariableDeclarationSegmentSymbol(assignment.getTarget().getImage(), assignment.getTarget().getType());
                        if (symbol.getVariables().size() > 1) {
                            if (assignment.getOperator().getImage().equals("=")) {
                                MethodDeclarationSymbol method = new MethodDeclarationSymbol();
                                if (catchVariableNames.contains(method.getName()) && catchVariableTypes.get(method.getName()).equals(method.getType())) {
                                    addViolation(this, catchClause, "Identical catch branches can be collapsed into a single multi-catch branch");
                                }
                            }
                        }
                    }
                }
            }
        }
    }

    @Override
    public Object visit(ASTAssignment node, Object data) {
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTAssignmentOperator node, Object data) {
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTCatches node, Object data) {
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTCatchClause node, Object data) {
        return super.visit(node, data);
    }
}

package net.sourceforge.pmd.lang.java.rule.codestyle;

import java.util.*;
import java.lang.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class IdenticalCatchBranchesRule extends AbstractJavaRulechainRule {

    public IdenticalCatchBranchesRule() {
        super(
            ATryStatement.class,
            ACatchClause.class
        );
    }

    @Override
    public Object visit(ACatchClause node, Object data) {
        List<ACatchClause> catchClauses = getPreviousCatchClauses(node);
        if (catchClauses.isEmpty()) {
            return super.visit(node, data);
        }
        List<String> exceptionNames = catchClauses.stream()
            .map(ACatchClause::getException)
            .map(ANormalClassDeclaration::getImage)
            .toList();
        if (exceptionNames.size() != 1 || !exceptionNames.get(0).equals(node.getException().getImage())) {
            return super.visit(node, data);
        }
        List<String> exceptionHandlingBlocks = catchClauses.stream()
            .map(ACatchClause::getBody)
            .map(Node::getImage)
            .toList();
        if (exceptionHandlingBlocks.size() == exceptionNames.size() && exceptionHandlingBlocks.contains(node.getBody().getImage())) {
            addViolation(this, data, node);
        }
        return super.visit(node, data);
    }

    private List<ACatchClause> getPreviousCatchClauses(ACatchClause node) {
        return (List<ACatchClause>) getPreviousNodes(node);
    }

    @Override
    public Object visit(ATryStatement node, Object data) {
        return super.visit(node, data);
    }
}

        }

        Set<String> exceptionNames = new HashSet<>();

            exceptionNames.add(catchBlock.getType().getImage());
        }

        if (exceptionNames.size() > 1 || exceptionNames.size() == 1 && exceptionNames.contains("Exception")) {
            return super.visit(node, data);

        for (int i = 1; i < node.getNumberOfChildNodesWithClass(catchStatement.class); i++) {
