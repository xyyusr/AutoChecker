package net.sourceforge.pmd.lang.java.rule.errorprone;

import java.util.*;
import java.lang.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class BrokenNullCheckRule extends AbstractJavaRulechainRule {

    public BrokenNullCheckRule() {
        super(MethodCallExpr.class, JavadocComment.class, Assignment.class, MethodCallExpr.class);
    }

    @Override
    public Object visit(MethodCallExpr node, Object data) {
        return super.visit(node, data);
    }

    @Override
    public Object visit(JavadocComment node, Object data) {
        return super.visit(node, data);
    }

    @Override
    public Object visit(Assignment node, Object data) {
        return super.visit(node, data);
    }

    @Override
    public Object visit(MethodCallExpr node, Object data) {
        if (node.getImage().contains("isNull") || node.getImage().contains("isNotNull")) {
            return super.visit(node, data);
        }

        List<MethodCallExpr> calls = new ArrayList<>();
        if (node.getParent() instanceof MethodCallExpr) {
            calls.add((MethodCallExpr) node.getParent());
        }

        for (MethodCallExpr call : calls) {
            if (call.getImage().contains("equals") && (call.getLastChild().getImage().contains("null") || call.getLastChild().getImage().contains("this"))) {
                return super.visit(node, data);
            }
        }

        if (node.getImage().contains("equals") && node.getParent() instanceof AssignExpr) {
            return super.visit(node, data);
        }

        return super.visit(node, data);
    }

    @Override
    public Object visit(Assignment node, Object data) {
        if (node.getLeftHandSide() instanceof AssignExpr && node.getLeftHandSide().getLastChild().getImage().contains("null")) {
            return super.visit(node, data);
        }

        if (node.getLeftHandSide() instanceof AssignExpr && node.getLeftHandSide().getLastChild().getImage().contains("this")) {
            return super.visit(node, data);
        }

        if (node.getLeftHandSide() instanceof AssignExpr && node.getLeftHandSide().getLastChild().getImage().contains("get") && node.getLeftHandSide().getLastChild().getImage().contains("is")) {
            return super.visit(node, data);
        }

        return super.visit(node, data);
    }

    @Override
    public Object visit(MethodCallExpr node, Object data) {
        if (node.getImage().contains("equals") && node.getParent() instanceof AssignExpr) {
            return super.visit(node, data);
        }

        if (node.getImage().contains("equals") && (node.getLastChild().getImage().contains("null") || node.getLastChild().getImage().contains("this"))) {
            return super.visit(node, data);
        }

        if (node.getImage().contains("hashCode") && (node.getLastChild().getImage().contains("null") || node.getLastChild().getImage().contains("this"))) {
            return super.visit(node, data);
        }

        if (node.getImage().contains("get") && node.getImage().contains("is") && (node.getLastChild().getImage().contains("null") || node.getLastChild().getImage().contains("this"))) {
            return super.visit(node, data);
        }

        return super.visit(node, data);
    }
}

public class BrokenNullCheckRule extends AbstractJavaRulechainRule {

    public BrokenNullCheckRule() {
        addRuleChainVisitor(this, 
                IfStatement.class, 
                MethodCallExpression.class, 
                ConditionalExpression.class, 
                BinaryExpression.class, 
                AssignExpression.class);
    }

    @Override
    public Object visit(IfStatement node, Object data) {
        Object result = super.visit(node, data);
        if (result == null) {
            result = visit(node.getCondition(), data);
        }
        return result;
    }

    @Override
    public Object visit(MethodCallExpression node, Object data) {
        Object result = super.visit(node, data);
        if (result == null && node.getMethodName().equals("equals")) {
            if (node.getTarget() instanceof NullLiteral) {
                result = "Method 'equals' can't be called with a null argument";
            }
        }
        return result;
    }

    @Override
    public Object visit(ConditionalExpression node, Object data) {
        Object result = super.visit(node, data);
        if (result == null) {
            result = visit(node.getLeftOperand(), data);
        }
        return result;
    }

    @Override
    public Object visit(BinaryExpression node, Object data) {
        Object result = super.visit(node, data);
        if (result == null) {
            if (node.getLeftOperand() instanceof NullLiteral || node.getRightOperand() instanceof NullLiteral) {
                result = checkBrokenNullCheck(node.getLeftOperand(), node.getRightOperand(), data);
            }
            if (result == null && node.getLeftOperand() instanceof MethodCallExpression && node.getRightOperand() instanceof NullLiteral) {
                result = checkBrokenNullCheck(node.getLeftOperand(), node.getRightOperand(), data);
            }
            if (result == null && node.getLeftOperand() instanceof NullLiteral && node.getRightOperand() instanceof MethodCallExpression) {
                result = checkBrokenNullCheck(node.getLeftOperand(), node.getRightOperand(), data);
            }
            if (result == null && node.getLeftOperand() instanceof AssignExpression && node.getRightOperand() instanceof NullLiteral) {
                result = checkBrokenNullCheck(node.getLeftOperand(), node.getRightOperand(), data);
            }
            if (result == null && node.getLeftOperand() instanceof NullLiteral && node.getRightOperand() instanceof AssignExpression) {
                result = checkBrokenNullCheck(node.getLeftOperand(), node.getRightOperand(), data);
            }
        }
        return result;
    }

    @Override
    public Object visit(AssignExpression node, Object data) {
        Object result = super.visit(node, data);
        if (result == null) {
            if (node.getLeftOperand() instanceof NullLiteral && node.getRightOperand() instanceof NullLiteral) {
                result = "Assigning null to null is not necessary";
            }
        }
        return result;
    }

    private Object checkBrokenNullCheck(Expression left, Expression right, Object data) {
        if (left instanceof NullLiteral && right instanceof NullLiteral) {
            return "Broken null check: both expressions are null";
        } else if (left instanceof NullLiteral && right instanceof MethodCallExpression) {
            if (((MethodCallExpression) right).getMethodName().equals("equals")) {
                return "Broken null check: calling equals on null";
            } else {
                return "Broken null check: calling method on null";
            }
        } else if (left instanceof MethodCallExpression && right instanceof NullLiteral) {
            if (((MethodCallExpression) left).getMethodName().equals("equals")) {
                return "Broken null check: calling equals with null argument";
            } else {
                return "Broken null check: calling method with null argument";
            }
        } else if (left instanceof AssignExpression && right instanceof NullLiteral) {
            return "Broken null check: assigning null to a variable";
        } else if (left instanceof NullLiteral && right instanceof AssignExpression) {
            return "Broken null check: assigning null to a variable";
        }
        return null;
    }
}

            return super.visit(node, data);

                addViolation(this, data, node);

            return super.visit(node, data);
        } else if (node.getOperator() == Operator.OR) {
            if (hasNullCheck((BinaryExpression) node.getLeftOperand(), data)) {
                addViolation(this, data, node);

            if (hasNullCheck((BinaryExpression) node.getRightOperand(), data)) {
                addViolation(this, data, node);
            }
            return super.visit(node, data);
        } else {
            return super.visit(node, data);
        }
    }

    public boolean hasNullCheck(BinaryExpression node, Object data) {
        if (node.getLeftOperand() instanceof NullLiteral || node.getRightOperand() instanceof NullLiteral) {
            return true;
        }
        if (node.getLeftOperand() instanceof NullLiteral) {
            if (node.getOperator() == Operator.AND) {
                return hasNullCheck((BinaryExpression) node.getRightOperand(), data);
            } else if (node.getOperator() == Operator.OR) {
                return true;
            }
        } else if (node.getRightOperand() instanceof NullLiteral) {
            if (node.getOperator() == Operator.AND) {
                return true;
            } else if (node.getOperator() == Operator.OR) {
                return false;
            }
        }
        if (node.getLeftOperand() instanceof PrefixExpression && ((PrefixExpression) node.getLeftOperand()).getOperator() == Operator.NOT) {

                return true;

            if (node.getLeftOperand() instanceof NullLiteral) {
                return true;
            }
        }
        return false;
    }
