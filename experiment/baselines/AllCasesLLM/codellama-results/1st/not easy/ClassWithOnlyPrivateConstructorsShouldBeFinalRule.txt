package net.sourceforge.pmd.lang.java.rule.design;

import java.util.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ClassWithOnlyPrivateConstructorsShouldBeFinalRule extends AbstractJavaRulechainRule {

    @Override
    public Object visit(ASTClassOrInterfaceDeclaration node, Object data) {
        if (node.isInterface() || !node.isPublic()) {
            return super.visit(node, data);
        }
        List<ASTMethodDeclaration> constructors = node.findDescendantsOfType(ASTConstructorDeclaration.class);
        List<ASTMethodDeclaration> publicConstructors = new ArrayList<>();
        for (ASTMethodDeclaration constructor : constructors) {
            if (constructor.modifiers().isPublic()) {
                publicConstructors.add(constructor);
            }
        }
        if (publicConstructors.isEmpty()) {
            List<ASTFieldDeclaration> fields = node.findDescendantsOfType(ASTFieldDeclaration.class);
            List<ASTFieldDeclaration> publicFields = new ArrayList<>();
            for (ASTFieldDeclaration field : fields) {
                if (field.modifiers().isPublic()) {
                    publicFields.add(field);
                }
            }
            if (publicFields.isEmpty()) {
                List<ASTMethodDeclaration> methods = node.findDescendantsOfType(ASTMethodDeclaration.class);
                List<ASTMethodDeclaration> publicMethods = new ArrayList<>();
                for (ASTMethodDeclaration method : methods) {
                    if (method.modifiers().isPublic()) {
                        publicMethods.add(method);
                    }
                }
                if (publicMethods.isEmpty()) {
                    return super.visit(node, data);
                }
            }
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTClassOrInterfaceBody node, Object data) {
        List<ASTFieldDeclaration> fields = node.findDescendantsOfType(ASTFieldDeclaration.class);
        List<ASTFieldDeclaration> publicFields = new ArrayList<>();
        for (ASTFieldDeclaration field : fields) {
            if (field.modifiers().isPublic()) {
                publicFields.add(field);
            }
        }
        if (publicFields.isEmpty()) {
            List<ASTMethodDeclaration> methods = node.findDescendantsOfType(ASTMethodDeclaration.class);
            List<ASTMethodDeclaration> publicMethods = new ArrayList<>();
            for (ASTMethodDeclaration method : methods) {
                if (method.modifiers().isPublic()) {
                    publicMethods.add(method);
                }
            }
            if (publicMethods.isEmpty()) {
                return super.visit(node, data);
            }
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTMethodInvocation node, Object data) {
        ASTPrimaryPrefix prefix = node.getPrimaryPrefix();
        if (prefix instanceof ASTNewClass) {
            ASTTypeDeclaration typeDecl = ((ASTNewClass) prefix).getTypeDeclaration();
            if (typeDecl instanceof ASTClassOrInterfaceDeclaration) {
                ASTClassOrInterfaceDeclaration classDecl = (ASTClassOrInterfaceDeclaration) typeDecl;
                if (classDecl.isPublic()) {
                    return super.visit(node, data);
                }
            }
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTPrimaryExpression node, Object data) {
        ASTPrimaryPrefix prefix = node.getPrimaryPrefix();
        if (prefix instanceof ASTNewArray) {
            ASTPrimaryExpression expr = ((ASTNewArray) prefix).getExpression();
            if (expr instanceof ASTNewClass) {
                ASTTypeDeclaration typeDecl = ((ASTNewClass) expr).getTypeDeclaration();
                if (typeDecl instanceof ASTClassOrInterfaceDeclaration) {
                    ASTClassOrInterfaceDeclaration classDecl = (ASTClassOrInterfaceDeclaration) typeDecl;
                    if (classDecl.isPublic()) {
                        return super.visit(node, data);
                    }
                }
            }
        }
        return super.visit(node, data);
    }

    public PrivateConstructorShouldBeFinalCheckerRule() {
        super(ClassOrInterfaceDeclaration.class);

    public Object visit(ClassOrInterfaceDeclaration n, Object data) {
        if (!n.isInterface() && !processedUnits.contains(n)) {
            processedUnits.add(n);
            if (hasOnlyPrivateConstructors(n) && !isFinal(n)) {
                addViolation(data, n);
            }

        return super.visit(n, data);

    private boolean hasOnlyPrivateConstructors(ClassOrInterfaceDeclaration n) {
        boolean hasPublicConstructor = false;
        boolean hasProtectedConstructor = false;
        boolean hasPackagePrivateConstructor = false;
        for (MethodDeclaration md : n.getBody().getMethods()) {
            if (md.isConstructor()) {
                if (md.getModifiers().isPublic()) {
                    hasPublicConstructor = true;
                } else if (md.getModifiers().isProtected()) {
                    hasProtectedConstructor = true;
                } else if (!md.getModifiers().isPrivate()) {
                    hasPackagePrivateConstructor = true;
                }
            }
        }
        return !hasPublicConstructor && !hasProtectedConstructor && !hasPackagePrivateConstructor;
    }

    private boolean hasSubclassUsingPrivateConstructor(ClassOrInterfaceDeclaration n) {
        for (CompilationUnitDeclaration cu : n.getCompilationUnit().getCompilationUnits()) {

                        if (md.isConstructor()) {
                            if (md.getModifiers().isPrivate()) {
                                return true;
                            }
                        }
                    }
                }
            }
        }
        return false;
    }

        return n.getModifiers().isFinal();

            return true;
        }
        return false;
    }

        for (MethodDeclaration method : node.getMethods()) {
            if (method.getImage().equals("public")) {
                return true;
            }
        }
        return false;
    }

    private boolean hasProtectedConstructor(ClassDeclaration node) {
        for (MethodDeclaration method : node.getMethods()) {
            if (method.getImage().equals("protected")) {
                return true;
            }
        }
        return false;
    }

    private boolean hasPackagePrivateConstructor(ClassDeclaration node) {
        for (MethodDeclaration method : node.getMethods()) {
            if (method.getImage().equals("default")) {
                return true;
            }
        }
        return false;
    }

    private boolean isAbstractClass(ClassDeclaration node) {
        if (node.getImage().equals("abstract")) {
            return true;
        }
        return false;
    }

        return node.getScope().contains(NodeStream.findFirst(node.getScope(), ClassDeclaration.class));
    }

    private boolean hasInnerClass(ClassDeclaration node) {
        return node.getScope().contains(NodeStream.findFirst(node.getScope(), InnerClassDeclaration.class));

                return true;
            }
        }
        return false;
    }

            if (innerClass.getImage().equals("private") && innerClass.getImage().equals("abstract")) {
                return true;
            }
        }
        return false;
    }

    private boolean hasAbstractMethods(ClassDeclaration node) {
        for (MethodDeclaration method : node.getMethods()) {
            if (method.getImage().equals("abstract")) {
                return true;
            }
        }
        return false;
    }

        super("ClassWithOnlyPrivateConstructorsShouldBeFinal", "Reports classes that may be made final because they cannot be extended from outside their compilation unit anyway.");
    }

    @Override
    public Object visit(AccessModifiers node, Object data) {
        if (!node.isFinal()) {
            return super.visit(node, data);
        }
        return false;
    }

    @Override
    public Object visit(ClassOrInterfaceDeclaration node, Object data) {
        if (node.isInterface() || node.isAbstract()) {
            return super.visit(node, data);
        }
        if (node.getConstructors().size() == 0) {
            addViolation(node, data);
            return false;
        }
        boolean hasPublicConstructor = false;
        for (ConstructorDeclaration constructor : node.getConstructors()) {
            if (constructor.getModifiers().isPublic()) {

                break;
            }
        }

        }
        return false;
