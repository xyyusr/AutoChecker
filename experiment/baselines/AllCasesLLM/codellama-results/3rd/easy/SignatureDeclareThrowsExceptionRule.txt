package net.sourceforge.pmd.lang.java.rule.design;

import java.util.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class SignatureDeclareThrowsExceptionRule extends AbstractJavaRulechainRule {
    private static final Set<String> JUNIT3_TEST_CASE_SUPER_CLASSES = Collections.singleton("junit.framework.TestCase");
    private static final Set<String> JUNIT4_TEST_CASE_SUPER_CLASSES = Collections.singleton("org.junit.Before");

    public SignatureDeclareThrowsExceptionRule() {
        addRuleMatch(JClassOrInterfaceDeclaration.class, this::visitClassOrInterfaceDeclaration);
    }

    private void visitClassOrInterfaceDeclaration(JClassOrInterfaceDeclaration node, Object data) {
        for (JMethodDeclaration method : node.getMethods()) {
            visitMethod(method, data);
        }
        for (JConstructorDeclaration constructor : node.getConstructors()) {
            visitConstructor(constructor, data);
        }
    }

    private void visitMethod(JMethodDeclaration node, Object data) {
        if (node.getThrows().size() > 0 && node.getThrows().get(0).toReference().getName().equals("Exception")) {
            addViolation(data, node);
        } else if (node.getThrows().size() > 0) {
            JType type = node.getThrows().get(0).toReference().getType();
            if (!isAllowedThrows(type)) {
                addViolation(data, node);
            }
        }
        visitClassOrInterfaceDeclaration(node, data);
    }

    private void visitConstructor(JConstructorDeclaration node, Object data) {
        if (node.getThrows().size() > 0 && node.getThrows().get(0).toReference().getName().equals("Exception")) {
            addViolation(data, node);
        } else if (node.getThrows().size() > 0) {
            JType type = node.getThrows().get(0).toReference().getType();
            if (!isAllowedThrows(type)) {
                addViolation(data, node);
            }
        }
        visitClassOrInterfaceDeclaration(node, data);
    }

    private boolean isAllowedThrows(JType type) {
        if (type instanceof JArrayType) {
            return isAllowedThrows(((JArrayType) type).getType());
        } else if (type instanceof JClassOrInterfaceType) {
            JClassOrInterfaceType refType = (JClassOrInterfaceType) type;
            if (refType.getName().equals("Exception")) {
                return false;
            } else {
                return true;
            }
        } else if (type instanceof JName) {
            JName name = (JName) type;
            if (name.getName().equals("Exception")) {
                return false;
            }
        }
        return true;
    }

    private boolean isJUnit3TestCase(JClassOrInterfaceDeclaration node) {
        String superclassName = node.getSuperclassName();
        return JUNIT3_TEST_CASE_SUPER_CLASSES.contains(superclassName);
    }

    private boolean isJUnit4TestCase(JClassOrInterfaceDeclaration node) {
        String superclassName = node.getSuperclassName();
        return JUNIT4_TEST_CASE_SUPER_CLASSES.contains(superclassName);
    }
}

                    if (parentClass.getQualifiedName().equals("junit.framework.TestCase")) {
                        isJunit3Setup = true;
                    }
                }
                if (parentClass != null && parentClass.getQualifiedName().equals("junit.framework.TestCase")) {
                    isTestCase = true;
                }
            }
        } catch (Exception e) {
            return false;
        }
        return isJunit3Setup || isJunit3TearDown || isJunit4Setup || isTestCase;
    }
