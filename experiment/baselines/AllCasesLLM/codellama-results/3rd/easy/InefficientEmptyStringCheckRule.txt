package net.sourceforge.pmd.lang.java.rule.performance;

import java.util.*;
import java.lang.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class InefficientEmptyStringCheckRule extends AbstractJavaRulechainRule {

    public InefficientEmptyStringCheckRule() {
        addRuleChain(new MethodCallExpression(node -> nodeToString(node)), node -> {
            if (node instanceof MethodInvocation) {
                if (node.getIdentifier().equals("trim") && node.getMethodName().equals("length")) {
                    return node;
                }
            } else if (node instanceof MethodCallExpression) {
                if (node.getMethodName().equals("isEmpty")) {
                    return node;
                }
            }
            return null;
        });
    }

    @Override
    public Object visit(MethodCallExpression node, Object data) {
        ASTNode result = (ASTNode) data;
        if (result instanceof MethodCallExpression) {
            MethodCallExpression methodCall = (MethodCallExpression) result;
            if (methodCall.getMethodName().equals("isEmpty")) {
                MethodDeclaration methodDecl = (MethodDeclaration) methodCall.getMethod();
                if (methodDecl.getName().equals("trim") && methodDecl.getTypeParameters().isEmpty() && methodDecl.getParameters().isEmpty()) {
                    addViolation(this, node);
                }
            } else if (methodCall.getMethodName().equals("length")) {
                MethodDeclaration methodDecl = (MethodDeclaration) methodCall.getMethod();
                if (methodDecl.getName().equals("trim") && methodDecl.getTypeParameters().isEmpty() && methodDecl.getParameters().isEmpty()) {
                    addViolation(this, node);
                }
            }
        } else if (result instanceof AssignmentExpression) {
            AssignmentExpression assignment = (AssignmentExpression) result;
            if (assignment.getVariable().getIdentifier().equals("length")) {
                MethodCallExpression methodCall = (MethodCallExpression) assignment.getExpression();
                if (methodCall.getMethodName().equals("length") && methodCall.getMethodName().equals("trim")) {
                    addViolation(this, node);
                }
            }
        } else if (result instanceof BinaryExpression) {
            BinaryExpression binary = (BinaryExpression) result;
            if (binary.getLeftOperand() instanceof MethodCallExpression && binary.getRightOperand() instanceof NumericLiteral) {
                MethodCallExpression methodCall = (MethodCallExpression) binary.getLeftOperand();
                if (methodCall.getMethodName().equals("length") && methodCall.getMethodName().equals("trim")) {
                    addViolation(this, node);
                }
            }
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(MethodInvocation node, Object data) {
        return visit((MethodCallExpression) node, data);
    }

    @Override
    public Object visit(MethodDeclaration node, Object data) {
        return super.visit(node, data);
    }

    @Override
    public Object visit(AssignmentExpression node, Object data) {
        return super.visit(node, data);
    }

    @Override
    public Object visit(BinaryExpression node, Object data) {
        return super.visit(node, data);
    }

    private MethodCallExpression nodeToString(MethodCallExpression node) {
        return node;
    }
}

import java.util.*;
import java.lang.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class InefficientEmptyStringCheckRule extends AbstractJavaRulechainRule {

    public InefficientEmptyStringCheckRule() {
        super(MethodCallExpr.class, NameExpr.class, MethodInvocationExpr.class);
    }

    @Override
    public Object visit(MethodCallExpr node, Object data) {
        if (node.getImage().equals("isEmpty") || node.getImage().equals("length")) {
            if (node.getArguments().size() == 1 && node.getArguments().get(0) instanceof MethodCallExpr) {
                MethodCallExpr trimMethod = (MethodCallExpr) node.getArguments().get(0);
                if (trimMethod.getImage().equals("trim")) {
                    addViolation(data, node, "Avoid calling trim() and then checking isEmpty() or length()");
                }
            } else if (node.getArguments().size() == 1 && node.getArguments().get(0) instanceof NameExpr) {
                NameExpr nameExpr = (NameExpr) node.getArguments().get(0);
                if (nameExpr.getImage().equals("0")) {
                    addViolation(data, node, "Avoid calling trim() and then comparing the result to 0");
                }
            }
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(NameExpr node, Object data) {
        if (node.getImage().equals("isEmpty")) {
            MethodCallExpr trimMethod = getTrimMethod(parent(node));
            if (trimMethod != null) {
                addViolation(data, trimMethod, "Avoid calling trim() and then checking isEmpty()");
            }
        } else if (node.getImage().equals("length")) {
            MethodCallExpr trimMethod = getTrimMethod(parent(node));
            if (trimMethod != null) {
                addViolation(data, trimMethod, "Avoid calling trim() and then checking length()");
            }
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(MethodInvocationExpr node, Object data) {
        if (node.getImage().equals("isEmpty") || node.getImage().equals("length")) {
            MethodCallExpr trimMethod = getTrimMethod(node);
            if (trimMethod == null) {
                trimMethod = getTrimMethod(parent(node));
            }
            if (trimMethod != null) {
                addViolation(data, trimMethod, "Avoid calling trim() and then checking isEmpty() or length()");
            }
        }
        return super.visit(node, data);
    }

    private MethodCallExpr getTrimMethod(Node node) {
        if (node instanceof MethodCallExpr && ((MethodCallExpr) node).getImage().equals("trim")) {
            return (MethodCallExpr) node;
        }
        for (JavaNode child : node.getJavaNodeChildren()) {
            MethodCallExpr trimMethod = getTrimMethod(child);
            if (trimMethod != null) {
                return trimMethod;
            }
        }
        return null;
    }

    private Node parent(Node node) {
        if (node instanceof JavaNode) {
            return ((JavaNode) node).getJavaParent();
        }
        return null;
    }
