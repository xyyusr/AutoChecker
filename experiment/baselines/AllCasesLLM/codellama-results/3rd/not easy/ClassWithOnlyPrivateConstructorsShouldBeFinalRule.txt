package net.sourceforge.pmd.lang.java.rule.design;

import java.util.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ClassWithOnlyPrivateConstructorsShouldBeFinalRule extends AbstractJavaRulechainRule {

    @Override
    public Object visit(ClassOrInterfaceDeclaration node, Object data) {
        if (node.isAbstract() || node.isInterface() || node.isEnum()) {
            return super.visit(node, data);
        }

        List<MethodDeclaration> constructors = extractConstructors(node);
        if (constructors.stream().noneMatch(constructor -> !constructor.isPrivate())) {
            addViolation(data, this, node);
        }

        return super.visit(node, data);
    }

    @Override
    public Object visit(AnonymousClassDeclaration node, Object data) {
        return visit((ClassOrInterfaceDeclaration) node.getParent(), data);
    }

    private List<MethodDeclaration> extractConstructors(ClassOrInterfaceDeclaration node) {
        List<MethodDeclaration> constructors = new ArrayList<>();
        for (MethodDeclaration method : node.getMethods()) {
            if (method.getNameAsString().equals("<init>") && !method.isPrivate()) {
                constructors.add(method);
            }
        }
        return constructors;
    }
}

package net.sourceforge.pmd.lang.java.rule.design;

import java.util.*;
import java.lang.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ClassWithOnlyPrivateConstructorsShouldBeFinalRule extends AbstractJavaRulechainRule {
    public ClassWithOnlyPrivateConstructorsShouldBeFinalRule() {
        addRuleChainElement(new JavaVisibilityCheck());
        addRuleChainElement(new JavaClassDeclarationCheck());
        addRuleChainElement(new JavaFieldDeclarationCheck());
        addRuleChainElement(new JavaConstructorDeclarationCheck());
        addRuleChainElement(new JavaMethodDeclarationCheck());
        addRuleChainElement(new JavaSubClassCheck());
        addRuleChainElement(new JavaInnerClassCheck());
        addRuleChainElement(new JavaConstructorsCheck());
    }

    @Override
    public Object visit(ASTClassOrInterfaceDeclaration node, Object data) {
        if (node.isClass() && !node.isInnerClass() && !node.isEnum() && !node.isNestedClass()) {
            setLanguageVersion(node);
            JavaVisibilityCheck visibilityCheck = (JavaVisibilityCheck) getRuleChainElement(0);
            if (visibilityCheck.isPrivateConstructor(node)) {
                JavaClassDeclarationCheck classDeclarationCheck = (JavaClassDeclarationCheck) getRuleChainElement(1);

package net.sourceforge.pmd.lang.java.rule.design;

import java.util.*;
import java.lang.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ClassWithOnlyPrivateConstructorsShouldBeFinalRule extends AbstractJavaRulechainRule {

    public ClassWithOnlyPrivateConstructorsShouldBeFinalRule() {
        addRuleChainVisitor(new ClassWithOnlyPrivateConstructorsShouldBeFinalRuleVisitor());
    }

    private class ClassWithOnlyPrivateConstructorsShouldBeFinalRuleVisitor extends RuleChainVisitor {

        @Override
        public Object visit(ASTClassOrInterfaceDeclaration node, Object data) {
            if (node.isClass() && !node.isInnerClass() && !node.isAnonymous() && hasPrivateConstructors(node)) {
                handleNode(node, "ClassWithOnlyPrivateConstructorsShouldBeFinal");
            }
            return super.visit(node, data);
        }

        @Override
        public Object visit(ASTInnerClassDeclaration node, Object data) {
            if (hasPrivateConstructors(node)) {
                handleNode(node, "ClassWithOnlyPrivateConstructorsShouldBeFinal");
            }
            return super.visit(node, data);
        }

        private boolean hasPrivateConstructors(Node node) {
            NodeStream stream = new NodeStream(node);
            stream.setRecurseThroughImports(true);
            for (Object object : stream) {
                if (object instanceof ASTConstructorDeclaration && ((ASTConstructorDeclaration) object).isPrivate()) {
                    return true;
                }
            }
            return false;
        }

        @Override
        public Object visit(ASTAbstractClassDeclaration node, Object data) {
            if (!isAbstractClassNodeIgnored(node)) {
                handleNode(node, "ClassWithOnlyPrivateConstructorsShouldBeFinal");
            }
            return super.visit(node, data);
        }

        @Override
        public Object visit(ASTAbstractInterfaceDeclaration node, Object data) {
            if (!isAbstractClassNodeIgnored(node)) {
                handleNode(node, "ClassWithOnlyPrivateConstructorsShouldBeFinal");
            }
            return super.visit(node, data);
        }

        private boolean isAbstractClassNodeIgnored(Node node) {
            NodeStream stream = new NodeStream(node);
            stream.setRecurseThroughImports(true);
            for (Object object : stream) {
                if (object instanceof ASTClassOrInterfaceDeclaration &&
                        ((ASTClassOrInterfaceDeclaration) object).isAbstract() &&
                        ((ASTClassOrInterfaceDeclaration) object).hasAbstractMethods() &&
                        !node.isInnerClass()) {
                    return true;
                }
            }
            return false;
        }
    }
}
