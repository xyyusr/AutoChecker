package net.sourceforge.pmd.lang.java.rule.design;

import java.util.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class AvoidThrowingNullPointerExceptionRule extends AbstractJavaRulechainRule {

    public AvoidThrowingNullPointerExceptionRule() {
        super("AvoidThrowingNullPointerExceptionRule", "Avoid throwing NullPointerExceptions manually.");
        addRuleMatch(ASTThrowStatement.class);
    }

    @Override
    public Object visit(ASTThrowStatement node, Object data) {
        ASTExpression expression = node.getExpression();
        if (expression == null) {
            return data;
        }
        if (expression instanceof ASTNewClass) {
            ASTNewClass newClass = (ASTNewClass) expression;
            if (newClass.getExpression() instanceof ASTClassOrInterfaceDeclaration) {
                ASTClassOrInterfaceDeclaration classDeclaration = (ASTClassOrInterfaceDeclaration) newClass.getExpression();
                if (classDeclaration.getImage().equals("NullPointerException")) {
                    if (!hasVariableBeenReassignedBeforeThrow(node, expression)) {
                        addViolation(data, node);
                    }
                }
            }
        }
        return data;
    }

    private boolean hasVariableBeenReassignedBeforeThrow(ASTThrowStatement node, ASTExpression expression) {
        ASTAssignmentExpression assignment = findPreviousAssignment(node, expression);
        if (assignment == null) {
            return false;
        }
        ASTExpression target = assignment.getleftHandSide();
        ASTExpression right = assignment.getRightHandSide();
        if (right instanceof ASTNewClass && ((ASTNewClass) right).getExpression() instanceof ASTClassOrInterfaceDeclaration
                && ((ASTClassOrInterfaceDeclaration) ((ASTNewClass) right).getExpression()).getImage().equals("NullPointerException")) {
            return true;
        }
        return false;
    }

    private ASTAssignmentExpression findPreviousAssignment(ASTThrowStatement node, ASTExpression expression) {
        JavaNode parent = node.getParent();
        for (int i = parent.getJavaASTIndex(); i >= 0; i--) {
            JavaNode child = parent.getJavaASTChild(i);
            if (child instanceof ASTForStatement || child instanceof ASTWhileStatement || child instanceof ASTTryStatement
                    || child instanceof ASTMethodDeclaration || child instanceof ASTReturnStatement) {
                break;
            }
            if (child instanceof ASTAssignmentExpression) {
                ASTAssignmentExpression assignment = (ASTAssignmentExpression) child;
                ASTExpression target = assignment.getleftHandSide();
                if (target.equals(expression)) {
                    return assignment;
                }
            }
        }
        return null;
    }
}

package net.sourceforge.pmd.lang.java.rule.design;

import java.util.*;
import java.util.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class AvoidThrowingNullPointerExceptionRule extends AbstractJavaRulechainRule {

    public AvoidThrowingNullPointerExceptionRule() {
        super.addRuleChainElement(new ASTMethodDeclaration());
    }

    @Override
    public Object visit(ASTMethodDeclaration node, Object data) {
        Object result = super.visit(node, data);
        if (result instanceof RuleViolation) {
            return result;
        }
        ASTThrowStatement throwStatement = findThrowStatement(node);
        if (throwStatement != null) {
            ASTExpression exceptionExpression = findExceptionExpression(throwStatement);
            if (exceptionExpression instanceof ASTNewClass) {
                ASTNewClass newClass = (ASTNewClass) exceptionExpression;
                if (newClass.getType().getImage().equals("NullPointerException")) {
                    reportViolations(node, throwStatement, exceptionExpression);
                }
            }
        }
        return result;
    }

    @Override
    public Object visit(ASTThrowStatement node, Object data) {
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTNewClass node, Object data) {
        return super.visit(node, data);
    }

    private ASTThrowStatement findThrowStatement(ASTMethodDeclaration node) {
        for (ASTStatement statement : node.getBody().getStatements()) {
            if (statement instanceof ASTThrowStatement) {
                return (ASTThrowStatement) statement;
            }
        }
        return null;
    }

    private ASTExpression findExceptionExpression(ASTThrowStatement throwStatement) {
        return throwStatement.getExpression();
    }

    private void reportViolations(ASTMethodDeclaration node, ASTThrowStatement throwStatement, ASTExpression exceptionExpression) {
        RuleViolation violation = new RuleViolation("Avoid throwing NullPointerExceptions manually.", node);
        addViolation(violation);
    }
}

package net.sourceforge.pmd.lang.java.rule.design;


import java.util.List;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class AvoidThrowingNullPointerExceptionRule extends AbstractJavaRulechainRule {

    @Override
    public Object visit(ASTThrowStatement node, Object data) {
        if (isNullPointerException(node.getExceptions())) {
            addViolation(data, this, node);
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTTryStatement node, Object data) {
        Object result = super.visit(node, data);
        if (node.getTryBlocks().size() == 0 && node.getExceptions().size() == 1) {
            JNode caughtException = node.getExceptions().get(0);
            if (isNullPointerException(caughtException) && isReassignedBeforeThrow(node, caughtException)) {
                addViolation(data, this, node);
            }
        }
        return result;
    }

    @Override
    public Object visit(ASTMethodDeclaration node, Object data) {
        Object result = super.visit(node, data);
        MethodDeclarationSymbolTable methodSymbolTable = node.getSymbol().getSymbolTable();
        if (methodSymbolTable != null && methodSymbolTable.getParameters().size() > 0) {
            JNode lastParameter = methodSymbolTable.getParameters().get(methodSymbolTable.getParameters().size() - 1);
            if (lastParameter != null && lastParameter.getType().equals(Type.getExceptionType())) {
                ASTExpressionList exceptions = node.getExceptions();
                if (exceptions != null && exceptions.size() > 0) {
                    JNode caughtException = exceptions.get(0);
                    if (isNullPointerException(caughtException) && isReassignedInMethod(node, caughtException)) {
                        addViolation(data, this, node);
                    }
                }
            }
        }
        return result;
    }

    private boolean isNullPointerException(JNode node) {
        if (node instanceof ASTNewClass) {
            ASTNewClass newClass = (ASTNewClass) node;
            if (newClass.getType().getSimpleName().equals("NullPointerException")) {
                return true;
            }
        }
        return false;
    }

    private boolean isReassignedBeforeThrow(ASTTryStatement node, JNode caughtException) {
        List<ASTStatement> statementsBeforeThrow = node.getTryBlocks().get(0).getStatements();
        for (ASTStatement statement : statementsBeforeThrow) {
            if (statement instanceof ASTThrowStatement) {
                return true;
            }
            if (statement instanceof ASTAssignment) {
                ASTAssignment assignment = (ASTAssignment);
                if (assignment.getVariables().size() == 1) {
                    ASTVariableDeclaratorId declaratorId = (ASTVariableDeclaratorId) assignment.getVariables().get(0);
                    if (declaratorId.getName().equals(caughtException.getType().getName())) {
                        return true;
                    }
                }
            }
        }
        return false;
    }

    private boolean isReassignedInMethod(ASTMethodDeclaration node, JNode caughtException) {
        List<ASTStatement> statements = node.getBody().getStatements();
        for (ASTStatement statement : statements) {
            if (statement instanceof ASTAssignment) {
                ASTAssignment assignment = (ASTAssignment);
                if (assignment.getVariables().size() == 1) {
                    ASTVariableDeclaratorId declaratorId = (ASTVariableDeclaratorId) assignment.getVariables().get(0);
                    if (declaratorId.getName().equals(caughtException.getType().getName())) {
                        return true;
                    }
                }
            }
            if (statement instanceof ASTThrowStatement) {
                return false;
            }
        }
        return false;
    }
}

public class AvoidThrowingNullPointerExceptionRule extends AbstractJavaRulechainRule {

    public AvoidThrowingNullPointerExceptionRule() {
        addRuleChainVisitor(this);
    }

    @Override
    public Object visit(ASTThrowStatement node, Object data) {
        ASTExpressionList exceptions = (ASTExpressionList) node.getNonTerminalChildNodes().get(0);
        for (Object exception : exceptions.getJavaNodeList()) {
            if (exception instanceof ASTNewExpression) {
                ASTNewExpression newExpression = (ASTNewExpression) exception;
                if (newExpression.getExpression() instanceof ASTClassOrInterfaceType) {
                    ASTClassOrInterfaceType classOrInterfaceType = (ASTClassOrInterfaceType) newExpression.getExpression();
                    if (classOrInterfaceType.getImage().equals(NullPointerException.class.getName())) {
                        if (hasNoAssignmentsBeforeThrow(node, exceptions.getJavaNodeList())) {
                            addViolation(this, data, node);
                            break;
                        }
                    }
                }
            }
        }
        return super.visit(node, data);
    }

    private boolean hasNoAssignmentsBeforeThrow(ASTThrowStatement node, List<JavaNode> exceptions) {
        for (int i = exceptions.size() - 1; i > 0; i--) {
            JavaNode exception = exceptions.get(i);
            if (exception instanceof ASTAssignExpr) {
                ASTAssignExpr assignExpr = (ASTAssignExpr) exception;
                if (assignExpr.getVariable().getImage().equals(((ASTName) exceptions.get(i - 1)).getImage())) {
                    return false;
                }
            }
        }
        return true;
    }
