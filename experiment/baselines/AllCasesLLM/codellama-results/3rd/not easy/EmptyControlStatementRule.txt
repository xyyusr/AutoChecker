You are an expert in writing java rule checkers and I need your help to generate a custom java rule checker in PMD tool version 7.0.0. 
I will give you a rule description and its several test cases.

The following is the description of the rule:
Rule description: Reports control statements whose body is empty, as well as empty initializers. The checked code constructs are the following: bodies of try statements、 finally clauses of try statements、 switch statements、 synchronized statements、 if statements、 loop statements: while, for, do .. while、initializers and blocks used as statements (for scoping).;

package net.sourceforge.pmd.lang.java.rule.codestyle;

import java.util.List;
import java.util.Map;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class EmptyControlStatementRule extends AbstractJavaRulechainRule {

    @Override
    public Object visit(ASTTryStatement node, Object data) {
        Object result = super.visit(node, data);
        if (node.getBody() != null && node.getBody().size() == 0) {
            addRuleViolation(node, "Empty try statement body");
        }
        if (node.getFinally() != null && node.getFinally().size() == 0) {
            addRuleViolation(node.getFinally().get(0), "Empty try statement finally body");
        }
        return result;
    }

    @Override
    public Object visit(ASTTryResource node, Object data) {
        Object result = super.visit(node, data);
        if (node.getVariableDeclaration() != null && node.getVariableDeclaration().size() == 0) {
            addRuleViolation(node, "Empty try-with-resource initializer");
        }
        return result;
    }

    @Override
    public Object visit(ASTFinallyStatement node, Object data) {
        Object result = super.visit(node, data);
        if (node.getBody() != null && node.getBody().size() == 0) {
            addRuleViolation(node, "Empty finally statement body");
        }
        return result;
    }

    @Override
    public Object visit(ASTEmptyStatement node, Object data) {
        Object result = super.visit(node, data);
        addRuleViolation(node, "Empty statement");
        return result;
    }

    @Override
    public Object visit(ASTEmptySwitchStatement node, Object data) {
        Object result = super.visit(node, data);
        addRuleViolation(node, "Empty switch statement");
        return result;
    }

    @Override
    public Object visit(ASTSynchronizedStatement node, Object data) {
        Object result = super.visit(node, data);
        if (node.getBody() != null && node.getBody().size() == 0) {
            addRuleViolation(node, "Empty synchronized statement body");
        }
        return result;
    }

    @Override
    public Object visit(ASTIfStatement node, Object data) {
        Object result = super.visit(node, data);
        if (node.getElseIf() == null && node.getElse() == null && node.getThen() == null || node.getThen().size() == 0) {
            addRuleViolation(node, "Empty if statement");
        }
        return result;
    }

    @Override
    public Object visit(ASTDoStatement node, Object data) {
        Object result = super.visit(node, data);
        if (node.getBody() != null && node.getBody().size() == 0) {
            addRuleViolation(node, "Empty do-while statement body");
        }
        return result;
    }

    @Override
    public Object visit(ASTForStatement node, Object data) {
        Object result = super.visit(node, data);
        if (node.getForInit() != null && node.getForInit().size() == 0) {
            addRuleViolation(node, "Empty for statement initializer");
        }
        if (node.getBody() != null && node.getBody().size() == 0) {
            addRuleViolation(node, "Empty for statement body");
        }
        return result;
    }

    @Override
    public Object visit(ASTForLoopStatement node, Object data) {
        Object result = super.visit(node, data);
        if (node.getBody() != null && node.getBody().size() == 0) {
            addRuleViolation(node, "Empty for loop statement body");
        }
        return result;
    }

    @Override
    public Object visit(ASTWhileStatement node, Object data) {
        Object result = super.visit(node, data);
        if (node.getBody() != null && node.getBody().size() == 0) {
            addRuleViolation(node, "Empty while statement body");
        }
        return result;
    }

    @Override
    public Object visit(ASTSwitchStatement node, Object data) {
        Object result = super.visit(node, data);
        if (node.getStatements() == null || node.getStatements().size() == 0) {
            addRuleViolation(node, "Empty switch statement");
        }
        return result;
    }

    @Override
    public Object visit(ASTBlock node, Object data) {
        Object result = super.visit(node, data);
        if (node.getStatements() != null && node.getStatements().size() == 0) {
            addRuleViolation(node, "Empty block");
        }
        return result;
    }

    @Override
    public Object visit(ASTStaticInitializer node, Object data) {
        Object result = super.visit(node, data);
        if (node.getBody() != null && node.getBody().size() == 0) {
            addRuleViolation(node, "Empty static initializer");
        }
        return result;
    }

    @Override
    public Object visit(ASTInitializer node, Object data) {
        Object result = super.visit(node, data);
        if (node.getBody() != null && node.getBody().size() == 0) {
            addRuleViolation(node, "Empty initializer");
        }
        return result;
    }
