package net.sourceforge.pmd.lang.java.rule.errorprone;

import java.util.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class BrokenNullCheckRule extends AbstractJavaRulechainRule {

    @Override
    public Object visit(AstIfStmt n, Object data) {
        if (!n.hasElse()) {
            AstExpression cond = n.getCondition();
            if (cond instanceof AstBinaryOperator) {
                AstBinaryOperator binOp = (AstBinaryOperator) cond;
                if (binOp.getLeftOperand() instanceof AstNullLiteral) {
                    if (binOp.getRightOperand() instanceof AstPrefixExpr
                            && ((AstPrefixExpr) binOp.getRightOperand()).getOperator() == AstPrefixExpr.Operator.NOT_EQUAL) {
                        checkNotEqual(binOp.getLeftOperand(), binOp.getRightOperand());
                    } else if (binOp.getRightOperand() instanceof AstPrefixExpr
                            && ((AstPrefixExpr) binOp.getRightOperand()).getOperator() == AstPrefixExpr.Operator.EQUAL) {
                        checkEqual(binOp.getLeftOperand(), binOp.getRightOperand());
                    }
                } else if (binOp.getLeftOperand() instanceof AstPrefixExpr
                        && ((AstPrefixExpr) binOp.getLeftOperand()).getOperator() == AstPrefixExpr.Operator.NOT_EQUAL) {
                    checkNotEqual(binOp.getLeftOperand(), binOp.getRightOperand());
                } else if (binOp.getLeftOperand() instanceof AstPrefixExpr
                        && ((AstPrefixExpr) binOp.getLeftOperand()).getOperator() == AstPrefixExpr.Operator.EQUAL) {
                    checkEqual(binOp.getLeftOperand(), binOp.getRightOperand());
                }
            }
        }
        return super.visit(n, data);
    }

    private void checkNotEqual(AstExpression left, AstExpression right) {
        if (left instanceof AstPrefixExpr && ((AstPrefixExpr) left).getOperator() == AstPrefixExpr.Operator.NOT_EQUAL) {
            checkNotEqual(left.getOperand(), right);
        } else if (left instanceof AstBinaryOperator) {
            AstBinaryOperator binOp = (AstBinaryOperator) left;
            if (binOp.getLeftOperand() instanceof AstNullLiteral) {
                if (binOp.getRightOperand() instanceof AstPrefixExpr
                        && ((AstPrefixExpr) binOp.getRightOperand()).getOperator() == AstPrefixExpr.Operator.EQUAL) {
                    checkEqual(binOp.getLeftOperand(), binOp.getRightOperand());
                }
            } else if (binOp.getRightOperand() instanceof AstPrefixExpr
                    && ((AstPrefixExpr) binOp.getRightOperand()).getOperator() == AstPrefixExpr.Operator.EQUAL) {
                checkEqual(binOp.getLeftOperand(), binOp.getRightOperand());
            } else if (binOp.getRightOperand() instanceof AstNullLiteral) {
                if (binOp.getLeftOperand() instanceof AstPrefixExpr
                        && ((AstPrefixExpr) binOp.getLeftOperand()).getOperator() == AstPrefixExpr.Operator.EQUAL) {
                    checkEqual(binOp.getLeftOperand(), binOp.getRightOperand());
                }
            }
        }
    }

    private void checkEqual(AstExpression left, AstExpression right) {
        if (left instanceof AstPrefixExpr && ((AstPrefixExpr) left).getOperator() == AstPrefixExpr.Operator.EQUAL) {
            checkEqual(left.getOperand(), right);
        } else if (left instanceof AstBinaryOperator) {
            AstBinaryOperator binOp = (AstBinaryOperator) left;
            if (binOp.getLeftOperand() instanceof AstNullLiteral) {
                if (binOp.getRightOperand() instanceof AstPrefixExpr
                        && ((AstPrefixExpr) binOp.getRightOperand()).getOperator() == AstPrefixExpr.Operator.NOT_EQUAL) {
                    checkNotEqual(binOp.getLeftOperand(), binOp.getRightOperand());
                }
            } else if (binOp.getRightOperand() instanceof AstPrefixExpr
                    && ((AstPrefixExpr) binOp.getRightOperand()).getOperator() == AstPrefixExpr.Operator.NOT_EQUAL) {
                checkNotEqual(binOp.getLeftOperand(), binOp.getRightOperand());

package net.sourceforge.pmd.lang.java.rule.errorprone;

import java.util.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class BrokenNullCheckRule extends AbstractJavaRulechainRule {

    public BrokenNullCheckRule() {
        addRuleChainVisitor(this);
    }

    @Override
    public Object visit(IfStatement n, Object data) {
        if (!isAndOrOperator(n)) {
            return super.visit(n, data);
        }

        List<ASTNode> conditions = getConditions(n);
        if (conditions.size() != 2) {
            return super.visit(n, data);
        }

        ASTNode condition1 = conditions.get(0);
        ASTNode condition2 = conditions.get(1);

        if ((isMemberAccess(condition1) && isMemberAccess(condition2)) || (isMethodCall(condition1) && isMethodCall(condition2))) {
            return super.visit(n, data);
        }

        if (isPrimitiveCondition(condition1) && isPrimitiveCondition(condition2) && areSameVariable(condition1, condition2)) {
            return super.visit(n, data);
        }

        if (isUnboxingCondition(condition1) || isUnboxingCondition(condition2)) {
            return super.visit(n, data);
        }

        return addViolation(this, n, data);
    }

    @Override
    public Object visit(MethodCallExpression n, Object data) {
        if (isNullCheck(n)) {
            return super.visit(n, data);
        }

        if (isMemberAccess(n)) {
            return super.visit(n, data);
        }

        return super.visit(n, data);
    }

    @Override
    public Object visit(MemberAccessExpression n, Object data) {
        if (isNullCheck(n)) {
            return super.visit(n, data);
        }

        if (isMemberAccess(n)) {
            return super.visit(n, data);
        }

        return super.visit(n, data);
    }

    private List<ASTNode> getConditions(IfStatement n) {
        List<ASTNode> conditions = new ArrayList<>();
        for (Object condition : n.getConditionList().getList()) {
            conditions.add((ASTNode) condition);
        }
        return conditions;
    }

    private boolean isAndOrOperator(IfStatement n) {
        return n.getConditionList().getOperator() == AndExpression.class || n.getConditionList().getOperator() == OrExpression.class;
    }

    private boolean isMemberAccess(ASTNode n) {
        return n instanceof MemberAccessExpression;
    }

    private boolean isMethodCall(ASTNode n) {
        return n instanceof MethodCallExpression;
    }

    private boolean isPrimitiveCondition(ASTNode n) {
        return (n instanceof PrimitiveExpression) || (n instanceof ArrayAccessExpression) || (n instanceof ArrayLengthExpression);
    }

    private boolean areSameVariable(ASTNode n1, ASTNode n2) {
        if (n1 instanceof VariableDeclaratorId && n2 instanceof VariableDeclaratorId) {
            String name1 = ((VariableDeclaratorId) n1).getImage();
            String name2 = ((VariableDeclaratorId) n2).getImage();
            return name1.equals(name2);
        }

        return false;
    }

    private boolean isUnboxingCondition(ASTNode n) {
        return n instanceof NullLiteral || n instanceof IntegerLiteral || n instanceof FloatLiteral || n instanceof BooleanLiteral || n instanceof CharacterLiteral || n instanceof StringLiteral;
    }

    private boolean isNullCheck(ASTNode n) {
        if (n instanceof NullLiteral) {
            return true;
        }

        if (n instanceof MethodCallExpression) {
            MethodCallExpression methodCall = (MethodCallExpression) n;
            if (methodCall.getImage().equals("equals") && methodCall.getArguments().size() == 1 && methodCall.getArguments().get(0) instanceof NullLiteral) {
                return true;
            }
        }

        return false;
    }
}
