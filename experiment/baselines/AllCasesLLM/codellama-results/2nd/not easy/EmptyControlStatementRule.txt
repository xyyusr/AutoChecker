You are an expert in writing java rule checkers and I need your help to generate a custom java rule checker in PMD tool version 7.0.0. 
I will give you a rule description and its several test cases.

The following is the description of the rule:
Rule description: Reports control statements whose body is empty, as well as empty initializers. The checked code constructs are the following: bodies of try statements、 finally clauses of try statements、 switch statements、 synchronized statements、 if statements、 loop statements: while, for, do .. while、initializers and blocks used as statements (for scoping).;

            }

package net.sourceforge.pmd.lang.java.rule.codestyle;

import java.util.*;
import java.lang.*;
import java.util.stream.Collectors;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class EmptyControlStatementRule extends AbstractJavaRulechainRule {

    public EmptyControlStatementRule() {
        super.addRuleMessage(CodeSmellRuleMessageDescriptor.getInstance()
                .getMessageTemplate("EmptyControlStatementRule-1"));
    }

    @Override
    public Object visit(ASTTryStatement node, Object data) {
        if (node.getTryBlock().getStatements().isEmpty()
                && (node.getFinallyBlock() == null || (node.getFinallyBlock() != null && node.getFinallyBlock().getStatements().isEmpty()))) {
            addViolation(this, data, node);
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTFinallyStatement node, Object data) {
        if (node.getStatements().isEmpty()) {
            addViolation(this, data, node);
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTSwitchStatement node, Object data) {
        if (node.getStatements().isEmpty()) {
            addViolation(this, data, node);
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASynchronizedStatement node, Object data) {
        if (node.getStatement().getStatements().isEmpty()) {
            addViolation(this, data, node);
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTIfStatement node, Object data) {
        if (node.getThenStatement() == null || (node.getThenStatement() != null && node.getThenStatement().getStatements().isEmpty())) {
            addViolation(this, data, node);
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTForStatement node, Object data) {
        if (node.getStatements().isEmpty()) {
            addViolation(this, data, node);
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTDoStatement node, Object data) {
        if (node.getBody().getStatements().isEmpty()) {
            addViolation(this, data, node);
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTForLoop node, Object data) {
        if (node.getStatements().isEmpty()) {
            addViolation(this, data, node);
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTEnhancedForLoop node, Object data) {
        if (node.getStatements().isEmpty()) {
            addViolation(this, data, node);
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTWhileStatement node, Object data) {
        if (node.getBody().getStatements().isEmpty()) {
            addViolation(this, data, node);
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(AScopeStatement node, Object data) {
        if (node.getBody().getStatements().isEmpty()) {
            addViolation(this, data, node);
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(AScope block, Object data) {
        if (block.getStatements().isEmpty()) {
            addViolation(this, data, block);
        }
        return super.visit(block, data);
    }

    @Override
    public Object visit(ASTBlockStatement node, Object data) {
        if (node.getStatements().isEmpty()) {
            addViolation(this, data, node);
        }
        return super.visit(node, data);
    }

    public Object visit(ASTInitializer node, Object data) {
        if (node.getStatements().isEmpty()) {

        } else if (node.getFinallyBlock() != null && node.getFinallyBlock().getChildren().isEmpty()) {

        } else {

    @Override
    public Object visit(ASTTryCatchFinallyStatement tryCatchFinallyStatement, Object data) {
        if (isTryBlockEmpty(tryCatchFinallyStatement.getTryBlock()) && isBlockedEmpty(tryCatchFinallyStatement.getFinallyBlock())) {
            addViolation(data, tryCatchFinallyStatement.getTryBlock());
            addViolation(data, tryCatchFinallyStatement.getFinallyBlock());
        }
        return super.visit(tryCatchFinallyStatement, data);
    }

    @Override
    public Object visit(ASTTryWithResourcesStatement tryWithResourcesStatement, Object data) {
        if (tryWithResourcesStatement.getTryBlock() == null || tryWithResourcesStatement.getTryBlock().isEmpty()) {
            addViolation(data, tryWithResourcesStatement.getTryBlock());
        }
        return super.visit(tryWithResourcesStatement, data);
    }

    @Override
    public Object visit(ASTEmptyStatement emptyStatement, Object data) {
        addViolation(data, emptyStatement);
        return super.visit(emptyStatement, data);
    }

    private boolean isBlockedEmpty(ASTBlock block) {
        return block.getStatements() == null || block.getStatements().isEmpty();
    }

    private boolean isTryBlockEmpty(ASTBlock tryBlock) {
        ASTBlock block = (ASTBlock) tryBlock;
        return block.getStatements() == null || block.getStatements().isEmpty();
    }

    private void addViolation(Object data, Object node) {
        if (node instanceof JavaNode) {
            addRuleViolation(data, (JavaNode) node);
        } else if (node instanceof ASTInitializers) {
            ASTInitializers initializers = (ASTInitializers) node;
            for (JavaNode statement : initializers.getStatements()) {
                addRuleViolation(data, statement);
            }
        }
    }

    private void addRuleViolation(Object data, JavaNode node) {
        addViolation(data, node.getImage());
    }
