import java.util.*;
import java.lang.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ExcessivePublicCountRule extends AbstractJavaRulechainRule {
    private int threshold = 2;

    public ExcessivePublicCountRule() {
        super(MethodDeclaration.class, FieldDeclaration.class);
    }

    @Override
    public Object visit(MethodDeclaration node, Object data) {
        if (node.isPublic() && node.getName().isIdentifier() && node.isStatic() && !node.isFinal()) {
            int methodCount = 1;
            MethodDeclaration parent = (MethodDeclaration) node.getParent();
            while (parent != null && parent instanceof MethodDeclaration) {
                if (((MethodDeclaration) parent).isPublic() && ((MethodDeclaration) parent).getName().isIdentifier() && !((MethodDeclaration) parent).isFinal()) {
                    methodCount++;
                }
                parent = (MethodDeclaration) parent.getParent();
            }
            if (methodCount > threshold) {
                addViolation(data, node);
            }
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(FieldDeclaration node, Object data) {
        if (node.isPublic() && node.getType() instanceof PrimitiveType) {
            int fieldCount = 1;
            FieldDeclaration parent = (FieldDeclaration) node.getParent();
            while (parent != null && parent instanceof FieldDeclaration) {
                if (((FieldDeclaration) parent).isPublic() && ((FieldDeclaration) parent).getType() instanceof PrimitiveType) {
                    fieldCount++;
                }
                parent = (FieldDeclaration) parent.getParent();
            }
            if (fieldCount > threshold) {
                addViolation(data, node);
            }
        }
        return super.visit(node, data);
    }
}

package net.sourceforge.pmd.lang.java.rule.design;

import java.util.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ExcessivePublicCountRule extends AbstractJavaRulechainRule {
    private int checkThreshold = 2;

    public ExcessivePublicCountRule() {
        addRuleMatch(ASTClassOrInterfaceDeclaration.class, this::visitClass);
    }

    @Override
    public Object visitClass(ASTClassOrInterfaceDeclaration node, Object data) {
        if (!isPublic(node)) {
            return data;
        }

        int publicFieldCount = 0;
        int publicMethodCount = 0;
        int staticFinalCount = 0;

        for (JavaNode child : node.jjtGetChildren()) {
            if (child instanceof ASTFieldDeclaration) {
                ASTFieldDeclaration field = (ASTFieldDeclaration) child;
                if (isPublic(field)) {
                    publicFieldCount++;
                } else if (isStaticFinal(field)) {
                    staticFinalCount++;
                }
            } else if (child instanceof ASTMethodDeclaration) {
                ASTMethodDeclaration method = (ASTMethodDeclaration) child;
                if (isPublic(method)) {
                    publicMethodCount++;
                }
            } else if (child instanceof ASTStaticInitializer) {
                ASTStaticInitializer initializer = (ASTStaticInitializer) child;
                if (isStaticFinal(initializer)) {
                    staticFinalCount++;
                }
            }
        }

        if (publicFieldCount > checkThreshold || publicMethodCount > checkThreshold) {
            addViolation(data, this, node);
        } else if (publicFieldCount > 0 || publicMethodCount > 0) {
            // Check if all fields and methods are static finals
            if (publicFieldCount == staticFinalCount && publicMethodCount == staticFinalCount) {
                if (staticFinalCount > checkThreshold) {
                    addViolation(data, this, node);
                }
            }
        }

        return data;
    }

    private boolean isPublic(JavaNode node) {
        SymbolTable symbolTable = node.getSymbolTable();
        SymbolType symbolType = symbolTable.lookup(node.getImage());
        return symbolType.isPublic();
    }

    private boolean isStaticFinal(JavaNode node) {
        SymbolTable symbolTable = node.getSymbolTable();
        SymbolType symbolType = symbolTable.lookup(node.getImage());
        return symbolType.isStatic() && symbolType.isFinal();
    }
}

package net.sourceforge.pmd.lang.java.rule.design;

import java.util.*;
import java.lang.*;
import java.util.stream.Collectors;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ExcessivePublicCountRule extends AbstractJavaRulechainRule {

    public ExcessivePublicCountRule() {
        super("excessive-public-count-rule", "Classes with large numbers of public methods and attributes require disproportionate testing efforts since combinational side effects grow rapidly and increase risk.");
    }

    @Override
    public Object visit(ClassOrInterfaceDeclaration node, Object data) {
        int publicCount = (int) data;

        if (node.modifiers().stream()
                .filter(ModifierDeclaration.class::isInstance)
                .map(ModifierDeclaration.class::cast)
                .anyMatch(modifier -> modifier.isPublic())) {
            publicCount++;
        }

        int methods = node.bodyDeclarations().stream()
                .filter(MethodDeclaration.class::isInstance)
                .map(MethodDeclaration.class::cast)
                .filter(method -> method.modifiers().stream()
                        .anyMatch(modifier -> modifier.isPublic()))
                .count();

        int fields = node.bodyDeclarations().stream()
                .filter(VariableDeclarator.class::isInstance)
                .map(VariableDeclarator.class::cast)
                .filter(field -> field.modifiers().stream()
                        .anyMatch(modifier -> modifier.isPublic()))
                .count();

        if (publicCount > 2) {
            addViolation(data, this, node, "Classes with large numbers of public methods and attributes require disproportionate testing efforts since combinational side effects grow rapidly and increase risk.");
        }

        return data;
    }

    @Override
    public Object visit(ModifierDeclaration node, Object data) {
        if (node.isPublic()) {
            data = (int) data + 1;
        }
        return data;
    }

    @Override
    public Object visit(MethodDeclaration node, Object data) {
        if (node.modifiers().stream()
                .anyMatch(modifier -> modifier.isPublic())) {
            data = (int) data + 1;
        }
        return data;
    }

    @Override
    public Object visit(VariableDeclarator node, Object data) {
        if (node.modifiers().stream()
                .anyMatch(modifier -> modifier.isPublic())) {
            data = (int) data + 1;
        }
        return data;
    }

    @Override
    public Object visit(ClassOrInterfaceType node, Object data) {
        data = 0;
        return data;
    }
}

package net.sourceforge.pmd.lang.java.rule.design;

import java.util.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ExcessivePublicCountRule extends AbstractJavaRulechainRule {

    private static final int DEFAULT_THRESHOLD = 2;

    public ExcessivePublicCountRule() {
        addRuleChainVisitorClass(JavaClass.class);
        addRuleChainVisitorClass(JavaField.class);
        addRuleChainVisitorClass(JavaMethod.class);
    }

    @Override
    public Object visit(JavaClass node, Object data) {
        if (node.isPublic() && hasTooManyPublicFields(node) || hasTooManyPublicMethods(node)) {
            addViolation(data, this, node);
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(JavaField node, Object data) {
        if (!node.isStatic() && !node.isFinal()) {
            return super.visit(node, data);
        }
        if (node.getScope() instanceof JavaClass) {
            JavaClass parentClass = (JavaClass) node.getScope();
            if (parentClass.isPublic() && hasTooManyPublicFields(parentClass)) {
                addViolation(data, this, node);
            }
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(JavaMethod node, Object data) {
        if (!node.isStatic() && !node.isFinal()) {
            return super.visit(node, data);
        }
        if (node.getScope() instanceof JavaClass) {
            JavaClass parentClass = (JavaClass) node.getScope();
            if (parentClass.isPublic() && hasTooManyPublicMethods(parentClass)) {
                addViolation(data, this, node);
            }
        }
        return super.visit(node, data);
    }

    private boolean hasTooManyPublicFields(JavaClass node) {
        int count = 0;
        for (Object member : node.getFields()) {
            JavaField field = (JavaField) member;
            if (field.isPublic()) {
                count++;
            }
        }
        return count > DEFAULT_THRESHOLD;
    }

    private boolean hasTooManyPublicMethods(JavaClass node) {
        int count = 0;
        for (Object member : node.getMethods()) {
            JavaMethod method = (JavaMethod) member;
            if (method.isPublic()) {
                count++;
            }
        }
        return count > DEFAULT_THRESHOLD;
    }
}
