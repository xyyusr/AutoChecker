package net.sourceforge.pmd.lang.java.rule.codestyle;

import java.util.*;
import java.lang.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class IdenticalCatchBranchesRule extends AbstractJavaRulechainRule {
    public IdenticalCatchBranchesRule() {
        super("IdenticalCatchBranchesRule", "Identical catch branches use up vertical space and increase the complexity of code without adding functionality.");
    }

    @Override
    public Object visit(CatchClause node, Object data) {
        ASTNode parent = node.getParent();
        if (parent instanceof TryStatement) {
            return super.visit(node, data);
        }

        Map<String, CatchClause> catchClauses = new HashMap<>();
        catchClauses.put(node.getException().getImage(), node);

        parent = parent.getParent();
        if (parent instanceof TryStatement) {
            ASTNode currentCatch = null;
            for (ASTNode child : parent.getChildren()) {
                if (child instanceof CatchClause) {
                    CatchClause catchClause = (CatchClause) child;
                    String exceptionType = catchClause.getException().getImage();
                    if (exceptionType.equals(node.getException().getImage())) {
                        if (currentCatch != null) {
                            catchClauses.merge(currentCatch.getCatchClauses(), catchClause.getCatchClauses(), (oldValue, newValue) -> {
                                oldValue.put(catchClause.getException().getImage(), catchClause);
                                return oldValue;
                            });
                        } else {
                            catchClauses.put(exceptionType, catchClause);
                        }
                        currentCatch = catchClause;
                    } else {
                        if (currentCatch != null) {
                            currentCatch = null;
                        }
                    }
                }
            }
            if (!catchClauses.values().stream().allMatch(catchClause -> catchClauses.values().stream().distinct().count() == 1)) {
                addViolation(this, node, data);
            }
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(TryStatement node, Object data) {
        return super.visit(node, data);
    }

    @Override
    public Object visit(ThrowableDeclaration node, Object data) {
        return super.visit(node, data);
    }

    @Override
    public Object visit(VariableDeclaration node, Object data) {
        return super.visit(node, data);
    }
}

package net.sourceforge.pmd.lang.java.rule.codestyle;

import java.util.*;
import java.lang.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class IdenticalCatchBranchesRule extends AbstractJavaRulechainRule {

    public IdenticalCatchBranchesRule() {
        addRuleMessage("Identical catch branches can be collapsed into a single multi-catch branch.");
        addRulePriority(RulePriority.NORMAL);
        addRulePattern("cs=ExceptionName");
    }

    @Override
    public Object visit(ASTTryStatementNode node, Object data) {
        if (node.isFinallyBlock() || node.isCatchBlock()) {
            return super.visit(node, data);
        }

        List<ASTCatchNode> catchNodes = node.findDescendants(ASTCatchNode.class);
        if (catchNodes != null && catchNodes.size() > 1) {
            Map<String, List<ASTCatchNode>> catchMap = new HashMap<>();
            for (ASTCatchNode catchNode : catchNodes) {
                String exceptionName = catchNode.findFirstDescendant(ASTClassOrInterfaceType.class).getImage();
                catchMap.computeIfAbsent(exceptionName, k -> new ArrayList<>()).add(catchNode);
            }

            for (List<ASTCatchNode> catchList : catchMap.values()) {
                if (catchList.size() > 1) {
                    List<ASTCatchNode> identicalCatchNodes = new ArrayList<>();
                    String firstException = catchList.get(0).findFirstDescendant(ASTClassOrInterfaceType.class).getImage();
                    for (ASTCatchNode catchNode : catchList) {
                        if (firstException.equals(catchNode.findFirstDescendant(ASTClassOrInterfaceType.class).getImage())) {
                            identicalCatchNodes.add(catchNode);
                        }
                    }
                    addViolation(data, node, identicalCatchNodes);
                }
            }
        }

        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTCatchesNode node, Object data) {
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTCatchNode node, Object data) {
        return super.visit(node, data);
    }
}

package net.sourceforge.pmd.lang.java.rule.codestyle;

import java.util.*;
import java.lang.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class IdenticalCatchBranchesRule extends AbstractJavaRulechainRule {
    public IdenticalCatchBranchesRule() {
        super(TRY_STATEMENT, CATCH_CLAUSE);
    }

    @Override
    public Object visit(TryStatement node, Object data) {
        if (isSingleCatch(node)) {
            return super.visit(node, data);
        }

        Map<String, List<ASTCatchClause>> catchClauses = new HashMap<>();
        for (ASTCatchClause catchClause : node.getCatchClauses()) {
            String exceptionName = catchClause.getException().getImage();
            if (catchClauses.containsKey(exceptionName)) {
                catchClauses.get(exceptionName).add(catchClause);
            } else {
                catchClauses.put(exceptionName, new ArrayList<>(Arrays.asList(catchClause)));
            }
        }

        for (List<ASTCatchClause> catchClausesList : catchClauses.values()) {
            if (catchClausesList.size() > 1 && areCatchClausesIdentical(catchClausesList)) {
                addViolation(this, node, data, "Identical catch branches can be collapsed into a single multi-catch branch");
            }
        }

        return super.visit(node, data);
    }

    private boolean isSingleCatch(TryStatement node) {
        return node.getCatchClauses().size() == 1;
    }

    private boolean areCatchClausesIdentical(List<ASTCatchClause> catchClausesList) {
        ASTCatch catchClause = catchClausesList.get(0).getCatch();
        for (int i = 1; i < catchClausesList.size(); i++) {
            ASTCatch otherCatchClause = catchClausesList.get(i).getCatch();
            if (!catchClause.getException().getImage().equals(otherCatchClause.getException().getImage())) {
                return false;
            }
            if (!throwExpressionEqual(catchClause.getThrowStatement(), otherCatchClause.getThrowStatement())) {
                return false;
            }
        }
        return true;
    }

    private boolean throwExpressionEqual(ThrowStatement throwStatement1, ThrowStatement throwStatement2) {
        if (throwStatement1 == null || throwStatement2 == null) {
            return throwStatement1 == throwStatement2;
        }
        if (!throwStatement1.getExpression().getImage().equals(throwStatement2.getExpression().getImage())) {
            return false;
        }
        if (throwStatement1.getExpression() instanceof ASTMethodInvocation && throwStatement2.getExpression() instanceof ASTMethodInvocation) {
            MethodInvocation mi1 = (MethodInvocation) throwStatement1.getExpression();

        NodeStream nbs = new NodeStream(node.getStatement().getFirstChild());

    @Override
    public Object visit(JavaTryBlock n, Object data) {
        List<JavaCatchClause> catchClauses = n.getCatchClauses();
        if (catchClauses == null) {
            return data;
        }
        Map<String, List<JavaCatchClause>> exceptions = new HashMap<>();
        for (JavaCatchClause catchClause : catchClauses) {

                addViolation(this, n, data);
            }
        }

    }
