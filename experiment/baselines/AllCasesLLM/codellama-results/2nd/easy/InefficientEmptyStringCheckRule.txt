package net.sourceforge.pmd.lang.java.rule.performance;

import java.util.*;
import java.lang.*;
import java.util.regex.Matcher;
import java.util.regex.Pattern;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class InefficientEmptyStringCheckRule extends AbstractJavaRulechainRule {
    public InefficientEmptyStringCheckRule() {
        super(MethodCallExpr.class, MethodInvocationExpr.class, StringLiteralExpr.class);
    }

    @Override
    public Object visit(MethodCallExpr node, Object data) {
        String methodName = node.getImage();
        if (methodName.equals("trim") || methodName.equals("isEmpty")) {
            Expression containingExpr = node.getExpression();
            if (containingExpr instanceof IdentifierExpr && ((IdentifierExpr) containingExpr).getImage().equals("length")) {
                addViolation(data, node);
            } else if (containingExpr instanceof LiteralExpr) {
                if (methodName.equals("isEmpty")) {
                    addViolation(data, node);
                }
            } else if (containingExpr instanceof MethodCallExpr) {
                String containingMethodName = ((MethodCallExpr) containingExpr).getImage();
                if (containingMethodName.equals("trim") && methodName.equals("isEmpty")) {
                    addViolation(data, node);
                }
                if (containingMethodName.equals("toString") && methodName.equals("isEmpty")) {
                    addViolation(data, node);
                }
            }
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(MethodInvocationExpr node, Object data) {
        Expression target = node.getTarget();
        String methodName = node.getImage();
        if (target instanceof MethodCallExpr) {
            return super.visit(node, data);
        }
        if (target instanceof IdentifierExpr && ((IdentifierExpr) target).getImage().equals("length")) {
            if (methodName.equals("isEmpty")) {
                addViolation(data, node);
            }
        }
        if (methodName.equals("isEmpty")) {
            if (target instanceof StringLiteralExpr) {
                addViolation(data, node);
            }
            if (target instanceof MethodCallExpr) {
                String containingMethodName = ((MethodCallExpr) target).getImage();
                if (containingMethodName.equals("trim") && methodName.equals("isEmpty")) {
                    addViolation(data, node);
                }
                if (containingMethodName.equals("toString") && methodName.equals("isEmpty")) {
                    addViolation(data, node);
                }
            }
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(StringLiteralExpr node, Object data) {
        Pattern p = Pattern.compile("\\.trim\\(\\)\\.isEmpty\\(\\)");
        Matcher m = p.matcher(node.getImage());
        if (m.find()) {
            addViolation(data, node);
        }
        return super.visit(node, data);
    }
}
