package net.sourceforge.pmd.lang.java.rule.errorprone;


import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class NullAssignmentRule extends AbstractJavaRulechainRule {
    public NullAssignmentRule() {
        super(ASTAssignmentExpression.class);
    }

    @Override
    public Object visit(ASTAssignmentExpression node, Object data) {
        ASTPrimaryExpression leftOperand = node.getLeftOperand();
        if (leftOperand instanceof ASTName) {
            ASTName name = (ASTName) leftOperand;
            JVariableSymbol symbol = name.getNameDeclaration().getSymbol();
            if (!(symbol.getNode() instanceof ASTVariableDeclarator) && !(symbol.getNode() instanceof ASTFieldDeclaration) && node.getRightOperand() instanceof ASTLiteral && "null".equals(((ASTLiteral) node.getRightOperand()).getImage())) {
                addViolation(data, node);
            }
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTTernaryExpression node, Object data) {
        if (node.getTrueExpression() instanceof ASTLiteral && "null".equals(((ASTLiteral) node.getTrueExpression()).getImage())) {
            addViolation(data, node);
        }
        if (node.getFalseExpression() instanceof ASTLiteral && "null".equals(((ASTLiteral) node.getFalseExpression()).getImage())) {
            addViolation(data, node);
        }
        return super.visit(node, data);
    }
}

package net.sourceforge.pmd.lang.java.rule.errorprone;


import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class NullAssignmentRule extends AbstractJavaRulechainRule {
    public NullAssignmentRule() {
        super(ASTAssignmentExpression.class, ASTMethodCall.class, ASTTernaryExpression.class, ASTVariableDeclarator.class);
    }

    @Override
    public Object visit(ASTAssignmentExpression node, Object data) {
        if (node.getType() == ASTAssignmentExpression.Type.ASSIGN && node.getRightOperand().isLiteralOrNull()) {
            if (node.getRightOperand().isLiteralOrNull() && node.getRightOperand().getValue() == null) {
                if (!(node.getLeftOperand() instanceof ASTVariableDeclarator)) {
                    addViolation(data, node);
                }
            }
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTMethodCall node, Object data) {
        for (int i = 0; i < node.getArgumentCount(); i++) {
            if (node.getArgument(i).isLiteralOrNull() && node.getArgument(i).getValue() == null) {
                addViolation(data, node);
            }
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTTernaryExpression node, Object data) {
        if (node.getFalseExpression().isLiteralOrNull() && node.getFalseExpression().getValue() == null) {
            addViolation(data, node);
        }
        if (node.getTrueExpression().isLiteralOrNull() && node.getTrueExpression().getValue() == null) {
            addViolation(data, node);
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTVariableDeclarator node, Object data) {
        if (node.getInitializer() != null && node.getInitializer().isLiteralOrNull() && node.getInitializer().getValue() == null) {
            return super.visit(node, data);
        }
        return super.visit(node, data);
    }
}

package net.sourceforge.pmd.lang.java.rule.errorprone;


import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class NullAssignmentRule extends AbstractJavaRulechainRule {

    public NullAssignmentRule() {
        super(ASTAssignmentExpression.class, ASTConditionalExpression.class, ASTVariableDeclarator.class);
    }

    @Override
    public Object visit(ASTAssignmentExpression node, Object data) {
        if (node.getLeftOperand() instanceof ASTVariableDeclarator) {
            return super.visit(node, data);
        }
        if (node.getRightOperand().isNullLiteral()) {
            JVariableSymbol symbol = node.getLeftOperand().getVariableSymbol();
            if (symbol != null && !isFinalAndInitialized(symbol)) {
                addViolation(data, node);
            }
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTConditionalExpression node, Object data) {
        if (node.getTrueExpression().isNullLiteral() || node.getFalseExpression().isNullLiteral()) {
            JVariableSymbol symbol = getAssignedVariable(node.getParent());
            if (symbol != null && !isFinalAndInitialized(symbol)) {
                addViolation(data, node);
            }
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTVariableDeclarator node, Object data) {
        if (node.getInitializer() != null && node.getInitializer().isNullLiteral()) {
            addViolation(data, node);
        }
        return super.visit(node, data);
    }

    private JVariableSymbol getAssignedVariable(JavaNode node) {
        while (node != null) {
            if (node instanceof ASTAssignmentExpression) {
                return ((ASTAssignmentExpression) node).getLeftOperand().getVariableSymbol();
            }
            node = node.getParent();
        }
        return null;
    }

    private boolean isFinalAndInitialized(JVariableSymbol symbol) {
        return symbol.isFinal() && symbol.hasExplicitInitializer();
    }
}
