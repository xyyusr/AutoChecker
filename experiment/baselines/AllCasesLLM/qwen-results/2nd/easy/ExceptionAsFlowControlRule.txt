You are an expert in writing java rule checkers and I need your help to generate a custom java rule checker in PMD tool version 7.0.0. 
I will give you a rule description and its several test cases.

The following is the description of the rule:
Rule description: This rule reports exceptions thrown and caught in an enclosing try statement. This use of exceptions as a form of goto statement is discouraged, as that may hide actual exceptions, and obscures control flow, especially when debugging. To fix a violation, add the necessary validation or use an alternate control structure.;

The following are the test cases of the rule:
1. This is a negative test case
```java
// failure case
public class Foo {
    void bar() {
        try {
            try {
            } catch (Exception e) {
                throw new WrapperException(e);
                // this is essentially a GOTO to the WrapperException catch block
            }
        } catch (WrapperException e) {
            // do some more stuff
        }
    }
}
class WrapperException extends Exception {}
```
2. This is a positive test case
```java
// normal throw catch
public class Foo {
    void bar() {
        try {} catch (Exception e) {}
    }
}
```

```java

        try {
        } catch (IOException e) {
            if (foo!=null)
                throw new IOException(foo.toString());
            else
                throw e;
        }
    }
}
```

package net.sourceforge.pmd.lang.java.rule.design;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ExceptionAsFlowControlRule extends AbstractJavaRulechainRule {
    public ExceptionAsFlowControlRule() {
        super(ASTTryStatement.class, ASTCatchStatement.class, ASTThrowStatement.class);
    }
    @Override
    public Object visit(ASTTryStatement node, Object data) {
        return super.visit(node, data);
    }
    @Override
    public Object visit(ASTCatchStatement node, Object data) {
        ASTTryStatement tryStatement = node.getFirstParentOfType(ASTTryStatement.class);
        if (tryStatement == null) {
            return super.visit(node, data);
        }
        JClassType caughtExceptionType = node.getFirstChildOfType(ASTClassOrInterfaceType.class).getType();
        List<ASTThrowStatement> throwStatements = NodeStream.of(tryStatement)
            .filter(ASTThrowStatement.class)
            .collect(Collectors.toList());
        for (ASTThrowStatement throwStatement : throwStatements) {
            JClassType thrownExceptionType = throwStatement.getFirstChildOfType(ASTClassOrInterfaceType.class).getType();
            if (thrownExceptionType.isSubtypeOf(caughtExceptionType)) {
                addViolation(data, throwStatement);
            }
        }
        return super.visit(node, data);
    }
    @Override
    public Object visit(ASTThrowStatement node, Object data) {
        return super.visit(node, data);
    }
}

public class ExceptionAsFlowControlRule extends AbstractJavaRulechainRule {
    public ExceptionAsFlowControlRule() {
        super(ASTTryStatement.class, ASTCatchStatement.class);
    }

    @Override
    public Object visit(ASTTryStatement node, Object data) {
        NodeStream<ASTCatchStatement> catchStatements = node.getCatchStatements();
        for (ASTCatchStatement catchStmt : catchStatements) {
            ASTThrowStatement throwStmt = catchStmt.getFirstDescendantOfType(ASTThrowStatement.class);
            if (throwStmt != null) {
                JClassType caughtExceptionType = catchStmt.getCatchVariable().getType();
                JClassType thrownExceptionType = throwStmt.getExpression().getType();
                if (thrownExceptionType.isSubtypeOf(caughtExceptionType)) {
                    if (!caughtExceptionType.equals(thrownExceptionType)) {
                        addViolation(data, throwStmt);
                    }
                }
            }
        }
        return super.visit(node, data);

import java.util.ArrayList;
import java.util.List;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;

public class ExceptionAsFlowControlRule extends AbstractJavaRulechainRule {
    public ExceptionAsFlowControlRule() {
        super(ASTTryStatement.class);
    }

    @Override
    public Object visit(ASTTryStatement node, Object data) {
        List<ASTCatchStatement> catchStatements = node.findDescendantsOfType(ASTCatchStatement.class);
        for (ASTCatchStatement catchStmt : catchStatements) {
            ASTThrowStatement throwStmt = catchStmt.getFirstDescendantOfType(ASTThrowStatement.class);
            if (throwStmt != null) {
                JType caughtExceptionType = catchStmt.getFirstChildOfType(ASTFormalParameter.class).getType();
                JType thrownExceptionType = throwStmt.getFirstChildOfType(ASTExpression.class).getType();
                if (thrownExceptionType != null && caughtExceptionType != null) {
                    TypeSystem typeSystem = node.getTypeSystem();
                    if (typeSystem.isA(thrownExceptionType, caughtExceptionType) && !thrownExceptionType.equals(caughtExceptionType)) {
                        addViolation(data, throwStmt);
                    }
                }
            }
        }
        return super.visit(node, data);
    }
}

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;

        super(ASTTryStatement.class);
    }

    @Override
    public Object visit(ASTTryStatement node, Object data) {
        for (ASTCatchStatement catchStmt : node.getCatchBlocks()) {
            JClassType caughtType = catchStmt.getFirstChildOfType(ASTClassOrInterfaceType.class).getType();
            for (ASTThrowStatement throwStmt : catchStmt.findDescendantsOfType(ASTThrowStatement.class)) {
                JClassType thrownType = TypeSystem.getInstance().findType(throwStmt.getExpression().getType());
                if (thrownType != null && caughtType != null && thrownType.isSubtypeOf(caughtType)) {

                }
            }

package net.sourceforge.pmd.lang.java.rule.design;


import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ExceptionAsFlowControlRule extends AbstractJavaRulechainRule {

    public ExceptionAsFlowControlRule() {
        super(ASTTryStatement.class, ASTCatchStatement.class, ASTThrowStatement.class);
    }

    @Override
    public Object visit(ASTTryStatement node, Object data) {
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTCatchStatement node, Object data) {
        ASTTryStatement tryBlock = node.getFirstParentOfType(ASTTryStatement.class);
        if (tryBlock != null) {
            for (ASTThrowStatement throwStmt : node.findDescendantsOfType(ASTThrowStatement.class)) {
                ASTConstructorCall constructorCall = throwStmt.getFirstDescendantOfType(ASTConstructorCall.class);
                ASTExplicitConstructorInvocation explicitConstructorInvocation = throwStmt.getFirstDescendantOfType(ASTExplicitConstructorInvocation.class);
                if (constructorCall != null || explicitConstructorInvocation != null) {
                    JClassType thrownType = getThrownExceptionType(throwStmt);
                    if (thrownType != null) {
                        for (ASTCatchStatement catchStmt : tryBlock.findDescendantsOfType(ASTCatchStatement.class)) {
                            JClassType caughtType = getCaughtExceptionType(catchStmt);
                            if (caughtType != null && caughtType.isSubtypeOf(thrownType)) {
                                addViolation(data, catchStmt);
                            }
                        }
                    }
                }
            }
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTThrowStatement node, Object data) {
        return super.visit(node, data);
    }

    private JClassType getThrownExceptionType(ASTThrowStatement throwStmt) {
        ASTConstructorCall constructorCall = throwStmt.getFirstDescendantOfType(ASTConstructorCall.class);
        if (constructorCall != null) {
            return constructorCall.getType();
        }
        ASTExplicitConstructorInvocation explicitConstructorInvocation = throwStmt.getFirstDescendantOfType(ASTExplicitConstructorInvocation.class);
        if (explicitConstructorInvocation != null) {
            return explicitConstructorInvocation.getType();
        }
        return null;
    }

    private JClassType getCaughtExceptionType(ASTCatchStatement catchStmt) {
        TypeNode typeNode = catchStmt.getTypeNode();
        if (typeNode != null) {
            return typeNode.getType();
        }
        return null;
    }
}
