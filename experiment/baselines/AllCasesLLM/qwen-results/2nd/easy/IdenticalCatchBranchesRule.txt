package net.sourceforge.pmd.lang.java.rule.codestyle;

import java.util.List;
import java.util.HashSet;
import java.util.Set;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class IdenticalCatchBranchesRule extends AbstractJavaRulechainRule {

    public IdenticalCatchBranchesRule() {
        super(ASTTryStatement.class);
    }

    @Override
    public Object visit(ASTTryStatement node, Object data) {
        List<ASTCatchClause> catchClauses = node.findDescendantsOfType(ASTCatchClause.class);
        Set<Integer> visitedIndices = new HashSet<>();
        for (int i = 0; i < catchClauses.size(); i++) {
            if (!visitedIndices.contains(i)) {
                ASTCatchClause currentCatch = catchClauses.get(i);
                ASTBlock currentBlock = currentCatch.getFirstChildOfType(ASTBlock.class);
                List<ASTStatement> currentStatements = currentBlock.findChildrenOfType(ASTStatement.class);
                for (int j = i + 1; j < catchClauses.size(); j++) {
                    ASTCatchClause comparedCatch = catchClauses.get(j);
                    ASTBlock comparedBlock = comparedCatch.getFirstChildOfType(ASTBlock.class);
                    List<ASTStatement> comparedStatements = comparedBlock.findChildrenOfType(ASTStatement.class);
                    if (areBlocksIdentical(currentStatements, comparedStatements)) {
                        visitedIndices.add(j);
                        addViolation(data, comparedCatch);
                    }
                }
            }
        }
        return super.visit(node, data);
    }

    private boolean areBlocksIdentical(List<ASTStatement> currentStatements, List<ASTStatement> comparedStatements) {
        if (currentStatements.size() != comparedStatements.size()) {
            return false;
        }
        for (int i = 0; i < currentStatements.size(); i++) {
            if (!currentStatements.get(i).getXPathNode().isSameNode(comparedStatements.get(i).getXPathNode())) {
                return false;
            }
        }
        return true;
    }
}

package net.sourceforge.pmd.lang.java.rule.codestyle;


import java.util.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class IdenticalCatchBranchesRule extends AbstractJavaRulechainRule {
    public IdenticalCatchBranchesRule() {
        super(ASTTryStatement.class);
    }

    @Override
    public Object visit(ASTTryStatement node, Object data) {
        Map<String, ASTCatchStatement> catchBranches = new HashMap<>();
        for (ASTCatchStatement catchStmt : node.getCatchStatements()) {
            ASTBlockStatement blockStmt = catchStmt.getFirstChildOfType(ASTBlockStatement.class);
            if (blockStmt != null) {
                String blockContent = blockStmt.getImage();
                if (catchBranches.containsKey(blockContent)) {
                    ASTCatchStatement previousCatchStmt = catchBranches.get(blockContent);
                    if (areExceptionsDistinct(previousCatchStmt, catchStmt)) {
                        addViolation(data, catchStmt);
                    }
                } else {
                    catchBranches.put(blockContent, catchStmt);
                }
            }
        }
        return super.visit(node, data);
    }

    private boolean areExceptionsDistinct(ASTCatchStatement catchStmt1, ASTCatchStatement catchStmt2) {
        Set<JClassSymbol> exceptions1 = getExceptionTypes(catchStmt1);
        Set<JClassSymbol> exceptions2 = getExceptionTypes(catchStmt2);
        return Collections.disjoint(exceptions1, exceptions2);
    }

    private Set<JClassSymbol> getExceptionTypes(ASTCatchStatement catchStmt) {
        Set<JClassSymbol> exceptionTypes = new HashSet<>();
        for (ASTType type : catchStmt.getParameter().getTypeNode().findDescendantsOfType(ASTClassOrInterfaceType.class)) {
            JClassSymbol classSymbol = type.getType().getSymbol();
            if (classSymbol != null) {
                exceptionTypes.add(classSymbol);
            }
        }
        return exceptionTypes;
    }
}

package net.sourceforge.pmd.lang.java.rule.codestyle;

import java.util.List;
import java.util.Objects;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class IdenticalCatchBranchesRule extends AbstractJavaRulechainRule {
    public IdenticalCatchBranchesRule() {
        super(ASTTryStatement.class);
    }

    @Override
    public Object visit(ASTTryStatement node, Object data) {
        List<ASTCatchClause> catchClauses = node.getCatchClauses();
        for (int i = 0; i < catchClauses.size(); i++) {
            ASTCatchClause currentCatch = catchClauses.get(i);
            ASTBlock currentBlock = currentCatch.getFirstChildOfType(ASTBlock.class);
            if (currentBlock == null) continue;
            List<ASTStatement> currentStatements = currentBlock.getStatements();

            for (int j = i + 1; j < catchClauses.size(); j++) {
                ASTCatchClause nextCatch = catchClauses.get(j);
                ASTBlock nextBlock = nextCatch.getFirstChildOfType(ASTBlock.class);
                if (nextBlock == null) continue;
                List<ASTStatement> nextStatements = nextBlock.getStatements();

                if (currentStatements.size() == nextStatements.size() && areStatementsIdentical(currentStatements, nextStatements)) {
                    addViolation(data, nextCatch);
                }
            }
        }
        return super.visit(node, data);
    }

    private boolean areStatementsIdentical(List<ASTStatement> currentStatements, List<ASTStatement> nextStatements) {
        for (int k = 0; k < currentStatements.size(); k++) {
            if (!Objects.equals(currentStatements.get(k).getXPathNode(), nextStatements.get(k).getXPathNode())) {
                return false;
            }
        }
        return true;
    }
}

package net.sourceforge.pmd.lang.java.rule.codestyle;


import java.util.ArrayList;
import java.util.List;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class IdenticalCatchBranchesRule extends AbstractJavaRulechainRule {

    public IdenticalCatchBranchesRule() {
        super(ASTTryStatement.class);
    }

    @Override
    public Object visit(ASTTryStatement node, Object data) {
        NodeStream<ASTCatchStatement> catchStatements = node.getCatchStatements();
        List<ASTCatchStatement> identicalCatches = new ArrayList<>();

        for (ASTCatchStatement catchStatement : catchStatements) {
            ASTBlockStatement blockStatement = catchStatement.getFirstDescendantOfType(ASTBlockStatement.class);
            if (blockStatement != null) {
                boolean isIdentical = false;
                for (ASTCatchStatement existingCatch : identicalCatches) {
                    if (blockStatement.isEqualTo(existingCatch.getFirstDescendantOfType(ASTBlockStatement.class))) {
                        isIdentical = true;
                        break;
                    }
                }
                if (isIdentical) {
                    identicalCatches.add(catchStatement);
                } else {
                    identicalCatches.clear();
                    identicalCatches.add(catchStatement);
                }
            }
        }

        if (identicalCatches.size() > 1) {
            ASTCatchStatement firstCatch = identicalCatches.get(0);
            String message = "Identical catch branches can be collapsed into a single multi-catch branch.";
            addViolationWithMessage(data, firstCatch, message);
        }

        return super.visit(node, data);
    }
}
