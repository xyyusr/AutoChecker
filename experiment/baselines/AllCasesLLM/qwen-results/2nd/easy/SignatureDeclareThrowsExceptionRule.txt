package net.sourceforge.pmd.lang.java.rule.design;


import java.util.List;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class SignatureDeclareThrowsExceptionRule extends AbstractJavaRulechainRule {

    public SignatureDeclareThrowsExceptionRule() {
        super(ASTConstructorDeclaration.class, ASTMethodDeclaration.class);
    }

    private boolean isJUnit3TestCase(ASTClassOrInterfaceDeclaration node) {
        JClassSymbol superClass = node.getSuperClass();
        return superClass != null && (superClass.getQualifiedName().equals(TestCase.class.getName()) || superClass.getQualifiedName().startsWith("net.sourceforge.pmd.lang.java.rule.design.signaturedeclarethrowsexception."));
    }

    private boolean isJUnit4SetupOrTeardown(ASTMethodDeclaration node) {
        return node.hasAnnotation(Before.class.getName()) || node.hasAnnotation(After.class.getName());
    }

    private boolean isJUnit3SetupOrTeardown(ASTMethodDeclaration node) {
        return node.getName().equals("setUp") || node.getName().equals("tearDown");
    }

    private boolean isTestMethod(ASTMethodDeclaration node) {
        return node.getName().startsWith("test");
    }

    private boolean isOverridingMethod(ASTMethodDeclaration node) {
        return node.isOverriding();
    }

    private boolean isTestMethodInJUnit3TestCase(ASTMethodDeclaration node) {
        return isJUnit3TestCase(node.getFirstParentOfType(ASTClassOrInterfaceDeclaration.class)) && isTestMethod(node);
    }

    @Override
    public Object visit(ASTConstructorDeclaration node, Object data) {
        if (node.hasThrowsClause() && isJUnit3TestCase(node.getFirstParentOfType(ASTClassOrInterfaceDeclaration.class))) {
            return data;
        }
        checkThrowsClause(node.getThrowsClause(), node, data);
        return data;
    }

    @Override
    public Object visit(ASTMethodDeclaration node, Object data) {
        if (isJUnit3SetupOrTeardown(node) && isJUnit3TestCase(node.getFirstParentOfType(ASTClassOrInterfaceDeclaration.class))) {
            return data;
        }
        if (isJUnit4SetupOrTeardown(node)) {
            return data;
        }
        if (isTestMethodInJUnit3TestCase(node)) {
            return data;
        }
        if (isOverridingMethod(node)) {
            return data;
        }
        checkThrowsClause(node.getThrowsClause(), node, data);
        return data;
    }

    private void checkThrowsClause(ASTThrowsClause throwsClause, JavaNode node, Object data) {
        if (throwsClause != null) {
            for (ASTName exceptionName : throwsClause.findDescendantsOfType(ASTName.class)) {
                JType exceptionType = TypeSystem.createReference(exceptionName.getImage()).resolve(node);
                if (exceptionType != null && TypeTestUtil.isExactlyA(exceptionType, Exception.class)) {
                    addViolation(data, node);
                }
            }
        }
    }
}

package net.sourceforge.pmd.lang.java.rule.design;


import java.util.List;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class SignatureDeclareThrowsExceptionRule extends AbstractJavaRulechainRule {

    public SignatureDeclareThrowsExceptionRule() {
        super(ASTConstructorDeclaration.class, ASTMethodDeclaration.class);
    }

    @Override
    public Object visit(ASTConstructorDeclaration node, Object data) {
        checkThrowsClause(node, data);
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTMethodDeclaration node, Object data) {
        if (!isJUnitSetupOrTeardown(node) && !isJUnitTestMethod(node) && !isTestNGAnnotationSetup(node) && !isOverridingMethod(node)) {
            checkThrowsClause(node, data);
        }
        return super.visit(node, data);
    }

    private void checkThrowsClause(ASTConstructorDeclaration node, Object data) {
        if (node.hasThrowsClause() && node.getThrows().stream().anyMatch(this::throwsGenericException)) {
            addViolation(data, node);
        }
    }

    private void checkThrowsClause(ASTMethodDeclaration node, Object data) {
        if (node.hasThrowsClause() && node.getThrows().stream().anyMatch(this::throwsGenericException)) {
            addViolation(data, node);
        }
    }

    private boolean throwsGenericException(ASTReferenceType refType) {
        ASTName name = refType.getFirstDescendantOfType(ASTName.class);
        if (name != null) {
            String exceptionName = name.getImage();
            return "Exception".equals(exceptionName) && !isIOException(name);
        }
        return false;
    }

    private boolean isIOException(ASTName name) {
        JClassSymbol symbol = name.getType().getSymbol();
        if (symbol != null) {
            JClassSymbol ioExceptionSymbol = TypeSystem.getInstance().findClass("java.io.IOException");
            return symbol.equals(ioExceptionSymbol) || symbol.isSubtypeOf(ioExceptionSymbol);
        }
        return false;
    }

    private boolean isJUnitSetupOrTeardown(ASTMethodDeclaration node) {
        String name = node.getName();
        return ("setUp".equals(name) || "tearDown".equals(name)) && isInJUnitTestCase(node);
    }

    private boolean isJUnitTestMethod(ASTMethodDeclaration node) {
        return node.getName().startsWith("test") && isInJUnitTestCase(node);
    }

    private boolean isInJUnitTestCase(ASTMethodDeclaration node) {
        ASTClassOrInterfaceDeclaration classDecl = node.getFirstParentOfType(ASTClassOrInterfaceDeclaration.class);
        if (classDecl != null) {
            JClassSymbol testCaseSymbol = TypeSystem.getInstance().findClass("junit.framework.TestCase");
            return classDecl.getType().getSymbol().isSubtypeOf(testCaseSymbol);
        }
        return false;
    }

    private boolean isOverridingMethod(ASTMethodDeclaration node) {
        JMethodSymbol methodSymbol = node.getSymbol();
        return methodSymbol != null && methodSymbol.isOverriding();
    }

    private boolean isTestNGAnnotationSetup(ASTMethodDeclaration node) {
        return node.hasAnnotation("org.testng.annotations.AfterClass") || node.hasAnnotation("org.testng.annotations.BeforeClass");

import java.util.List;
import java.util.Optional;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class SignatureDeclareThrowsExceptionRule extends AbstractJavaRulechainRule {
    public SignatureDeclareThrowsExceptionRule() {
        super(ASTMethodDeclaration.class, ASTConstructorDeclaration.class);
    }

    @Override
    public Object visit(ASTMethodDeclaration node, Object data) {
        if (!node.getName().startsWith("test") && !isJUnitSetupOrTeardown(node) && !isOverride(node)) {
            checkThrows(node, data);
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTConstructorDeclaration node, Object data) {
        if (!isJUnitSetupOrTeardown(node)) {
            checkThrows(node, data);
        }
        return super.visit(node, data);
    }

    private void checkThrows(JavaNode node, Object data) {
        node.getFormalExceptions().stream()
            .map(ASTReferenceType::getType)
            .filter(type -> type != null && "java.lang.Exception".equals(type.getQualifiedName()))
            .findFirst()
            .ifPresent(type -> addViolation(data, node));
    }

    private boolean isJUnitSetupOrTeardown(JavaNode node) {
        ASTClassOrInterfaceDeclaration classDecl = node.getNearestAncestorOfType(ASTClassOrInterfaceDeclaration.class);
        if (classDecl == null) {
            return false;
        }

        Optional<JClassSymbol> symbol = classDecl.getType().getSymbol();
        if (!symbol.isPresent()) {
            return false;
        }

        TypeSystem typeSystem = classDecl.getTypeSystem();
        JType testCaseType = typeSystem.forName("junit.framework.TestCase");
        JType myTestCaseType = typeSystem.forName("net.sourceforge.pmd.lang.java.rule.design.signaturedeclarethrowsexception.MyTestCase");

        if (symbol.get().isSubtypeOf(testCaseType) || symbol.get().isSubtypeOf(myTestCaseType)) {
            String methodName = node.getName();
            return "setUp".equals(methodName) || "tearDown".equals(methodName);
        }

        return false;
    }

    private boolean isOverride(JavaNode node) {
        if (node instanceof ASTMethodDeclaration) {
            ASTMethodDeclaration methodDecl = (ASTMethodDeclaration) node;
            return methodDecl.isOverride();
        }
        return false;
    }

package net.sourceforge.pmd.lang.java.rule.design;


import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class SignatureDeclareThrowsExceptionRule extends AbstractJavaRulechainRule {
    public SignatureDeclareThrowsExceptionRule() {
        super(ASTMethodDeclaration.class, ASTConstructorDeclaration.class);
    }

    @Override
    public Object visit(ASTMethodDeclaration node, Object data) {
        if (node.getName().startsWith("test") && isJUnit3TestCase(node)) {
            return data;
        }
        if (isJUnit3TestCase(node) && (node.getName().equals("setUp") || node.getName().equals("tearDown"))) {
            return data;
        }
        if (isJUnit4AnnotationPresent(node) || isTestNGAnnotationPresent(node)) {
            return data;
        }
        if (isOverridingMethod(node)) {
            return data;
        }
        checkForGenericException(node.getThrowsList(), node, data);
        return data;
    }

    @Override
    public Object visit(ASTConstructorDeclaration node, Object data) {
        if (isJUnit3TestCase(node)) {
            return data;
        }
        if (isJUnit4AnnotationPresent(node) || isTestNGAnnotationPresent(node)) {
            return data;
        }
        checkForGenericException(node.getThrowsList(), node, data);
        return data;
    }

    private boolean isJUnit3TestCase(ASTClassOrInterfaceDeclaration node) {
        return node.isExtending("junit.framework.TestCase") || node.isExtending("net.sourceforge.pmd.lang.java.rule.design.signaturedeclarethrowsexception.MyTestCase");
    }

    private boolean isJUnit3TestCase(ASTMethodDeclaration node) {
        return node.getEnclosingClass().isExtending("junit.framework.TestCase") || node.getEnclosingClass().isExtending("net.sourceforge.pmd.lang.java.rule.design.signaturedeclarethrowsexception.MyTestCase");
    }

    private boolean isJUnit4AnnotationPresent(ASTMethodDeclaration node) {
        return node.hasAnnotation("org.junit.Before") || node.hasAnnotation("org.junit.After") || node.hasAnnotation("org.junit.Test");
    }

    private boolean isTestNGAnnotationPresent(ASTMethodDeclaration node) {
        return node.hasAnnotation("org.testng.annotations.BeforeClass") || node.hasAnnotation("org.testng.annotations.AfterClass") || node.hasAnnotation("org.testng.annotations.Test");
    }

    private boolean isOverridingMethod(ASTMethodDeclaration node) {
        JMethodSymbol methodSymbol = node.getSymbol();
        if (methodSymbol == null) {
            return false;
        }
        for (JClassSymbol superClass : methodSymbol.getDeclaringClass().getSuperClasses()) {
            for (JMethodSymbol superMethod : superClass.getDeclaredMethods()) {
                if (methodSymbol.overrides(superMethod)) {
                    return true;
                }
            }
        }
        return false;
    }

    private void checkForGenericException(ASTNameList throwsList, JavaNode node, Object data) {
        for (ASTClassOrInterfaceType exceptionType : throwsList.childrenOfType(ASTClassOrInterfaceType.class)) {
            Type type = exceptionType.getType();
            if (type instanceof JClassTypeSymbol && "java.lang.Exception".equals(((JClassTypeSymbol) type).getQualifiedName())) {
                addViolation(data, node);
            }
        }
    }
}

package net.sourceforge.pmd.lang.java.rule.design;

import java.util.List;
import java.util.Objects;
import java.util.Optional;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class SignatureDeclareThrowsExceptionRule extends AbstractJavaRulechainRule {
    public SignatureDeclareThrowsExceptionRule() {
        super(ASTMethodDeclaration.class, ASTConstructorDeclaration.class);
    }

    @Override
    public Object visit(ASTMethodDeclaration node, Object data) {
        if (isJUnitMethod(node) || isTestNGMethod(node)) {
            return data;
        }
        return checkThrowsClause(node, data);
    }

    @Override
    public Object visit(ASTConstructorDeclaration node, Object data) {
        if (isJUnitConstructor(node)) {
            return data;
        }
        return checkThrowsClause(node, data);
    }

    private Object checkThrowsClause(ASTAnyTypeDeclaration node, Object data) {
        ASTThrowsClause throwsClause = node.getFirstChildOfType(ASTThrowsClause.class);
        if (throwsClause != null) {
            for (ASTName exceptionType : throwsClause.findDescendantsOfType(ASTName.class)) {
                JType type = TypeHelper.getType(exceptionType);
                if (type != null && type.getSymbol() instanceof JClassSymbol) {
                    JClassSymbol symbol = (JClassSymbol) type.getSymbol();
                    if (Objects.equals(symbol.getQualifiedName(), "java.lang.Exception") && !isOverridingMethod(node)) {
                        addViolation(data, node);
                    }
                }
            }
        }
        return data;
    }

    private boolean isJUnitMethod(ASTMethodDeclaration method) {
        String methodName = method.getName();
        if (methodName.startsWith("test")) {
            return true;
        }
        return isJUnitSetupOrTeardownMethod(method, "junit.framework.TestCase") ||
               isJUnitSetupOrTeardownMethod(method, "net.sourceforge.pmd.lang.java.rule.design.signaturedeclarethrowsexception.MyTestCase");
    }

    private boolean isJUnitConstructor(ASTConstructorDeclaration constructor) {
        return isJUnitSetupOrTeardownMethod(constructor, "junit.framework.TestCase");
    }

    private boolean isJUnitSetupOrTeardownMethod(ASTAnyTypeDeclaration method, String testCaseSuperClass) {
        JClassSymbol classSymbol = method.getEnclosingClass();
        if (classSymbol != null) {
            return classSymbol.getSuperClass().map(s -> s.getQualifiedName().equals(testCaseSuperClass)).orElse(false);
        }
        return false;
    }

    private boolean isTestNGMethod(ASTMethodDeclaration method) {
        Optional<JAnnotationSymbol> annotation = method.getFirstDescendantOfType(ASTAnnotation.class)
                .map(ASTAnnotation::getSymbol);
        return annotation.isPresent() && annotation.get().getQualifiedName().startsWith("org.testng.annotations.");
    }

    private boolean isOverridingMethod(ASTAnyTypeDeclaration method) {
        JMethodSymbol methodSymbol = method.getSymbol();
        return methodSymbol != null && methodSymbol.isOverriding();
    }
}
