package net.sourceforge.pmd.lang.java.rule.errorprone;


import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class BrokenNullCheckRule extends AbstractJavaRulechainRule {

    public BrokenNullCheckRule() {
        super(ASTIfStatement.class);
    }

    @Override
    public Object visit(ASTIfStatement node, Object data) {
        ASTExpression condition = node.getFirstChildOfType(ASTExpression.class);
        if (condition != null) {
            checkCondition(condition);
        }
        return super.visit(node, data);
    }

    private void checkCondition(ASTExpression condition) {
        if (condition instanceof ASTLogicalOr) {
            checkBrokenNullCheck((ASTLogicalOr) condition, false);
        } else if (condition instanceof ASTLogicalAnd) {
            checkBrokenNullCheck((ASTLogicalAnd) condition, true);
        }
    }

    private void checkBrokenNullCheck(ASTExpression condition, boolean isAnd) {
        NodeStream<ASTEqualityExpression> nullChecks = condition.descendants(ASTEqualityExpression.class)
                .filter(e -> e.getLeftOperand() instanceof ASTPrimaryExpression || e.getRightOperand() instanceof ASTPrimaryExpression)
                .filter(e -> e.getLeftOperand().containsDescendantOfType(ASTNullLiteral.class) || e.getRightOperand().containsDescendantOfType(ASTNullLiteral.class));

        nullChecks.forEach(e -> {
            ASTExpression otherOperand = e.getLeftOperand() instanceof ASTNullLiteral ? e.getRightOperand() : e.getLeftOperand();
            if (otherOperand.containsDescendantOfType(ASTPrimarySuffix.class) || otherOperand.containsDescendantOfType(ASTMethodCall.class) || otherOperand.containsDescendantOfType(ASTMemberSelect.class) || otherOperand.containsDescendantOfType(ASTArrayAccess.class)) {
                if (isAnd && otherOperand.containsDescendantOfType(ASTPrimaryExpression.class)) {
                    addViolation(condition);
                } else if (!isAnd && otherOperand.containsDescendantOfType(ASTPrimaryExpression.class)) {
                    addViolation(condition);
                }
            }
        });
    }
}

    public BrokenNullCheckRule() {
        super(ASTIfStatement.class);
    }

    @Override
    public Object visit(ASTIfStatement node, Object data) {
        NodeStream<ASTLogicalOrExpression> orExpressions = node.findDescendantsOfType(ASTLogicalOrExpression.class);
        NodeStream<ASTLogicalAndExpression> andExpressions = node.findDescendantsOfType(ASTLogicalAndExpression.class);

        orExpressions.forEach(expr -> checkLogicalExpression(expr, true, data));
        andExpressions.forEach(expr -> checkLogicalExpression(expr, false, data));

        return super.visit(node, data);
    }

    private void checkLogicalExpression(ASTLogicalExpression node, boolean isOr, Object data) {
        JType nullType = TypeSystem.getInstance().NULL_TYPE;

        ASTPrimaryExpression left = node.getFirstPrimaryExpression();
        ASTPrimaryExpression right = node.getSecondPrimaryExpression();

        boolean leftIsNullCheck = isNullCheck(left);
        boolean rightIsNullCheck = isNullCheck(right);

        if (leftIsNullCheck && !rightIsNullCheck && willThrowNPE(right)) {
            if (isOr) {
                addViolation(data, node);
            }
        } else if (!leftIsNullCheck && rightIsNullCheck && willThrowNPE(left)) {
            if (isOr) {
                addViolation(data, node);
            }
        } else if (leftIsNullCheck && rightIsNullCheck) {
            // Both are null checks, no issue
        } else if (!leftIsNullCheck && !rightIsNullCheck) {
            if (isOr && willThrowNPE(left) && willThrowNPE(right)) {
                addViolation(data, node);
            } else if (!isOr && (willThrowNPE(left) || willThrowNPE(right))) {
                addViolation(data, node);
            }
        }
    }

    private boolean isNullCheck(ASTPrimaryExpression expr) {
        return expr.toString().matches(".+==\\s*null|.+!=\\s*null");
    }

    private boolean willThrowNPE(ASTPrimaryExpression expr) {
        if (expr.hasDescendantMatching(ASTPrimarySuffix.class, suffix -> suffix.getImage().equals("hashCode"))) {
            return true;
        }
        if (expr.hasDescendantMatching(ASTPrimarySuffix.class, suffix -> suffix.getImage().equals("equals"))) {
            return true;
        }
        if (expr.hasDescendantMatching(ASTPrimarySuffix.class, suffix -> suffix.getImage().equals("length"))) {
            return true;
        }
        if (expr.hasDescendantMatching(ASTPrimarySuffix.class, suffix -> suffix.getImage().equals("toString"))) {
            return true;
        }
        if (expr.hasDescendantMatching(ASTPrimarySuffix.class, suffix -> suffix.getImage().equals("new"))) {
            return true;
        }
        if (expr.hasDescendantMatching(ASTPrimarySuffix.class, suffix -> suffix.getImage().equals("<<"))) {
            return true;
        }
        if (expr.hasDescendantMatching(ASTPrimarySuffix.class, suffix -> suffix.getImage().equals("+"))) {
            return true;
        }
        if (expr.hasDescendantMatching(ASTPrimarySuffix.class, suffix -> suffix.getImage().equals("isEmpty"))) {
            return true;
        }

        return false;
    }

package net.sourceforge.pmd.lang.java.rule.errorprone;


import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class BrokenNullCheckRule extends AbstractJavaRulechainRule {
    public BrokenNullCheckRule() {
        super(ASTIfStatement.class);
    }

    @Override
    public Object visit(ASTIfStatement node, Object data) {
        NodeStream<ASTBinaryExpression> conditions = node.getCondition().children(ASTBinaryExpression.class);
        conditions.forEach(condition -> {
            if (condition.getOperator() == ASTBinaryExpression.Operator.OR) {
                checkForBrokenNullCheck(condition, ASTBinaryExpression.Operator.AND);
            } else if (condition.getOperator() == ASTBinaryExpression.Operator.AND) {
                checkForBrokenNullCheck(condition, ASTBinaryExpression.Operator.OR);
            }
        });
        return super.visit(node, data);
    }

    private void checkForBrokenNullCheck(ASTBinaryExpression condition, ASTBinaryExpression.Operator expectedOperator) {
        ASTExpression left = condition.getLeftOperand();
        ASTExpression right = condition.getRightOperand();

        if (isNullCheck(left) && isPotentiallyThrowing(right, expectedOperator)) {
            addViolation(data, condition);
        } else if (isPotentiallyThrowing(left, expectedOperator) && isNullCheck(right)) {
            addViolation(data, condition);
        }
    }

    private boolean isNullCheck(ASTExpression expr) {
        return expr instanceof ASTBinaryExpression &&
                (((ASTBinaryExpression) expr).getOperator() == ASTBinaryExpression.Operator.EQUAL ||
                 ((ASTBinaryExpression) expr).getOperator() == ASTBinaryExpression.Operator.NOT_EQUAL) &&
                ((ASTBinaryExpression) expr).getLeftOperand() instanceof ASTVariableAccess &&
                ((ASTBinaryExpression) expr).getRightOperand() instanceof ASTLiteral &&
                ((ASTLiteral) ((ASTBinaryExpression) expr).getRightOperand()).getValue() == null;
    }

    private boolean isPotentiallyThrowing(ASTExpression expr, ASTBinaryExpression.Operator expectedOperator) {
        if (expr instanceof ASTMethodCall || expr instanceof ASTFieldAccess || expr instanceof ASTArrayAccess || expr instanceof ASTMethodReference) {
            return true;
        } else if (expr instanceof ASTBinaryExpression) {
            ASTBinaryExpression binExpr = (ASTBinaryExpression) expr;
            if (binExpr.getOperator() == expectedOperator) {
                return isPotentiallyThrowing(binExpr.getLeftOperand(), expectedOperator) || isPotentiallyThrowing(binExpr.getRightOperand(), expectedOperator);
            }
        }
        return false;
    }
}

    public BrokenNullCheckRule() {
        super(ASTIfStatement.class, ASTBinaryExpression.class);
    }

    @Override
    public Object visit(ASTIfStatement node, Object data) {
        return super.visit(node, data);

    @Override
    public Object visit(ASTBinaryExpression node, Object data) {
        if (node.isUsingOperator(ASTBinaryExpression.Operator.OR) || node.isUsingOperator(ASTBinaryExpression.Operator.AND)) {
            ASTBinaryExpression left = node.getLeftOperand();
            ASTBinaryExpression right = node.getRightOperand();

            if (left instanceof ASTBinaryExpression && right instanceof ASTPrimaryExpression) {
                checkExpression(left, right, node, data);
            } else if (right instanceof ASTBinaryExpression && left instanceof ASTPrimaryExpression) {
                checkExpression(right, left, node, data);
            }
        }
        return super.visit(node, data);
    }

    private void checkExpression(ASTBinaryExpression binaryExpression, ASTPrimaryExpression primaryExpression, ASTBinaryExpression node, Object data) {
        if (binaryExpression.isUsingOperator(ASTBinaryExpression.Operator.EQUAL) || binaryExpression.isUsingOperator(ASTBinaryExpression.Operator.NOT_EQUAL)) {
            ASTPrimaryExpression left = binaryExpression.getLeftOperand();
            ASTPrimaryExpression right = binaryExpression.getRightOperand();

            if (isNullCheck(left) && isMethodCallOrFieldAccess(primaryExpression)) {
                boolean isOr = node.isUsingOperator(ASTBinaryExpression.Operator.OR);
                if (isOr) {
                    addViolation(data, node);
                }
            } else if (isNullCheck(right) && isMethodCallOrFieldAccess(primaryExpression)) {
                boolean isOr = node.isUsingOperator(ASTBinaryExpression.Operator.OR);
                if (isOr) {
                    addViolation(data, node);
                }
            }
        }
    }

    private boolean isNullCheck(ASTPrimaryExpression primaryExpression) {
        ASTPrimaryPrefix prefix = primaryExpression.getFirstChildOfType(ASTPrimaryPrefix.class);
        if (prefix != null) {
            ASTLiteral literal = prefix.getFirstChildOfType(ASTLiteral.class);
            if (literal != null && literal.getValue() == null) {
                ASTBinaryExpression.Operator op = primaryExpression.getParent() instanceof ASTBinaryExpression ? ((ASTBinaryExpression) primaryExpression.getParent()).getOperator() : null;
                return op == ASTBinaryExpression.Operator.EQUAL || op == ASTBinaryExpression.Operator.NOT_EQUAL;
            }
        }
        return false;
    }

    private boolean isMethodCallOrFieldAccess(ASTPrimaryExpression primaryExpression) {
        ASTPrimaryPrefix prefix = primaryExpression.getFirstChildOfType(ASTPrimaryPrefix.class);
        if (prefix != null) {
            ASTPrimarySuffix suffix = prefix.getChildOfType(ASTPrimarySuffix.class);
            return suffix != null && (suffix.isMethodCall() || suffix.isFieldAccess());
        }
        return false;
    }
