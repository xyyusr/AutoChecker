package net.sourceforge.pmd.lang.java.rule.codestyle;

import java.util.*;
import java.util.stream.Collectors;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class UnnecessaryImportRule extends AbstractJavaRulechainRule {
    public UnnecessaryImportRule() {
        super(ASTImportDeclaration.class, ASTCompilationUnit.class);
    }

    private Map<String, List<ASTImportDeclaration>> importMap = new HashMap<>();
    private Set<String> usedImports = new HashSet<>();
    private Set<String> samePackageImports = new HashSet<>();
    private Set<String> javaLangImports = new HashSet<>();

    @Override
    public Object visit(ASTImportDeclaration node, Object data) {
        String qualifiedName = node.getImportedName();
        if (node.isStarImport()) {
            importMap.computeIfAbsent(qualifiedName, k -> new ArrayList<>()).add(node);
        } else {
            usedImports.add(qualifiedName);
            importMap.computeIfAbsent(qualifiedName.substring(0, qualifiedName.lastIndexOf('.')), k -> new ArrayList<>()).add(node);
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTCompilationUnit node, Object data) {
        String packageName = node.getPackageName();
        if (!packageName.isEmpty()) {
            samePackageImports.addAll(node.findDescendantsOfType(ASTClassOrInterfaceDeclaration.class).stream()
                    .map(ASTClassOrInterfaceDeclaration::getQualifiedName)
                    .collect(Collectors.toList()));
        }
        javaLangImports.addAll(node.findDescendantsOfType(ASTClassOrInterfaceDeclaration.class).stream()
                .map(ASTClassOrInterfaceDeclaration::getImage)
                .collect(Collectors.toList()));

        node.findDescendantsOfType(ASTClassOrInterfaceType.class).forEach(this::processType);
        node.findDescendantsOfType(ASTTypeExpression.class).forEach(this::processTypeExpression);
        node.findDescendantsOfType(ASTAllocationExpression.class).forEach(this::processAllocationExpression);
        node.findDescendantsOfType(ASTCatchClause.class).forEach(this::processCatchClause);
        node.findDescendantsOfType(ASTThrowsClause.class).forEach(this::processThrowsClause);
        node.findDescendantsOfType(ASTAnnotation.class).forEach(this::processAnnotation);
        node.findDescendantsOfType(ASTJavadoc.class).forEach(this::processJavadoc);

        importMap.values().forEach(importList -> {
            if (importList.size() > 1 && !importList.get(0).isStarImport()) {
                for (int i = 1; i < importList.size(); i++) {
                    addViolation(data, importList.get(i));
                }
            }
            String importKey = importList.get(0).getImportedName();
            if (importList.get(0).isStarImport()) {
                importKey = importList.get(0).getImportedName() + ".*";
                if (!usedImports.stream().anyMatch(used -> used.startsWith(importList.get(0).getImportedName()))) {
                    addViolation(data, importList.get(0));
                }
            } else if (!usedImports.contains(importKey) &&
                       !samePackageImports.contains(importKey) &&
                       !javaLangImports.contains(importKey.substring(importKey.lastIndexOf('.') + 1))) {
                addViolation(data, importList.get(0));
            }
        });

        importMap.clear();
        usedImports.clear();
        samePackageImports.clear();
        javaLangImports.clear();

        return super.visit(node, data);
    }

    private void processType(ASTClassOrInterfaceType type) {
        String qualifiedName = type.getQualifiedName();
        usedImports.add(qualifiedName);
        if (qualifiedName.startsWith("java.lang")) {
            javaLangImports.add(qualifiedName.substring("java.lang.".length()));
        }
    }

    private void processTypeExpression(ASTTypeExpression typeExpression) {
        String qualifiedName = typeExpression.getTypeNode().getQualifiedName();
        usedImports.add(qualifiedName);
        if (qualifiedName.startsWith("java.lang")) {
            javaLangImports.add(qualifiedName.substring("java.lang.".length()));
        }
    }

    private void processAllocationExpression(ASTAllocationExpression allocationExpression) {
        String qualifiedName = allocationExpression.getTypeNode().getQualifiedName();
        usedImports.add(qualifiedName);
        if (qualifiedName.startsWith("java.lang")) {
            javaLangImports.add(qualifiedName.substring("java.lang.".length()));
        }
    }

    private void processCatchClause(ASTCatchClause catchClause) {
        String qualifiedName = catchClause.getParameter().getTypeNode().getQualifiedName();
        usedImports.add(qualifiedName);
        if (qualifiedName.startsWith("java.lang")) {
            javaLangImports.add(qualifiedName.substring("java.lang.".length()));
        }
    }

    private void processThrowsClause(ASTThrowsClause throwsClause) {
        throwsClause.findDescendantsOfType(ASTClassOrInterfaceType.class).forEach(type -> {
            String qualifiedName = type.getQualifiedName();
            usedImports.add(qualifiedName);
            if (qualifiedName.startsWith("java.lang")) {
                javaLangImports.add(qualifiedName.substring("java.lang.".length()));
            }
        });
    }

    private void processAnnotation(ASTAnnotation annotation) {
        String qualifiedName = annotation.getTypeNode().getQualifiedName();
        usedImports.add(qualifiedName);
        if (qualifiedName.startsWith("java.lang")) {
            javaLangImports.add(qualifiedName.substring("java.lang.".length()));
        }
    }

    private void processJavadoc(ASTJavadoc javadoc) {
        String text = javadoc.getImage();
        usedImports.addAll(findImportsInJavadoc(text));
    }

    private Set<String> findImportsInJavadoc(String text) {
        Set<String> imports = new HashSet<>();
        for (String importKey : importMap.keySet()) {
            String pattern = importKey + "\\.";
            if (text.contains(pattern)) {
                imports.add(importKey);
            }
        }
        return imports;
    }

package net.sourceforge.pmd.lang.java.rule.codestyle;

import java.util.HashSet;
import java.util.List;
import java.util.Set;
import java.util.stream.Collectors;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class UnnecessaryImportRule extends AbstractJavaRulechainRule {

    public UnnecessaryImportRule() {
        super(ASTCompilationUnit.class, ASTImportDeclaration.class, ASTFieldDeclaration.class,
              ASTFormalParameter.class, ASTMethodDeclaration.class, ASTConstructorDeclaration.class,
              ASTAnnotation.class, ASTExtendsList.class, ASTImplementsList.class, ASTStaticInitializer.class,
              ASTAnnotationTypeDeclaration.class, ASTCatchClause.class, ASTThrowStatement.class,
              ASTTryStatement.class, ASTDoStatement.class, ASTWhileStatement.class, ASTForStatement.class,
              ASTEnhancedForStatement.class, ASTSwitchStatement.class, ASTForInit.class, ASTAnnotationMethodOrFieldDeclaration.class);

    private Set<ASTImportDeclaration> imports;
    private Set<String> usedImports;
    private JPackageSymbol currentPackage;

    @Override
    public Object visit(ASTCompilationUnit node, Object data) {
        imports = new HashSet<>();
        usedImports = new HashSet<>();
        currentPackage = node.getPackageDeclaration() != null ? node.getPackageDeclaration().getPackageSymbol() : null;
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTImportDeclaration node, Object data) {
        imports.add(node);
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTFieldDeclaration node, Object data) {
        addUsedImportsFromType(node.getType());
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTFormalParameter node, Object data) {
        addUsedImportsFromType(node.getType());
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTMethodDeclaration node, Object data) {
        addUsedImportsFromType(node.getReturnType());
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTConstructorDeclaration node, Object data) {
        addUsedImportsFromType(node.getExtendedType());
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTAnnotation node, Object data) {
        usedImports.add(node.getType().getImage());
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTExtendsList node, Object data) {
        node.getExtendedTypes().forEach(type -> addUsedImportsFromType(type));
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTImplementsList node, Object data) {
        node.getImplementedTypes().forEach(type -> addUsedImportsFromType(type));
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTStaticInitializer node, Object data) {
        addUsedImportsFromBlock(node);
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTAnnotationTypeDeclaration node, Object data) {
        addUsedImportsFromBlock(node);
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTCatchClause node, Object data) {
        addUsedImportsFromType(node.getExceptionType());
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTThrowStatement node, Object data) {
        if (node.getExpression() != null) {
            addUsedImportsFromType(node.getExpression().getType());
        }
        return super.visit(node, data);

    @Override
    public Object visit(ASTTryStatement node, Object data) {
        addUsedImportsFromBlock(node);
        return super.visit(node, data);

    @Override
    public Object visit(ASTDoStatement node, Object data) {
        addUsedImportsFromBlock(node);
        return super.visit(node, data);

    @Override
    public Object visit(ASTWhileStatement node, Object data) {

        return super.visit(node, data);

    @Override
    public Object visit(ASTForStatement node, Object data) {
        addUsedImportsFromBlock(node);
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTEnhancedForStatement node, Object data) {

        addUsedImportsFromBlock(node);
        return super.visit(node, data);

    @Override
    public Object visit(ASTSwitchStatement node, Object data) {
        addUsedImportsFromBlock(node);

        addUsedImportsFromBlock(node);

        return super.visit(node, data);

    @Override
    public Object visit(ASTClassOrInterfaceBody node, Object data) {
        addUsedImportsFromBlock(node);
        return super.visit(node, data);

    private void addUsedImportsFromBlock(ASTClassOrInterfaceBody node) {
        node.getDeclarations().forEach(decl -> addUsedImportsFromDeclaration(decl));
    }

    private void addUsedImportsFromBlock(ASTTryStatement node) {
        node.getCatchClauses().forEach(clause -> addUsedImportsFromDeclaration(clause));
        addUsedImportsFromBlock(node.getFinallyBlock());
        addUsedImportsFromBlock(node.getTryBlock());
    }

    private void addUsedImportsFromBlock(ASTDoStatement node) {
        addUsedImportsFromBlock(node.getBlock());
    }

    private void addUsedImportsFromBlock(ASTWhileStatement node) {
        addUsedImportsFromBlock(node.getBlock());
    }

    private void addUsedImportsFromBlock(ASTForStatement node) {
        addUsedImportsFromBlock(node.getBlock());
    }

    private void addUsedImportsFromBlock(ASTEnhancedForStatement node) {
        addUsedImportsFromBlock(node.getBlock());
    }

    private void addUsedImportsFromBlock(ASTStaticInitializer node) {
        addUsedImportsFromBlock(node.getBlock());
    }

    private void addUsedImportsFromBlock(ASTAnnotationTypeDeclaration node) {
        addUsedImportsFromBlock(node.getBlock());
    }

    private void addUsedImportsFromBlock(ASTSwitchStatement node) {
        node.getSwitchLabels().forEach(label -> label.getDeclarations().forEach(decl -> addUsedImportsFromDeclaration(decl)));
    }

    private void addUsedImportsFromBlock(ASTForInit node) {
        node.getStatements().forEach(stmt -> addUsedImportsFromDeclaration(stmt));
    }

    private void addUsedImportsFromBlock(ASTClassOrInterfaceBody node) {
        if (node != null) {
            node.getDeclarations().forEach(decl -> addUsedImportsFromDeclaration(decl));
        }
    }

    private void addUsedImportsFromDeclaration(ASTClassOrInterfaceBodyDeclaration decl) {
        if (decl instanceof ASTFieldDeclaration) {
            ((ASTFieldDeclaration) decl).getVariables().forEach(variable -> addUsedImportsFromType(variable.getType()));
        } else if (decl instanceof ASTMethodDeclaration) {
            addUsedImportsFromMethodOrConstructor((ASTMethodDeclaration) decl);
        } else if (decl instanceof ASTConstructorDeclaration) {
            addUsedImportsFromMethodOrConstructor((ASTConstructorDeclaration) decl);
        } else if (decl instanceof ASTStaticInitializer) {
            addUsedImportsFromBlock((ASTStaticInitializer) decl);
        } else if (decl instanceof ASTBlockStatement) {
            addUsedImportsFromBlock(((ASTBlockStatement) decl).getBlock());
        }
    }

    private void addUsedImportsFromMethodOrConstructor(ASTMethodOrConstructorNode method) {
        addUsedImportsFromType(method.getReturnType());
        method.getFormalParameters().forEach(param -> addUsedImportsFromType(param.getType()));
        addUsedImportsFromBlock(method.getBlock());
    }

    private void addUsedImportsFromType(ASTType type) {
        if (type != null) {
            type.findDescendantsOfType(ASTReferenceType.class).forEach(refType -> addUsedImportsFromReferenceType(refType));
        }
    }

    private void addUsedImportsFromReferenceType(ASTReferenceType refType) {
        JClassSymbol classSymbol = refType.getTypeReference().getReferencedClassSymbol();
        if (classSymbol != null) {
            usedImports.add(classSymbol.getQualifiedName().toString());
        }
    }

    private void addUsedImportsFromBlock(ASTBlock block) {
        if (block != null) {
            block.getStatements().forEach(stmt -> addUsedImportsFromNode(stmt));
        }
    }

    private void addUsedImportsFromNode(ASTNode node) {
        node.findDescendantsOfType(ASTType.class).forEach(type -> addUsedImportsFromType(type));
        node.findDescendantsOfType(ASTName.class).forEach(name -> addUsedImportsFromName(name));
    }

    private void addUsedImportsFromName(ASTName name) {
        SymbolTable symbolTable = name.getSymbolTable();
        Scope scope = symbolTable.getCurrentScope();
        JClassSymbol classSymbol = scope.resolveClass(name.getImage());
        if (classSymbol != null) {
            usedImports.add(classSymbol.getQualifiedName().toString());
        }
    }

    private void addUsedImportsFromJavadocComments(ASTNode node) {
        node.getJavadocComments().stream()
            .flatMap(comment -> comment.getTags().stream())
            .filter(tag -> "link".equals(tag.getName()) || "see".equals(tag.getName()) || "value".equals(tag.getName()) || "snippet".equals(tag.getName()))
            .flatMap(tag -> tag.findDescendantsOfType(ASTName.class).stream())
            .map(ASTName::getImage)
            .forEach(name -> usedImports.add(name));
    }

    @Override
    public Object end(ASTCompilationUnit node, Object data) {
        Set<String> implicitImports = new HashSet<>(Arrays.asList(
            "java.lang.Object",
            "java.lang.String",
            "java.lang.Integer",
            "java.lang.Long",
            "java.lang.Double",
            "java.lang.Float",
            "java.lang.Boolean",
            "java.lang.Character",
            "java.lang.Byte",
            "java.lang.Short",
            "java.lang.Void",
            "java.lang.Thread",
            "java.lang.ThreadLocal",
            "java.lang.Exception",
            "java.lang.RuntimeException",
            "java.lang.Error",
            "java.lang.StackTraceElement",
            "java.lang.Throwable",
            "java.lang.Math",
            "java.lang.StrictMath",
            "java.lang.System",
            "java.lang.ProcessBuilder.Redirect",
            "java.lang.invoke.MethodHandles"
        ));

        if (currentPackage != null) {
            implicitImports.addAll(currentPackage.getDirectMemberSymbols().stream()
                .filter(symbol -> symbol instanceof JClassSymbol)
                .map(symbol -> ((JClassSymbol) symbol).getQualifiedName().toString())
                .collect(Collectors.toList()));
        }

        Set<String> allImports = imports.stream()
            .map(this::getImportName)
            .collect(Collectors.toSet());

        allImports.removeAll(usedImports);
        allImports.removeAll(implicitImports);

        Set<String> processedImports = new HashSet<>();
        for (ASTImportDeclaration importDecl : imports) {
            String importName = getImportName(importDecl);
            if (importDecl.isStarImport()) {
                processedImports.addAll(processedImports.stream()
                    .filter(name -> name.startsWith(importName))
                    .collect(Collectors.toSet()));
                processedImports.addAll(allImports.stream()
                    .filter(name -> name.startsWith(importName))
                    .collect(Collectors.toList()));
            } else {
                if (allImports.contains(importName) && !processedImports.contains(importName)) {
                    addViolation(data, importDecl);
                    processedImports.add(importName);
                }
            }
        }

        return data;
    }

    private String getImportName(ASTImportDeclaration importDecl) {
        if (importDecl.isStarImport()) {
            return importDecl.getImportedName().getImage();
        } else {
            return importDecl.getImportedName().getQualifiedName().toString();
        }
    }

package net.sourceforge.pmd.lang.java.rule.codestyle;
import java.util.HashMap;
import java.util.HashSet;
import java.util.List;
import java.util.Map;
import java.util.Set;
import java.util.regex.Matcher;
import java.util.regex.Pattern;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class UnnecessaryImportRule extends AbstractJavaRulechainRule {
    public UnnecessaryImportRule() {
        super(ASTCompilationUnit.class, ASTImportDeclaration.class);
    }

    private Set<String> usedTypes;
    private Set<String> usedStaticMembers;
    private Set<String> wildcardImports;
    private String currentPackageName;

    @Override
    public Object visit(ASTCompilationUnit node, Object data) {
        usedTypes = new HashSet<>();
        usedStaticMembers = new HashSet<>();
        wildcardImports = new HashSet<>();
        currentPackageName = node.getPackageName();
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTImportDeclaration node, Object data) {
        if (node.isStatic()) {
            if (node.isStarImport()) {
                wildcardImports.add(node.getImportedName());
            } else {
                usedStaticMembers.add(node.getImportedName());
            }
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTClassOrInterfaceDeclaration node, Object data) {
        if (node.getPackageName().equals(currentPackageName)) {
            usedTypes.add(node.getName());
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTAnnotation node, Object data) {
        usedTypes.add(node.getType().getTypeMirror().getSymbol().getQualifiedName());
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTConstructorDeclaration node, Object data) {
        for (ASTFormalParameter param : node.findDescendantsOfType(ASTFormalParameter.class)) {
            usedTypes.add(param.getType().getTypeMirror().getSymbol().getQualifiedName());
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTFieldDeclaration node, Object data) {
        for (ASTVariableDeclaratorId id : node.findDescendantsOfType(ASTVariableDeclaratorId.class)) {
            usedTypes.add(node.getType().getTypeMirror().getSymbol().getQualifiedName());
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTFormalParameter node, Object data) {
        usedTypes.add(node.getType().getTypeMirror().getSymbol().getQualifiedName());
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTLocalVariableDeclaration node, Object data) {
        usedTypes.add(node.getType().getTypeMirror().getSymbol().getQualifiedName());
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTMethodBody node, Object data) {
        for (ASTReferenceType ref : node.findDescendantsOfType(ASTReferenceType.class)) {
            JClassSymbol sym = (JClassSymbol) ref.getTypeMirror().getSymbol();
            usedTypes.add(sym.getQualifiedName());
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTMethodDeclaration node, Object data) {
        if (node.hasThrowsClause()) {
            for (ASTReferenceType ref : node.getThrowsClause().findDescendantsOfType(ASTReferenceType.class)) {
                JClassSymbol sym = (JClassSymbol) ref.getTypeMirror().getSymbol();
                usedTypes.add(sym.getQualifiedName());
            }
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTTypeParameter node, Object data) {
        for (ASTType type : node.getBounds()) {
            usedTypes.add(type.getTypeMirror().getSymbol().getQualifiedName());
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTJavadoc node, Object data) {
        for (ASTJavadocComment comment : node.findDescendantsOfType(ASTJavadocComment.class)) {
            Pattern pattern = Pattern.compile("(?:@link|@see|@linkplain|@value)\\s+([\\w\\.]+(?:\\[[\\w\\.]+\\])*)");
            Matcher matcher = pattern.matcher(comment.getImage());
            while (matcher.find()) {
                String matchedType = matcher.group(1).split("#")[0].replace("[", "").replace("]", "");
                if (!matchedType.contains("*")) {
                    usedTypes.add(matchedType);
                }
            }
        }
        for (ASTJavadocTag tag : node.findDescendantsOfType(ASTJavadocTag.class)) {
            if (tag.isName("snippet")) {
                Pattern pattern = Pattern.compile("(@link|@linkplain)\\s+substring=\"([\\w\\.]+)\"\\s+target=\"([\\w\\.]+)\"");
                Matcher matcher = pattern.matcher(tag.getValue());
                while (matcher.find()) {
                    usedTypes.add(matcher.group(2));
                }
            }
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTTryStatement node, Object data) {
        for (ASTCatchClause catchClause : node.findDescendantsOfType(ASTCatchClause.class)) {
            usedTypes.add(catchClause.getParameter().getType().getTypeMirror().getSymbol().getQualifiedName());
        }
        return super.visit(node, data);
    }

    @Override
    public void addViolation(Object data, ASTNode node) {
        if (node instanceof ASTImportDeclaration) {
            ASTImportDeclaration importDecl = (ASTImportDeclaration) node;
            String importName = importDecl.getImportedName();
            if (importDecl.isStatic()) {
                if (!importDecl.isStarImport() && !usedStaticMembers.contains(importName)) {
                    super.addViolation(data, node);
                }
            } else if (!importDecl.isStarImport()) {
                if (!usedTypes.contains(importName) && !isImplicitlyImported(importName)) {
                    super.addViolation(data, node);
                }
            } else if (wildcardImports.stream().anyMatch(i -> i.startsWith(importName + "."))) {
                super.addViolation(data, node);
            }
        }
    }

    private boolean isImplicitlyImported(String importName) {
        return importName.startsWith("java.lang.") || importName.equals(currentPackageName);
    }

package net.sourceforge.pmd.lang.java.rule.codestyle;

import java.util.*;
import java.util.stream.Collectors;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class UnnecessaryImportRule extends AbstractJavaRulechainRule {
    public UnnecessaryImportRule() {
        super(ASTCompilationUnit.class, ASTImportDeclaration.class);
    }

    @Override
    public Object visit(ASTCompilationUnit node, Object data) {
        Set<ASTImportDeclaration> usedImports = new HashSet<>();
        Set<ASTImportDeclaration> allImports = new HashSet<>(node.findDescendantsOfType(ASTImportDeclaration.class));
        Set<ASTImportDeclaration> onDemandImports = allImports.stream().filter(ASTImportDeclaration::isOnDemand).collect(Collectors.toSet());

        for (ASTImportDeclaration imp : allImports) {
            if (!imp.isOnDemand()) {
                usedImports.add(imp);
            }
        }

        for (ASTImportDeclaration imp : onDemandImports) {
            checkOnDemandImport(imp, node, usedImports);
        }

        for (ASTImportDeclaration imp : allImports) {
            if (!isUsed(imp, node, usedImports)) {
                addViolation(data, imp);
            }
        }

        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTImportDeclaration node, Object data) {
        return super.visit(node, data);
    }

    private void checkOnDemandImport(ASTImportDeclaration imp, ASTCompilationUnit node, Set<ASTImportDeclaration> usedImports) {
        String importedPackage = imp.getImportedPackage();
        for (ASTImportDeclaration usedImp : usedImports) {
            if (usedImp.getImportedPackage().startsWith(importedPackage)) {
                usedImports.add(imp);
            }
        }
    }

    private boolean isUsed(ASTImportDeclaration imp, ASTCompilationUnit node, Set<ASTImportDeclaration> usedImports) {
        if (imp.isStatic() && imp.isOnDemand()) {
            return isStaticOnDemandImportUsed(imp, node);
        } else if (imp.isStatic() && !imp.isOnDemand()) {
            return isStaticImportUsed(imp, node);
        } else if (imp.isOnDemand()) {
            return usedImports.contains(imp);
        } else {
            String importedName = imp.getImportedName();
            String importedPackage = imp.getImportedPackage();
            String fullyQualifiedName = importedPackage + "." + importedName;

            if (isJavaLangImport(imp) || isFromSamePackage(imp, node)) {
                return true;
            }

            NodeStream<ASTName> names = node.descendants(ASTName.class);

            for (ASTName name : names) {
                if (name.getImage().equals(importedName) && isImportUsed(name, fullyQualifiedName, node)) {
                    return true;
                }
            }

            NodeStream<ASTAnnotation> annotations = node.descendants(ASTAnnotation.class);
            for (ASTAnnotation annotation : annotations) {
                if (annotation.getTypeName().getImage().equals(importedName) && annotation.getType().getFullyQualifiedName().equals(fullyQualifiedName)) {
                    return true;
                }
            }

            NodeStream<ASTMethodDeclaration> methodDeclarations = node.descendants(ASTMethodDeclaration.class);
            for (ASTMethodDeclaration method : methodDeclarations) {
                for (JMethodSig sig : method.getFormalParameters()) {
                    if (sig.getType().getFullyQualifiedName().equals(fullyQualifiedName)) {
                        return true;
                    }
                }
            }

            NodeStream<ASTJavadocComment> javadocComments = node.descendants(ASTJavadocComment.class);
            for (ASTJavadocComment javadoc : javadocComments) {
                if (isJavadocLinkUsingImport(javadoc, fullyQualifiedName)) {
                    return true;
                }
            }

            return false;
        }
    }

    private boolean isStaticOnDemandImportUsed(ASTImportDeclaration imp, ASTCompilationUnit node) {
        String importedPackage = imp.getImportedPackage();
        NodeStream<ASTPrimaryExpression> primaryExpressions = node.descendants(ASTPrimaryExpression.class);
        for (ASTPrimaryExpression expr : primaryExpressions) {
            for (ASTName name : expr.findDescendantsOfType(ASTName.class)) {
                if (name.getImage().startsWith(importedPackage)) {
                    return true;
                }
            }
        }
        return false;
    }

    private boolean isStaticImportUsed(ASTImportDeclaration imp, ASTCompilationUnit node) {
        String importedName = imp.getImportedName();
        NodeStream<ASTPrimaryExpression> primaryExpressions = node.descendants(ASTPrimaryExpression.class);
        for (ASTPrimaryExpression expr : primaryExpressions) {
            for (ASTName name : expr.findDescendantsOfType(ASTName.class)) {
                if (name.getImage().equals(importedName)) {
                    return true;
                }
            }
        }
        return false;
    }

    private boolean isImportUsed(ASTName name, String fullyQualifiedName, ASTCompilationUnit node) {
        if (name.hasType()) {
            JType type = name.getType();
            return type.getFullyQualifiedName().equals(fullyQualifiedName);
        }

        ASTPrimaryExpression primaryExpression = name.getFirstParentOfType(ASTPrimaryExpression.class);
        if (primaryExpression == null) {
            return false;
        }

        JType type = primaryExpression.getType();
        if (type == null) {
            return false;
        }

        if (type instanceof JArrayType) {
            type = ((JArrayType) type).getComponentType();
        }

        return type.getFullyQualifiedName().equals(fullyQualifiedName);
    }

    private boolean isJavaLangImport(ASTImportDeclaration imp) {
        return imp.getImportedPackage().equals("java.lang");
    }

    private boolean isFromSamePackage(ASTImportDeclaration imp, ASTCompilationUnit node) {
        ASTPackageDeclaration packageDeclaration = node.getFirstDescendantOfType(ASTPackageDeclaration.class);
        String packageName = packageDeclaration != null ? packageDeclaration.getPackageName() : "";
        String importedPackage = imp.getImportedPackage();
        String importedName = imp.getImportedName();
        String fullyQualifiedName = packageName + "." + importedName;

        NodeStream<ASTClassOrInterfaceDeclaration> classDeclarations = node.descendants(ASTClassOrInterfaceDeclaration.class);
        for (ASTClassOrInterfaceDeclaration classDecl : classDeclarations) {
            if (classDecl.getName().equals(importedName)) {
                return true;
            }
        }

        NodeStream<ASTName> names = node.descendants(ASTName.class);
        for (ASTName name : names) {
            if (name.getImage().equals(importedName) && name.getType().getFullyQualifiedName().equals(fullyQualifiedName)) {
                return true;
            }
        }

        return false;
    }

    private boolean isJavadocLinkUsingImport(ASTJavadocComment javadoc, String fullyQualifiedName) {
        String javadocText = javadoc.getComment().getText();
        return javadocText.contains(fullyQualifiedName) || javadocText.contains(fullyQualifiedName.split("\\.")[fullyQualifiedName.split("\\.").length - 1]);
    }

package net.sourceforge.pmd.lang.java.rule.codestyle;

import java.util.List;
import java.util.Map;
import java.util.Set;
import java.util.HashSet;
import java.util.HashMap;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class UnnecessaryImportRule extends AbstractJavaRulechainRule {

    public UnnecessaryImportRule() {
        super(ASTCompilationUnit.class, ASTImportDeclaration.class, ASTStaticImportDeclaration.class, ASTClassOrInterfaceDeclaration.class, ASTMethodDeclaration.class, ASTFieldDeclaration.class, ASTReferenceType.class, ASTTypeReference.class, ASTVariableDeclaratorId.class, ASTResource.class, ASTName.class);
    }

    private Set<String> usedImports = new HashSet<>();
    private Set<String> usedStaticImports = new HashSet<>();
    private Map<String, Integer> importCount = new HashMap<>();
    private Map<String, Integer> staticImportCount = new HashMap<>();
    private String currentPackageName = "";

    @Override
    public Object visit(ASTCompilationUnit node, Object data) {
        usedImports.clear();
        usedStaticImports.clear();
        importCount.clear();
        staticImportCount.clear();
        currentPackageName = node.getPackageDeclaration().getImage();
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTImportDeclaration node, Object data) {
        String importedName = node.getImportedName();
        importCount.put(importedName, importCount.getOrDefault(importedName, 0) + 1);
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTStaticImportDeclaration node, Object data) {
        String importedName = node.getImportedName();
        staticImportCount.put(importedName, staticImportCount.getOrDefault(importedName, 0) + 1);
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTTypeDeclaration node, Object data) {
        for (NameOccurrence occ : node.getUsagesForName(node.getImage())) {
            usedImports.add(node.getImage());
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTMethodDeclaration node, Object data) {
        for (NameOccurrence occ : node.getUsagesForName(node.getImage())) {
            usedImports.add(node.getImage());
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTFieldDeclaration node, Object data) {
        for (NameOccurrence occ : node.getUsagesForName(node.getImage())) {
            usedImports.add(node.getImage());
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTReferenceType node, Object data) {
        JType type = node.getType();
        if (type instanceof JClassType) {
            String qualifiedName = ((JClassType) type).getQualifiedName();
            usedImports.add(qualifiedName);
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTTypeReference node, Object data) {
        JType type = node.getType();
        if (type instanceof JClassType) {
            String qualifiedName = ((JClassType) type).getQualifiedName();
            usedImports.add(qualifiedName);
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTVariableDeclaratorId node, Object data) {
        JType type = node.getType();
        if (type instanceof JClassType) {
            String qualifiedName = ((JClassType) type).getQualifiedName();
            usedImports.add(qualifiedName);
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTResource node, Object data) {
        JType type = node.getType();
        if (type instanceof JClassType) {
            String qualifiedName = ((JClassType) type).getQualifiedName();
            usedImports.add(qualifiedName);
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTName node, Object data) {
        String name = node.getImage();
        if (node.getUsagePMD().isClassReference()) {
            for (JClassType type : TypeSystem.getInstance().findClasses(name)) {
                usedImports.add(type.getQualifiedName());
            }
        }
        if (node.getUsagePMD().isStaticReference()) {
            usedStaticImports.add(name);
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTAnnotation node, Object data) {
        String name = node.getTypeName();
        if (name != null) {
            usedImports.add(name);
        }
        return super.visit(node, data);
    }

    @Override
    public void end(ASTCompilationUnit node, Object data) {
        for (Map.Entry<String, Integer> entry : importCount.entrySet()) {
            String importedName = entry.getKey();
            if (entry.getValue() > 1) {
                continue;
            }
            if (!isImplicitlyInScope(importedName, node) && !usedImports.contains(importedName)) {
                addViolation(data, node.getImportDeclarationFor(importedName));
            }
        }
        for (Map.Entry<String, Integer> entry : staticImportCount.entrySet()) {
            String importedName = entry.getKey();
            if (entry.getValue() > 1) {
                continue;
            }
            if (!isImplicitlyInScope(importedName, node) && !usedStaticImports.contains(importedName)) {
                addViolation(data, node.getStaticImportDeclarationFor(importedName));
            }
        }
    }

    private boolean isImplicitlyInScope(String importedName, ASTCompilationUnit node) {
        if (importedName.startsWith(currentPackageName)) {
            return true;
        }
        if (importedName.startsWith("java.lang")) {
            return true;
        }
        return false;
    }
}
