package net.sourceforge.pmd.lang.java.rule.performance;


import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class AvoidInstantiatingObjectsInLoopsRule extends AbstractJavaRulechainRule {

    public AvoidInstantiatingObjectsInLoopsRule() {
        super(ASTForStatement.class, ASTWhileStatement.class, ASTDoStatement.class);
    }

    @Override
    public Object visit(ASTForStatement node, Object data) {
        checkLoopBody(node.getStatement(), node);
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTWhileStatement node, Object data) {
        checkLoopBody(node.getStatement(), node);
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTDoStatement node, Object data) {
        checkLoopBody(node.getStatement(), node);
        return super.visit(node, data);
    }

    private void checkLoopBody(JavaNode node, JavaNode loopNode) {
        if (node instanceof ASTBlockStatement) {
            node.findDescendantsOfType(ASTAllocationExpression.class).stream()
                .filter(allocation -> !isAllowed(allocation))
                .forEach(allocation -> addViolation(data, allocation, loopNode));
        } else if (node instanceof ASTStatementExpression) {
            node.findDescendantsOfType(ASTAllocationExpression.class).stream()
                .filter(allocation -> !isAllowed(allocation))
                .forEach(allocation -> addViolation(data, allocation, loopNode));
        }
    }

    private boolean isAllowed(ASTAllocationExpression allocation) {
        return allocation.getFirstParentOfType(ASTReturnStatement.class) != null ||
               allocation.getFirstParentOfType(ASTThrowStatement.class) != null ||
               allocation.getFirstParentOfType(ASTCatchClause.class) != null ||
               allocation.getFirstParentOfType(ASTForStatement.class) != null ||
               allocation.getFirstParentOfType(ASTStatementExpression.class) != null &&
               (allocation.getFirstParentOfType(ASTStatementExpression.class).getParent() instanceof ASTForStatement ||
                allocation.getFirstParentOfType(ASTStatementExpression.class).getParent() instanceof ASTWhileStatement ||
                allocation.getFirstParentOfType(ASTStatementExpression.class).getParent() instanceof ASTDoStatement);
    }

    private void addViolation(Object data, ASTAllocationExpression allocation, JavaNode loopNode) {
        if (allocation.getFirstParentOfType(ASTForStatement.class) != null &&
            allocation.getFirstParentOfType(ASTForStatement.class).getExpression() != null &&
            allocation.getFirstParentOfType(ASTForStatement.class).getExpression().equals(allocation)) {
            return;
        }
        addViolationWithMessage(data, allocation, "Avoid instantiating objects inside loops.");
    }
}

import java.util.List;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class AvoidInstantiatingObjectsInLoopsRule extends AbstractJavaRulechainRule {
    public AvoidInstantiatingObjectsInLoopsRule() {
        super(ASTForStatement.class, ASTForeachStatement.class, ASTWhileStatement.class, ASTDoStatement.class);
    }

    @Override
    public Object visit(ASTForStatement node, Object data) {
        return checkLoop(node, data);
    }

    @Override
    public Object visit(ASTForeachStatement node, Object data) {
        return checkLoop(node, data);
    }

    @Override
    public Object visit(ASTWhileStatement node, Object data) {
        return checkLoop(node, data);
    }

    @Override
    public Object visit(ASTDoStatement node, Object data) {
        return checkLoop(node, data);
    }

    private Object checkLoop(ASTStatement node, Object data) {
        List<ASTBlockStatement> statements = node.getBody().getStatements();
        for (ASTBlockStatement statement : statements) {
            if (statement instanceof ASTStatement) {
                checkStatement((ASTStatement) statement, data);
            }
        }
        return data;
    }

    private void checkStatement(ASTStatement statement, Object data) {
        if (statement instanceof ASTAllocationExpression) {
            ASTAllocationExpression allocation = (ASTAllocationExpression) statement;
            if (allocation.getConstructorCall() instanceof ASTConstructorCall) {
                addViolation(data, allocation);
            }
        } else if (statement instanceof ASTThrowStatement) {
            // No violation
        } else if (statement instanceof ASTReturnStatement) {
            // No violation
        } else if (statement instanceof ASTBreakStatement) {
            // No violation
        } else if (statement instanceof ASTForeachStatement) {
            // No violation
        } else if (statement instanceof ASTBlockStatement) {
            for (ASTStatement child : ((ASTBlockStatement) statement).getStatements()) {
                checkStatement(child, data);
            }
        }
    }

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class AvoidInstantiatingObjectsInLoopsRule extends AbstractJavaRulechainRule {
    public AvoidInstantiatingObjectsInLoopsRule() {
        super(ASTForStatement.class, ASTWhileStatement.class, ASTDoStatement.class);
    }

    @Override
    public Object visit(ASTForStatement node, Object data) {
        NodeStream<ASTAllocationExpression> allocations = node.getBody().findDescendantsOfType(ASTAllocationExpression.class);
        for (ASTAllocationExpression allocation : allocations) {
            if (!isAllowedInLoop(allocation)) {
                addViolation(data, allocation);
            }
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTWhileStatement node, Object data) {
        NodeStream<ASTAllocationExpression> allocations = node.getBody().findDescendantsOfType(ASTAllocationExpression.class);
        for (ASTAllocationExpression allocation : allocations) {
            if (!isAllowedInLoop(allocation)) {
                addViolation(data, allocation);
            }
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTDoStatement node, Object data) {
        NodeStream<ASTAllocationExpression> allocations = node.getBody().findDescendantsOfType(ASTAllocationExpression.class);
        for (ASTAllocationExpression allocation : allocations) {
            if (!isAllowedInLoop(allocation)) {
                addViolation(data, allocation);
            }
        }
        return super.visit(node, data);
    }

    private boolean isAllowedInLoop(ASTAllocationExpression allocation) {
        JavaNode parent = allocation.getParent();
        while (parent != null) {
            if (parent instanceof ASTReturnStatement || parent instanceof ASTThrowStatement) {
                return true;
            }
            if (parent instanceof ASTBlockStatement) {
                JavaNode grandParent = parent.getParent();
                if (grandParent instanceof ASTTryStatement) {
                    return true;
                }
            }
            parent = parent.getParent();
        }
        return false;
    }
}

public class AvoidInstantiatingObjectsInLoopsRule extends AbstractJavaRulechainRule {
    public AvoidInstantiatingObjectsInLoopsRule() {
        super(ASTForStatement.class, ASTWhileStatement.class, ASTDoStatement.class, ASTForeachStatement.class);
    }
    @Override
    public Object visit(ASTForStatement node, Object data) {
        NodeStream<ASTAllocationExpression> allocations = node.findDescendantsOfType(ASTAllocationExpression.class);
        for (ASTAllocationExpression allocation : allocations) {
            if (isInsideLoopBody(allocation, node) && !isAllowed(allocation)) {
                addViolation(data, allocation);
            }
        }
        return super.visit(node, data);
    }
    @Override
    public Object visit(ASTWhileStatement node, Object data) {
        NodeStream<ASTAllocationExpression> allocations = node.findDescendantsOfType(ASTAllocationExpression.class);
        for (ASTAllocationExpression allocation : allocations) {
            if (isInsideLoopBody(allocation, node) && !isAllowed(allocation)) {
                addViolation(data, allocation);
            }
        }
        return super.visit(node, data);
    }
    @Override

        NodeStream<ASTAllocationExpression> allocations = node.findDescendantsOfType(ASTAllocationExpression.class);
        for (ASTAllocationExpression allocation : allocations) {
            if (isInsideLoopBody(allocation, node) && !isAllowed(allocation)) {
                addViolation(data, allocation);
            }
        }
        return super.visit(node, data);
    }

        NodeStream<ASTAllocationExpression> allocations = node.findDescendantsOfType(ASTAllocationExpression.class);
        for (ASTAllocationExpression allocation : allocations) {
            if (isInsideLoopBody(allocation, node) && !isAllowed(allocation)) {
                addViolation(data, allocation);
            }
        }
        return super.visit(node, data);
    }
    private boolean isInsideLoopBody(JavaNode node, JavaNode loopNode) {
        JavaNode parent = node.getParent();
        while (parent != null) {
            if (parent == loopNode) {
                return true;
            }
            parent = parent.getParent();
        }
        return false;
    }
    private boolean isAllowed(ASTAllocationExpression allocation) {
        JavaNode parent = allocation.getParent();
        while (parent != null && !(parent instanceof ASTMethodDeclaration) && !(parent instanceof ASTConstructorDeclaration)) {
            if (parent instanceof ASTBreakStatement || parent instanceof ASTReturnStatement) {
                return true;
            }
            if (parent instanceof ASTCatchClause) {
                return true;

                return true;
            }
            parent = parent.getParent();
        }
