package net.sourceforge.pmd.lang.java.rule.codestyle;

import java.util.List;
import java.util.Arrays;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class MethodNamingConventionsRule extends AbstractJavaRulechainRule {

    private static final List<String> JUNIT3_PREFIXES = Arrays.asList("junit.framework.TestCase", "junit.framework.Assert");
    private static final String JUNIT4_ANNOTATION = "org.junit.Test";
    private static final String JUNIT5_ANNOTATION = "org.junit.jupiter.api.Test";
    private static final String JUNIT5_PARAMETERIZED_ANNOTATION = "org.junit.jupiter.params.ParameterizedTest";

    public MethodNamingConventionsRule() {
        super(ASTMethodDeclaration.class);
    }

    @Override
    public Object visit(ASTMethodDeclaration node, Object data) {
        String methodName = node.getName();
        if (node.isNative()) {
            if (!methodName.matches("[a-z][a-zA-Z0-9]*")) {
                addViolation(data, node);
            }
        } else if (node.isStatic()) {
            if (!methodName.matches("[a-z][a-zA-Z0-9]*")) {
                addViolation(data, node);
            }
        } else if (isJUnit3TestMethod(node)) {
            if (!methodName.matches("test[A-Z0-9][a-zA-Z0-9]*")) {
                addViolation(data, node);
            }
        } else if (isJUnit4TestMethod(node) || isJUnit5TestMethod(node)) {
            if (!methodName.matches("[a-z][a-zA-Z0-9]*")) {
                addViolation(data, node);
            }
        } else {
            if (!methodName.matches("[a-z][a-zA-Z0-9]*")) {
                addViolation(data, node);
            }
        }
        return super.visit(node, data);
    }

    private boolean isJUnit3TestMethod(ASTMethodDeclaration node) {
        ASTClassOrInterfaceDeclaration classNode = node.getFirstParentOfType(ASTClassOrInterfaceDeclaration.class);
        if (classNode != null) {
            for (String prefix : JUNIT3_PREFIXES) {
                if (classNode.getSuperClassType() != null && classNode.getSuperClassType().isSubtypeOf(prefix)) {
                    return true;
                }
            }
        }
        return false;
    }

    private boolean isJUnit4TestMethod(ASTMethodDeclaration node) {
        return hasAnnotation(node, JUNIT4_ANNOTATION);
    }

    private boolean isJUnit5TestMethod(ASTMethodDeclaration node) {
        return hasAnnotation(node, JUNIT5_ANNOTATION) || hasAnnotation(node, JUNIT5_PARAMETERIZED_ANNOTATION);
    }

    private boolean hasAnnotation(ASTMethodDeclaration node, String annotation) {
        TypeSystem typeSystem = node.getTopLevelNode().getTypeSystem();
        JClassType annotationType = typeSystem.forName(annotation);
        if (annotationType != null) {
            for (ASTAnnotation ann : node.getAnnotations()) {
                if (ann.getType().isSubtypeOf(annotationType)) {
                    return true;
                }
            }
        }
        return false;
    }
}

package net.sourceforge.pmd.lang.java.rule.codestyle;


import java.util.regex.Pattern;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class MethodNamingConventionsRule extends AbstractJavaRulechainRule {
    private static final Pattern INSTANCE_METHOD_PATTERN = Pattern.compile("[a-z][a-zA-Z0-9]*");
    private static final Pattern STATIC_METHOD_PATTERN = Pattern.compile("[a-z][a-zA-Z0-9]*");
    private static final Pattern NATIVE_METHOD_PATTERN = Pattern.compile("[a-z][a-zA-Z0-9]*");
    private static final Pattern JUNIT3_TEST_METHOD_PATTERN = Pattern.compile("test[A-Z0-9][a-zA-Z0-9]*");
    private static final Pattern JUNIT4_TEST_METHOD_PATTERN = Pattern.compile("[a-z][a-zA-Z0-9]*");
    private static final Pattern JUNIT5_TEST_METHOD_PATTERN = Pattern.compile("[a-z][a-zA-Z0-9]*");

    public MethodNamingConventionsRule() {
        super(ASTMethodDeclaration.class, ASTClassOrInterfaceDeclaration.class);
    }

    @Override
    public Object visit(ASTClassOrInterfaceDeclaration node, Object data) {
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTMethodDeclaration node, Object data) {
        JMethodSymbol methodSymbol = node.getSymbol();
        String methodName = methodSymbol.getName();
        boolean isStatic = methodSymbol.isStatic();
        boolean isNative = methodSymbol.isNative();

        if (isJUnit3TestMethod(node)) {
            if (!JUNIT3_TEST_METHOD_PATTERN.matcher(methodName).matches()) {
                addViolation(data, node);
            }
        } else if (isJUnit4TestMethod(node)) {
            if (!JUNIT4_TEST_METHOD_PATTERN.matcher(methodName).matches()) {
                addViolation(data, node);
            }
        } else if (isJUnit5TestMethod(node)) {
            if (!JUNIT5_TEST_METHOD_PATTERN.matcher(methodName).matches()) {
                addViolation(data, node);
            }
        } else if (isNative) {
            if (!NATIVE_METHOD_PATTERN.matcher(methodName).matches()) {
                addViolation(data, node);
            }
        } else if (isStatic) {
            if (!STATIC_METHOD_PATTERN.matcher(methodName).matches()) {
                addViolation(data, node);
            }
        } else {
            if (!INSTANCE_METHOD_PATTERN.matcher(methodName).matches()) {
                addViolation(data, node);
            }
        }

        return data;
    }

    private boolean isJUnit3TestMethod(ASTMethodDeclaration node) {
        ASTClassOrInterfaceDeclaration classNode = node.getNearestParent(ASTClassOrInterfaceDeclaration.class);
        if (classNode != null) {
            JClassType superClassType = classNode.getType().getSuperClass();
            while (superClassType != null) {
                if ("junit.framework.TestCase".equals(superClassType.getQualifiedName())) {
                    return true;
                }
                superClassType = superClassType.getSuperClass();
            }
        }
        return false;
    }

    private boolean isJUnit4TestMethod(ASTMethodDeclaration node) {
        return node.hasAnnotation("org.junit.Test");
    }

    private boolean isJUnit5TestMethod(ASTMethodDeclaration node) {
        return node.hasAnnotation("org.junit.jupiter.api.Test") || node.hasAnnotation("org.junit.jupiter.params.ParameterizedTest");
    }
}

import java.util.List;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class MethodNamingConventionsRule extends AbstractJavaRulechainRule {
    private static final String INSTANCE_METHOD_REGEX = "[a-z][a-zA-Z0-9]*";
    private static final String STATIC_METHOD_REGEX = "[a-z][a-zA-Z0-9]*";
    private static final String NATIVE_METHOD_REGEX = "[a-z][a-zA-Z0-9]*";
    private static final String JUNIT3_TEST_METHOD_REGEX = "test[A-Z0-9][a-zA-Z0-9]*";
    private static final String JUNIT4_TEST_METHOD_REGEX = "[a-z][a-zA-Z0-9]*";
    private static final String JUNIT5_TEST_METHOD_REGEX = "[a-z][a-zA-Z0-9]*";
    private static final String JUNIT4_TEST_ANNOTATION = "org.junit.Test";
    private static final String JUNIT5_TEST_ANNOTATION = "org.junit.jupiter.api.Test";
    private static final String JUNIT5_PARAMETERIZED_TEST_ANNOTATION = "org.junit.jupiter.params.ParameterizedTest";
    private static final String JUNIT3_TESTCASE_CLASS = "junit.framework.TestCase";

    public MethodNamingConventionsRule() {
        super(ASTMethodDeclaration.class, ASTClassOrInterfaceDeclaration.class);
    }

    @Override
    public Object visit(ASTClassOrInterfaceDeclaration node, Object data) {
        TypeSystem typeSystem = node.getTypeSystem();
        JClassSymbol classSymbol = node.getSymbol();
        if (classSymbol != null) {
            JClassType testCaseType = typeSystem.forName(JUNIT3_TESTCASE_CLASS);
            if (testCaseType != null && classSymbol.isSubTypeOf(testCaseType)) {
                for (ASTMethodDeclaration method : node.getDeclaredMethods()) {
                    checkJUnit3TestMethods(method);
                }
            }
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTMethodDeclaration node, Object data) {
        if (node.isStatic()) {
            checkMethodName(node, STATIC_METHOD_REGEX);
        } else if (node.isNative()) {
            checkMethodName(node, NATIVE_METHOD_REGEX);
        } else if (isJUnit4Test(node) || isJUnit5Test(node)) {
            checkMethodName(node, JUNIT4_TEST_METHOD_REGEX);
        } else {
            checkMethodName(node, INSTANCE_METHOD_REGEX);
        }
        return super.visit(node, data);
    }

    private void checkMethodName(ASTMethodDeclaration node, String regex) {
        String methodName = node.getName();
        if (!methodName.matches(regex)) {
            addViolation(data, node);
        }
    }

    private void checkJUnit3TestMethods(ASTMethodDeclaration node) {
        String methodName = node.getName();
        if (!methodName.matches(JUNIT3_TEST_METHOD_REGEX)) {
            addViolation(data, node);
        }
    }

    private boolean isJUnit4Test(ASTMethodDeclaration node) {
        return node.hasAnnotation(JUNIT4_TEST_ANNOTATION);
    }

    private boolean isJUnit5Test(ASTMethodDeclaration node) {
        List<ASTAnnotation> annotations = node.getAnnotations();
        for (ASTAnnotation annotation : annotations) {
            if (annotation.getTypeName().equals(JUNIT5_TEST_ANNOTATION) || annotation.getTypeName().equals(JUNIT5_PARAMETERIZED_TEST_ANNOTATION)) {
                return true;
            }
        }
        return false;
    }
}
