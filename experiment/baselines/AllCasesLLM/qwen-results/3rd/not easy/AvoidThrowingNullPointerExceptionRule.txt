import java.util.HashSet;
import java.util.Set;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class AvoidThrowingNullPointerExceptionRule extends AbstractJavaRulechainRule {
    public AvoidThrowingNullPointerExceptionRule() {
        super(ASTThrowStatement.class, ASTAssignmentExpression.class);
    }

    private Set<String> npeVariables = new HashSet<>();

    @Override
    public Object visit(ASTThrowStatement node, Object data) {
        ASTExpression expression = node.getExpression();
        if (expression instanceof ASTVariableDeclaratorId) {
            ASTVariableDeclaratorId vdi = (ASTVariableDeclaratorId) expression;
            if (npeVariables.contains(vdi.getName())) {
                addViolation(data, node);
            }
        } else if (expression instanceof ASTClassOrInterfaceType) {
            ASTClassOrInterfaceType type = (ASTClassOrInterfaceType) expression;
            JClassType classType = TypeSystem.getTypeFromNode(type);
            if (classType != null && classType.getSymbol() instanceof JClassSymbol) {
                JClassSymbol symbol = (JClassSymbol) classType.getSymbol();
                if (symbol.getQualifiedName().equals("java.lang.NullPointerException")) {
                    addViolation(data, node);
                }
            }
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTAssignmentExpression node, Object data) {
        ASTVariableDeclaratorId lhs = node.getLeftOperand().getFirstDescendantOfType(ASTVariableDeclaratorId.class);
        ASTExpression rhs = node.getRightOperand();
        if (lhs != null && rhs instanceof ASTClassOrInterfaceType) {
            ASTClassOrInterfaceType type = (ASTClassOrInterfaceType) rhs;
            JClassType classType = TypeSystem.getTypeFromNode(type);
            if (classType != null && classType.getSymbol() instanceof JClassSymbol) {
                JClassSymbol symbol = (JClassSymbol) classType.getSymbol();
                if (symbol.getQualifiedName().equals("java.lang.NullPointerException")) {
                    npeVariables.add(lhs.getName());
                }
            }
        } else if (lhs != null && rhs instanceof ASTVariableDeclaratorId) {
            ASTVariableDeclaratorId rhsVdi = (ASTVariableDeclaratorId) rhs;
            if (npeVariables.contains(rhsVdi.getName())) {
                npeVariables.add(lhs.getName());
            } else {
                npeVariables.remove(lhs.getName());
            }
        } else if (lhs != null) {
            npeVariables.remove(lhs.getName());
        }
        return super.visit(node, data);
    }
}

package net.sourceforge.pmd.lang.java.rule.design;

import java.util.HashSet;
import java.util.Set;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class AvoidThrowingNullPointerExceptionRule extends AbstractJavaRulechainRule {
    public AvoidThrowingNullPointerExceptionRule() {
        super(ASTClassOrInterfaceDeclaration.class, ASTMethodDeclaration.class, ASTThrowStatement.class, ASTVariableDeclaratorId.class, ASTAssignmentExpression.class);
    }

    private Set<String> npeVariables = new HashSet<>();

    @Override
    public Object visit(ASTClassOrInterfaceDeclaration node, Object data) {
        npeVariables.clear();
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTMethodDeclaration node, Object data) {
        npeVariables.clear();
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTThrowStatement node, Object data) {
        ASTPrimaryExpression expr = node.getFirstChildOfType(ASTPrimaryExpression.class);
        if (expr != null) {
            ASTPrimaryPrefix prefix = expr.getFirstChildOfType(ASTPrimaryPrefix.class);
            if (prefix != null) {
                ASTAllocationExpression alloc = prefix.getFirstChildOfType(ASTAllocationExpression.class);
                if (alloc != null) {
                    ASTName name = alloc.getFirstChildOfType(ASTName.class);
                    if (name != null && "NullPointerException".equals(name.getImage())) {
                        addViolation(data, node);
                    } else {
                        ASTPrimaryPrefix varPrefix = prefix.getFirstChildOfType(ASTPrimaryPrefix.class);
                        if (varPrefix != null && varPrefix.getNamingContext().getNameDeclaration().getImage() != null) {
                            String varName = varPrefix.getNamingContext().getNameDeclaration().getImage();
                            if (npeVariables.contains(varName)) {
                                addViolation(data, node);
                            }
                        }
                    }
                }
            }
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTVariableDeclaratorId node, Object data) {
        ASTAllocationExpression alloc = node.getParent().getFirstChildOfType(ASTAllocationExpression.class);
        if (alloc != null) {
            ASTName name = alloc.getFirstChildOfType(ASTName.class);
            if (name != null && "NullPointerException".equals(name.getImage())) {
                npeVariables.add(node.getImage());
            }
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTAssignmentExpression node, Object data) {
        ASTPrimaryExpression leftExpr = node.getFirstChildOfType(ASTPrimaryExpression.class);
        ASTPrimaryExpression rightExpr = node.getFirstDescendantOfType(ASTPrimaryExpression.class);
        if (leftExpr != null && rightExpr != null) {
            ASTPrimaryPrefix leftPrefix = leftExpr.getFirstChildOfType(ASTPrimaryPrefix.class);
            ASTAllocationExpression rightAlloc = rightExpr.getFirstChildOfType(ASTAllocationExpression.class);
            if (leftPrefix != null && rightAlloc != null) {
                ASTName rightName = rightAlloc.getFirstChildOfType(ASTName.class);
                if (rightName != null && "NullPointerException".equals(rightName.getImage())) {
                    npeVariables.add(leftPrefix.getNamingContext().getNameDeclaration().getImage());
                } else {
                    ASTPrimaryPrefix rightPrefix = rightExpr.getFirstChildOfType(ASTPrimaryPrefix.class);
                    if (rightPrefix != null) {
                        String varName = rightPrefix.getNamingContext().getNameDeclaration().getImage();
                        if (!npeVariables.contains(varName)) {
                            npeVariables.remove(leftPrefix.getNamingContext().getNameDeclaration().getImage());
                        }
                    }
                }
            }
        }
        return super.visit(node, data);
    }
}

package net.sourceforge.pmd.lang.java.rule.design;

import java.util.HashMap;
import java.util.Map;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class AvoidThrowingNullPointerExceptionRule extends AbstractJavaRulechainRule {
    private Map<JVariableSymbol, JClassType> variableTypes;

    public AvoidThrowingNullPointerExceptionRule() {
        super(ASTClassOrInterfaceDeclaration.class, ASTMethodDeclaration.class, ASTThrowStatement.class, ASTVariableDeclarator.class);
        variableTypes = new HashMap<>();
    }

    @Override
    public Object visit(ASTClassOrInterfaceDeclaration node, Object data) {
        variableTypes.clear();
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTMethodDeclaration node, Object data) {
        variableTypes.clear();
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTVariableDeclarator node, Object data) {
        JVariableSymbol variableSymbol = node.getVariableSymbol();
        ASTPrimaryExpression initializer = node.getFirstChildOfType(ASTPrimaryExpression.class);
        if (initializer != null) {
            JClassType type = getNullPointerExceptionType(initializer);
            if (type != null) {
                variableTypes.put(variableSymbol, type);
            }
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTThrowStatement node, Object data) {
        ASTExpression expression = node.getFirstChildOfType(ASTExpression.class);
        if (expression != null) {
            JClassType type = getNullPointerExceptionType(expression);
            if (type != null) {
                addViolation(data, node);
            } else {
                JVariableSymbol variableSymbol = getVariableSymbolFromExpression(expression);
                if (variableSymbol != null && variableTypes.containsKey(variableSymbol)) {
                    addViolation(data, node);
                }
            }
        }
        return super.visit(node, data);
    }

    private JClassType getNullPointerExceptionType(ASTExpression expression) {
        ASTPrimaryPrefix prefix = expression.getFirstChildOfType(ASTPrimaryPrefix.class);
        if (prefix != null) {
            ASTConstructorCall constructorCall = prefix.getFirstChildOfType(ASTConstructorCall.class);
            if (constructorCall != null) {
                ASTName name = constructorCall.getFirstChildOfType(ASTName.class);
                if (name != null && "NullPointerException".equals(name.getImage())) {
                    TypeSystem typeSystem = getRuleContext().getTypeSystem();
                    return typeSystem.forName("java.lang.NullPointerException").getType();
                }
            }
        }
        return null;
    }

    private JVariableSymbol getVariableSymbolFromExpression(ASTExpression expression) {
        ASTPrimaryPrefix prefix = expression.getFirstChildOfType(ASTPrimaryPrefix.class);
        if (prefix != null) {
            ASTName name = prefix.getFirstChildOfType(ASTName.class);
            if (name != null) {
                return name.getQualifier();
            }
        }
        return null;
    }
}

package net.sourceforge.pmd.lang.java.rule.design;

import java.util.HashMap;
import java.util.Map;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class AvoidThrowingNullPointerExceptionRule extends AbstractJavaRulechainRule {
    private Map<String, Type> variableTypes = new HashMap<>();

    public AvoidThrowingNullPointerExceptionRule() {
        super(ASTLocalVariableDeclaration.class, ASTThrowStatement.class, ASTAssignmentExpression.class);
    }

    @Override
    public Object visit(ASTLocalVariableDeclaration node, Object data) {
        for (ASTVariableDeclaratorId id : node.findDescendantsOfType(ASTVariableDeclaratorId.class)) {
            ASTVariableInitializer initializer = id.getFirstChildOfType(ASTVariableInitializer.class);
            if (initializer != null) {
                ASTExpression expression = initializer.getFirstChildOfType(ASTExpression.class);
                if (expression instanceof ASTConstructorCall) {
                    ASTConstructorCall constructorCall = (ASTConstructorCall) expression;
                    if ("java.lang.NullPointerException".equals(constructorCall.getType().getQualifiedName())) {
                        variableTypes.put(id.getImage(), constructorCall.getType());
                    }
                }
            }
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTThrowStatement node, Object data) {
        ASTExpression expression = node.getFirstChildOfType(ASTExpression.class);
        if (expression instanceof ASTVariableDeclaratorId) {
            ASTVariableDeclaratorId id = (ASTVariableDeclaratorId) expression;
            Type type = variableTypes.get(id.getImage());
            if (type != null && "java.lang.NullPointerException".equals(type.getQualifiedName())) {
                addViolation(data, node);
            }
        } else if (expression instanceof ASTConstructorCall) {
            ASTConstructorCall constructorCall = (ASTConstructorCall) expression;
            if ("java.lang.NullPointerException".equals(constructorCall.getType().getQualifiedName())) {
                addViolation(data, node);
            }
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTAssignmentExpression node, Object data) {
        ASTVariableDeclaratorId left = node.getLeftOperand().getFirstChildOfType(ASTVariableDeclaratorId.class);
        ASTExpression right = node.getRightOperand().getFirstChildOfType(ASTExpression.class);
        if (right instanceof ASTConstructorCall) {
            ASTConstructorCall constructorCall = (ASTConstructorCall) right;
            if ("java.lang.NullPointerException".equals(constructorCall.getType().getQualifiedName())) {
                variableTypes.put(left.getImage(), constructorCall.getType());
            }
        } else {
            variableTypes.remove(left.getImage());
        }
        return super.visit(node, data);
    }
}

package net.sourceforge.pmd.lang.java.rule.design;


import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class AvoidThrowingNullPointerExceptionRule extends AbstractJavaRulechainRule {
    public AvoidThrowingNullPointerExceptionRule() {
        super(ASTThrowStatement.class, ASTLocalVariableDeclaration.class);
    }

    @Override
    public Object visit(ASTThrowStatement node, Object data) {
        if (node.getExpression() instanceof ASTConstructorCall) {
            ASTConstructorCall call = (ASTConstructorCall) node.getExpression();
            if (TypeTestUtil.isA(call.getType(), NullPointerException.class)) {
                addViolation(data, node);
            }
        } else if (node.getExpression() instanceof ASTName) {
            ASTName name = (ASTName) node.getExpression();
            VariableNameDeclaration decl = name.getNameDeclaration();
            if (decl != null) {
                for (ASTBlockStatement stmt : decl.getNode().getASTBlock().findDescendantsOfType(ASTBlockStatement.class)) {
                    if (stmt.getFirstChildOfType(ASTAssignmentExpression.class) != null) {
                        ASTAssignmentExpression assignment = stmt.getFirstChildOfType(ASTAssignmentExpression.class);
                        if (assignment.getLeftOperand() instanceof ASTName && ((ASTName) assignment.getLeftOperand()).getImage().equals(name.getImage())) {
                            if (assignment.getRightOperand() instanceof ASTConstructorCall) {
                                ASTConstructorCall call = (ASTConstructorCall) assignment.getRightOperand();
                                if (TypeTestUtil.isA(call.getType(), NullPointerException.class)) {
                                    addViolation(data, node);
                                }
                            }
                        }
                    }
                }
            }
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTLocalVariableDeclaration node, Object data) {
        for (ASTVariableDeclaratorId id : node.findDescendantsOfType(ASTVariableDeclaratorId.class)) {
            for (ASTBlockStatement stmt : node.getASTBlock().findDescendantsOfType(ASTBlockStatement.class)) {
                if (stmt.getFirstChildOfType(ASTAssignmentExpression.class) != null) {
                    ASTAssignmentExpression assignment = stmt.getFirstChildOfType(ASTAssignmentExpression.class);
                    if (assignment.getLeftOperand() instanceof ASTName && ((ASTName) assignment.getLeftOperand()).getImage().equals(id.getImage())) {
                        if (assignment.getRightOperand() instanceof ASTConstructorCall) {
                            ASTConstructorCall call = (ASTConstructorCall) assignment.getRightOperand();
                            if (TypeTestUtil.isA(call.getType(), NullPointerException.class)) {
                                for (ASTBlockStatement throwStmt : node.getASTBlock().findDescendantsOfType(ASTBlockStatement.class)) {

                                        ASTThrowStatement throwStatement = throwStmt.getFirstChildOfType(ASTThrowStatement.class);
                                        if (throwStatement.getExpression() instanceof ASTName && ((ASTName) throwStatement.getExpression()).getImage().equals(id.getImage())) {
                                            addViolation(data, throwStatement);
                                        }
                                    }
                                }
                            }
                        }
                    }
                }
            }
        }
        return super.visit(node, data);
    }
