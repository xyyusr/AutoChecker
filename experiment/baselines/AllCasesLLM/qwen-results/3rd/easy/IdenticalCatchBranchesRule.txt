import java.util.List;
import java.util.Objects;
import java.util.stream.Collectors;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class IdenticalCatchBranchesRule extends AbstractJavaRulechainRule {
    public IdenticalCatchBranchesRule() {
        super(ASTTryStatement.class);
    }

    @Override
    public Object visit(ASTTryStatement node, Object data) {
        List<ASTCatchStatement> catchStatements = node.findDescendantsOfType(ASTCatchStatement.class);
        if (catchStatements.size() < 2) {
            return super.visit(node, data);
        }

        for (int i = 0; i < catchStatements.size() - 1; i++) {
            ASTCatchStatement currentCatch = catchStatements.get(i);
            ASTCatchStatement nextCatch = catchStatements.get(i + 1);

            if (areCatchBranchesIdentical(currentCatch, nextCatch)) {
                addViolation(data, currentCatch);
            }
        }

        return super.visit(node, data);
    }

    private boolean areCatchBranchesIdentical(ASTCatchStatement catch1, ASTCatchStatement catch2) {
        List<ASTStatement> statements1 = catch1.findDescendantsOfType(ASTStatement.class);
        List<ASTStatement> statements2 = catch2.findDescendantsOfType(ASTStatement.class);

        if (statements1.size() != statements2.size()) {
            return false;
        }

        for (int i = 0; i < statements1.size(); i++) {
            if (!Objects.equals(statements1.get(i).getXPathNodeName(), statements2.get(i).getXPathNodeName())) {
                return false;
            }
            if (!statements1.get(i).isSimilar(statements2.get(i))) {
                return false;
            }
        }

        return true;
    }
}

package net.sourceforge.pmd.lang.java.rule.codestyle;

import java.util.List;
import java.util.Objects;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class IdenticalCatchBranchesRule extends AbstractJavaRulechainRule {
    public IdenticalCatchBranchesRule() {
        super(ASTTryStatement.class);
    }

    @Override
    public Object visit(ASTTryStatement node, Object data) {
        List<ASTCatchStatement> catchStatements = node.getCatchStatements();
        for (int i = 0; i < catchStatements.size(); i++) {
            for (int j = i + 1; j < catchStatements.size(); j++) {
                ASTCatchStatement catch1 = catchStatements.get(i);
                ASTCatchStatement catch2 = catchStatements.get(j);
                if (areCatchBranchesIdentical(catch1, catch2)) {
                    addViolation(data, catch2);
                }
            }
        }
        return super.visit(node, data);
    }

    private boolean areCatchBranchesIdentical(ASTCatchStatement catch1, ASTCatchStatement catch2) {
        ASTBlock block1 = catch1.getFirstChildOfType(ASTBlock.class);
        ASTBlock block2 = catch2.getFirstChildOfType(ASTBlock.class);
        if (block1 == null || block2 == null) {
            return false;
        }

        List<ASTStatement> statements1 = block1.findDescendantsOfType(ASTStatement.class);
        List<ASTStatement> statements2 = block2.findDescendantsOfType(ASTStatement.class);

        if (statements1.size() != statements2.size()) {
            return false;
        }

        for (int i = 0; i < statements1.size(); i++) {
            if (!areStatementsIdentical(statements1.get(i), statements2.get(i), catch1, catch2)) {
                return false;
            }
        }
        return true;
    }

    private boolean areStatementsIdentical(ASTStatement statement1, ASTStatement statement2, ASTCatchStatement catch1, ASTCatchStatement catch2) {
        if (!statement1.getXPathNodeName().equals(statement2.getXPathNodeName())) {
            return false;
        }

        if (statement1 instanceof ASTThrowStatement && statement2 instanceof ASTThrowStatement) {
            ASTPrimaryExpression expr1 = ((ASTThrowStatement) statement1).getFirstDescendantOfType(ASTPrimaryExpression.class);
            ASTPrimaryExpression expr2 = ((ASTThrowStatement) statement2).getFirstDescendantOfType(ASTPrimaryExpression.class);
            return areExpressionsIdentical(expr1, expr2, catch1, catch2);
        }

        return Objects.equals(statement1.getImage(), statement2.getImage());
    }

    private boolean areExpressionsIdentical(ASTPrimaryExpression expr1, ASTPrimaryExpression expr2, ASTCatchStatement catch1, ASTCatchStatement catch2) {
        if (expr1 == null || expr2 == null) {
            return false;
        }

        if (expr1.getFirstChildOfType(ASTName.class) != null && expr2.getFirstChildOfType(ASTName.class) != null) {
            String name1 = expr1.getFirstDescendantOfType(ASTName.class).getImage();
            String name2 = expr2.getFirstDescendantOfType(ASTName.class).getImage();
            if (isCatchParam(name1, catch1) && isCatchParam(name2, catch2)) {
                return true;
            }
        }

        return expr1.getImage().equals(expr2.getImage());
    }

    private boolean isCatchParam(String name, ASTCatchStatement catchStatement) {
        JVariableSymbol param = catchStatement.getParameter().getSymbol();
        return param != null && param.getName().equals(name);
    }
}

package net.sourceforge.pmd.lang.java.rule.codestyle;


import java.util.List;
import java.util.Objects;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class IdenticalCatchBranchesRule extends AbstractJavaRulechainRule {

    public IdenticalCatchBranchesRule() {
        super(ASTTryStatement.class);
    }

    @Override
    public Object visit(ASTTryStatement node, Object data) {
        List<ASTCatchStatement> catchStatements = node.getCatchClauses().toList();
        for (int i = 0; i < catchStatements.size(); i++) {
            ASTCatchStatement currentCatch = catchStatements.get(i);
            ASTBlock currentBlock = currentCatch.getFirstChildOfType(ASTBlock.class);
            if (currentBlock == null) {
                continue;
            }
            List<ASTStatement> currentStatements = currentBlock.findDescendantsOfType(ASTStatement.class);

            for (int j = i + 1; j < catchStatements.size(); j++) {
                ASTCatchStatement nextCatch = catchStatements.get(j);
                ASTBlock nextBlock = nextCatch.getFirstChildOfType(ASTBlock.class);
                if (nextBlock == null) {
                    continue;
                }
                List<ASTStatement> nextStatements = nextBlock.findDescendantsOfType(ASTStatement.class);

                if (currentStatements.size() != nextStatements.size()) {
                    continue;
                }

                boolean areIdentical = true;
                for (int k = 0; k < currentStatements.size(); k++) {
                    if (!Objects.equals(currentStatements.get(k).getXPathNode(), nextStatements.get(k).getXPathNode())) {
                        areIdentical = false;
                        break;
                    }

import java.util.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class IdenticalCatchBranchesRule extends AbstractJavaRulechainRule {
    public IdenticalCatchBranchesRule() {
        super(ASTTryStatement.class);
    }

    @Override
    public Object visit(ASTTryStatement node, Object data) {
        List<ASTCatchClause> catchClauses = node.findDescendantsOfType(ASTCatchClause.class);
        Map<ASTBlock, List<ASTCatchClause>> catchBranches = new HashMap<>();

        for (ASTCatchClause catchClause : catchClauses) {
            ASTBlock catchBlock = catchClause.getFirstChildOfType(ASTBlock.class);
            catchBranches.computeIfAbsent(catchBlock, k -> new ArrayList<>()).add(catchClause);
        }

        for (List<ASTCatchClause> clauses : catchBranches.values()) {
            if (clauses.size() > 1 && allBranchesIdentical(clauses)) {
                addViolation(data, clauses.get(0));
            }
        }

        return super.visit(node, data);
    }

    private boolean allBranchesIdentical(List<ASTCatchClause> clauses) {
        ASTBlock firstBlock = clauses.get(0).getFirstChildOfType(ASTBlock.class);
        List<JavaNode> firstChildren = firstBlock.getChildren();

        for (int i = 1; i < clauses.size(); i++) {
            ASTBlock currentBlock = clauses.get(i).getFirstChildOfType(ASTBlock.class);
            List<JavaNode> currentChildren = currentBlock.getChildren();

            if (!firstChildren.equals(currentChildren)) {
                return false;
            }
        }
        return true;
    }
}

public class IdenticalCatchBranchesRule extends AbstractJavaRulechainRule {
    public IdenticalCatchBranchesRule() {
        super(ASTTryCatchFinallyStatement.class);
    }

    @Override
    public Object visit(ASTTryCatchFinallyStatement node, Object data) {
        List<ASTCatchClause> catchClauses = node.findDescendantsOfType(ASTCatchClause.class);
        for (int i = 0; i < catchClauses.size(); i++) {
            ASTCatchClause currentClause = catchClauses.get(i);
            for (int j = i + 1; j < catchClauses.size(); j++) {
                ASTCatchClause nextClause = catchClauses.get(j);
                if (areCatchBranchesIdentical(currentClause, nextClause)) {
                    addViolation(data, nextClause);
                }
            }
        }
        return super.visit(node, data);
    }

    private boolean areCatchBranchesIdentical(ASTCatchClause clause1, ASTCatchClause clause2) {
        List<ASTBlockStatement> statements1 = clause1.getFirstChildOfType(ASTBlockStatement.class).findDescendantsOfType(ASTBlockStatement.class);
        List<ASTBlockStatement> statements2 = clause2.getFirstChildOfType(ASTBlockStatement.class).findDescendantsOfType(ASTBlockStatement.class);
        return statements1.size() == statements2.size() && statements1.containsAll(statements2) && statements2.containsAll(statements1);
    }
