    public Object visit(ASTAssignmentExpression node, Object data) {
        if (node.isCompound()) {
            ASTVariableDeclaratorId lhs = node.getLeftOperand().getFirstDescendantOfType(ASTVariableDeclaratorId.class);
            if (lhs != null) {
                Type type = lhs.getType();
                if (type.isSubtypeOf(JType.STRING_TYPE)) {

                    int nodeLine = node.getBeginLine();
                    stringAppendCounts.putIfAbsent(variableName, new HashSet<>());

public class UseStringBufferForStringAppendsRule extends AbstractJavaRulechainRule {
    public UseStringBufferForStringAppendsRule() {
        super(ASTAssignmentExpression.class, ASTConstructorDeclaration.class, ASTMethodDeclaration.class, ASTStaticInitializationBlock.class);

    private Set<JVariableSymbol> stringVariablesInScope(ASTBlock block) {
        Set<JVariableSymbol> stringVariables = new HashSet<>();
        for (ASTFieldDeclaration field : block.getAncestorsOfType(ASTClassOrInterfaceDeclaration.class).getFirstDescendantOfType(ASTFieldDeclaration.class).getSiblingsOfType(ASTFieldDeclaration.class)) {
            JType fieldType = field.getTypeNode().getType();
            if (fieldType.isA(TypeSystem.STRING)) {
                stringVariables.add(field.getVariableDeclaratorId().getSymbol());
            }
        }
        for (ASTVariableDeclaration varDecl : block.getDescendantsOfType(ASTVariableDeclaration.class)) {
            JType varType = varDecl.getTypeNode().getType();
            if (varType.isA(TypeSystem.STRING)) {
                for (ASTVariableDeclaratorId varId : varDecl.getDescendantsOfType(ASTVariableDeclaratorId.class)) {
                    stringVariables.add(varId.getSymbol());
                }
            }
        }
        return stringVariables;
    }

    @Override
    public Object visit(ASTAssignmentExpression node, Object data) {
        if (node.isCompounded() && node.getLeftOperand() instanceof ASTPrimaryExpression) {
            ASTPrimaryExpression primaryExpression = (ASTPrimaryExpression) node.getLeftOperand();
            if (primaryExpression.getVariableAccess() != null) {
                JVariableSymbol symbol = primaryExpression.getVariableAccess().getSymbol();
                if (symbol.getType().isA(TypeSystem.STRING)) {
                    ASTBlock block = JavaAstUtils.getEnclosingBlock(node);
                    if (block != null) {
                        int count = 0;
                        for (ASTAssignmentExpression expr : block.getDescendantsOfType(ASTAssignmentExpression.class)) {
                            if (expr.isCompounded() && expr.getLeftOperand() instanceof ASTPrimaryExpression) {
                                ASTPrimaryExpression exprPrimaryExpression = (ASTPrimaryExpression) expr.getLeftOperand();
                                if (exprPrimaryExpression.getVariableAccess() != null && exprPrimaryExpression.getVariableAccess().getSymbol().equals(symbol)) {
                                    count++;
                                }
                            }
                        }
                        if (count > 1) {
                            addViolation(data, node);
                        }
                    }
                }
            }
        }
        return super.visit(node, data);
    }

package net.sourceforge.pmd.lang.java.rule.performance;

import java.util.HashSet;
import java.util.List;
import java.util.Set;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class UseStringBufferForStringAppendsRule extends AbstractJavaRulechainRule {
    public UseStringBufferForStringAppendsRule() {
        super(ASTMethodDeclaration.class, ASTConstructorDeclaration.class, ASTInitializer.class);
    }

    @Override
    public Object visit(ASTMethodDeclaration node, Object data) {
        return visitStatements(node, data);
    }

    @Override
    public Object visit(ASTConstructorDeclaration node, Object data) {
        return visitStatements(node, data);
    }

    @Override
    public Object visit(ASTInitializer node, Object data) {
        return visitStatements(node, data);
    }

    private Object visitStatements(ASTStatements node, Object data) {
        Set<String> stringVariables = new HashSet<>();
        for (JavaNode child : node.children()) {
            if (child instanceof ASTVariableDeclaratorId) {
                ASTVariableDeclaratorId varNode = (ASTVariableDeclaratorId) child;
                if (isStringType(varNode)) {
                    stringVariables.add(varNode.getImage());
                }
            } else if (child instanceof ASTBlockStatement) {
                ASTBlockStatement blockStatement = (ASTBlockStatement) child;
                if (blockStatement.getFirstChildOfType(ASTAssignmentExpression.class) != null) {
                    ASTAssignmentExpression assignment = blockStatement.getFirstChildOfType(ASTAssignmentExpression.class);
                    if (assignment.isCompound() && assignment.getLeftOperand() instanceof ASTVariableDeclaratorId) {
                        ASTVariableDeclaratorId varNode = (ASTVariableDeclaratorId) assignment.getLeftOperand();
                        if (stringVariables.contains(varNode.getImage()) && isStringConcatenation(assignment)) {
                            addViolation(data, assignment);
                        }
                    }
                }
            } else if (child instanceof ASTForStatement || child instanceof ASTEnhancedForStatement) {
                visitLoop((ASTBlockStatement) child, data, stringVariables);
            }
        }
        return data;
    }

    private void visitLoop(ASTBlockStatement loopNode, Object data, Set<String> stringVariables) {
        Set<String> loopVariables = new HashSet<>(stringVariables);
        ASTVariableDeclaratorId loopVar = loopNode.getFirstDescendantOfType(ASTVariableDeclaratorId.class);
        if (loopVar != null && isStringType(loopVar)) {
            loopVariables.add(loopVar.getImage());
        }
        for (JavaNode child : loopNode.children()) {
            if (child instanceof ASTBlockStatement) {
                ASTBlockStatement blockStatement = (ASTBlockStatement) child;
                if (blockStatement.getFirstChildOfType(ASTAssignmentExpression.class) != null) {
                    ASTAssignmentExpression assignment = blockStatement.getFirstChildOfType(ASTAssignmentExpression.class);
                    if (assignment.isCompound() && assignment.getLeftOperand() instanceof ASTVariableDeclaratorId) {
                        ASTVariableDeclaratorId varNode = (ASTVariableDeclaratorId) assignment.getLeftOperand();
                        if (loopVariables.contains(varNode.getImage()) && isStringConcatenation(assignment)) {
                            addViolation(data, assignment);
                        }
                    }
                }
            }
        }
    }

    private boolean isStringType(ASTVariableDeclaratorId varNode) {
        Symbol symbol = varNode.getSymbol();
        if (symbol != null && symbol.getType() instanceof JType) {
            JType type = (JType) symbol.getType();
            return type.getType() == TypeSystem.STRING_TYPE;
        }
        return false;
    }

    private boolean isStringConcatenation(ASTAssignmentExpression assignment) {
        Type type = assignment.getRightOperand().getType();
        return type != null && type.isSubtypeOf(TypeSystem.STRING_TYPE);
    }
}

import java.util.HashSet;
import java.util.Set;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class UseStringBufferForStringAppendsRule extends AbstractJavaRulechainRule {
    public UseStringBufferForStringAppendsRule() {
        super(ASTBlockStatement.class, ASTConstructorDeclaration.class, ASTMethodDeclaration.class, ASTStaticInitializer.class, ASTDoStatement.class, ASTForStatement.class, ASTWhileStatement.class);
    }

    private Set<String> stringVariables = new HashSet<>();

    @Override
    public Object visit(ASTBlockStatement node, Object data) {
        if (node.jjtGetChild(0) instanceof ASTVariableDeclaratorId && ((ASTVariableDeclaratorId) node.jjtGetChild(0)).getTypeImage().equals("String")) {
            stringVariables.add(((ASTVariableDeclaratorId) node.jjtGetChild(0)).getName());
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTConstructorDeclaration node, Object data) {
        stringVariables.clear();
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTMethodDeclaration node, Object data) {
        stringVariables.clear();
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTStaticInitializer node, Object data) {
        stringVariables.clear();
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTDoStatement node, Object data) {
        checkLoop(node.getBody(), node);
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTForStatement node, Object data) {
        checkLoop(node.getBody(), node);
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTWhileStatement node, Object data) {
        checkLoop(node.getBody(), node);
        return super.visit(node, data);
    }

    private void checkLoop(ASTBlock body, JavaNode loopNode) {
        for (ASTBlockStatement statement : body.findDescendantsOfType(ASTBlockStatement.class)) {
            if (statement.jjtGetChild(0) instanceof ASTStatementExpression) {
                ASTStatementExpression expr = (ASTStatementExpression) statement.jjtGetChild(0);
                if (expr.jjtGetChild(0) instanceof ASTAssignmentExpression) {
                    ASTAssignmentExpression assignment = (ASTAssignmentExpression) expr.jjtGetChild(0);
                    if (assignment.isCompound() && assignment.getOperator() == ASTAssignmentExpression.Operator.PLUS) {
                        ASTAdditiveExpression additive = (ASTAdditiveExpression) assignment.getRightOperand();
                        JTypeMirror type = TypeSystem.getInstance().findType("java.lang.String");
                        if (additive.getLeftOperand().getType() == type || additive.getRightOperand().getType() == type) {
                            if (stringVariables.contains(((ASTVariableDeclaratorId) assignment.getLeftOperand()).getName())) {
                                addViolation(data, loopNode);
                                break;
                            }
                        }
                    }
                }
            }
        }
    }

    @Override
    public Object visit(ASTAssignmentExpression node, Object data) {
        if (node.isCompound() && node.getOperator() == ASTAssignmentExpression.Operator.PLUS) {
            ASTAdditiveExpression additive = (ASTAdditiveExpression) node.getRightOperand();
            JTypeMirror type = TypeSystem.getInstance().findType("java.lang.String");

                }
            }
        }

    }
