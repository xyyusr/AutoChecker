package net.sourceforge.pmd.lang.java.rule.design;


import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class SignatureDeclareThrowsExceptionRule extends AbstractJavaRulechainRule {

    public SignatureDeclareThrowsExceptionRule() {
        super(ASTMethodDeclaration.class, ASTConstructorDeclaration.class);
    }

    @Override
    public Object visit(ASTMethodDeclaration node, Object data) {
        if (isJUnitMethod(node) || isTestNGMethod(node) || isOverridingMethod(node)) {
            return data;
        }
        checkThrowsClause(node, data);
        return data;
    }

    @Override
    public Object visit(ASTConstructorDeclaration node, Object data) {
        if (isJUnitConstructor(node)) {
            return data;
        }
        checkThrowsClause(node, data);
        return data;
    }

    private void checkThrowsClause(ASTThrowsClauseOwner node, Object data) {
        ASTThrowsClause throwsClause = node.getThrowsClause();
        if (throwsClause != null) {
            for (ASTReferenceType referenceType : throwsClause.getFormalParameters()) {
                JClassType type = referenceType.getType();
                if (type != null && type.getSymbol() != null && "java.lang.Exception".equals(type.getSymbol().getQualifiedName())) {
                    addViolation(data, node);
                }
            }
        }
    }

    private boolean isJUnitMethod(ASTMethodDeclaration node) {
        if (node.getName().startsWith("test")) {
            return true;
        }
        JClassSymbol declaringClass = node.getDeclaringClass().getSymbol();
        return declaringClass != null && (isJUnit3Class(declaringClass) || isJUnit4Class(declaringClass));
    }

    private boolean isJUnitConstructor(ASTConstructorDeclaration node) {
        JClassSymbol declaringClass = node.getDeclaringClass().getSymbol();
        return declaringClass != null && isJUnit3Class(declaringClass);
    }

    private boolean isJUnit3Class(JClassSymbol declaringClass) {
        return declaringClass.extendsClass("junit.framework.TestCase");
    }

    private boolean isJUnit4Class(JClassSymbol declaringClass) {
        return declaringClass.getAnnotation("org.junit.Test") != null;
    }

    private boolean isTestNGMethod(ASTMethodDeclaration node) {
        return node.getFirstDescendantOfType(ASTAnnotation.class) != null && "org.testng.annotations".equals(node.getFirstDescendantOfType(ASTAnnotation.class).getType().getSymbol().getPackageName());
    }

    private boolean isOverridingMethod(ASTMethodDeclaration node) {
        JMethodSymbol methodSymbol = node.getSymbol();
        if (methodSymbol != null && methodSymbol.isOverriding()) {
            return methodSymbol.getOverriddenSymbol().getDeclaringClass().getQualifiedName().equals("java.lang.Exception");
        }
        return false;
    }
}

package net.sourceforge.pmd.lang.java.rule.design;

import java.util.List;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class SignatureDeclareThrowsExceptionRule extends AbstractJavaRulechainRule {
    public SignatureDeclareThrowsExceptionRule() {
        super(ASTMethodDeclaration.class, ASTConstructorDeclaration.class);
    }

    @Override
    public Object visit(ASTMethodDeclaration node, Object data) {
        if (isJUnitMethod(node) || isOverridingMethod(node)) {
            return data;
        }
        checkThrowsClause(node, data);
        return data;
    }

    @Override
    public Object visit(ASTConstructorDeclaration node, Object data) {
        if (isJUnitConstructor(node)) {
            return data;
        }
        checkThrowsClause(node, data);
        return data;
    }

    private void checkThrowsClause(JavaNode node, Object data) {
        ASTThrowsClause throwsClause = node.getFirstDescendantOfType(ASTThrowsClause.class);
        if (throwsClause != null) {
            for (ASTType exceptionType : throwsClause.findDescendantsOfType(ASTType.class)) {
                if (exceptionType.getType().isSubtypeOf(TypeSystem.current().forClass(Exception.class)) && !exceptionType.getType().equals(TypeSystem.current().forClass(Exception.class))) {
                    addViolation(data, node);
                    break;
                } else if (exceptionType.getType().equals(TypeSystem.current().forClass(Exception.class))) {
                    addViolation(data, node);
                    break;
                }
            }
        }
    }

    private boolean isJUnitMethod(ASTMethodDeclaration node) {
        String methodName = node.getName();
        if (methodName.startsWith("test")) {
            return true;
        }
        ASTClassOrInterfaceDeclaration classNode = node.getFirstParentOfType(ASTClassOrInterfaceDeclaration.class);
        if (classNode != null) {
            JClassSymbol classSymbol = classNode.getSymbol();
            return isSubclassOfJUnitTestCase(classSymbol) || isAnnotatedWithJUnit(classNode);
        }
        return false;
    }

    private boolean isJUnitConstructor(ASTConstructorDeclaration node) {
        ASTClassOrInterfaceDeclaration classNode = node.getFirstParentOfType(ASTClassOrInterfaceDeclaration.class);
        if (classNode != null) {
            JClassSymbol classSymbol = classNode.getSymbol();
            return isSubclassOfJUnitTestCase(classSymbol);
        }
        return false;
    }

    private boolean isSubclassOfJUnitTestCase(JClassSymbol classSymbol) {
        if (classSymbol == null) {
            return false;
        }
        TypeSystemContext ctx = classSymbol.getProcessingContext();
        return ctx.getClassSymbol("junit.framework.TestCase").isAncestorOf(classSymbol) ||
               ctx.getClassSymbol("net.sourceforge.pmd.lang.java.rule.design.signaturedeclarethrowsexception.MyTestCase").isAncestorOf(classSymbol);
    }

    private boolean isAnnotatedWithJUnit(ASTClassOrInterfaceDeclaration classNode) {
        return classNode.hasDescendantMatching(ASTAnnotation.class, annotation -> 
            "org.junit.Before".equals(annotation.getTypeReference().getType().getQualifiedName()) ||
            "org.junit.After".equals(annotation.getTypeReference().getType().getQualifiedName()) ||

import java.util.List;
import java.util.Arrays;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class SignatureDeclareThrowsExceptionRule extends AbstractJavaRulechainRule {
    private static final List<String> JUNIT_SETUP_METHODS = Arrays.asList("setUp", "tearDown");
    private static final List<String> TESTNG_ANNOTATIONS = Arrays.asList("BeforeClass", "AfterClass", "BeforeMethod", "AfterMethod");

    public SignatureDeclareThrowsExceptionRule() {
        super(ASTMethodDeclaration.class, ASTConstructorDeclaration.class);
    }

    private boolean isJUnitMethod(ASTMethodDeclaration node) {
        ASTClassOrInterfaceDeclaration classNode = node.getNearestAncestorOfType(ASTClassOrInterfaceDeclaration.class);
        if (classNode != null) {
            String className = classNode.getImage();
            if (className.endsWith("Test")) {
                return true;
            }
            JType superClass = classNode.getSuperClass();
            if (superClass != null && superClass.getSymbol() instanceof JClassSymbol) {
                JClassSymbol superClassSymbol = (JClassSymbol) superClass.getSymbol();
                return superClassSymbol.getQualifiedName().equals("junit.framework.TestCase") || isJUnitSetupMethod(node);
            }
        }
        return false;
    }

    private boolean isJUnitSetupMethod(ASTMethodDeclaration node) {
        String methodName = node.getName();
        return JUNIT_SETUP_METHODS.contains(methodName);
    }

    private boolean isTestNgAnnotatedMethod(ASTMethodDeclaration node) {
        return node.hasDescendantMatching(ASTAnnotation.class, annotation -> {
            QualifiedName name = annotation.getTypeName();
            return name != null && TESTNG_ANNOTATIONS.contains(name.getFullyQualifiedName().getSimpleName());
        });
    }

    private boolean isOverridingMethod(ASTMethodDeclaration node) {
        return node.isOverriding();
    }

    @Override
    public Object visit(ASTMethodDeclaration node, Object data) {
        if (!isJUnitMethod(node) && !isTestNgAnnotatedMethod(node) && !isOverridingMethod(node)) {
            ASTThrowsClause throwsClause = node.getFirstChildOfType(ASTThrowsClause.class);
            if (throwsClause != null) {
                for (ASTName exceptionType : throwsClause.findDescendantsOfType(ASTName.class)) {
                    if (exceptionType.getImage().equals("Exception")) {
                        addViolation(data, node);
                    }
                }
            }
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTConstructorDeclaration node, Object data) {
        if (!node.getNearestAncestorOfType(ASTClassOrInterfaceDeclaration.class).getImage().endsWith("Test")) {
            ASTThrowsClause throwsClause = node.getFirstChildOfType(ASTThrowsClause.class);

                    }
                }
            }
        }

package net.sourceforge.pmd.lang.java.rule.design;


import java.util.List;
import java.util.Set;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class SignatureDeclareThrowsExceptionRule extends AbstractJavaRulechainRule {

    public SignatureDeclareThrowsExceptionRule() {
        super(ASTMethodDeclaration.class, ASTConstructorDeclaration.class);
    }

    @Override
    public Object visit(ASTMethodDeclaration node, Object data) {
        if (isJUnitMethod(node) || isOverridingMethod(node)) {
            return super.visit(node, data);
        }
        checkThrowsClause(node.getThrowsClause(), data);
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTConstructorDeclaration node, Object data) {
        if (isJUnitConstructor(node)) {
            return super.visit(node, data);
        }
        checkThrowsClause(node.getThrowsClause(), data);
        return super.visit(node, data);
    }

    private void checkThrowsClause(ASTThrowsClause throwsClause, Object data) {
        if (throwsClause != null) {
            for (JType type : throwsClause.getExceptions()) {
                if (TypeSystem.isExactlyA(type, Exception.class)) {
                    addViolation(data, throwsClause);
                }
            }
        }
    }

    private boolean isJUnitMethod(ASTMethodDeclaration method) {
        String methodName = method.getName();
        ASTClassOrInterfaceDeclaration classNode = method.getFirstParentOfType(ASTClassOrInterfaceDeclaration.class);
        if (classNode == null) {
            return false;
        }
        JClassSymbol classSymbol = classNode.getSymbol();
        if (classSymbol == null) {
            return false;
        }
        if (methodName.startsWith("test") && isJUnitTest(classSymbol)) {
            return true;
        }
        return isJUnitSetupOrTearDown(method, classSymbol);
    }

    private boolean isJUnitSetupOrTearDown(ASTMethodDeclaration method, JClassSymbol classSymbol) {
        String methodName = method.getName();
        return (methodName.equals("setUp") || methodName.equals("tearDown")) && isJUnitTest(classSymbol);
    }

    private boolean isJUnitTest(JClassSymbol classSymbol) {
        Set<JClassSymbol> supertypes = classSymbol.getSuperTypes();
        for (JClassSymbol supertype : supertypes) {
            if (supertype.getQualifiedName().equals("junit.framework.TestCase")) {
                return true;
            }
        }
        return false;
    }

    private boolean isJUnitConstructor(ASTConstructorDeclaration constructor) {
        ASTClassOrInterfaceDeclaration classNode = constructor.getFirstParentOfType(ASTClassOrInterfaceDeclaration.class);
        if (classNode == null) {
            return false;
        }
        JClassSymbol classSymbol = classNode.getSymbol();
        if (classSymbol == null) {
            return false;
        }
        Set<JClassSymbol> supertypes = classSymbol.getSuperTypes();
        for (JClassSymbol supertype : supertypes) {
            if (supertype.getQualifiedName().equals("junit.framework.TestCase")) {
                return true;
            }
        }
        return false;
    }

    private boolean isOverridingMethod(ASTMethodDeclaration method) {
        JMethodSymbol methodSymbol = method.getSymbol();
        if (methodSymbol == null) {
            return false;
        }
        return methodSymbol.isOverriding();
    }
}

package net.sourceforge.pmd.lang.java.rule.design;


import java.util.List;
import java.util.Objects;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class SignatureDeclareThrowsExceptionRule extends AbstractJavaRulechainRule {

    public SignatureDeclareThrowsExceptionRule() {
        super(ASTMethodDeclaration.class, ASTConstructorDeclaration.class);
    }

    @Override
    public Object visit(ASTMethodDeclaration node, Object data) {
        JMethodSymbol methodSymbol = node.getSymbol();
        if (!methodSymbol.isAbstract() && !methodSymbol.isDefault() && !methodSymbol.isPrivate()) {
            if (node.getName().startsWith("test")) {
                return super.visit(node, data);
            }
            if (isJUnit3SetupOrTearDown(node) || isJUnit4SetupOrTearDown(node) || isTestNGSetupOrTearDown(node)) {
                return super.visit(node, data);
            }
            if (isOverridingMethod(node)) {
                return super.visit(node, data);
            }
            checkThrowsDeclaration(node.getFormalExceptions(), node, data);
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTConstructorDeclaration node, Object data) {
        if (isJUnit3SetupOrTearDown(node) || isJUnit4SetupOrTearDown(node) || isTestNGSetupOrTearDown(node)) {
            return super.visit(node, data);
        }
        checkThrowsDeclaration(node.getFormalExceptions(), node, data);
        return super.visit(node, data);
    }

    private void checkThrowsDeclaration(List<ASTType> exceptions, JavaNode node, Object data) {
        for (ASTType exceptionType : exceptions) {
            JClassType type = exceptionType.getType();
            if (Objects.equals(type.getQualifiedName(), "java.lang.Exception")) {
                addViolation(data, node);
            }
        }
    }

    private boolean isJUnit3SetupOrTearDown(ASTMethodDeclaration node) {
        return node.getName().equals("setUp") || node.getName().equals("tearDown");
    }

    private boolean isJUnit4SetupOrTearDown(ASTMethodDeclaration node) {
        return node.hasSingleAnnotation("org.junit.Before") || node.hasSingleAnnotation("org.junit.After");
    }

    private boolean isTestNGSetupOrTearDown(ASTMethodDeclaration node) {
        return node.hasSingleAnnotation("org.testng.annotations.BeforeClass") || node.hasSingleAnnotation("org.testng.annotations.AfterClass");
    }

    private boolean isOverridingMethod(ASTMethodDeclaration node) {
        JMethodSymbol methodSymbol = node.getSymbol();
        return methodSymbol.isOverriding() && methodSymbol.getOverriddenMethods().stream().anyMatch(sym -> sym.getFormalExceptions().stream().anyMatch(ex -> ex.getQualifiedName().equals("java.lang.Exception")));
    }

    private boolean isJUnit3SetupOrTearDown(ASTConstructorDeclaration node) {
        return false; // Constructors cannot be setUp or tearDown in JUnit3
    }

    private boolean isJUnit4SetupOrTearDown(ASTConstructorDeclaration node) {
        return false; // Constructors cannot be setUp or tearDown in JUnit4
    }

    private boolean isTestNGSetupOrTearDown(ASTConstructorDeclaration node) {
        return false; // Constructors cannot be setUp or tearDown in TestNG
    }

}
