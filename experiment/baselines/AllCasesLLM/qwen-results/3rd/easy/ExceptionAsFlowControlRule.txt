Rule description: This rule reports exceptions thrown and caught in an enclosing try statement. package net.sourceforge.pmd.lang.java.rule.design;


import java.util.List;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ExceptionAsFlowControlRule extends AbstractJavaRulechainRule {
    public ExceptionAsFlowControlRule() {
        super(ASTTryStatement.class, ASTCatchStatement.class);
    }

    @Override
    public Object visit(ASTTryStatement node, Object data) {
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTCatchStatement node, Object data) {
        ASTTryStatement tryStatement = node.getFirstParentOfType(ASTTryStatement.class);
        if (tryStatement == null) {
            return data;
        }

        List<ASTThrowStatement> throwStatements = node.findDescendantsOfType(ASTThrowStatement.class);
        for (ASTThrowStatement throwStatement : throwStatements) {
            JType caughtType = node.getCatchParameter().getType();
            JType thrownType = throwStatement.getExpression().getType();
            if (thrownType != null && caughtType != null && isSubtype(thrownType, caughtType)) {
                addViolation(data, throwStatement);
            }
        }

        return data;
    }

    private boolean isSubtype(JType thrownType, JType caughtType) {
        if (thrownType instanceof JClassType && caughtType instanceof JClassType) {
            JClassType thrownClassType = (JClassType) thrownType;
            JClassType caughtClassType = (JClassType) caughtType;
            return thrownClassType.isSubtypeOf(caughtClassType);
        }
        return false;
    }
}

package net.sourceforge.pmd.lang.java.rule.design;


import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ExceptionAsFlowControlRule extends AbstractJavaRulechainRule {
    public ExceptionAsFlowControlRule() {
        super(ASTTryStatement.class, ASTCatchStatement.class);
    }

    @Override
    public Object visit(ASTTryStatement node, Object data) {
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTCatchStatement node, Object data) {
        ASTThrowStatement throwStmt = node.getFirstDescendantOfType(ASTThrowStatement.class);
        if (throwStmt != null) {
            AType catchType = node.getParameter().getType();
            AType throwType = throwStmt.getExpression().getType();
            if (catchType != null && throwType != null && isSubtype(throwType, catchType)) {
                addViolation(data, throwStmt);
            }
        }
        return super.visit(node, data);
    }

    private boolean isSubtype(AType throwType, AType catchType) {
        TypeSystem typeSystem = throwType.getTypeSystem();
        if (throwType instanceof JTypeVar) {
            throwType = ((JTypeVar) throwType).getErasure(typeSystem);
        }
        if (catchType instanceof JTypeVar) {
            catchType = ((JTypeVar) catchType).getErasure(typeSystem);
        }
        return typeSystem.isSubtype(throwType, catchType);
    }
}

package net.sourceforge.pmd.lang.java.rule.design;


import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ExceptionAsFlowControlRule extends AbstractJavaRulechainRule {
    public ExceptionAsFlowControlRule() {
        super(ASTTryStatement.class, ASTCatchStatement.class);
    }

    @Override
    public Object visit(ASTTryStatement node, Object data) {
        NodeStream<ASTCatchStatement> catchStatements = node.getCatchStatements();
        TypeSystem typeSystem = node.getTypeSystem();
        JClassType exceptionType = typeSystem.forName(Exception.class.getName());

        catchStatements.forEach(catchStatement -> {
            JClassType caughtType = catchStatement.getFirstChildOfType(ASTClassOrInterfaceType.class).getType();
            if (exceptionType.isAssignableFrom(caughtType)) {
                NodeStream<ASTThrowStatement> throwStatements = catchStatement.findDescendantsOfType(ASTThrowStatement.class);
                throwStatements.forEach(throwStatement -> {
                    JClassType thrownType = throwStatement.getExpression().getType();
                    if (thrownType != null && exceptionType.isAssignableFrom(thrownType) && !caughtType.isAssignableFrom(thrownType)) {
                        addViolation(data, throwStatement);
                    }
                });
            }
        });

        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTCatchStatement node, Object data) {
        return super.visit(node, data);
    }
}

package net.sourceforge.pmd.lang.java.rule.design;


import java.util.List;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ExceptionAsFlowControlRule extends AbstractJavaRulechainRule {
    public ExceptionAsFlowControlRule() {
        super(ASTTryStatement.class, ASTCatchStatement.class);
    }

    @Override
    public Object visit(ASTTryStatement node, Object data) {
        List<ASTCatchStatement> catchStatements = node.getCatchBlocks();
        for (ASTCatchStatement catchStatement : catchStatements) {
            JClassSymbol exceptionType = catchStatement.getFirstException().getType().getSymbol();
            ASTTryStatement parentTry = node.getFirstParentOfType(ASTTryStatement.class);
            if (parentTry != null && parentTry.getCatchBlocks().stream().anyMatch(cs -> cs.getFirstException().getType().getSymbol().equals(exceptionType))) {
                addViolation(data, node);
            }
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTCatchStatement node, Object data) {
        JClassSymbol exceptionType = node.getFirstException().getType().getSymbol();
        ASTTryStatement parentTry = node.getFirstParentOfType(ASTTryStatement.class);
        if (parentTry != null) {
            for (ASTThrowStatement throwStatement : parentTry.findDescendantsOfType(ASTThrowStatement.class)) {
                JClassType thrownExceptionType = throwStatement.getExpression().getType();
                if (thrownExceptionType != null && JavaTypeSystem.isA(thrownExceptionType, exceptionType)) {
                    addViolation(data, node);
                    break;
                }
            }
        }
        return super.visit(node, data);
    }
}

package net.sourceforge.pmd.lang.java.rule.design;


import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ExceptionAsFlowControlRule extends AbstractJavaRulechainRule {
    public ExceptionAsFlowControlRule() {
        super(ASTTryStatement.class);
    }

    @Override
    public Object visit(ASTTryStatement node, Object data) {
        ASTBlock tryBlock = node.getFirstChildOfType(ASTBlock.class);
        if (tryBlock != null) {
            List<ASTThrowStatement> throwStatements = tryBlock.findDescendantsOfType(ASTThrowStatement.class);
            for (ASTThrowStatement throwStatement : throwStatements) {
                ASTTryStatement outerTry = node.getFirstParentOfType(ASTTryStatement.class);
                if (outerTry != null) {
                    List<ASTCatchStatement> catchStatements = outerTry.findDescendantsOfType(ASTCatchStatement.class);
                    for (ASTCatchStatement catchStatement : catchStatements) {
                        ASTType catchType = catchStatement.getFirstChildOfType(ASTType.class);
                        ASTName throwExceptionName = throwStatement.getFirstChildOfType(ASTName.class);
                        if (throwExceptionName != null && catchType != null) {
                            TypeSystem typeSystem = node.getTypeSystem();
                            JClassSymbol catchSymbol = typeSystem.resolveType(catchType.getType()).getSymbol();
                            JClassSymbol throwSymbol = typeSystem.resolveType(throwExceptionName.getType()).getSymbol();
                            if (TypeTestUtil.isA(throwSymbol, catchSymbol)) {
                                if (!throwSymbol.equals(catchSymbol)) {
                                    addViolation(data, throwStatement);
                                }
                            }
                        }
                    }
                }
            }
        }
        return super.visit(node, data);
    }
}
