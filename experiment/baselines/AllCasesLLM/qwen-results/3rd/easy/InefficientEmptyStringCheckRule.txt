package net.sourceforge.pmd.lang.java.rule.performance;


import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class InefficientEmptyStringCheckRule extends AbstractJavaRulechainRule {
    public InefficientEmptyStringCheckRule() {
        super(ASTMethodCall.class, ASTEqualityExpression.class);
    }

    @Override
    public Object visit(ASTMethodCall node, Object data) {
        if (isTrimAndLengthOrIsEmpty(node)) {
            addViolation(data, node);
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTEqualityExpression node, Object data) {
        if (isTrimAndLengthComparison(node)) {
            addViolation(data, node);
        }
        return super.visit(node, data);
    }

    private boolean isTrimAndLengthOrIsEmpty(ASTMethodCall node) {
        ASTPrimaryExpression primaryExpression = node.getFirstParentOfType(ASTPrimaryExpression.class);
        if (primaryExpression != null) {
            ASTPrimaryPrefix prefix = primaryExpression.getFirstChildOfType(ASTPrimaryPrefix.class);
            if (prefix != null) {
                ASTPrimarySuffix suffix = prefix.getFirstChildOfType(ASTPrimarySuffix.class);
                if (suffix != null && suffix.getImage().equals("trim()")) {
                    NodeStream<ASTPrimarySuffix> suffixes = node.children(ASTPrimarySuffix.class);
                    return suffixes.anyMatch(s -> s.getImage().equals("isEmpty()") || s.getImage().equals("length()"));
                }
            }
        }
        return false;
    }

    private boolean isTrimAndLengthComparison(ASTEqualityExpression node) {
        ASTExpression leftOperand = node.getLeftOperand();
        ASTExpression rightOperand = node.getRightOperand();

        if (rightOperand instanceof ASTPrimaryExpression) {
            ASTPrimaryExpression primaryExpression = (ASTPrimaryExpression) rightOperand;
            ASTPrimaryPrefix prefix = primaryExpression.getFirstChildOfType(ASTPrimaryPrefix.class);
            if (prefix != null) {
                ASTPrimarySuffix suffix = prefix.getFirstChildOfType(ASTPrimarySuffix.class);
                if (suffix != null && suffix.getImage().equals("length()")) {
                    NodeStream<ASTPrimarySuffix> suffixes = leftOperand.children(ASTPrimarySuffix.class);
                    return suffixes.anyMatch(s -> s.getImage().equals("trim()"));
                }
            }
        }

        if (leftOperand instanceof ASTPrimaryExpression) {
            ASTPrimaryExpression primaryExpression = (ASTPrimaryExpression) leftOperand;
            ASTPrimaryPrefix prefix = primaryExpression.getFirstChildOfType(ASTPrimaryPrefix.class);
            if (prefix != null) {
                ASTPrimarySuffix suffix = prefix.getFirstChildOfType(ASTPrimarySuffix.class);
                if (suffix != null && suffix.getImage().equals("length()")) {
                    NodeStream<ASTPrimarySuffix> suffixes = rightOperand.children(ASTPrimarySuffix.class);
                    return suffixes.anyMatch(s -> s.getImage().equals("trim()"));
                }
            }
        }

        return false;
    }
}

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class InefficientEmptyStringCheckRule extends AbstractJavaRulechainRule {
    public InefficientEmptyStringCheckRule() {
        super(ASTIfStatement.class, ASTWhileStatement.class);
    }

    @Override
    public Object visit(ASTIfStatement node, Object data) {
        checkExpression(node.getCondition(), data);
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTWhileStatement node, Object data) {
        checkExpression(node.getCondition(), data);
        return super.visit(node, data);
    }

    private void checkExpression(ASTExpression expression, Object data) {
        if (expression instanceof ASTMethodCall) {
            ASTMethodCall methodCall = (ASTMethodCall) expression;
            if (isTrimLengthCall(methodCall) || isTrimIsEmptyCall(methodCall)) {
                addViolation(data, methodCall);
            }
        }
    }

    private boolean isTrimLengthCall(ASTMethodCall methodCall) {
        List<ASTExpression> arguments = methodCall.getArguments().getArguments();
        if (arguments.size() == 1 && arguments.get(0) instanceof ASTMethodCall) {
            ASTMethodCall innerMethodCall = (ASTMethodCall) arguments.get(0);
            return isTrimCall(innerMethodCall) && methodCall.getName().equals("length");
        }
        return false;
    }

    private boolean isTrimIsEmptyCall(ASTMethodCall methodCall) {
        return isTrimCall(methodCall) && methodCall.getName().equals("isEmpty");
    }

    private boolean isTrimCall(ASTMethodCall methodCall) {
        return methodCall.getName().equals("trim") && methodCall.getType() == TypeSystem.STRING;
    }

public class InefficientEmptyStringCheckRule extends AbstractJavaRulechainRule {
    public InefficientEmptyStringCheckRule() {
        super(ASTExpression.class, ASTMethodCall.class);
    }

    @Override
    public Object visit(ASTExpression node, Object data) {
        if (node instanceof BinaryOperation) {
            BinaryOperation bo = (BinaryOperation) node;
            if (bo.getOperator() == BinaryOperation.Operator.EQUALS) {
                List<ASTExpression> operands = bo.getChildrenOfType(ASTExpression.class);
                if (operands.size() == 2) {
                    ASTExpression left = operands.get(0);
                    ASTExpression right = operands.get(1);
                    if (isTrimLengthCall(left) && isZeroLiteral(right)) {
                        addViolation(data, left);
                    } else if (isTrimLengthCall(right) && isZeroLiteral(left)) {
                        addViolation(data, right);
                    }
                }
            }
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTMethodCall node, Object data) {
        JMethodSymbol methodSymbol = node.getSymbol();
        if (methodSymbol != null && methodSymbol.getName().equals("isEmpty")) {
            ASTPrimaryExpression primaryExpression = node.getFirstChildOfType(ASTPrimaryExpression.class);
            if (primaryExpression != null) {
                List<ASTPrimarySuffix> suffixes = primaryExpression.getChildrenOfType(ASTPrimarySuffix.class);
                if (!suffixes.isEmpty() && suffixes.get(suffixes.size() - 1).getImage().equals("trim")) {
                    addViolation(data, node);
                }

        return super.visit(node, data);
    }
