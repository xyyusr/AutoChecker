package net.sourceforge.pmd.lang.java.rule.design;

import java.util.HashMap;
import java.util.Map;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class AvoidThrowingNullPointerExceptionRule extends AbstractJavaRulechainRule {
    private Map<String, Type> variableTypes;

    public AvoidThrowingNullPointerExceptionRule() {
        super(ASTMethodDeclaration.class, ASTThrowStatement.class, ASTVariableDeclarator.class, ASTAssignmentExpression.class);
    }

    @Override
    public Object visit(ASTMethodDeclaration node, Object data) {
        variableTypes = new HashMap<>();
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTThrowStatement node, Object data) {
        ASTExpression expression = node.getExpression();
        if (expression instanceof ASTVariableAccess) {
            String variableName = ((ASTVariableAccess) expression).getName();
            Type variableType = variableTypes.get(variableName);
            if (variableType != null && variableType.isA(TypeSystem.getInstance().NULL_POINTER_EXCEPTION)) {
                addViolation(data, node);
            }
        } else if (expression instanceof ASTClassOrInterfaceType) {
            String className = ((ASTClassOrInterfaceType) expression).getImage();
            if ("NullPointerException".equals(className)) {
                addViolation(data, node);
            }
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTVariableDeclarator node, Object data) {
        ASTVariableDeclaratorId id = node.getVariableId();
        ASTVariableInitializer initializer = node.getInitializer();
        if (initializer != null && initializer instanceof ASTAllocationExpression) {
            ASTClassOrInterfaceType type = ((ASTAllocationExpression) initializer).getType();
            if (type != null && "NullPointerException".equals(type.getImage())) {
                variableTypes.put(id.getImage(), type.getType());
            }
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTAssignmentExpression node, Object data) {
        ASTExpression leftExpression = node.getLeftOperand();
        ASTExpression rightExpression = node.getRightOperand();
        if (leftExpression instanceof ASTVariableAccess && rightExpression instanceof ASTAllocationExpression) {
            ASTClassOrInterfaceType type = ((ASTAllocationExpression) rightExpression).getType();
            if (type != null && "NullPointerException".equals(type.getImage())) {
                String variableName = ((ASTVariableAccess) leftExpression).getName();
                variableTypes.put(variableName, type.getType());
            }
        }
        return super.visit(node, data);
    }
}

import java.util.HashMap;
import java.util.Map;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class AvoidThrowingNullPointerExceptionRule extends AbstractJavaRulechainRule {
    public AvoidThrowingNullPointerExceptionRule() {
        super(ASTClassOrInterfaceDeclaration.class, ASTMethodDeclaration.class, ASTThrowStatement.class, ASTVariableInitializer.class);
    }

    private Map<JVariableSymbol, Type> variableTypes = new HashMap<>();

    @Override
    public Object visit(ASTClassOrInterfaceDeclaration node, Object data) {
        variableTypes.clear();
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTMethodDeclaration node, Object data) {
        variableTypes.clear();
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTVariableInitializer node, Object data) {
        ASTExpression initializer = node.getFirstChildOfType(ASTExpression.class);
        if (initializer != null) {
            ASTConstructorCall constructorCall = initializer.getFirstChildOfType(ASTConstructorCall.class);
            if (constructorCall != null) {
                JConstructorSymbol constructorSymbol = constructorCall.getConstructorSymbol();
                if (constructorSymbol != null) {
                    JClassSymbol classSymbol = constructorSymbol.getDeclaringClassSymbol();
                    if ("java.lang.NullPointerException".equals(classSymbol.getQualifiedName())) {
                        JVariableSymbol variableSymbol = node.getPreviousSiblingOfType(ASTVariableDeclaratorId.class).getVariableSymbol();
                        variableTypes.put(variableSymbol, classSymbol.getType());
                    }
                }
            }
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTThrowStatement node, Object data) {
        ASTExpression expression = node.getFirstChildOfType(ASTExpression.class);
        if (expression != null) {
            Type type = expression.getType();
            if (type instanceof JClassType) {
                JClassType classType = (JClassType) type;
                if ("java.lang.NullPointerException".equals(classType.getName())) {
                    addViolation(node, data);
                } else if (expression instanceof ASTVariableDeclaratorId) {
                    JVariableSymbol variableSymbol = ((ASTVariableDeclaratorId) expression).getVariableSymbol();
                    if (variableTypes.containsKey(variableSymbol) && "java.lang.NullPointerException".equals(variableTypes.get(variableSymbol).getName())) {
                        addViolation(node, data);
                    }
                }
            }
        }
        return super.visit(node, data);
    }
}

    public AvoidThrowingNullPointerExceptionRule() {
        super(ASTClassOrInterfaceDeclaration.class, ASTMethodDeclaration.class, ASTThrowStatement.class, ASTVariableDeclaratorId.class, ASTConstructorCall.class);

    @Override
    public Object visit(ASTClassOrInterfaceDeclaration node, Object data) {
        npeVariables.clear();

    }

    @Override
    public Object visit(ASTMethodDeclaration node, Object data) {
        npeVariables.clear();
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTConstructorCall node, Object data) {
        if (node.getType() instanceof JClassType) {

            if ("java.lang.NullPointerException".equals(symbol.getQualifiedName())) {
                npeVariables.add(node.getPreviousSibling().getFirstChildOfType(ASTVariableDeclaratorId.class).getImage());
            }
        }
        return super.visit(node, data);
    }
