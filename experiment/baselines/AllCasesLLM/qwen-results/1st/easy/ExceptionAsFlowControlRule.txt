The following is the description of the rule:
Rule description: This rule reports exceptions thrown and caught in an enclosing try statement. This use of exceptions as a form of goto statement is discouraged, as that may hide actual exceptions, and obscures control flow, especially when debugging. To fix a violation, add the necessary validation or use an alternate control structure.;

```java
// failure case
public class Foo {
    void bar() {
        try {
            try {
            } catch (Exception e) {
                throw new WrapperException(e);
                // this is essentially a GOTO to the WrapperException catch block
            }
        } catch (WrapperException e) {
            // do some more stuff
        }
    }
}
class WrapperException extends Exception {}
```

package net.sourceforge.pmd.lang.java.rule.design;

import java.util.List;
import java.util.HashSet;
import java.util.Set;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ExceptionAsFlowControlRule extends AbstractJavaRulechainRule {
    public ExceptionAsFlowControlRule() {
        super(ASTTryStatement.class);
    }

    @Override
    public Object visit(ASTTryStatement node, Object data) {
        List<ASTCatchStatement> catches = node.getCatchBlocks();
        Set<JType> caughtExceptions = new HashSet<>();
        for (ASTCatchStatement catchStmt : catches) {
            JType caughtType = catchStmt.getFirstExceptionHandler().getParameterType();
            caughtExceptions.add(caughtType);
        }

        node.findDescendantsOfType(ASTThrowStatement.class).forEach(throwStmt -> {
            JType thrownType = throwStmt.getExpression().getType();
            for (JType caughtType : caughtExceptions) {
                if (caughtType.isSubtypeOf(thrownType) && !thrownType.equals(caughtType)) {
                    addViolation(data, throwStmt);
                }
            }
        });

        return super.visit(node, data);
    }
}

package net.sourceforge.pmd.lang.java.rule.design;

import java.util.List;
import java.util.stream.Collectors;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ExceptionAsFlowControlRule extends AbstractJavaRulechainRule {
    public ExceptionAsFlowControlRule() {
        super(ASTTryStatement.class, ASTCatchStatement.class);
    }

    @Override
    public Object visit(ASTTryStatement node, Object data) {
        List<ASTCatchStatement> catchStatements = node.findDescendantsOfType(ASTCatchStatement.class);
        List<ASTThrowStatement> throwStatements = node.findDescendantsOfType(ASTThrowStatement.class);

        TypeSystem typeSystem = node.getTypeSystem();
        List<Type> caughtExceptions = catchStatements.stream()
                .map(ASTCatchStatement::getParameter)
                .map(ASTFormalParameter::getType)
                .map(typeSystem::resolve)
                .collect(Collectors.toList());

        for (ASTThrowStatement throwStatement : throwStatements) {
            Type thrownException = typeSystem.resolve(throwStatement.getExpression().getType());
            for (Type caughtException : caughtExceptions) {
                if (thrownException.isSubtypeOf(caughtException)) {
                    addViolation(data, throwStatement);
                    break;
                }
            }
        }

        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTCatchStatement node, Object data) {
        return super.visit(node, data);
    }
}

package net.sourceforge.pmd.lang.java.rule.design;
import java.util.List;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ExceptionAsFlowControlRule extends AbstractJavaRulechainRule {
    public ExceptionAsFlowControlRule() {
        super(ASTClassOrInterfaceDeclaration.class, ASTMethodDeclaration.class, ASTTryStatement.class, ASTCatchStatement.class, ASTThrowStatement.class);
    }
    @Override
    public Object visit(ASTClassOrInterfaceDeclaration node, Object data) {
        return super.visit(node, data);
    }
    @Override
    public Object visit(ASTMethodDeclaration node, Object data) {
        return super.visit(node, data);
    }
    @Override
    public Object visit(ASTTryStatement node, Object data) {
        return super.visit(node, data);
    }
    @Override
    public Object visit(ASTCatchStatement node, Object data) {
        ASTTryStatement tryStatement = node.getFirstParentOfType(ASTTryStatement.class);
        if (tryStatement == null) {
            return data;
        }
        Type caughtType = node.getCatchParameter().getType();
        List<ASTThrowStatement> throwStatements = node.findDescendantsOfType(ASTThrowStatement.class);
        for (ASTThrowStatement throwStatement : throwStatements) {
            Type thrownType = getThrownExceptionType(throwStatement);
            if (thrownType != null && isSubtypeOrEqual(thrownType, caughtType)) {
                addViolation(data, throwStatement);
            }
        }
        return data;
    }
    @Override
    public Object visit(ASTThrowStatement node, Object data) {
        return data;
    }
    private Type getThrownExceptionType(ASTThrowStatement throwStatement) {
        ASTExpression expression = throwStatement.getExpression();
        if (expression instanceof ASTConstructorCall) {
            ASTConstructorCall constructorCall = (ASTConstructorCall) expression;
            return constructorCall.getType();
        }
        return expression.getType();
    }
    private boolean isSubtypeOrEqual(Type thrownType, Type caughtType) {
        TypeSystem typeSystem = getSymbolFacade().getTypeSystem();

public class ExceptionAsFlowControlRule extends AbstractJavaRulechainRule {
    public ExceptionAsFlowControlRule() {
        super(ASTTryStatement.class, ASTCatchStatement.class);
    }

    @Override
    public Object visit(ASTTryStatement node, Object data) {
        List<ASTCatchStatement> catchStatements = node.getCatchStatements();
        for (ASTCatchStatement catchStatement : catchStatements) {
            if (containsThrowInCatch(catchStatement)) {
                addViolation(data, catchStatement);
            }
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTCatchStatement node, Object data) {
        return super.visit(node, data);
    }

    private boolean containsThrowInCatch(ASTCatchStatement catchStatement) {
        return catchStatement.findDescendantsOfType(ASTThrowStatement.class).stream()
                .anyMatch(throwStatement -> {
                    ASTCatchFormalParameter catchParam = catchStatement.getFirstChildOfType(ASTCatchFormalParameter.class);
                    if (catchParam == null) {
                        return false;
                    }
                    JType catchType = catchParam.getType();
                    ASTThrowStatement.ThrowExpression throwExpression = throwStatement.getThrowExpression();
                    JExpression throwExpr = throwExpression.getExpression();
                    JType throwType = throwExpr.getType();

                    return TypeSystem.isA(throwType, catchType) && !TypeTest.isSameType(catchType, throwType);
                });
    }

public class ExceptionAsFlowControlRule extends AbstractJavaRulechainRule {
    public ExceptionAsFlowControlRule() {
        super(ASTTryStatement.class, ASTCatchStatement.class);
    }

    private Set<JClassType> caughtExceptions = new HashSet<>();

    @Override
    public Object visit(ASTTryStatement node, Object data) {
        caughtExceptions.clear();
        for (ASTCatchStatement catchStmt : node.getCatchBlocks()) {
            caughtExceptions.add(catchStmt.getExceptionType().getType());
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTCatchStatement node, Object data) {
        JClassType caughtType = node.getExceptionType().getType();
        for (ASTThrowStatement throwStmt : node.getBody().findDescendantsOfType(ASTThrowStatement.class)) {
            JClassType thrownType = throwStmt.getExpression().getType();
            if (thrownType != null && caughtExceptions.contains(thrownType)) {
                addViolation(data, throwStmt);
            }
        }
        return super.visit(node, data);
    }
