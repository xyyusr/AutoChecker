package net.sourceforge.pmd.lang.java.rule.performance;


import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class InefficientEmptyStringCheckRule extends AbstractJavaRulechainRule {
    public InefficientEmptyStringCheckRule() {
        super(ASTEqualityExpression.class, ASTMethodCall.class);
    }

    @Override
    public Object visit(ASTEqualityExpression node, Object data) {
        if (node.getLeftOperand() instanceof ASTPrimaryExpression && node.getRightOperand() instanceof ASTPrimaryExpression) {
            ASTPrimaryExpression left = (ASTPrimaryExpression) node.getLeftOperand();
            ASTPrimaryExpression right = (ASTPrimaryExpression) node.getRightOperand();
            if (left.getPrimaryPrefix().getFirstChildOfType(ASTMethodCall.class) != null && right.getPrimaryPrefix().getFirstChildOfType(ASTPrimaryExpression.class) != null) {
                ASTMethodCall methodCall = left.getPrimaryPrefix().getFirstChildOfType(ASTMethodCall.class);
                if ("trim".equals(methodCall.getName()) && "length".equals(methodCall.getFirstChildOfType(ASTPrimarySuffix.class).getImage()) && node.getOperator().equals(ASTEqualityExpression.EQUAL)) {
                    ASTPrimaryExpression rightPrefix = (ASTPrimaryExpression) right.getPrimaryPrefix().getFirstChildOfType(ASTPrimaryExpression.class);
                    if (rightPrefix.getPrimaryPrefix().getFirstChildOfType(ASTPrimarySuffix.class) != null && rightPrefix.getPrimaryPrefix().getFirstChildOfType(ASTPrimarySuffix.class).getImage().equals("")) {
                        addViolation(data, node);
                    }
                }
            }
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTMethodCall node, Object data) {
        if ("trim".equals(node.getName()) && node.getFirstChildOfType(ASTPrimarySuffix.class).getImage().equals("isEmpty")) {
            addViolation(data, node);
        }
        return super.visit(node, data);
    }
}

public class InefficientEmptyStringCheckRule extends AbstractJavaRulechainRule {

    public InefficientEmptyStringCheckRule() {
        super(ASTIfStatement.class, ASTExpression.class);
    }

    @Override
    public Object visit(ASTIfStatement node, Object data) {
        NodeStream<ASTExpression> expressions = node.getCondition().findDescendantsOfType(ASTExpression.class);
        expressions.forEach(expr -> checkExpression(expr));
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTExpression node, Object data) {
        checkExpression(node);
        return super.visit(node, data);
    }

    private void checkExpression(ASTExpression expr) {
        NodeStream<ASTMethodCall> methodCalls = expr.findDescendantsOfType(ASTMethodCall.class);
        methodCalls.forEach(this::checkMethodCall);
    }

    private void checkMethodCall(ASTMethodCall methodCall) {
        Type returnType = methodCall.getType();
        if (returnType != null && returnType.equals(JPrimitiveType.INT)) {
            return;
        }

        ASTPrimaryExpression primaryExpression = methodCall.getFirstChildOfType(ASTPrimaryExpression.class);
        if (primaryExpression == null) {
            return;
        }

        ASTPrimaryPrefix primaryPrefix = primaryExpression.getFirstChildOfType(ASTPrimaryPrefix.class);
        if (primaryPrefix == null) {
            return;
        }

        ASTPrimarySuffix primarySuffix = primaryPrefix.getFirstChildOfType(ASTPrimarySuffix.class);
        if (primarySuffix == null) {
            return;
        }

        if ("trim".equals(primarySuffix.getImage())) {
            ASTPrimarySuffix nextSuffix = methodCall.getNextSiblingOfType(ASTPrimarySuffix.class);
            if (nextSuffix != null) {
                String nextImage = nextSuffix.getImage();
                if ("length".equals(nextImage)) {
                    ASTPrimarySuffix subsequentSuffix = nextSuffix.getNextSiblingOfType(ASTPrimarySuffix.class);
                    if (subsequentSuffix != null && "=".equals(subsequentSuffix.getImage())) {
                        ASTExpression subsequentExpression = methodCall.getNextSiblingOfType(ASTExpression.class);
                        if (subsequentExpression != null && "0".equals(subsequentExpression.getImage())) {
                            addViolation(methodCall);
                        }
                    }
                } else if ("isEmpty".equals(nextImage)) {
                    addViolation(methodCall);
                }
            }
        }
    }
}
