package net.sourceforge.pmd.lang.java.rule.design;


import java.util.List;
import java.util.Optional;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class SignatureDeclareThrowsExceptionRule extends AbstractJavaRulechainRule {

    public SignatureDeclareThrowsExceptionRule() {
        super(ASTMethodDeclaration.class, ASTConstructorDeclaration.class);
    }

    @Override
    public Object visit(ASTMethodDeclaration node, Object data) {
        if (shouldCheckMethod(node)) {
            checkThrowsClause(node.getThrowsClause(), node, data);
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTConstructorDeclaration node, Object data) {
        if (!(node.getQualifiedName().endsWith("setUp") || node.getQualifiedName().endsWith("tearDown"))) {
            checkThrowsClause(node.getThrowsClause(), node, data);
        }
        return super.visit(node, data);
    }

    private void checkThrowsClause(ASTThrowsClause throwsClause, JavaNode node, Object data) {
        if (throwsClause != null) {
            for (ASTType type : throwsClause.findDescendantsOfType(ASTType.class)) {
                Optional<JClassSymbol> symbol = type.getType().getSymbol();
                if (symbol.isPresent() && symbol.get().getQualifiedName().equals("java.lang.Exception")) {
                    addViolation(data, node);
                    break;
                }
            }
        }
    }

    private boolean shouldCheckMethod(ASTMethodDeclaration method) {
        String methodName = method.getName();
        if (methodName.startsWith("test")) {
            return false;
        }
        JClassSymbol classSymbol = method.getEnclosingClass();
        if (classSymbol != null) {
            return !isJUnitSetupOrTeardownMethod(method, classSymbol) && !isOverridingMethod(method);
        }
        return true;
    }

    private boolean isJUnitSetupOrTeardownMethod(ASTMethodDeclaration method, JClassSymbol classSymbol) {
        String methodName = method.getName();
        if (methodName.equals("setUp") || methodName.equals("tearDown")) {
            return isSubclassOf(classSymbol, "junit.framework.TestCase") ||
                   isSubclassOf(classSymbol, "net.sourceforge.pmd.lang.java.rule.design.signaturedeclarethrowsexception.MyTestCase");
        }
        return false;
    }

    private boolean isSubclassOf(JClassSymbol classSymbol, String superclass) {
        JClassSymbol current = classSymbol;
        while (current != null) {
            if (current.getQualifiedName().equals(superclass)) {
                return true;
            }
            current = current.getSuperClassSymbol();
        }
        return false;
    }

    private boolean isOverridingMethod(ASTMethodDeclaration method) {
        JMethodSymbol methodSymbol = method.getSymbol();
        if (methodSymbol != null) {
            return methodSymbol.getOverriddenSymbol().isPresent();
        }
        return false;
    }
}

package net.sourceforge.pmd.lang.java.rule.design;


import java.util.List;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class SignatureDeclareThrowsExceptionRule extends AbstractJavaRulechainRule {

    public SignatureDeclareThrowsExceptionRule() {
        super(ASTMethodDeclaration.class, ASTConstructorDeclaration.class);
    }

    @Override
    public Object visit(ASTMethodDeclaration node, Object data) {
        if (!isJUnit4SetUp(node) && !isJUnit3SetUpOrTearDown(node) && !isJUnit3Test(node) && !isOverridingMethod(node)) {
            checkThrowsClause(node.getThrowsClause().getReferences(), node);
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTConstructorDeclaration node, Object data) {
        if (!isJUnit3SetUpOrTearDown(node)) {
            checkThrowsClause(node.getThrowsClause().getReferences(), node);
        }
        return super.visit(node, data);
    }

    private void checkThrowsClause(List<ASTReferenceType> references, JavaNode node) {
        for (ASTReferenceType reference : references) {
            ASTType type = reference.getType();
            if (type instanceof ASTName && "Exception".equals(((ASTName) type).getImage())) {
                addViolation(data, node);
            }
        }
    }

    private boolean isJUnit4SetUp(ASTMethodDeclaration node) {
        NodeStream<ASTMarkerAnnotation> annotations = node.markerAnnotations();
        return annotations.anyMatch(a -> a.getType().getName().equals("Before"));
    }

    private boolean isJUnit3SetUpOrTearDown(ASTMethodDeclaration node) {
        String name = node.getName();
        return ("setUp".equals(name) || "tearDown".equals(name)) && isJUnit3TestCase(node);
    }

    private boolean isJUnit3Test(ASTMethodDeclaration node) {
        return node.getName().startsWith("test") && isJUnit3TestCase(node);
    }

    private boolean isJUnit3TestCase(ASTMethodDeclaration node) {
        ASTClassOrInterfaceDeclaration classNode = node.getFirstParentOfType(ASTClassOrInterfaceDeclaration.class);
        return classNode != null && isJUnit3TestCase(classNode);
    }

    private boolean isJUnit3TestCase(ASTClassOrInterfaceDeclaration node) {
        JClassSymbol classSymbol = node.getTypeDeclaration().getSymbol();
        return classSymbol != null && classSymbol.getSuperclass() != null && classSymbol.getSuperclass().getQualifiedName().equals("junit.framework.TestCase");
    }

    private boolean isOverridingMethod(ASTMethodDeclaration node) {
        JMethodSymbol methodSymbol = node.getSymbol();
        return methodSymbol != null && methodSymbol.isOverriding();
    }
}

    public SignatureDeclareThrowsExceptionRule() {
        super(ASTMethodDeclaration.class, ASTConstructorDeclaration.class);
    }

    @Override
    public Object visit(ASTMethodDeclaration node, Object data) {
        if (isJUnitMethod(node) || isTestNGMethod(node) || isOverridingMethod(node)) {
            return data;

        checkThrowsClause(node.getThrowsClause(), data);
        return data;

    @Override

        if (isJUnitConstructor(node) || isOverridingMethod(node)) {

        return data;

            for (ASTType type : throwsClause.getThrownExceptions()) {
                JType exceptionType = TypeHelper.facetOf(type, JType.class);
                if (exceptionType != null && "java.lang.Exception".equals(exceptionType.getQualifiedName())) {
                    addViolation(data, type);
                }
            }
        }

        return isJUnitSetupOrTeardown(node) || isTestMethod(node);

        String methodName = node.getName();
        return ("setUp".equals(methodName) || "tearDown".equals(methodName)) && isInJUnitTestCase(node);
    }

    private boolean isJUnitConstructor(ASTConstructorDeclaration node) {
        return isInJUnitTestCase(node);
    }

    private boolean isInJUnitTestCase(ASTMethodDeclaration node) {
        ASTClassOrInterfaceDeclaration classDecl = node.getFirstParentOfType(ASTClassOrInterfaceDeclaration.class);
        return isInJUnitTestCase(classDecl);
    }

    private boolean isInJUnitTestCase(ASTConstructorDeclaration node) {
        ASTClassOrInterfaceDeclaration classDecl = node.getFirstParentOfType(ASTClassOrInterfaceDeclaration.class);
        return isInJUnitTestCase(classDecl);
    }

    private boolean isInJUnitTestCase(ASTClassOrInterfaceDeclaration classDecl) {
        if (classDecl != null) {
            JClassSymbol classSymbol = classDecl.getSymbol();
            if (classSymbol != null) {
                return classSymbol.getSuperclassSymbol().map(superclass -> superclass.getQualifiedName().equals("junit.framework.TestCase") ||
                        superclass.getQualifiedName().equals("net.sourceforge.pmd.lang.java.rule.design.signaturedeclarethrowsexception.MyTestCase")).orElse(false);
            }
        }
        return false;
    }

    private boolean isTestMethod(ASTMethodDeclaration node) {
        String methodName = node.getName();
        return methodName.startsWith("test") && isInJUnitTestCase(node);
    }

    private boolean isOverridingMethod(JavaNode node) {
        if (node instanceof ASTMethodDeclaration method) {
            return method.getOverriddenMethod().isPresent();
        }
        return false;
    }

public class SignatureDeclareThrowsExceptionRule extends AbstractJavaRulechainRule {

    public SignatureDeclareThrowsExceptionRule() {
        super(ASTConstructorDeclaration.class, ASTMethodDeclaration.class);
    }

    @Override
    public Object visit(ASTConstructorDeclaration node, Object data) {
        JClassType declaringType = node.getType();
        JClassSymbol superClass = declaringType.getSuperClassSymbol();
        if (superClass != null && (superClass.getQualifiedName().equals("junit.framework.TestCase") || superClass.getQualifiedName().equals("net.sourceforge.pmd.lang.java.rule.design.signaturedeclarethrowsexception.MyTestCase"))) {
            return super.visit(node, data);
        }
        if (throwsGenericException(node)) {
            addViolation(data, node);
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTMethodDeclaration node, Object data) {
        JClassType declaringType = node.getType();
        JClassSymbol superClass = declaringType.getSuperClassSymbol();
        if (superClass != null && superClass.getQualifiedName().equals("junit.framework.TestCase")) {
            if (node.getName().toString().equals("setUp") || node.getName().toString().equals("tearDown")) {
                return super.visit(node, data);
            }
        }
        if (isJUnit4Setup(node) || isJUnit3Test(node) || isOverridingMethod(node)) {
            return super.visit(node, data);
        }
        if (throwsGenericException(node)) {
            addViolation(data, node);
        }
        return super.visit(node, data);
    }

    private boolean throwsGenericException(ASTConstructorDeclaration node) {
        for (ASTReferenceType type : node.findDescendantsOfType(ASTReferenceType.class)) {
            for (ASTName name : type.findDescendantsOfType(ASTName.class)) {
                if (name.getImage().equals("Exception")) {
                    return true;
                }
            }
        }
        return false;
    }

    private boolean throwsGenericException(ASTMethodDeclaration node) {
        for (ASTReferenceType type : node.findDescendantsOfType(ASTReferenceType.class)) {
            for (ASTName name : type.findDescendantsOfType(ASTName.class)) {
                if (name.getImage().equals("Exception")) {
                    return true;
                }
            }
        }
        return false;
    }

    private boolean isJUnit4Setup(ASTMethodDeclaration node) {
        List<ASTAnnotation> annotations = node.findDescendantsOfType(ASTAnnotation.class);
        for (ASTAnnotation annotation : annotations) {
            if (annotation.getType().getType().getQualifiedName().equals("org.junit.Before")) {
                return true;
            }
        }
        return false;
    }

    private boolean isJUnit3Test(ASTMethodDeclaration node) {
        if (node.getName().toString().startsWith("test")) {
            return true;
        }
        return false;
    }

    private boolean isOverridingMethod(ASTMethodDeclaration node) {
        JMethodSymbol symbol = node.getSymbol();
        if (symbol.isOverride()) {
            for (JMethodSymbol method : symbol.getOverriddenSymbols()) {
                if (method.getExceptionTypes().stream().anyMatch(t -> t.getQualifiedName().equals("java.lang.Exception"))) {
                    return true;
                }
            }
        }
        return false;
    }

package net.sourceforge.pmd.lang.java.rule.design;


import java.util.List;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class SignatureDeclareThrowsExceptionRule extends AbstractJavaRulechainRule {

    public SignatureDeclareThrowsExceptionRule() {
        super(ASTConstructorDeclaration.class, ASTMethodDeclaration.class);
    }

    @Override
    public Object visit(ASTConstructorDeclaration node, Object data) {
        checkThrowsClause(node, data);
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTMethodDeclaration node, Object data) {
        checkThrowsClause(node, data);
        return super.visit(node, data);
    }

    private void checkThrowsClause(ASTClassOrInterfaceBodyDeclaration node, Object data) {
        ASTThrowsClause throwsClause = node.getFirstChildOfType(ASTThrowsClause.class);
        if (throwsClause != null) {
            for (ASTName exceptionType : throwsClause.findDescendantsOfType(ASTName.class)) {
                if ("Exception".equals(exceptionType.getImage()) && !isJUnitSetUpOrTearDown(node) && !isTestNGAnnotation(node) && !isOverridingMethod(node)) {
                    addViolation(data, node);
                }
            }
        }
    }

    private boolean isJUnitSetUpOrTearDown(ASTClassOrInterfaceBodyDeclaration node) {
        ASTClassOrInterfaceDeclaration classDecl = node.getFirstParentOfType(ASTClassOrInterfaceDeclaration.class);
        if (classDecl == null) {
            return false;
        }
        JClassSymbol classSymbol = classDecl.getSymbol();
        if (classSymbol == null) {
            return false;
        }
        Scope scope = classDecl.getScope();
        JClassSymbol testCaseSymbol = scope.getSymbolDeclarations("junit.framework.TestCase").stream()
                .filter(JClassSymbol.class::isInstance)
                .map(JClassSymbol.class::cast)
                .findFirst()
                .orElse(null);

        if (testCaseSymbol != null && classSymbol.inheritsFrom(testCaseSymbol)) {
            String methodName = node.getName();
            return "setUp".equals(methodName) || "tearDown".equals(methodName);
        }

        return false;
    }

    private boolean isTestNGAnnotation(ASTClassOrInterfaceBodyDeclaration node) {
        return node.getFirstDescendantOfType(ASTAnnotation.class).map(ASTAnnotation::getTypeName).orElse("").contains("org.testng.annotations");
    }

    private boolean isOverridingMethod(ASTClassOrInterfaceBodyDeclaration node) {
        if (node instanceof ASTMethodDeclaration methodDecl) {
            JMethodSymbol methodSymbol = methodDecl.getSymbol();
            if (methodSymbol != null && methodSymbol.isOverride()) {
                return true;
            }
        }
        return false;
    }
}
