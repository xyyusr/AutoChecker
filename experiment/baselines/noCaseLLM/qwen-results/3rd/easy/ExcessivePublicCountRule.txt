public class ExcessivePublicCountRule extends AbstractJavaRulechainRule {
    public ExcessivePublicCountRule() {
        super(ASTClassOrInterfaceDeclaration.class);
    }

    @Override
    public Object visit(ASTClassOrInterfaceDeclaration node, Object data) {
        if (node.isInterface()) {
            return super.visit(node, data);
        }

        List<ASTFieldDeclaration> publicFields = node.findDescendantsOfType(ASTFieldDeclaration.class)
                .stream()

                .toList();

        List<ASTMethodDeclaration> publicMethods = node.findDescendantsOfType(ASTMethodDeclaration.class)
                .stream()
                .filter(ASTMethodDeclaration::isPublic)
                .toList();

        int publicCount = publicFields.size() + publicMethods.size();
        if (publicCount > 2) {
            addViolation(data, node);
        }

        return super.visit(node, data);
    }

package net.sourceforge.pmd.lang.java.rule.design;

import java.util.List;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ExcessivePublicCountRule extends AbstractJavaRulechainRule {
    public ExcessivePublicCountRule() {
        super(ASTClassOrInterfaceDeclaration.class);
    }

    @Override
    public Object visit(ASTClassOrInterfaceDeclaration node, Object data) {
        int publicMethodCount = 0;
        int publicFieldCount = 0;

        List<ASTMethodDeclaration> methods = node.findDescendantsOfType(ASTMethodDeclaration.class);
        for (ASTMethodDeclaration method : methods) {
            if (method.isPublic()) {
                publicMethodCount++;
            }
        }

        List<ASTFieldDeclaration> fields = node.findDescendantsOfType(ASTFieldDeclaration.class);
        for (ASTFieldDeclaration field : fields) {
            if (field.isPublic()) {
                publicFieldCount++;
            }
        }

        int publicCount = publicMethodCount + publicFieldCount;
        if (publicCount > 2) {
            addViolationWithMessage(data, node, "Class " + node.getName() + " has excessive number of public methods and fields: " + publicCount);
        }

        return super.visit(node, data);
    }
}

package net.sourceforge.pmd.lang.java.rule.design;
import java.util.List;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ExcessivePublicCountRule extends AbstractJavaRulechainRule {
    public ExcessivePublicCountRule() {
        super(ASTClassOrInterfaceDeclaration.class);
    }
    @Override
    public Object visit(ASTClassOrInterfaceDeclaration node, Object data) {
        int publicMethodCount = countPublicMethods(node);
        int publicFieldCount = countPublicFields(node);
        if (publicMethodCount + publicFieldCount > 2) {
            addViolation(data, node);
        }
        return super.visit(node, data);
    }
    private int countPublicMethods(ASTClassOrInterfaceDeclaration node) {
        List<ASTMethodDeclaration> methods = node.findDescendantsOfType(ASTMethodDeclaration.class);
        int count = 0;
        for (ASTMethodDeclaration method : methods) {
            if (method.isPublic()) {
                count++;
            }
        }
        return count;
    }
    private int countPublicFields(ASTClassOrInterfaceDeclaration node) {
        List<ASTFieldDeclaration> fields = node.findDescendantsOfType(ASTFieldDeclaration.class);
        int count = 0;
        for (ASTFieldDeclaration field : fields) {
            if (field.isPublic()) {
                count++;
            }
        }
        return count;
    }
}
