package net.sourceforge.pmd.lang.java.rule.errorprone;
import java.util.HashSet;
import java.util.Set;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class AssignmentToNonFinalStaticRule extends AbstractJavaRulechainRule {
    public AssignmentToNonFinalStaticRule() {
        super(ASTAssignmentExpression.class, ASTConstructorDeclaration.class);
    }

    private Set<String> staticFieldNames = new HashSet<>();

    @Override
    public Object visit(ASTConstructorDeclaration node, Object data) {
        staticFieldNames.clear();
        for (ASTFieldDeclaration field : node.getTopLevelClassNode().getFields()) {
            if (field.isStatic() && !field.isFinal()) {
                for (ASTVariableDeclaratorId varId : field.getVariableDeclaratorIds()) {
                    staticFieldNames.add(varId.getName());
                }
            }
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTAssignmentExpression node, Object data) {
        ASTPrimaryExpression primaryExpression = node.getLeftOperand();
        ASTPrimaryPrefix primaryPrefix = primaryExpression.getFirstChildOfType(ASTPrimaryPrefix.class);
        if (primaryPrefix != null && primaryPrefix.getQualifier() != null) {
            String fieldName = primaryPrefix.getQualifier().getImage();
            if (staticFieldNames.contains(fieldName)) {
                addViolation(data, node);
            }
        }
        return super.visit(node, data);
    }
}
