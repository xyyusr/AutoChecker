import net.sourceforge.pmd.lang.java.ast. *;

import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class StringInstantiationRule extends AbstractJavaRulechainRule {
    public StringInstantiationRule() {
        super(ASTAllocationExpression.class);
    }

    @Override
    public Object visit(ASTAllocationExpression node, Object data) {
        ASTClassOrInterfaceType type = node.getType();
        if (type != null) {
            JType jtype = type.getType();
            if (jtype != null && jtype.getQualifiedName().equals(String.class.getName()) && !isStringBufferOrStringBuilderInstantiation(node)) {
                addViolation(data, node);
            }
        }
        return super.visit(node, data);
    }

    private boolean isStringBufferOrStringBuilderInstantiation(ASTAllocationExpression node) {
        ASTClassOrInterfaceType type = node.getType();
        if (type != null) {
            JType jtype = type.getType();
            return jtype != null && (jtype.getQualifiedName().equals(StringBuffer.class.getName()) || jtype.getQualifiedName().equals(StringBuilder.class.getName()));
        }
        return false;
    }
}
