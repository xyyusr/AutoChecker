package net.sourceforge.pmd.lang.java.rule.codestyle;

import java.util.List;
import java.util.stream.Collectors;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class IdenticalCatchBranchesRule extends AbstractJavaRulechainRule {

    public IdenticalCatchBranchesRule() {
        super(ASTTryStatement.class);
    }

    @Override
    public Object visit(ASTTryStatement node, Object data) {
        List<ASTCatchStatement> catchStatements = node.findChildrenOfType(ASTCatchStatement.class);
        List<String> blockStatements = catchStatements.stream()
                .map(ASTCatchStatement::getBlock)
                .map(ASTBlock::getImage)
                .collect(Collectors.toList());

        for(int i = 0; i < blockStatements.size() - 1; i++){
            for(int j = i + 1; j < blockStatements.size(); j++){
                if(blockStatements.get(i).equals(blockStatements.get(j))){
                    addViolation(data, catchStatements.get(i));
                    addViolation(data, catchStatements.get(j));
                }
            }
        }
        return super.visit(node, data);
    }
}

    public IdenticalCatchBranchesRule() {
        super(ASTTryStatement.class);

        Map<String, ASTCatchStatement> catchBlocks = new HashMap<>();
        NodeStream<ASTCatchStatement> catchStatements = node.findDescendantsOfType(ASTCatchStatement.class).toStream();

        catchStatements.forEach(catchStatement -> {
            String catchBlock = catchStatement.getBlock().getFirstChildOfType(Node.class).getImage();
            if (catchBlocks.containsKey(catchBlock)) {
                addViolation(data, catchStatement);
            } else {
                catchBlocks.put(catchBlock, catchStatement);
            }
        });

        return super.visit(node, data);
    }

        super(ASTTryStatement.class);

        List<ASTCatchStatement> catchStmts = node.findDescendantsOfType(ASTCatchStatement.class);
        Map<String, ASTCatchStatement> map = new HashMap<>();
        for (ASTCatchStatement stmt : catchStmts) {

            if (block != null) {
                String blockCode = block.jjtGetChild(0).jjtGetChild(0).getImage();
                if (map.containsKey(blockCode)) {
                    addViolation(data, stmt);
                } else {
                    map.put(blockCode, stmt);
                }
            }
        }
        return super.visit(node, data);
    }
}
