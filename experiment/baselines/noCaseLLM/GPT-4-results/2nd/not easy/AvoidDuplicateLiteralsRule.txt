package net.sourceforge.pmd.lang.java.rule.errorprone;


import java.util.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class AvoidDuplicateLiteralsRule extends AbstractJavaRulechainRule {

    private static final int THRESHOLD = 4;
    private static final int STRING_LENGTH = 3;
    private Map<String, List<Node>> nodesWithSameString = new HashMap<>();
    private Set<String> reportedLiterals = new HashSet<>();
    
    public AvoidDuplicateLiteralsRule() {
        super(ASTLiteral.class);
    }
    
    @Override
    public void start(RootNode rootNode) {
        nodesWithSameString.clear();
        reportedLiterals.clear();
        super.start(rootNode);
    }
    
    @Override
    public Object visit(ASTLiteral node, Object data) {
        if (node.isStringLiteral()) {
            String image = node.getImage();
            if (image.length() >= STRING_LENGTH) {
                List<Node> occurrences = nodesWithSameString.computeIfAbsent(image, k -> new ArrayList<>());
                occurrences.add(node);

                if (occurrences.size() == THRESHOLD && reportedLiterals.add(image)) {
                    addViolation(data, node, new Object[]{image});
                }
            }
        }
        return super.visit(node, data);
    }
}

package net.sourceforge.pmd.lang.java.rule.errorprone;

import java.util.HashMap;
import java.util.Map;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class AvoidDuplicateLiteralsRule extends AbstractJavaRulechainRule {
    private static final int THRESHOLD = 4;
    private static final int MINIMUM_LENGTH = 3;
    private Map<String, Integer> stringLiterals = new HashMap<>();

    public AvoidDuplicateLiteralsRule() {
        super(ASTLiteral.class);
    }

    @Override
    public Object visit(ASTLiteral node, Object data) {
        if (node.isStringLiteral()) {
            String image = node.getImage();
            if(image.length() >= MINIMUM_LENGTH) {
                int count = stringLiterals.containsKey(image) ? stringLiterals.get(image) : 0;
                stringLiterals.put(image, count + 1);
            }
        }
        return super.visit(node, data);
    }

    @Override
    public void start(RootNode rootNode) {
        stringLiterals.clear();
        super.start(rootNode);
    }

    @Override
    public void end(RootNode rootNode, Object data) {
        for (Map.Entry<String, Integer> entry : stringLiterals.entrySet()) {
            if (entry.getValue() >= THRESHOLD) {
                addViolation(data, rootNode, entry.getKey());
            }
        }
        super.end(rootNode, data);
    }
}

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class AvoidDuplicateLiteralsRule extends AbstractJavaRulechainRule {
    
    private static final int THRESHOLD = 4;
    private static final int LENGTH = 3;

    private Map<String, Integer> stringOccurrences;

    public AvoidDuplicateLiteralsRule() {
        super(ASTLiteral.class);

                int count = stringOccurrences.containsKey(literal) ? stringOccurrences.get(literal) : 0;
                stringOccurrences.put(literal, count + 1);
                if (stringOccurrences.get(literal) == THRESHOLD) {
                    addViolation(data, node);
                }
            }
        }
        return super.visit(node, data);
    }
}
