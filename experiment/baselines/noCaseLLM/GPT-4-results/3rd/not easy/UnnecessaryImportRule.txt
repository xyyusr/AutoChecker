package net.sourceforge.pmd.lang.java.rule.codestyle;


import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;

import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class UnnecessaryImportRule extends AbstractJavaRulechainRule {

    public UnnecessaryImportRule() {
        super(ASTImportDeclaration.class);
    }

    @Override
    public Object visit(ASTImportDeclaration node, Object data) {
        if(isUnnecessary(node)) {
            addViolation(data, node);
        }
        return super.visit(node, data);
    }

    private boolean isUnnecessary(ASTImportDeclaration node) {
        String importedName = node.getImportedName();
        if (importedName != null) {
            Node parent = node.getNthParent(2);
            if (parent instanceof ASTCompilationUnit) {
                ASTCompilationUnit acu = (ASTCompilationUnit) parent;
                String pakkageName = acu.getPackageDeclaration().getPackageName();
                if (importedName.startsWith(pakkageName + ".") ||
                    importedName.startsWith("java.lang.") ||
                    isDuplicated(node, importedName)) {
                    return true;
                }
            }
        }
        return false;
    }

    private boolean isDuplicated(ASTImportDeclaration node, String importedName) {
        Node parent = node.getNthParent(2);
        if (parent instanceof ASTCompilationUnit) {
            ASTCompilationUnit acu = (ASTCompilationUnit) parent;
            List<ASTImportDeclaration> imports = acu.findDescendantsOfType(ASTImportDeclaration.class);
            for (ASTImportDeclaration anImport : imports) {
                if (anImport.getBeginLine() < node.getBeginLine() && anImport.getImportedName().equals(importedName)) {
                    return true;
                }
            }
        }
        return false;
    }
}

package net.sourceforge.pmd.lang.java.rule.codestyle;

import java.util.HashSet;
import java.util.Set;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class UnnecessaryImportRule extends AbstractJavaRulechainRule {

    public UnnecessaryImportRule() {
        super(ASTCompilationUnit.class, ASTImportDeclaration.class);
    }

    @Override
    public Object visit(ASTCompilationUnit node, Object data) {
        node.jjtAccept(this, data);
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTImportDeclaration node, Object data) {
        Set<String> imports = new HashSet<>();

        NodeStream<ASTImportDeclaration> declarations = node.getDescendants(ASTImportDeclaration.class);
        declarations.forEach(declaration -> {
            String importString = declaration.getImportedName();
            if (imports.contains(importString) || isJavaLangImport(importString) || isSamePackageImport(importString, node)) {
                addViolationWithMessage(data, declaration, "Unnecessary import");
            } else {
                imports.add(importString);
            }
        });

        return super.visit(node, data);
    }

    private boolean isJavaLangImport(String importString) {
        return importString.startsWith("java.lang.");
    }

    private boolean isSamePackageImport(String importString, ASTImportDeclaration node) {
        String packageName = node.getParentsOfType(ASTCompilationUnit.class).get(0).getPackageDeclaration().getPackageNameImage();
        return importString.startsWith(packageName + ".");
    }
}

package net.sourceforge.pmd.lang.java.rule.codestyle;

import java.util.HashSet;
import java.util.Set;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class UnnecessaryImportRule extends AbstractJavaRulechainRule {

    private Set<String> singleTypeImports;
    private Set<String> onDemandImports;

    public UnnecessaryImportRule() {
        super(ASTCompilationUnit.class);
    }

    @Override
    public Object visit(ASTCompilationUnit node, Object data) {
        singleTypeImports = new HashSet<>();
        onDemandImports = new HashSet<>();
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTImportDeclaration node, Object data) {
        String packageName = node.getPackageName();
        if (node.isImportOnDemand()) {
            onDemandImports.add(packageName);
        } else {
            singleTypeImports.add(packageName + '.' + node.getImportedSimpleName());
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTPackageDeclaration node, Object data) {
        onDemandImports.add(node.getPackageName());
        return super.visit(node, data);
    }

    @Override
    public Object visit(AbstractJavaTypeNode node, Object data) {
        String fullyQualifiedName = node.getType().getName();
        String packageName = fullyQualifiedName.substring(0, fullyQualifiedName.lastIndexOf('.'));
        String typeName = fullyQualifiedName.substring(fullyQualifiedName.lastIndexOf('.') + 1);
        if (!singleTypeImports.remove(packageName + '.' + typeName) && !onDemandImports.contains(packageName)) {
            addViolation(data, node);
        }
        return super.visit(node, data);
    }
}
