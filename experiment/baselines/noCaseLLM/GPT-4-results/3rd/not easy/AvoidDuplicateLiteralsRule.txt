public class AvoidDuplicateLiteralsRule extends AbstractJavaRulechainRule {

    private static final int THRESHOLD = 4;
    private static final int STRING_LENGTH = 3;

    private Map<String, Integer> duplicateStrings = new HashMap<>();

    public AvoidDuplicateLiteralsRule() {
        super(RootNode.class);

                if (count >= THRESHOLD) {
                    addViolation(data, node);
                }
            }
        }
        return super.visit(node, data);
    }

package net.sourceforge.pmd.lang.java.rule.errorprone;


import java.util.HashMap;
import java.util.Map;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class AvoidDuplicateLiteralsRule extends AbstractJavaRulechainRule {

    private static final int THRESHOLD = 4;
    private static final int MIN_STRING_LENGTH = 3;

    public AvoidDuplicateLiteralsRule() {
        super(ASTLiteral.class);
    }

    @Override
    public void start(RuleContext ctx) {
        ctx.setAttribute("literalsCounts", new HashMap<String, Integer>());
    }

    @Override
    public void end(RuleContext ctx) {
        Map<String, Integer> literalsCounts = (Map<String, Integer>) ctx.getAttribute("literalsCounts");
        for (Map.Entry<String, Integer> entry : literalsCounts.entrySet()) {
            if (entry.getValue() >= THRESHOLD) {
                addViolation(ctx, entry.getKey());
            }
        }
    }

    @Override
    public Object visit(ASTLiteral node, Object data) {
        RuleContext ctx = (RuleContext) data;
        Map<String, Integer> literalsCounts = (Map<String, Integer>) ctx.getAttribute("literalsCounts");
        String literal = node.getImage();
        if (literal.length() >= MIN_STRING_LENGTH) {
            int count = literalsCounts.getOrDefault(literal, 0);
            literalsCounts.put(literal, count + 1);
        }
        return super.visit(node, data);
    }

    private void addViolation(RuleContext ctx, String literal) {
        ParametricRuleViolation<Node> violation = new ParametricRuleViolation<>(this, getMessage(), (RootNode) ctx.getRootNode(), literal);
        ctx.getReport().addRuleViolation(violation);
    }
}

package net.sourceforge.pmd.lang.java.rule.errorprone;


import java.util.HashMap;
import java.util.Map;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class AvoidDuplicateLiteralsRule extends AbstractJavaRulechainRule {

    private static final int THRESHOLD = 4;
    private static final int MINIMUM_LENGTH = 3;

    public AvoidDuplicateLiteralsRule() {
        super(ASTCompilationUnit.class);
    }

    @Override
    public Object visit(ASTCompilationUnit node, Object data) {
        Map<String, Integer> stringOccurrences = new HashMap<>();
        NodeStream<ASTLiteral> literals = node.descendants(ASTLiteral.class);

        literals.forEachRemaining(literalNode -> {
            if (literalNode.isStringLiteral()) {
                String stringLiteral = literalNode.getImage();
                if (stringLiteral.length() >= MINIMUM_LENGTH) {
                    stringOccurrences.put(stringLiteral, stringOccurrences.getOrDefault(stringLiteral, 0) + 1);
                }
            }
        });

        stringOccurrences.forEach((key, value) -> {
            if (value >= THRESHOLD) {
                addViolation(data, node);
            }
        });

        return super.visit(node, data);
    }
}
