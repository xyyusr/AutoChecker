import net.sourceforge.pmd.lang.ast.NodeStream;
public class LiteralsFirstInComparisonsRule extends AbstractJavaRulechainRule {

    public LiteralsFirstInComparisonsRule() {
        super(ASTClassOrInterfaceDeclaration.class);
    }

    @Override
    public Object visit(ASTClassOrInterfaceDeclaration node, Object data) {
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTMethodDeclaration node, Object data) {
        for (ASTPrimaryExpression expr : node.findDescendantsOfType(ASTPrimaryExpression.class)) {
            checkExpression(expr, data);
        }
        return super.visit(node, data);
    }

    private void checkExpression(ASTPrimaryExpression node, Object data) {
        if (node.jjtGetNumChildren() != 2) {
            return;
        }
        ASTPrimaryPrefix prefix = node.getFirstDescendantOfType(ASTPrimaryPrefix.class);
        ASTPrimarySuffix suffix = node.getFirstDescendantOfType(ASTPrimarySuffix.class);
        if (prefix == null || suffix == null) {
            return;
        }
        String methodName = suffix.getImage();
        if (methodName == null || !methodName.matches("equals|equalsIgnoreCase|compareTo|compareToIgnoreCase|contentEquals")) {
            return;
        }
        if (prefix.usesThisModifier()) {
            addViolation(data, node);
        }
    }
}
