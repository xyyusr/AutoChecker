package net.sourceforge.pmd.lang.java.rule.errorprone;


import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class AssignmentToNonFinalStaticRule extends AbstractJavaRulechainRule {

    public AssignmentToNonFinalStaticRule() {
        super(ASTConstructorDeclaration.class);
    }

    @Override
    public Object visit(ASTConstructorDeclaration node, Object data) {
        final Set<String> nonFinalStaticFields = new HashSet<>();
        final Node parent = node.jjtGetParent().jjtGetParent();
        final int num = parent.jjtGetNumChildren();

        for (int ix = 0; ix < num; ix++) {
            Node sib = parent.jjtGetChild(ix);

            if (sib instanceof ASTClassOrInterfaceBodyDeclaration) {
                ASTFieldDeclaration field = sib.getFirstChildOfType(ASTFieldDeclaration.class);
                
                if (field != null && field.isStatic() && !field.isFinal()) {
                    ASTVariableDeclaratorId var = field.getFirstChildOfType(ASTVariableDeclaratorId.class);
                    nonFinalStaticFields.add(var.getImage());
                }
            }
        }

        return visitConstructorDeclaration(node, data, nonFinalStaticFields);
    }
    
    private Object visitConstructorDeclaration(ASTConstructorDeclaration node, Object data, Set<String> nonFinalStaticFields) {
        final int num = node.jjtGetNumChildren();

        for (int ix = 0; ix < num; ix++) {
            Node child = node.jjtGetChild(ix);
            if (child instanceof ASTPrimaryExpression) {
                ASTPrimaryExpression primaryExpression = (ASTPrimaryExpression) child;
                ASTName name = primaryExpression.getFirstChildOfType(ASTName.class);
                
                if (name != null && nonFinalStaticFields.contains(name.getImage())) {
                    ASTAssignmentOperator assignmentOperator = primaryExpression.getFirstChildOfType(ASTAssignmentOperator.class);
                    
                    if (assignmentOperator != null) {
                        addViolation(data, name);
                        break;
                    }
                }
            }
        }
        
        return super.visit(node, data);
    }
}
