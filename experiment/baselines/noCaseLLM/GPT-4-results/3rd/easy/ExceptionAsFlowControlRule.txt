public class ExceptionAsFlowControlRule extends AbstractJavaRulechainRule {

    public ExceptionAsFlowControlRule() {
        super(ASTTryStatement.class);
    }

    @Override
    public Object visit(final ASTTryStatement node, Object data) {

        for (ASTCatchStatement catchStatement : node.findDescendantsOfType(ASTCatchStatement.class)) {
            if (catchStatement.jjtGetParent() == node) {
                checkForThrowInTry(node.getBlock(), catchStatement, data);
            }
        }

        return super.visit(node, data);
    }

    private void checkForThrowInTry(ASTBlock tryBlock, ASTCatchStatement catchStatement, Object data) {
        for (ASTThrowStatement throwStatement : tryBlock.findDescendantsOfType(ASTThrowStatement.class)) {
            if (catchStatement.getCaughtExceptionType().equals(throwStatement.getFirstChildOfType(ASTClassOrInterfaceType.class))) {
                addViolation(data, throwStatement);
            }
        }
    }
}

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ExceptionAsFlowControlRule extends AbstractJavaRulechainRule {

    public ExceptionAsFlowControlRule() {
        super(ASTTryStatement.class);
    }

    @Override
    public Object visit(ASTTryStatement node, Object data) {
        List<ASTThrowStatement> throwStatements = node.findDescendantsOfType(ASTThrowStatement.class);
        List<ASTCatchStatement> catchStatements = node.findDescendantsOfType(ASTCatchStatement.class);

        for (ASTThrowStatement throwStatement : throwStatements) {
            for (ASTCatchStatement catchStatement : catchStatements) {
                if (throwStatement.getFirstParentOfType(ASTTryStatement.class) == catchStatement.getFirstParentOfType(ASTTryStatement.class)) {
                    addViolation(data, throwStatement);
                    break;
                }
            }
        }

        return super.visit(node, data);
    }
}
