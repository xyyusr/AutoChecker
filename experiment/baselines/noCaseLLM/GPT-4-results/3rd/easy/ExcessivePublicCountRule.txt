import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ExcessivePublicCountRule extends AbstractJavaRulechainRule {

    private static final int THRESHOLD = 2;

    public ExcessivePublicCountRule() {
        super(ASTAnyTypeDeclaration.class);
    }

    @Override
    public Object visit(ASTAnyTypeDeclaration node, Object data) {
        int publicCount = 0;
        List<ASTClassOrInterfaceBodyDeclaration> declarations = node.findDescendantsOfType(ASTClassOrInterfaceBodyDeclaration.class);
        for (ASTClassOrInterfaceBodyDeclaration declaration : declarations) {
            ASTMethodDeclaration methodDeclaration = declaration.getFirstDescendantOfType(ASTMethodDeclaration.class);
            ASTFieldDeclaration fieldDeclaration = declaration.getFirstDescendantOfType(ASTFieldDeclaration.class);
            if ((methodDeclaration != null && methodDeclaration.isPublic()) ||
                (fieldDeclaration != null && fieldDeclaration.isPublic())) {
                publicCount += 1;
            }
            if (publicCount > THRESHOLD) {
                addViolation(data, node);
                break;
            }
        }
        return super.visit(node, data);
    }
}

package net.sourceforge.pmd.lang.java.rule.design;


import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ExcessivePublicCountRule extends AbstractJavaRulechainRule {

    private static final int EXCESSIVE_PUBLIC_COUNT = 2;

    public ExcessivePublicCountRule() {
        super(ASTClassOrInterfaceDeclaration.class);
    }

    @Override
    public Object visit(ASTClassOrInterfaceDeclaration node, Object data) {
        int publicCount = node.findDescendantsOfType(ASTMethodDeclaration.class, true).stream()
                .filter(ASTMethodDeclaration::isPublic)
                .mapToInt(md -> 1)
                .sum();

        publicCount += node.findDescendantsOfType(ASTFieldDeclaration.class, true).stream()
                .filter(ASTFieldDeclaration::isPublic)
                .mapToInt(fd -> 1)
                .sum();

        if (publicCount > EXCESSIVE_PUBLIC_COUNT) {
            addViolation(data, node);
        }

        return super.visit(node, data);
    }
}
