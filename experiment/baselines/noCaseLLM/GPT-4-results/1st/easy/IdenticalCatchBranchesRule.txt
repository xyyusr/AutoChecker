        int numCatchBlocks = node.getNumCatchBlocks();
        Set<String> uniqueCatchBlocks = new HashSet<>();

        for (int i = 0; i < numCatchBlocks; i++) {
            ASTCatchStatement catchStatement = node.getCatchBlock(i);

                uniqueCatchBlocks.add(catchBlockCode);
            }
