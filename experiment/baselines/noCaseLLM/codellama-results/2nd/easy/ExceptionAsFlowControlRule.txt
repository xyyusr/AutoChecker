package net.sourceforge.pmd.lang.java.rule.design;

import java.util.*;
import java.util.stream.Collectors;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ExceptionAsFlowControlRule extends AbstractJavaRulechainRule {

    public ExceptionAsFlowControlRule() {
        addRuleChainVisitor(this);
    }

    @Override
    public Object visitTryStatement(TryStatement node, Object data) {
        visitTryBlock(node.getTryBlock(), data);
        node.getCatchBlocks().stream()
                .filter(catchBlock -> catchBlock.getCaughtExceptions().size() == 1)
                .forEach(catchBlock -> {
                    JavaBlock throwBlock = catchBlock.getThrowStatement();
                    if (throwBlock != null) {
                        throwBlock.getFirstChild(Node.TYPE).ifPresent(exception -> {
                            addViolation(this, catchBlock, "Exception used as flow control");
                        });
                    }
                });
        return super.visitTryStatement(node, data);
    }

    @Override
    public Object visitThrowStatement(ThrowStatement node, Object data) {
        addViolation(this, node, "Exception used as flow control");
        return super.visitThrowStatement(node, data);
    }

    @Override
    public Object visitFinallyBlock(FinallyBlock node, Object data) {
        node.getStatements().stream()
                .filter(statement -> statement instanceof ThrowStatement)
                .forEach(statement -> {
                    addViolation(this, statement, "Exception used as flow control");
                });
        return super.visitFinallyBlock(node, data);
    }
}

package net.sourceforge.pmd.lang.java.rule.design;

import java.util.*;
import java.lang.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ExceptionAsFlowControlRule extends AbstractJavaRulechainRule {

    public ExceptionAsFlowControlRule() {
        addRuleMatch("design", "exceptionasflowcontrol", "Exceptions thrown and caught in an enclosing try statement are discouraged.", "java");
    }

    @Override
    public Object visit(ASTTryStmt n, Object data) {
        List<ASTCatcher> catchBlocks = n.findDescendants(ASTCatcher.class);
        if (catchBlocks.size() > 0) {
            for (ASTCatcher catchBlock : catchBlocks) {
                List<ASTExpression> exceptions = catchBlock.findDescendants(ASTExpression.class);
                if (exceptions.size() > 0) {
                    for (ASTExpression exception : exceptions) {
                        if (exception instanceof ASTThrow) {
                            addViolation(this, n, "Exceptions thrown and caught in an enclosing try statement are discouraged.");
                        }
                    }
                }
            }
        }
        return super.visit(n, data);
    }

    @Override
    public Object visit(ASTThrow n, Object data) {
        ASTTryStmt tryStmt = (ASTTryStmt) n.getParent();
        if (tryStmt != null) {
            addViolation(this, n, "Exceptions thrown and caught in an enclosing try statement are discouraged.");
        }
        return super.visit(n, data);
    }
}

package net.sourceforge.pmd.lang.java.rule.design;

import java.util.*;
import java.lang.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ExceptionAsFlowControlRule extends AbstractJavaRulechainRule {
    public ExceptionAsFlowControlRule() {
        addRuleChainVisitor(this);
    }

    @Override
    public Object visit(BlockStmt node, Object data) {
        Set<JavaIdentifier> caughtExceptions = new HashSet<>();
        for (Expression excExpr : node.getExceptions()) {
            caughtExceptions.add((JavaIdentifier) excExpr);
        }
        if (caughtExceptions.isEmpty() && node.getBody().getStatements().size() > 0) {
            for (Statement statement : node.getBody().getStatements()) {
                if (statement instanceof ThrowStmt) {
                    JavaIdentifier exception = ((ThrowExpr) ((ThrowStmt) statement).getExpression()).getException();
                    if (caughtExceptions.contains(exception)) {
                        addViolation(data, this, node);
                    }
                }
            }
        }
        return data;
    }

    @Override
    public Object visit(ThrowStmt node, Object data) {
        return data;
    }

    @Override
    public Object visit(ThrowExpr node, Object data) {
        return data;
    }

    @Override
    public Object visit(BlockStmtBody node, Object data) {
        return data;
    }

    @Override
    public Object visit(ThrowExprWithException node, Object data) {
        return data;
    }
}
