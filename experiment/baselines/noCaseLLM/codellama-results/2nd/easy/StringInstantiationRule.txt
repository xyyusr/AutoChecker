import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class StringInstantiationRule extends AbstractJavaRulechainRule {

    public StringInstantiationRule() {
        super(NewExpression.class);
    }

    @Override
    public Object visit(NewExpression node, Object data) {
        if (node.getType() instanceof PrimitiveType) {
            PrimitiveType primitiveType = (PrimitiveType) node.getType();
            if (primitiveType.getName().equals("String")) {
                addViolation(data, node);
            }
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ClassOrInterfaceType node, Object data) {
        String name = node.getImage();
        if (name.endsWith("String")) {
            NewExpression newExpression = findParent(node, NewExpression.class);
            if (newExpression != null) {
                addViolation(data, newExpression);
            }
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(MethodCallExpression node, Object data) {
        String name = node.getImage();
        if (name.equals("String.valueOf") || name.equals("String.format") || name.equals("String.concat")) {
            NewExpression newExpression = findParent(node, NewExpression.class);
            if (newExpression != null) {
                addViolation(data, newExpression);
            }
        }
        return super.visit(node, data);
    }

    private NewExpression findParent(Node node, Class<?> clazz) {
        while (node != null) {
            if (clazz.isInstance(node)) {
                return (NewExpression) node;
            }
            node = node.getParentNode();
        }
        return null;
    }
}
