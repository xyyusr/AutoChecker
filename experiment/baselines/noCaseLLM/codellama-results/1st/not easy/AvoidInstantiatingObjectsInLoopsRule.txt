package net.sourceforge.pmd.lang.java.rule.performance;

import java.util.*;
import java.lang.*;
import java.util.regex.Pattern;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class AvoidInstantiatingObjectsInLoopsRule extends AbstractJavaRulechainRule {

    public AvoidInstantiatingObjectsInLoopsRule() {
        super(NodeKind.CLASS_DECL, NodeKind.METHOD_DECL, NodeKind.FOR_STATEMENT);
    }

    @Override
    public Object visit(ClassOrInterfaceDeclaration n, Object data) {
        return super.visit(n, data);
    }

    @Override
    public Object visit(MethodDeclaration n, Object data) {
        if (n.getBody() != null) {
            Object result = super.visit(n, data);
            visitForStatements(n.getBody(), data);
            return result;
        }
        return super.visit(n, data);
    }

    private void visitForStatements(Node body, Object data) {
        ForStatement forStatement = findFirstDescendant(body, ForStatement.class);
        if (forStatement != null) {
            visitForLoopVariables(forStatement.getInit(), data);
            visitForLoopVariables(forStatement.getUpdate(), data);
        }
    }

    private void visitForLoopVariables(Node init, Object data) {
        if (init != null) {
            ExpressionList expressions = findDescendants(init, Expression.class);
            for (Expression expression : expressions.getExpressions()) {
                visitExpression(expression, data);
            }
        }
    }

    private void visitExpression(Expression n, Object data) {
        if (n instanceof NewClass) {
            addViolation(data, n);
        } else if (n instanceof ArrayAccess) {
            ArrayAccess arrayAccess = (ArrayAccess) n;
            if (arrayAccess.getArray() instanceof NewArray) {
                NewArray newArray = (NewArray) arrayAccess.getArray();
                addViolation(data, newArray);
            }
        } else if (n instanceof MethodInvocation) {
            MethodInvocation methodInvocation = (MethodInvocation) n;
            if (methodInvocation.getExpression() instanceof NewClass) {
                NewClass newClass = (NewClass) methodInvocation.getExpression();
                addViolation(data, newClass);
            }
        }
    }

    private ForStatement findFirstDescendant(Node node, Class<?> type) {
        if (node instanceof ForStatement) {
            return (ForStatement) node;
        }
        return node.findFirstDescendant(type);
    }

    private ExpressionList findDescendants(Node node, Class<?> type) {
        NodeList<Expression> expressions = node.findDescendants(type);
        return new ExpressionList(expressions);
    }
}

package net.sourceforge.pmd.lang.java.rule.performance;

import java.util.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class AvoidInstantiatingObjectsInLoopsRule extends AbstractJavaRulechainRule {

    public AvoidInstantiatingObjectsInLoopsRule() {
        super(ForStatement.class, WhileStatement.class);
    }

    @Override
    public Object visit(ForStatement node, Object data) {
        return visitLoop(node, data);
    }

    @Override
    public Object visit(WhileStatement node, Object data) {
        return visitLoop(node, data);
    }

    private Object visitLoop(StatementLoop node, Object data) {
        Iterator<Statement> statements = node.getBody().getStatements();
        while (statements.hasNext()) {
            Statement statement = statements.next();
            if (statement instanceof ExpressionStatement) {
                Expression expression = ((ExpressionStatement) statement).getExpression();
                if (expression instanceof NewClassExpression) {
                    addViolation(data, this, node);
                }
            }
        }
        return data;
    }
}
