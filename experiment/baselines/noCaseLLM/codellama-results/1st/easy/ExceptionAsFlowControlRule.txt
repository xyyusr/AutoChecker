        addRuleChainVisitor(this);

        return super.visit(node, data);

    public Object visit(TryStatement node, Object data) {

        for (Statement statement : body) {

                CatchClause catchClause = (CatchClause) statement;

                            addViolation(data, node);
                        }
                    }
                }
            }
        }
        return super.visit(node, data);
    }

    public Object visit(Block node, Object data) {

package net.sourceforge.pmd.lang.java.rule.design;


import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;

import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ExceptionAsFlowControlRule extends AbstractJavaRulechainRule {

    public ExceptionAsFlowControlRule() {
        addRuleMatch("java", "method");
        addRuleMatch("java", "method", "return");

        addRuleMatch("java", "try");
        addRuleMatch("java", "try", "catch");

        setRulePriority(PMD.LOW);
    }

    @Override
    public Object visit(ASTTryStatement node, Object data) {
        for (JavaNode child : node.jjtGetChildren()) {
            if (child instanceof ASTCatchClause) {
                ASTCatchClause catchClause = (ASTCatchClause) child;
                if (hasSameEnclosingTry(node, catchClause)) {
                    addViolation(data, node);
                }
            }
        }
        return super.visit(node, data);
    }

    private boolean hasSameEnclosingTry(ASTTryStatement outerTry, ASTCatchClause innerCatch) {
        AbstractNode parent = innerCatch.getParent();
        while (parent != null) {
            if (parent instanceof ASTTryStatement && parent != outerTry) {
                return true;
            }
            parent = parent.getParent();
        }
        return false;
    }
}

import java.util.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;

public class ExceptionAsFlowControlRule extends AbstractJavaRulechainRule {

    public ExceptionAsFlowControlRule() {
        addRuleMatcher(new TryBlockRuleMatcher(this));
    }

    @Override
    public Object visit(TryBlock node, Object data) {
        List<CaughtException> caughtExceptions = node.getCaughtExceptions();
        for (CaughtException caughtException : caughtExceptions) {
            if (caughtException.getExceptionType() instanceof ClassOrInterfaceType) {
                ClassOrInterfaceType exceptionType = (ClassOrInterfaceType) caughtException.getExceptionType();
                if (hasEnclosingTryBlock(node)) {
                    addViolation(this, node, "Exceptions thrown and caught in an enclosing try statement");
                }
            }
        }
        return data;
    }

    private boolean hasEnclosingTryBlock(TryBlock node) {
        JavaNode parent = node.getParent();
        while (parent != null) {
            if (parent instanceof TryBlock) {
                return true;
            }
            parent = parent.getParent();
        }
        return false;
    }

    private static class TryBlockRuleMatcher extends RuleChainVisitor {
        private final ExceptionAsFlowControlRule rule;

        public TryBlockRuleMatcher(ExceptionAsFlowControlRule rule) {
            this.rule = rule;
        }

        @Override
        public Object visit(TryBlock node, Object data) {
            return rule.visit(node, data);
        }
    }
}

import java.util.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;

import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ExceptionAsFlowControlRule extends AbstractJavaRulechainRule {

    public ExceptionAsFlowControlRule() {
        super(CompiledClass.class);
    }

    @Override
    public Object visit(CompiledClass node, Object data) {
        return super.visit(node, data);
    }

    @Override
    public Object visit(CompilationUnit node, Object data) {
        List<MethodDeclaration> methods = node.findDescendants(MethodDeclaration.class);
        for (MethodDeclaration method : methods) {
            List<TryStatement> tryStatements = method.findDescendants(TryStatement.class);
            for (TryStatement tryNode : tryStatements) {
                List<CatchClause> catchClauses = tryNode.findDescendants(CatchClause.class);
                for (CatchClause catchNode : catchClauses) {
                    List<MethodCallExpr> methodCalls = catchNode.findDescendants(MethodCallExpr.class);
                    for (MethodCallExpr call : methodCalls) {
                        if (call.getName().toString().equals("throws")) {
                            addViolation(this, data, call);
                        }
                    }
                }
            }
        }
        return super.visit(node, data);
    }
}

                    addViolation(data, node);

        return super.visit(node, data);

        return super.visit(node, data);
