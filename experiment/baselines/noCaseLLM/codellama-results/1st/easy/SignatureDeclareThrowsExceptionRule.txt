public class SignatureDeclareThrowsExceptionRule extends AbstractJavaRulechainRule {

    public SignatureDeclareThrowsExceptionRule() {
        addRuleMatch(ASTMethodDeclaration.class, this::visitMethodDeclaration);
    }

    @Override
    public Object visitMethodDeclaration(MethodDeclaration node, Object data) {
        if (node.getThrowsClause() != null && hasExplicitExceptionType(node.getThrowsClause())) {
            addViolation(this, data, node);
        }
        return super.visitMethodDeclaration(node, data);
    }

        for (Type type : node.getTypes()) {
            if (type.isQualifiedName() && "java.lang".equals(type.getImage().substring(0, 10)) && type.getImage().length() == 12 && type.getImage().endsWith("Exception")) {
                return true;
            }
        }
        return false;
    }
}
