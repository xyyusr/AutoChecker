public class InefficientEmptyStringCheckRule extends AbstractJavaRulechainRule {

    public InefficientEmptyStringCheckRule() {
        addRuleChainVisitorClass(this);
    }

    @Override
    public Object visit(ASTCallExpression node, Object data) {
        if (isStringTrimCall(node)) {
            List<JavaNode> args = node.getChildrenByField("arguments");
            if (args.size() == 1) {
                ASTMethodInvocation methodInvocation = (ASTMethodInvocation) args.get(0);
                if (methodInvocation.getImage().equals("length") || methodInvocation.getImage().equals("isEmpty")) {
                    addViolation(data, node);
                }
            }
        }
        return super.visit(node, data);
    }

    private boolean isStringTrimCall(ASTCallExpression node) {
        List<JavaNode> args = node.getChildrenByField("arguments");
        if (args.size() != 1) {
            return false;
        }
        ASTMethodInvocation methodInvocation = (ASTMethodInvocation) args.get(0);
        if (!methodInvocation.getImage().equals("trim")) {
            return false;
        }
        return methodInvocation.getImage().equals("trim");
    }

public class InefficientEmptyStringCheckRule extends AbstractJavaRulechainRule {

    public InefficientEmptyStringCheckRule() {
        addRuleMatch(JDK8.get("java.lang.String.trim()"));
        addRuleMatch(JDK8.get("java.lang.String.isEmpty()"));
    }

    @Override
    public Object visit(ASTCallExpression node, Object data) {
        if (node.getType() == ASTCallExpression.CALL_TYPE_METHOD_INVOCATION) {
            ASTMethodInvocation methodInvocation = (ASTMethodInvocation) node.getFirstChildByType(ASTMethodInvocation.class);
            if (methodInvocation != null) {
                String methodName = methodInvocation.getMethodName().getImage();
                if (methodName.equals("trim") || methodName.equals("isEmpty")) {
                    ASTExpression target = methodInvocation.getTarget();
                    if (target.getType() == ASTExpression.EXPR_TYPE_METHOD_INVOCATION || target.getType() == ASTExpression.EXPR_TYPE_METHOD_REFERENCE) {
                        ASTExpressionList args = methodInvocation.getArguments();
                        if (args != null && args.size() == 0) {
                            addViolation(this, node, "Using String.trim().length() or String.isEmpty() is inefficient, use isEmpty() instead.");
                        }
                    }
                }
            }
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTMethodInvocation node, Object data) {
        if (node.getMethodName().getImage().equals("length")) {
            ASTExpression target = node.getTarget();
            if (target.getType() == ASTExpression.EXPR_TYPE_METHOD_INVOCATION) {
