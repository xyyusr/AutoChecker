    }

    @Override
    public Object visit(Assign node, Object data) {
        Object result = super.visit(node, data);
        if (result == null) {
            Expression target = node.getTarget();
            if (target instanceof FieldAccess) {
                FieldAccess fieldAccess = (FieldAccess) target;
                if (fieldAccess.getTarget() != null && fieldAccess.getTarget().getImage().startsWith("global")) {
                    // Ignore global variables
                    return result;
                }

                // Ignore variable declarations
                return result;
            }
            if (node.getRightHandSide() instanceof NullLiteral) {
                addRuleBreak(node, MESSAGE);
            }
        }
        return result;
    }

            if (node.getInitializer() != null && node.getInitializer() instanceof NullLiteral) {
                addRuleBreak(node, MESSAGE);
            }
        }

package net.sourceforge.pmd.lang.java.rule.errorprone;

import java.util.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class NullAssignmentRule extends AbstractJavaRulechainRule {

    public NullAssignmentRule() {
        super(
                InternalNodeTypes.ASSIGNMENT_EXPRESSION,
                JavaTypes.NULL_LITERAL
        );
    }

    @Override
    public Object visit(ASTAssignmentExpression node, Object data) {
        for (JavaNode child : node.jjtGetChildren()) {
            if (child instanceof ASTIdentifier) {
                ASTIdentifier id = (ASTIdentifier) child;
                if (!id.getImage().equals(node.getImage())) {
                    SymbolTable symbolTable = getSymbolTable(node);
                    if (symbolTable != null && symbolTable.isSymbolDeclared(id.getImage())) {
                        JavaNode variableDeclaration = getVariableDeclaration(id.getImage(), node);
                        if (variableDeclaration != null) {
                            ASTVariableDeclaratorId variableDeclaratorId = (ASTVariableDeclaratorId) variableDeclaration.jjtGetChild(0);
                            if (variableDeclaratorId != null) {
                                if (!variableDeclaratorId.getImage().equals(id.getImage())) {
                                    continue;
                                }
                            }
                            if (variableDeclaration instanceof ASTVariableDeclarator) {
                                ASTLiteral literal = (ASTLiteral) variableDeclaration.jjtGetChild(0);
                                if (literal != null && literal.getImage().equals(node.getImage())) {
                                    continue;
                                }
                            }
                            addViolation(this, data, node);
                        }
                    }
                }
            }
        }
        return super.visit(node, data);
    }

    private SymbolTable getSymbolTable(JavaNode node) {
        return node.getSymbolTable();
    }

    private JavaNode getVariableDeclaration(String name, JavaNode node) {
        JavaNode variableDeclaration = null;
        for (JavaNode child : node.jjtGetChildren()) {
            if (child instanceof ASTVariableDeclarator) {
                ASTIdentifier identifier = (ASTIdentifier) child.jjtGetChild(0);
                if (identifier != null && identifier.getImage().equals(name)) {
                    variableDeclaration = child;
                    break;
                }
            }
        }
        return variableDeclaration;
    }
}
