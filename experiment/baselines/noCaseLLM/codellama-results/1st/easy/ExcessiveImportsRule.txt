import net.sourceforge.pmd.lang.java.symbols. *;

public class ExcessiveImportsRule extends AbstractJavaRulechainRule {

    private static final int DEFAULT_THRESHOLD = 3;
    private static final String RULE_NAME = "ExcessiveImports";

    public ExcessiveImportsRule() {
        addRuleChainElement(new NodeMatcher("import ( ~ImportDeclaration )", "method Invoking Import"));
    }

    @Override
    public Object visit(ImportDeclaration node, Object data) {
        int importCount = (int) data;

        if (importCount < DEFAULT_THRESHOLD) {
            return null;
        }

        addViolation(this, node, getRulePriority());
        return null;
    }

    @Override
    public Object visit(CompilationUnit javaNode, Object data) {
        int importCount = ((ImportDeclaration[]) javaNode.jjtGetChild(0)).length;

        if (importCount > DEFAULT_THRESHOLD) {
            addViolation(this, javaNode, getRulePriority());
        }

        return new Object[] { importCount };
    }
}

public class ExcessiveImportsRule extends AbstractJavaRulechainRule {
    @Override
    public Object visit(ImportDeclaration node, Object data) {
        if (!node.isStatic()) {
            int totalImports = 0;
            for (ImportDeclaration importNode : node.getAST().getImportDeclarations()) {
                if (!importNode.isStatic()) {
                    totalImports++;
                }
            }
            if (totalImports > 3) {
                addViolation(data, node);
            }
        }
        return super.visit(node, data);
    }
}
