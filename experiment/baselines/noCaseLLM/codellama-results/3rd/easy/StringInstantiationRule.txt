package net.sourceforge.pmd.lang.java.rule.performance;

import java.util.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class StringInstantiationRule extends AbstractJavaRulechainRule {
    public StringInstantiationRule() {
        addRuleMatch(ASTNewClassExpression.class);
    }

    @Override
    public Object visit(ASTNewClassExpression node, Object data) {
        String className = node.getImage();
        if (className.equals("java.lang.String")) {
            addViolation(data, node);
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTCallExpression node, Object data) {
        String methodName = node.getImage();
        if ("valueOf".equals(methodName)) {
            ASTPrimaryPrefix prefix = (ASTPrimaryPrefix) node.getFirstChildByType(ASTPrimaryPrefix.class);
            ASTStringLiteral literal = (ASTStringLiteral) prefix.getFirstChildByType(ASTStringLiteral.class);
            if (literal != null) {
                addViolation(data, node);
            }
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTMethodCallExpression node, Object data) {
        String methodName = node.getImage();
        if ("constructor".equals(methodName)) {
            ASTArgumentList arguments = (ASTArgumentList) node.getFirstChildByType(ASTArgumentList.class);
            if (arguments != null && arguments.size() > 0) {
                ASTArgument argument = arguments.getFirstChildByType(ASTArgument.class);
                ASTStringLiteral literal = (ASTStringLiteral) argument.getExpression().getFirstChildByType(ASTStringLiteral.class);
                if (literal != null) {
                    addViolation(data, node);
                }
            }
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTStringConstructor node, Object data) {
        addViolation(data, node);
        return super.visit(node, data);
    }
}
