import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class AvoidUsingOctalValuesRule extends AbstractJavaRulechainRule {
    public AvoidUsingOctalValuesRule() {
        super(LiteralNumeric.class);
    }

    @Override
    public Object visit(LiteralNumeric node, Object data) {
        if (node instanceof IntegerLiteral) {
            IntegerLiteral intNode = (IntegerLiteral) node;
            String value = intNode.getImage();
            if (value.startsWith("0")) {
                BigInteger bigInt = new BigInteger(value);
                if (bigInt.toString(10).equals(value) || !bigInt.toString(8).equals(value)) {
                    addViolation(data, new RuleViolation(intNode, "Integer literal should not start with zero"));
                }
            }
        }
        return super.visit(node, data);
    }
}
