import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class LiteralsFirstInComparisonsRule extends AbstractJavaRulechainRule {
    public LiteralsFirstInComparisonsRule() {
        super(ASTMethodCall.class);
    }

    @Override
    public Object visit(ASTMethodCall node, Object data) {
        String methodName = node.getImage();
        if (!"equals".equals(methodName) && !"equalsIgnoreCase".equals(methodName) && !"compareTo".equals(methodName) && !"compareToIgnoreCase".equals(methodName) && !"contentEquals".equals(methodName)) {
            return super.visit(node, data);
        }

        ASTPrimaryPrefix prefix = node.getFirstDescendantOfType(ASTPrimaryPrefix.class);
        ASTPrimaryExpression primaryExpression = prefix.getFirstChildOfType(ASTPrimaryExpression.class);
        ASTExpression firstArgument = primaryExpression.getFirstChildOfType(ASTExpression.class);

        ASTPrimarySuffix suffix = node.getFirstDescendantOfType(ASTPrimarySuffix.class);
        ASTLiteral secondArgument = suffix.arguments().getFirstDescendantOfType(ASTLiteral.class);

        if (secondArgument instanceof ASTNullLiteral) {
            addViolation(data, node);
        } else if (secondArgument instanceof ASTStringLiteral && firstArgument instanceof ASTStringLiteral) {
            int comparison = ((ASTStringLiteral) firstArgument).compareLocation(secondArgument);
            if (comparison > 0) {
                addViolation(data, node);
            }
        } else if (secondArgument instanceof ASTStringLiteral && !(firstArgument instanceof ASTStringLiteral)) {
            addViolation(data, node);
        }

        return super.visit(node, data);
    }
}

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class LiteralsFirstInComparisonsRule extends AbstractJavaRulechainRule {
    public LiteralsFirstInComparisonsRule() {
        super(ASTMethodCall.class);
    }

    @Override
    public Object visit(ASTMethodCall node, Object data) {
        if (isStringComparisonMethod(node)) {
            List<ASTExpression> arguments = node.getArguments();
            if (arguments.size() == 2) {
                ASTExpression firstArg = arguments.get(0);
                ASTExpression secondArg = arguments.get(1);
                if (!isStringLiteral(firstArg) && isStringLiteral(secondArg)) {
                    addViolation(data, node);
                }
                if (JavaAstUtils.isNullLiteral(secondArg)) {
                    addViolation(data, node);
                }
            }
        }
        return super.visit(node, data);
    }

    private boolean isStringComparisonMethod(ASTMethodCall node) {
        String methodName = node.getName();
        ASTExpression receiver = node.getFirstChildOfType(ASTExpression.class);
        return receiver != null && receiver.getType() == null || receiver.getType().isSubtypeOf("java.lang.String") &&
                ("equals".equals(methodName) ||
                 "equalsIgnoreCase".equals(methodName) ||
                 "compareTo".equals(methodName) ||
                 "compareToIgnoreCase".equals(methodName) ||
                 "contentEquals".equals(methodName));
    }

    private boolean isStringLiteral(ASTExpression node) {
        return node instanceof ASTStringLiteral;
    }
}

    public LiteralsFirstInComparisonsRule() {
        super(ASTMethodCall.class);
    }

    @Override
    public Object visit(ASTMethodCall node, Object data) {

        if (isComparisonMethod(methodName)) {
            List<ASTExpression> arguments = node.getArguments();
            if (arguments.size() == 2) {
                ASTExpression firstArgument = arguments.get(0);
                ASTExpression secondArgument = arguments.get(1);
                if (secondArgument instanceof ASTNullLiteral) {

                }

    }

    private boolean isComparisonMethod(String methodName) {

               "equalsIgnoreCase".equals(methodName) ||

               "compareToIgnoreCase".equals(methodName) || 
               "contentEquals".equals(methodName);
