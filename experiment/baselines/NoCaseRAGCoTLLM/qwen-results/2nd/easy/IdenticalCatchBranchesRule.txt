    public IdenticalCatchBranchesRule() {
        super(ASTTryStatement.class);
    }

    @Override
    public Object visit(ASTTryStatement node, Object data) {
        Map<String, List<ASTCatchClause>> catchBranchesMap = new HashMap<>();
        for (ASTCatchClause catchClause : node.getCatchClauses()) {
            StringBuilder sb = new StringBuilder();
            catchClause.getDescendantsOrSelf().forEach(n -> sb.append(n.getImage()).append(" "));
            String key = sb.toString().trim();
            catchBranchesMap.computeIfAbsent(key, k -> new ArrayList<>()).add(catchClause);
        }

        for (List<ASTCatchClause> catchClauses : catchBranchesMap.values()) {
            if (catchClauses.size() > 1) {
                boolean allSingleCatch = true;

                        allSingleCatch = false;
                        break;
                    }
                }
                if (allSingleCatch) {
                    for (int i = 1; i < catchClauses.size(); i++) {
                        addViolation(data, catchClauses.get(i));
                    }
                }
            }
        }

        return super.visit(node, data);
    }

package net.sourceforge.pmd.lang.java.rule.codestyle;

import java.util.HashSet;
import java.util.Set;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class IdenticalCatchBranchesRule extends AbstractJavaRulechainRule {
    public IdenticalCatchBranchesRule() {
        super(ASTTryStatement.class);
    }

    @Override
    public Object visit(ASTTryStatement node, Object data) {
        Set<Set<ASTCatchClause>> catchClauseGroups = new HashSet<>();
        for (ASTCatchClause catchClause : node.getCatchClauses()) {
            Set<ASTCatchClause> identicalGroup = new HashSet<>();
            for (ASTCatchClause otherCatchClause : node.getCatchClauses()) {
                if (!otherCatchClause.equals(catchClause) && catchClausesAreIdentical(catchClause, otherCatchClause)) {
                    identicalGroup.add(otherCatchClause);
                }
            }
            if (!identicalGroup.isEmpty()) {
                identicalGroup.add(catchClause);
                catchClauseGroups.add(identicalGroup);
            }
        }

        for (Set<ASTCatchClause> group : catchClauseGroups) {
            if (group.size() > 1) {
                for (ASTCatchClause catchClause : group) {
                    addViolation(data, catchClause);
                }
            }
        }

        return super.visit(node, data);
    }

    private boolean catchClausesAreIdentical(ASTCatchClause catchClause1, ASTCatchClause catchClause2) {
        ASTBlock block1 = catchClause1.getFirstChildOfType(ASTBlock.class);
        ASTBlock block2 = catchClause2.getFirstChildOfType(ASTBlock.class);
        if (block1 == null || block2 == null) return false;
        List<ASTStatement> statements1 = block1.findDescendantsOfType(ASTStatement.class);
        List<ASTStatement> statements2 = block2.findDescendantsOfType(ASTStatement.class);
        if (statements1.size() != statements2.size()) return false;

        for (int i = 0; i < statements1.size(); i++) {
            if (!statements1.get(i).equals(statements2.get(i))) return false;
        }

        return true;
    }
}

package net.sourceforge.pmd.lang.java.rule.codestyle;

import java.util.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class IdenticalCatchBranchesRule extends AbstractJavaRulechainRule {
    public IdenticalCatchBranchesRule() {
        super(ASTTryStatement.class);
    }

    @Override
    public Object visit(ASTTryStatement node, Object data) {
        List<ASTCatchClause> catchClauses = node.findDescendantsOfType(ASTCatchClause.class);
        Map<String, List<ASTCatchClause>> branchMap = new HashMap<>();

        for (ASTCatchClause catchClause : catchClauses) {
            String branchCode = catchClause.getBody().getImage();
            branchMap.computeIfAbsent(branchCode, k -> new ArrayList<>()).add(catchClause);
        }

        for (List<ASTCatchClause> branches : branchMap.values()) {
            if (branches.size() > 1) {
                Set<String> exceptionTypes = new HashSet<>();
                for (ASTCatchClause branch : branches) {
                    exceptionTypes.addAll(getExceptionTypes(branch));
                }
                if (exceptionTypes.size() > 1) {
                    StringBuilder multicatchTypes = new StringBuilder();
                    for (String type : exceptionTypes) {
                        if (multicatchTypes.length() > 0) {
                            multicatchTypes.append(" | ");
                        }
                        multicatchTypes.append(type);
                    }
                    addViolation(data, branches.get(0), multicatchTypes.toString());
                }
