import java.util.HashMap;
import java.util.Map;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class UseStringBufferForStringAppendsRule extends AbstractJavaRulechainRule {
    private Map<JVariableSymbol, Integer> stringConcatCountMap = new HashMap<>();

    public UseStringBufferForStringAppendsRule() {
        super(ASTAssignmentExpression.class);
    }

    @Override
    public void start(NodeStream stream) {
        stringConcatCountMap.clear();
    }

    @Override
    public Object visit(ASTAssignmentExpression node, Object data) {
        if (node.getOperator() == ASTAssignmentExpression.Operator.PLUS_ASSIGN) {
            ASTVariableDeclaratorId target = JavaAstUtils.getVariableDeclaratorIdFromLhs(node.getLeftOperand());
            if (target != null) {
                JVariableSymbol symbol = target.getVariableSymbol();
                if (symbol != null) {
                    JType type = symbol.getTypeInSignature();
                    if (TypeSystem.STRING_TYPE.equals(type)) {
                        ASTExpression rhs = node.getRightOperand();
                        if (JavaAstUtils.isStringConcatExpr(rhs)) {
                            stringConcatCountMap.put(symbol, stringConcatCountMap.getOrDefault(symbol, 0) + 1);
                            if (stringConcatCountMap.get(symbol) > 1) {
                                addViolation(data, node);
                            }
                        }
                    }
                }
            }
        }
        return super.visit(node, data);
    }
}
