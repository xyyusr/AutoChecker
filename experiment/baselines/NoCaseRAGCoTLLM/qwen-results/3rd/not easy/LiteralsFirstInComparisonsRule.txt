    public LiteralsFirstInComparisonsRule() {
        super(ASTMethodCallExpression.class);

        JMethodSymbol methodSymbol = node.getInvokedMethodSymbol();
        if (methodSymbol != null && "java.lang.String".equals(methodSymbol.getOwnerType().getQualifiedName())) {
            String methodName = methodSymbol.getName();

                "compareTo".equals(methodName) || "compareToIgnoreCase".equals(methodName) || 
                "contentEquals".equals(methodName)) {
                List<ASTExpression> arguments = node.getArguments();

                    ASTExpression firstArg = arguments.get(0);
                    ASTExpression secondArg = arguments.get(1);

                        return data;

                        return data;

                            ASTPrimaryExpression primaryExpr = (ASTPrimaryExpression) firstArg;

                                addViolation(data, node);
                            }
                        } else if (!(firstArg instanceof ASTStringLiteral)) {
                            addViolation(data, node);
                        }
                    }
                }
            }
        }

                }
            }
