import net.sourceforge.pmd.lang.java.ast. *;

public class UnnecessaryImportRule extends AbstractJavaRulechainRule {
    public UnnecessaryImportRule() {
        super(ASTCompilationUnit.class);
    }
    @Override
    public Object visit(ASTCompilationUnit node, Object data) {
        Set<String> usedImports = new HashSet<>();
        Set<String> implicitImports = new HashSet<>(Arrays.asList("Object", "String", "Boolean", "Character", "Byte", "Short", "Integer", "Long", "Float", "Double", "Void", "Thread", "Runnable"));
        String currentPackage = getCurrentPackage(node);

        node.descendants(ASTImportDeclaration.class).forEach(importDecl -> {
            if (importDecl.getImportedSimpleName() != null) {
                if (currentPackage != null && importDecl.getImportedName().startsWith(currentPackage)) {
                    return;
                }
                if (implicitImports.contains(importDecl.getImportedSimpleName())) {
                    return;
                }
            }
            usedImports.add(importDecl.getImportedName());
        });

        node.descendants(ASTName.class).forEach(nameNode -> {
            JType type = nameNode.getType();
            if (type != null && type.getTypeKind() == JType.Kind.DECLARED) {
                usedImports.add(((JClassSymbol) type.getSymbol()).getQualifiedName());
            }
        });

        node.descendants(ASTImportDeclaration.class).forEach(importDecl -> {
            if (!usedImports.contains(importDecl.getImportedName()) || importDecl.isStatic()) {
                return;
            }
            if (importDecl.isOnDemand()) {
                String packageName = importDecl.getImportedName();
                if (node.descendants(ASTName.class).noneMatch(nameNode -> {
                    JType type = nameNode.getType();
                    return type != null && type.getTypeKind() == JType.Kind.DECLARED && ((JClassSymbol) type.getSymbol()).getQualifiedName().startsWith(packageName);
                })) {
                    addViolation(data, importDecl);
                }
            } else {
                addViolation(data, importDecl);
            }
        });

        return super.visit(node, data);
    }
    private String getCurrentPackage(ASTCompilationUnit node) {
        ASTPackageDeclaration packageDecl = node.getFirstDescendantOfType(ASTPackageDeclaration.class);
        return packageDecl != null ? packageDecl.getPackageName() : null;
    }

package net.sourceforge.pmd.lang.java.rule.codestyle;

import java.util.HashSet;
import java.util.List;
import java.util.Set;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class UnnecessaryImportRule extends AbstractJavaRulechainRule {
    public UnnecessaryImportRule() {
        super(ASTCompilationUnit.class);
    }

    @Override
    public Object visit(ASTCompilationUnit node, Object data) {
        Set<String> usedImports = new HashSet<>();
        Set<String> implicitImports = new HashSet<>(Arrays.asList("Object", "String", "Integer", "Boolean", "Double", "Float", "Long", "Short", "Character", "Byte", "Void", "Exception", "Error"));
        JSymbolTable symbolTable = node.getSymbolTable();

        // Collect all used imports
        for (ASTImportDeclaration importDeclaration : node.findDescendantsOfType(ASTImportDeclaration.class, false)) {
            if (!importDeclaration.isStatic()) {
                String importName = importDeclaration.getImportedSimpleName();
                if (importName != null) {
                    usedImports.add(importName);
                }
            }
        }

        // Collect all used names
        for (ASTName name : node.findDescendantsOfType(ASTName.class, false)) {
            String nameImage = name.getImage();
            if (!nameImage.contains(".")) {
                usedImports.remove(nameImage);
            }
        }

        // Collect all used static imports
        for (ASTImportDeclaration importDeclaration : node.findDescendantsOfType(ASTImportDeclaration.class, false)) {
            if (importDeclaration.isStatic()) {
                String importName = importDeclaration.getImportedSimpleName();
                if (importName != null) {
                    String fullName = importDeclaration.getImportedName();
                    for (ASTPrimaryExpression expr : node.findDescendantsOfType(ASTPrimaryExpression.class, false)) {
                        if (expr.getFirstDescendantOfType(ASTName.class).getImage().equals(importName)) {
                            usedImports.add(fullName);
                        }
                    }
                }
            }
        }

        // Collect all implicit imports
        ASTPackageDeclaration packageDeclaration = node.getFirstDescendantOfType(ASTPackageDeclaration.class);
        if (packageDeclaration != null) {
            implicitImports.add(packageDeclaration.getPackageNameImage());
        }
        implicitImports.add("java.lang");

        // Collect all used types in the code
        for (JType type : symbolTable.getReferences(JType.class)) {
            if (type != null && type.getTypeKind() == JType.TypeKind.CLASS) {
                JClassSymbol classSymbol = (JClassSymbol) type;
                Scope scope = classSymbol.getScope();
                if (scope instanceof JSymbolTable) {
                    usedImports.add(classSymbol.getSimpleName());
                }
            }
        }

        // Check for unused, duplicated, or implicit imports
        for (ASTImportDeclaration importDeclaration : node.findDescendantsOfType(ASTImportDeclaration.class, false)) {
            String importName = importDeclaration.getImportedName();
            String simpleName = importDeclaration.getImportedSimpleName();

            if (simpleName != null && !usedImports.contains(simpleName) || implicitImports.contains(importName)) {
                addViolation(data, importDeclaration);
            }
        }

        return super.visit(node, data);
    }
}

package net.sourceforge.pmd.lang.java.rule.codestyle;

import java.util.HashSet;
import java.util.List;
import java.util.Set;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class UnnecessaryImportRule extends AbstractJavaRulechainRule {
    public UnnecessaryImportRule() {
        super(ASTCompilationUnit.class);
    }

    @Override
    public Object visit(ASTCompilationUnit node, Object data) {
        Set<String> usedTypes = new HashSet<>();
        Set<String> javaLangTypes = getJavaLangTypes();
        String currentPackage = getCurrentPackage(node);

        node.descendants(ASTName.class)
            .filter(ASTName::isImported)
            .forEach(name -> usedTypes.add(name.getImage()));

        node.descendants(ASTPrimaryPrefix.class)
            .filter(primaryPrefix -> primaryPrefix.getFirstChildOfType(ASTPrimaryExpression.class) != null)
            .forEach(primaryPrefix -> {
                ASTPrimaryExpression expr = primaryPrefix.getFirstChildOfType(ASTPrimaryExpression.class);
                if (expr != null && expr.getType() != null) {
                    usedTypes.add(expr.getType().getTypeName());
                }
            });

        node.descendants(ASTImportDeclaration.class)
            .forEach(importDeclaration -> {
                if (importDeclaration.isStatic()) {
                    return;
                }

                String importName = importDeclaration.getImportedSimpleName();
                String fullName = importDeclaration.getName().getImage();

                if (importName == null) {
                    return;
                }

                if (usedTypes.contains(importName) || javaLangTypes.contains(fullName)) {
                    return;
                }

                if (fullName.startsWith(currentPackage)) {
                    return;
                }

                if (isDuplicated(importDeclaration, node)) {
                    return;
                }

                addViolation(data, importDeclaration);
            });

        return data;
    }

    private Set<String> getJavaLangTypes() {
        Set<String> javaLangTypes = new HashSet<>();
        javaLangTypes.add("java.lang.Object");
        javaLangTypes.add("java.lang.String");

        super(ASTCompilationUnit.class, ASTImportDeclaration.class);
    }

    private String currentPackageName;

    @Override
    public Object visit(ASTCompilationUnit node, Object data) {
        currentPackageName = Optional.ofNullable(node.getFirstChildOfType(ASTPackageDeclaration.class))
                                     .map(ASTPackageDeclaration::getPackageName)
                                     .orElse("");
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTImportDeclaration node, Object data) {
        if (node.isStatic()) {
            return data;
        }

        String importedName = node.getImportedSimpleName();
        String importPackageName = node.getImportedName() != null ? node.getImportedName().getPackageName() : null;
        JSymbolTable symbolTable = node.getSymbolTable();
        TypeSystem typeSystem = symbolTable.getTypeSystem();

            addViolation(data, node);
            return data;

        Set<String> usedClasses = new HashSet<>();
        node.getNthParent(1).descendantsOrSelf(ASTName.class)
                           .filter(ASTName::isFullyQualified)

package net.sourceforge.pmd.lang.java.rule.codestyle;
import java.util.HashSet;
import java.util.List;
import java.util.Set;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class UnnecessaryImportRule extends AbstractJavaRulechainRule {
    public UnnecessaryImportRule() {
        super(ASTCompilationUnit.class);
    }

    @Override
    public Object visit(ASTCompilationUnit node, Object data) {
        Set<String> usedImports = new HashSet<>();
        Set<String> onDemandImports = new HashSet<>();
        JSymbolTable symbolTable = node.getSymbolTable();

        NodeStream<ASTImportDeclaration> imports = node.descendants(ASTImportDeclaration.class);
        for (ASTImportDeclaration imp : imports) {
            String importedName = imp.getImportedName();
            if (imp.isOnDemand()) {
                onDemandImports.add(importedName.substring(0, importedName.lastIndexOf('.')));
            } else {
                usedImports.add(importedName);
            }
        }

        NodeStream<ASTName> names = node.descendants(ASTName.class);
        for (ASTName name : names) {
            String fullName = name.getImage();
            int lastDotIndex = fullName.lastIndexOf('.');
            if (lastDotIndex != -1) {
                String simpleName = fullName.substring(lastDotIndex + 1);
                String packageName = fullName.substring(0, lastDotIndex);

                if (isImplicitlyInScope(simpleName, symbolTable)) {
                    usedImports.remove(fullName);
                    continue;
                }

                if (onDemandImports.contains(packageName)) {
                    usedImports.remove(fullName);
                    continue;
                }

                List<ASTPrimaryExpression> primaryExpressions = name.getFirstParentOfType(ASTPrimaryExpression.class).descendants(ASTPrimaryPrefix.class).filter(prefix -> prefix.getFirstChildOfType(ASTName.class).getImage().equals(simpleName)).collectToList();
                for (ASTPrimaryExpression expr : primaryExpressions) {
                    JType type = expr.getType();
                    if (type != null && type instanceof JClassSymbol) {
                        JClassSymbol classSymbol = (JClassSymbol) type;
                        usedImports.remove(classSymbol.getQualifiedName());
                    }
                }
            }
        }

        for (ASTImportDeclaration imp : imports) {
            String importedName = imp.getImportedName();
            if (!usedImports.contains(importedName)) {
                addViolation(data, imp);
            }
        }

        return super.visit(node, data);
    }

    private boolean isImplicitlyInScope(String name, JSymbolTable symbolTable) {
        TypeSystem typeSystem = symbolTable.getTypeSystem();
        JClassSymbol javaLangClass = typeSystem.lookupClass(name);
        if (javaLangClass != null && javaLangClass.getPackageName().equals("java.lang")) {
            return true;
        }

        ASTCompilationUnit compilationUnit = symbolTable.getScopeRoot();
        ASTPackageDeclaration packageDeclaration = compilationUnit.getFirstChildOfType(ASTPackageDeclaration.class);
        if (packageDeclaration != null) {
            String packageName = packageDeclaration.getPackageName();
            JClassSymbol packageClass = typeSystem.lookupClass(packageName + "." + name);
            if (packageClass != null) {
                return true;
            }
        }

        return false;
    }
