import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class BrokenNullCheckRule extends AbstractJavaRulechainRule {
    public BrokenNullCheckRule() {
        super(ASTBinaryExpression.class);
    }

    @Override
    public Object visit(ASTBinaryExpression node, Object data) {
        if (isLogicalOr(node) || isLogicalAnd(node)) {
            ASTExpression left = node.getLeftOperand();
            ASTExpression right = node.getRightOperand();

            if (isNullLiteral(left) && !isSafeNullCheck(left, right) || 
                isNullLiteral(right) && !isSafeNullCheck(right, left)) {
                addViolation(data, node);
            }
        }

        return super.visit(node, data);
    }

    private boolean isLogicalOr(ASTBinaryExpression node) {
        return node.getOperator() == ASTBinaryExpression.Operator.LOGICAL_OR;
    }

    private boolean isLogicalAnd(ASTBinaryExpression node) {
        return node.getOperator() == ASTBinaryExpression.Operator.LOGICAL_AND;
    }

    private boolean isNullLiteral(ASTExpression expr) {
        return JavaAstUtils.isNullLiteral(expr);
    }

    private boolean isSafeNullCheck(ASTExpression nullLiteral, ASTExpression otherExpr) {
        return isNullChecked(otherExpr) || isNullLiteral(otherExpr);
    }

    private boolean isNullChecked(ASTExpression expr) {
        return JavaRuleUtil.isNullChecked(expr);
    }
}

package net.sourceforge.pmd.lang.java.rule.errorprone;


import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class BrokenNullCheckRule extends AbstractJavaRulechainRule {
    public BrokenNullCheckRule() {
        super(ASTAndExpression.class, ASTOrExpression.class);
    }

    @Override
    public Object visit(ASTAndExpression node, Object data) {
        if (isBrokenNullCheck(node)) {
            addViolation(data, node);
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTOrExpression node, Object data) {
        if (isBrokenNullCheck(node)) {
            addViolation(data, node);
        }
        return super.visit(node, data);
    }

    private boolean isBrokenNullCheck(ASTExpression node) {
        if (!(node instanceof ASTEqualityExpression)) {
            return false;
        }

        ASTEqualityExpression equalityExpression = (ASTEqualityExpression) node;
        boolean isOrExpression = equalityExpression.getParent() instanceof ASTOrExpression;
        boolean isAndExpression = equalityExpression.getParent() instanceof ASTAndExpression;

        ASTExpression leftOperand = equalityExpression.getLeftOperand();
        ASTExpression rightOperand = equalityExpression.getRightOperand();

        boolean leftIsNullLiteral = JavaAstUtils.isNullLiteral(leftOperand);
        boolean rightIsNullLiteral = JavaAstUtils.isNullLiteral(rightOperand);

        if (isOrExpression && (leftIsNullLiteral || rightIsNullLiteral)) {
            return !JavaRuleUtil.isNullChecked(leftOperand) || !JavaRuleUtil.isNullChecked(rightOperand);
        }

        if (isAndExpression && (leftIsNullLiteral || rightIsNullLiteral)) {
            return JavaRuleUtil.isNullChecked(leftOperand) && JavaRuleUtil.isNullChecked(rightOperand);
        }

        return false;
    }
}
