package net.sourceforge.pmd.lang.java.rule.performance;


import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class InefficientEmptyStringCheckRule extends AbstractJavaRulechainRule {
    public InefficientEmptyStringCheckRule() {
        super(ASTPrimaryExpression.class);
    }

    @Override
    public Object visit(ASTPrimaryExpression node, Object data) {
        NodeStream<ASTMethodCall> methodCalls = node.descendants(ASTMethodCall.class);
        methodCalls.forEach(methodCall -> {
            if (isInefficientEmptyCheck(methodCall)) {
                addViolation(data, methodCall);
            }
        });
        return super.visit(node, data);
    }

    private boolean isInefficientEmptyCheck(ASTMethodCall methodCall) {
        ASTPrimaryExpression primaryExpression = (ASTPrimaryExpression) methodCall.jjtGetParent().jjtGetParent();
        NodeStream<ASTMethodCall> nestedMethodCalls = primaryExpression.descendants(ASTMethodCall.class);
        List<String> methodNames = nestedMethodCalls.map(ASTMethodCall::getImage).toList();

        return (methodNames.contains("trim") && methodNames.contains("length"))
            || (methodNames.contains("trim") && methodNames.contains("isEmpty"));
    }
}

    public InefficientEmptyStringCheckRule() {
        super(ASTEqualityExpression.class);
    }

    @Override
    public Object visit(ASTEqualityExpression node, Object data) {
        if (node.getOperator() == ASTEqualityExpression.Operator.EQ || node.getOperator() == ASTEqualityExpression.Operator.NEQ) {
            ASTExpression leftOperand = node.getLeftOperand();
            ASTExpression rightOperand = node.getRightOperand();

            if (rightOperand instanceof ASTLiteral && ((ASTLiteral) rightOperand).getValue().equals(0)) {
                if (isTrimLengthCall(leftOperand)) {
                    addViolation(data, node);
                }
            } else if (rightOperand instanceof ASTLiteral && ((ASTLiteral) rightOperand).getValue().equals(false)) {
                if (isTrimIsEmptyCall(leftOperand)) {
                    addViolation(data, node);
                }
            } else if (leftOperand instanceof ASTLiteral && ((ASTLiteral) leftOperand).getValue().equals(0)) {
                if (isTrimLengthCall(rightOperand)) {
                    addViolation(data, node);
                }
            } else if (leftOperand instanceof ASTLiteral && ((ASTLiteral) leftOperand).getValue().equals(false)) {
                if (isTrimIsEmptyCall(rightOperand)) {
                    addViolation(data, node);
                }
            }
        }
        return super.visit(node, data);
    }

    private boolean isTrimLengthCall(ASTExpression expression) {
        if (expression instanceof ASTPrimaryExpression) {
            ASTPrimaryExpression primaryExpression = (ASTPrimaryExpression) expression;
            ASTPrimaryPrefix prefix = primaryExpression.getFirstChildOfType(ASTPrimaryPrefix.class);
            ASTPrimarySuffix suffix = prefix.getFirstChildOfType(ASTPrimarySuffix.class);

            if (suffix != null && suffix.getImage().equals(".trim")) {
                ASTPrimarySuffix nextSuffix = suffix.getNextSiblingOfType(ASTPrimarySuffix.class);
                if (nextSuffix != null && nextSuffix.getImage().equals(".length")) {
                    return true;
                }
            }
        }
        return false;
    }

    private boolean isTrimIsEmptyCall(ASTExpression expression) {
        if (expression instanceof ASTPrimaryExpression) {
            ASTPrimaryExpression primaryExpression = (ASTPrimaryExpression) expression;
            ASTPrimaryPrefix prefix = primaryExpression.getFirstChildOfType(ASTPrimaryPrefix.class);

            if (suffix != null && suffix.getImage().equals(".trim")) {

                if (nextSuffix != null && nextSuffix.getImage().equals(".isEmpty")) {
                    return true;
                }
            }
        }
        return false;
    }

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;

import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;

public class InefficientEmptyStringCheckRule extends AbstractJavaRulechainRule {
    public InefficientEmptyStringCheckRule() {
        super(ASTEqualityExpression.class);
    }

    @Override
    public Object visit(ASTEqualityExpression node, Object data) {
        if (node.getNumChildren() != 2) {
            return super.visit(node, data);
        }

        ASTExpression leftOperand = node.getChild(0);
        ASTExpression rightOperand = node.getChild(1);

        if (isInefficientCheck(leftOperand, rightOperand) || isInefficientCheck(rightOperand, leftOperand)) {
            addViolation(data, node);
        }

        return super.visit(node, data);
    }

    private boolean isInefficientCheck(ASTExpression operand1, ASTExpression operand2) {
        if (!(operand1 instanceof ASTPrimaryExpression) || !(operand2 instanceof ASTLiteral)) {
            return false;
        }

        ASTPrimaryExpression primaryExpression = (ASTPrimaryExpression) operand1;
        ASTLiteral literal = (ASTLiteral) operand2;

        if (!literal.isIntegerLiteral() || !literal.getImage().equals("0")) {
            return false;
        }

        ASTPrimaryPrefix primaryPrefix = primaryExpression.getFirstChildOfType(ASTPrimaryPrefix.class);
        if (primaryPrefix == null) {
            return false;
        }

        ASTPrimarySuffix primarySuffix = primaryPrefix.getFirstChildOfType(ASTPrimarySuffix.class);
        if (primarySuffix == null || !primarySuffix.getImage().equals("length")) {
            return false;
        }

        ASTPrimaryExpression trimExpression = primaryPrefix.getFirstChildOfType(ASTPrimaryExpression.class);
        if (trimExpression == null || trimExpression.getNumChildren() != 2) {
            return false;
        }

        ASTPrimaryPrefix trimPrefix = trimExpression.getFirstChildOfType(ASTPrimaryPrefix.class);
        if (trimPrefix == null) {
            return false;
        }

        ASTPrimarySuffix trimSuffix = trimPrefix.getFirstChildOfType(ASTPrimarySuffix.class);
        if (trimSuffix == null || !trimSuffix.getImage().equals("trim")) {
            return false;
        }

        ASTExpression stringExpression = trimPrefix.getFirstChildOfType(ASTExpression.class);
        if (stringExpression == null) {
            return false;
        }

        JType type = stringExpression.getType();
        return type != null && type.equalsTo(TypeSystem.STRING);
    }
