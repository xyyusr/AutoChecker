import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class AssignmentToNonFinalStaticRule extends AbstractJavaRulechainRule {

    public AssignmentToNonFinalStaticRule() {
        addRuleChainVisit(ASTConstructorDeclaration.class);
    }

    @Override
    public Object visit(ASTConstructorDeclaration node, Object data) {
        node.findDescendantsOfType(ASTAssignmentOperator.class).forEach(assignNode -> {
            Node leftHandSideNode = assignNode.jjtGetChild(0);
            if (leftHandSideNode instanceof ASTPrimaryExpression) {
                ASTPrimaryExpression primaryExpression = (ASTPrimaryExpression) leftHandSideNode;
                if (primaryExpression.jjtGetNumChildren() > 0 && primaryExpression.jjtGetChild(0) instanceof ASTPrimaryPrefix) {
                    ASTPrimaryPrefix primaryPrefix = (ASTPrimaryPrefix) primaryExpression.jjtGetChild(0);
                    if (primaryPrefix.jjtGetNumChildren() > 0 && primaryPrefix.jjtGetChild(0) instanceof ASTName) {
                        ASTName name = (ASTName) primaryPrefix.jjtGetChild(0);
                        if (name.getType() != null && name.getType().isField() && !name.getType().isFinal() && name.getType().isStatic()) {
                            addViolation(data, name);
                        }
                    }
                }
            }
        });
        return super.visit(node, data);
    }
}

    private void checkAssignment(ASTAssignmentOperator assignment, Object data) {

            ASTStatementExpression statement = (ASTStatementExpression) parentNode;
            Node potentialField = statement.jjtGetChild(0);

                if (potentialPrefix instanceof ASTPrimaryPrefix) {
                    ASTPrimaryPrefix primaryPrefix = (ASTPrimaryPrefix) potentialPrefix;
                    if (primaryPrefix.usesThisModifier()) {
                        reportViolationIfNonFinalStatic(primaryPrefix, data);
                    }
                }
            }
        }
    }

    private void reportViolationIfNonFinalStatic(ASTPrimaryPrefix primaryPrefix, Object data) {
        Node potentialField = primaryPrefix.jjtGetChild(0);
        if (potentialField instanceof ASTName) {
            ASTName name = (ASTName) potentialField;
            if (name.getType() != null && name.getType().isField()) {
                ASTFieldDeclaration field = (ASTFieldDeclaration) name.getTypeDeclaration().getNode();
                if (field.isStatic() && !field.isFinal()) {
                    addViolation(data, name);
                }
            }
        }
    }
}
