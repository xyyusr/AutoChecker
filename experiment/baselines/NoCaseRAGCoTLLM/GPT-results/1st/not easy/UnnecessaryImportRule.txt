public class UnnecessaryImportRule extends AbstractJavaRulechainRule {

    public UnnecessaryImportRule() {
        addRuleChainVisit(ASTCompilationUnit.class);
    }

    @Override
    public Object visit(ASTCompilationUnit node, Object data) {
        for (ASTImportDeclaration importDeclaration : node.findDescendantsOfType(ASTImportDeclaration.class)) {
            if (importDeclaration.isStatic()) {
                continue;
            }

            String importedName = importDeclaration.getImportedName();
            if (importedName.startsWith("java.lang.") || importedName.startsWith(node.getPackage().getPackageName() + ".")) {
                addViolation(data, importDeclaration);
                continue;
            }

            if (!isUsed(importDeclaration, node)) {
                addViolation(data, importDeclaration);
                continue;
            }

            if (isDuplicate(importDeclaration, node)) {
                addViolation(data, importDeclaration);
            }
        }

        return super.visit(node, data);
    }

    private boolean isUsed(ASTImportDeclaration importDeclaration, ASTCompilationUnit node) {
        // Implement logic to check if the imported class or package is used in the code.
        // This may involve traversing the AST and checking if any identifiers match the import.
        // This is left as an exercise for the reader.
        return false;
    }

    private boolean isDuplicate(ASTImportDeclaration importDeclaration, ASTCompilationUnit node) {
        for (ASTImportDeclaration otherImportDeclaration : node.findDescendantsOfType(ASTImportDeclaration.class)) {

package net.sourceforge.pmd.lang.java.rule.codestyle;


import java.util.HashSet;
import java.util.Set;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class UnnecessaryImportRule extends AbstractJavaRulechainRule {
    private Set<String> imports = new HashSet<>();

    public UnnecessaryImportRule() {
        addRuleChainVisit(ASTCompilationUnit.class);
    }

    @Override
    public Object visit(ASTCompilationUnit node, Object data) {
        imports.clear();
        super.visit(node, data);
        return data;
    }

    @Override
    public Object visit(ASTImportDeclaration node, Object data) {
        String packageName = node.getPackageName();
        String importName = node.getImportedName();

        if (packageName.equals("java.lang") || packageName.equals(node.getScope().getEnclosingSourceFileScope().getPackageName())) {
            addViolation(data, node);
        } else if (imports.contains(importName)) {
            addViolation(data, node);
        } else if (!isUsed(importName, node.jjtGetParent())) {
            addViolation(data, node);
        } else {
            imports.add(importName);
        }

        return data;
    }

    private boolean isUsed(String importName, Node node) {
        for (int i = 0; i < node.jjtGetNumChildren(); i++) {
            Node child = node.jjtGetChild(i);
            if (child.getImage() != null && child.getImage().equals(importName)) {
                return true;
            }

            if (isUsed(importName, child)) {
                return true;
            }
        }
        return false;
    }
}

package net.sourceforge.pmd.lang.java.rule.codestyle;
import java.util.List;
import java.util.ArrayList;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class UnnecessaryImportRule extends AbstractJavaRulechainRule {
    private List<String> imports;

    public UnnecessaryImportRule() {
        super(ASTImportDeclaration.class);
        imports = new ArrayList<>();
    }

    @Override
    public Object visit(ASTImportDeclaration node, Object data) {
        String importedName = node.getImportedName();
        if (isJavaLangOrCurrentPackage(importedName) || isStaticAndUnused(node) || isDuplicate(importedName)) {
            addViolation(data, node);
        }
        return super.visit(node, data);
    }

    private boolean isJavaLangOrCurrentPackage(String importedName) {
        return importedName.startsWith("java.lang.") || isInCurrentPackage(importedName);
    }

    private boolean isInCurrentPackage(String importedName) {
        Node node = getRootNode();
        if (node instanceof ASTCompilationUnit) {
            ASTPackageDeclaration pkg = ((ASTCompilationUnit) node).getPackageDeclaration();
            return pkg != null && importedName.startsWith(pkg.getPackageName());
        }
        return false;
    }

    private boolean isStaticAndUnused(ASTImportDeclaration node) {
        return node.isStatic() && node.getUsages().isEmpty();
    }

    private boolean isDuplicate(String importedName) {
        boolean isDuplicate = imports.contains(importedName);
        imports.add(importedName);
        return isDuplicate;
    }
}
