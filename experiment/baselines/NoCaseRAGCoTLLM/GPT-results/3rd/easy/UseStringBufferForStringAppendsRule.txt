        super(ASTLocalVariableDeclaration.class);

                        if (count > 1) {
                            addViolation(data, assign);
                        }
                    }
                });
            }
        }
        return super.visit(node, data);

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class UseStringBufferForStringAppendsRule extends AbstractJavaRulechainRule {
    
    public UseStringBufferForStringAppendsRule() {
        addRuleChainVisit(ASTAssignmentOperator.class);
    }

    @Override
    public Object visit(ASTAssignmentOperator node, Object data) {
        if (!"+=".equals(node.getImage())) {
            return super.visit(node, data);
        }

        ASTStatementExpression statement = node.getFirstParentOfType(ASTStatementExpression.class);
        if (statement == null) {
            return super.visit(node, data);
        }

        ASTName leftSide = statement.getFirstDescendantOfType(ASTName.class);
        if (leftSide == null || !TypeTestUtil.isA(String.class, leftSide)) {
            return super.visit(node, data);
        }

        ASTExpression rightSide = statement.getFirstDescendantOfType(ASTExpression.class);
        if (rightSide != null && rightSide.isStringConcatenation()) {
            addViolation(data, node);
        }

        return super.visit(node, data);
    }
}
