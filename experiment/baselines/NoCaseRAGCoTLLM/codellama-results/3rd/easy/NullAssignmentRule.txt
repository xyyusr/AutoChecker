package net.sourceforge.pmd.lang.java.rule.errorprone;

import java.util.List;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class NullAssignmentRule extends AbstractJavaRulechainRule {

    public void apply(List<? extends JavaNode> nodes, JavaRuleContext context) {
        for (JavaNode node : nodes) {
            if (node instanceof ASTAssignment) {
                ASTAssignment assignment = (ASTAssignment) node;
                ASTVariableDeclarator declarator = (ASTVariableDeclarator) assignment.getFirstChild();
                if (declarator != null && declarator.getFirstChild() instanceof ASTVariableDeclaratorId) {
                    ASTVariableDeclaratorId declaratorId = (ASTVariableDeclaratorId) declarator.getFirstChild();
                    ASTExpression expression = assignment.getLastChild();
                    if (expression instanceof ASTNullLiteral && !JavaRuleUtil.isNullCheck(expression, declaratorId.getSymbol())) {
                        reportRuleViolation(context, node);
                    }
                }
            } else if (node instanceof ASTClassOrInterfaceBody) {
                ASTClassOrInterfaceBody classBody = (ASTClassOrInterfaceBody) node;
                ASTVariableDeclarator declarator = (ASTVariableDeclarator) classBody.getFirstChild();
                if (declarator != null && declarator.getFirstChild() instanceof ASTVariableDeclaratorId) {
                    ASTVariableDeclaratorId declaratorId = (ASTVariableDeclaratorId) declarator.getFirstChild();
                    ASTExpression expression = classBody.getLastChild();
                    if (expression instanceof ASTNullLiteral && !JavaRuleUtil.isNullCheck(expression, declaratorId.getSymbol())) {
                        reportRuleViolation(context, node);
                    }
                }
            }
        }
    }
}

package net.sourceforge.pmd.lang.java.rule.errorprone;


import java.util.Iterator;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class NullAssignmentRule extends AbstractJavaRulechainRule {

    @Override
    public Object visit(ASTAssignment node, Object data) {
        if (node.getOp() == ASTAssignment.OP_EQ) {
            ASTExpression target = node.getTarget();
            if (target instanceof ASTVariableDeclaratorId) {
                JVariableSymbol variable = SymbolTable.getVariable(((ASTVariableDeclaratorId) target).getImage());
                if (variable != null && !isNullCheck(variable, node.getAssignment())) {
                    addViolation(this, data, node);
                }
            }
        }
        return super.visit(node, data);
    }

    private boolean isNullCheck(JVariableSymbol variable, ASTExpression expression) {
        if (!variable.isAssigned()) {
            return false;
        }
        Iterator<ASTExpression> children = expression.children();
        while (children.hasNext()) {
            ASTExpression child = children.next();
            if (child instanceof ASTNullLiteral && ASTNullLiteral.singleOrNull(child.children()) == null) {
                return true;
            }
        }
        return false;
    }
}
