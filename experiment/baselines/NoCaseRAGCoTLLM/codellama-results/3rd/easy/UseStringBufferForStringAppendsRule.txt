        if (result == null) {
            return result;
        }

        }
        return result;
    }

        return JavaAstUtils.isStringConcatExpr(expression);
    }

        ASTExpression parent = getParentWithExplicitStringBuffer(expression);
        return parent != null;
    }

    private ASTExpression getParentWithExplicitStringBuffer(ASTExpression expression) {
        ASTNode parentNode = getParent(expression);
        ASTExpression parentExpression;
        while (parentNode != null) {
            if (parentNode instanceof ASTExpression) {
                parentExpression = (ASTExpression) parentNode;
                if (JavaRuleUtil.isStringBuilderCtorOrAppend(parentExpression)) {
                    return parentExpression;
                }
            }
            parentNode = getParent(parentNode);
        }
        return null;
    }

        ASTExpression parent;
        do {
            parent = getParent(expression);
            if (parent == null) {

                if (JavaRuleUtil.isStringBuilderCtorOrAppend(expressionParent)) {
                    return true;
                }
            }
            expression = parent;
        } while (parent != null);
        return false;
    }

package net.sourceforge.pmd.lang.java.rule.performance;

import java.util.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class UseStringBufferForStringAppendsRule extends AbstractJavaRulechainRule {

    public UseStringBufferForStringAppendsRule() {
        addRuleMatch("expression",
                new Object[] { "ASTStringLiteral" });
    }

    @Override
    public Object visit(ASTStringLiteral node, Object data) {
        if (!node.isCompileTimeConstant()) {
            return super.visit(node, data);
        }

        ASTExpression context = (ASTExpression) node.getEnclosingType();
        if (context == null) {
            context = node.getParent();
        }

        ASTExpression appendExpression = null;
        if (context instanceof ASTExpressionList) {
            ASTExpressionList list = (ASTExpressionList) context;
            ASTExpression appendExpr = JavaAstUtils.isReferenceToSameVar(list.getFirstChild(0), node) ? list.getFirstChild(1) : null;
            if (appendExpr != null) {
                appendExpression = appendExpr;
            }
        } else if (context instanceof ASTStatement) {
            ASTStatement statement = (ASTStatement) context;
            ASTExpressionList list = statement.getStatementExpressionList();
            ASTExpression appendExpr = JavaAstUtils.isReferenceToSameVar(list.getFirstChild(0), node) ? list.getFirstChild(1) : null;
            if (appendExpr != null) {
                appendExpression = appendExpr;
            }
        }

        if (appendExpression != null) {
            ASTMethod call = (ASTMethod) appendExpression.getFirstChild(0);
            if (call != null && call.getName().getImage().equals("+=")) {
                ASTType type = call.getType();
                if (type != null && (type.getImage().equals("java.lang.StringBuffer") || type.getImage().equals("java.lang.StringBuilder"))) {
                    ASTMethod appendMethod = (ASTMethod) call.getFirstChild(1);
                    if (appendMethod.getImage().equals("append")) {
                        return super.visit(node, data);
                    }
                }
            }
        }

        return super.visit(node, data);
    }
}
