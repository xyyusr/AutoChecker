import org.apache.commons.lang3.StringUtils;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class LiteralsFirstInComparisonsRule extends AbstractJavaRulechainRule {

    public LiteralsFirstInComparisonsRule() {
        addRuleMessage("Position literals first in all String comparison methods.");
        addRulePriority(PMDRulePriority.MAX);
        addRuleCategory("Best Practices");
        addRuleDescription("Position literals first in all String comparison methods.");
        addRuleExample("should be", "should not be");
        addRuleExample("Method should be foo(String a, String b) { return a.equals(b); }", "Method should not be foo(String a, String b) { return b.equals(a); }");
        addRuleExample("Method should be foo(String a, String b) { return a.compareTo(b); }", "Method should not be foo(String a, String b) { return b.compareTo(a); }");
    }

    @Override
    public Object visit(MethodCallNode node, Object data) {
        if (node != null && (isEqualsMethod(node) || isCompareToMethod(node) || isEqualsIgnoreCaseMethod(node) || isCompareToIgnoreCaseMethod(node) || isContentEqualsMethod(node))) {
            String methodName = node.getImage();
            String[] methodSignature = methodName.split("\\(");
            if (methodSignature.length > 1 && methodSignature[1].startsWith("java.lang.String")) {
                ASTNode args = node.getChildren().findFirst().get();
                if (args != null && args instanceof ASTArrayInitializer) {
                    List<ASTNode> children = ((ASTArrayInitializer) args).getChildren();
                    if (children != null && children.size() > 0) {
                        ASTNode arg = children.get(0);
                        if (arg != null && (arg instanceof ASTStringLiteral || arg instanceof ASTNullLiteral || arg instanceof ASTCharLiteral || arg instanceof ASTBooleanLiteral)) {
                            if (arg instanceof ASTNullLiteral) {
                                if (arg.getPreviousSibling() != null && arg.getPreviousSibling() instanceof ASTStringLiteral) {
                                    ast.addRuleBreaking(node);
                                }
                            } else {
                                if (arg.getNextSibling() != null && arg.getNextSibling() instanceof ASTNullLiteral) {
                                    ast.addRuleBreaking(node);
                                }
                            }
                        }
                    }
                }
            }
        }
        return super.visit(node, data);
    }

    private boolean isEqualsMethod(MethodCallNode node) {
        return node.getImage().equals("equals") && node.getChildren().get(0) != null && node.getChildren().get(0) instanceof ASTNullLiteral;
    }

    private boolean isCompareToMethod(MethodCallNode node) {
        return node.getImage().equals("compareTo");
    }

    private boolean isEqualsIgnoreCaseMethod(MethodCallNode node) {
        return node.getImage().equals("equalsIgnoreCase");
    }

    private boolean isCompareToIgnoreCaseMethod(MethodCallNode node) {
        return node.getImage().equals("compareToIgnoreCase");
    }

    private boolean isContentEqualsMethod(MethodCallNode node) {
        return node.getImage().equals("contentEquals");
    }
}
