public class UnnecessaryImportRule extends AbstractJavaRulechainRule {

    public UnnecessaryImportRule() {
        super(ASTImportDeclaration.class);
    }

    @Override
    public Object visit(ASTImportDeclaration importDeclaration, Object data) {
        if (importDeclaration.isStatic()) {
            if (importDeclaration.getSymbolTable().isMemberPresent(importDeclaration.getImportedSimpleName())) {
                return false;
            }
        }

        if (importDeclaration.findDescendantsOfType(ASTImportDeclaration.class).size() > 1) {
            return false;
        }

        if (importDeclaration.getImportedSimpleName().equals("java.lang")) {
            return false;
        }

        if (importDeclaration.findDescendantsOfType(ASTClassOrInterfaceDeclaration.class).size() > 0) {
            return false;
        }

        if (importDeclaration.findDescendantsOfType(ASTMethodDeclaration.class).size() > 0) {
            return false;
        }

        if (importDeclaration.getSymbolTable().isMemberPresent(importDeclaration.getImportedSimpleName())) {
            return false;
        }

        ASTPackageDeclaration packageDeclaration = (ASTPackageDeclaration) importDeclaration.getParent();
        if (packageDeclaration.getSymbolTable().isMemberPresent(importDeclaration.getImportedSimpleName())) {
            return false;
        }

        List<ASTClassOrInterfaceDeclaration> classesOrInterfaces = importDeclaration.getParent().getDeclaredClassesOrInterfaces();
        for (ASTClassOrInterfaceDeclaration classOrInterface : classesOrInterfaces) {
            if (importDeclaration.getSymbolTable().isMemberPresent(classOrInterface.getName(), importDeclaration.getImportedSimpleName())) {
                return false;
            }
        }

        return true;
    }

package net.sourceforge.pmd.lang.java.rule.codestyle;

import java.util.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class UnnecessaryImportRule extends AbstractJavaRulechainRule {

    public UnnecessaryImportRule() {
        addRuleChainElement(new ASTImportDeclaration());
    }

    @Override
    public Object visit(ASTImportDeclaration node, Object data) {
        if (isImportUnnecessary(node)) {
            addViolation(data, node);
        }
        return super.visit(node, data);
    }

    private boolean isImportUnnecessary(ASTImportDeclaration node) {
        if (node.isStatic()) {
            return isStaticImportUnnecessary(node);
        }
        return isDynamicImportUnnecessary(node);
    }

    private boolean isStaticImportUnnecessary(ASTImportDeclaration node) {
        String importedSimpleName = node.getImportedSimpleName();
        if (importedSimpleName == null) {
            return false;
        }
        return isMemberAlreadyInScope(importedSimpleName);
    }

    private boolean isDynamicImportUnnecessary(ASTImportDeclaration node) {
        List<String> importedFullNames = node.findChildrenOfType(ASTImportEntry.class).stream()
                .map(ASTImportEntry::getImportedFullname)
                .collect(Collectors.toList());
        return importedFullNames.stream().allMatch(this::isMemberAlreadyInScope);
    }

    private boolean isMemberAlreadyInScope(String memberName) {
        try {
            JavaSourceUtil.findMember(java.lang.Class.forName("java.lang"), memberName);
            return true;
        } catch (ClassNotFoundException e) {
            // ignore
        }
        return false;
    }

    private boolean isMemberAlreadyInScope(String memberName, Class<?> clazz) {
        try {
            JavaSourceUtil.findMember(clazz, memberName);
            return true;
        } catch (ClassNotFoundException e) {
            // ignore
        }
        return false;
    }
}

package net.sourceforge.pmd.lang.java.rule.codestyle;

import java.util.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class UnnecessaryImportRule extends AbstractJavaRulechainRule {

    @Override
    public Object visit(ASTImportDeclaration node, Object data) {
        Set<String> usedNames = new HashSet<>();
        Set<String> importedNames = new HashSet<>();
        Set<String> staticImportedNames = new HashSet<>();

        if (node.isStatic()) {
            staticImportedNames.add(node.getImportedSimpleName());
        } else {
            importedNames.add(node.getImportedSimpleName());
        }

        ASTPackageDeclaration packageDecl = (ASTPackageDeclaration) node.getParent();
        if (packageDecl != null) {
            for (ASTImportDeclaration impDecl : node.findChildrenOfType(ASTImportDeclaration.class)) {
                if (impDecl.isStatic()) {
                    staticImportedNames.add(impDecl.getImportedSimpleName());
                } else {
                    importedNames.add(impDecl.getImportedSimpleName());
                }
            }

            for (ASTMemberDeclaration member : packageDecl.getDeclaredTypes()) {
                if (member instanceof ASTClassOrInterfaceDeclaration) {
                    for (ASTMethodDeclaration method : member.getDeclaredMethods()) {
                        usedNames.add(method.getName());
                    }
                }
            }
        }

        usedNames.addAll(importedNames);
        usedNames.addAll(staticImportedNames);

        for (ASTImportDeclaration impDecl : node.findChildrenOfType(ASTImportDeclaration.class)) {
            if (impDecl.isStatic()) {
                if (!staticImportedNames.contains(impDecl.getImportedSimpleName())
                        && !usedNames.contains(impDecl.getImportedSimpleName())
                        && !impDecl.isFindBoundary()) {
                    addViolation(this, impDecl.getReportLocation(), "Unnecessary static import");
                }
            } else {
                if (!importedNames.contains(impDecl.getImportedSimpleName())
                        && !usedNames.contains(impDecl.getImportedSimpleName())
                        && !impDecl.isFindBoundary()) {
                    addViolation(this, impDecl.getReportLocation(), "Unnecessary import");
                }
            }
        }

        return super.visit(node, data);
    }
}

package net.sourceforge.pmd.lang.java.rule.codestyle;

import java.util.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class UnnecessaryImportRule extends AbstractJavaRulechainRule {

    public UnnecessaryImportRule() {
        super(ASTImportDeclaration.class);
    }

    @Override
    public Object visit(ASTImportDeclaration node, Object data) {
        if (node.isStatic()) {
            return super.visit(node, data);
        }
        ASTCompilationUnit compilationUnit = (ASTCompilationUnit) node.getParent();
        if (compilationUnit == null) {
            return super.visit(node, data);
        }
        ASTPackageDeclaration packageDeclaration = compilationUnit.getPackageDeclaration();
        if (packageDeclaration == null) {
            return super.visit(node, data);
        }
        String packageName = packageDeclaration.getImage();
        String importedSimpleName = node.getImportedSimpleName();
        String importName = node.getImage();
        boolean isImplicitlyAvailable = false;
        if (importName.equals("java.lang." + importedSimpleName) || importName.equals(packageName + "." + importedSimpleName)) {
            isImplicitlyAvailable = true;
        }
        boolean isDuplicated = false;
        for (ASTImportDeclaration otherImport : compilationUnit.getImports()) {
            if (otherImport.equals(node)) {
                continue;
            }
            if (otherImport.getImportedSimpleName().equals(importedSimpleName) && otherImport.isStatic() == node.isStatic()) {
                isDuplicated = true;
                break;
            }
        }
        if (isImplicitlyAvailable || isDuplicated) {
            addViolation(this, node, "Unused import statement");
        } else {
            ASTCompilationUnit unit = (ASTCompilationUnit) node.getCompilationUnit();
            ASTMethodDeclaration method = firstChild(unit, ASTMethodDeclaration.class);
            if (method != null) {
                for (ASTVariableDeclaratorId declaratorId : method.getParameterList().getDeclaratorIds()) {
                    if (declaratorId.getImage().equals(importedSimpleName)) {
                        addViolation(this, node, "Unused import statement");
                        break;
                    }
                }
            }
        }
        return super.visit(node, data);
    }
}
