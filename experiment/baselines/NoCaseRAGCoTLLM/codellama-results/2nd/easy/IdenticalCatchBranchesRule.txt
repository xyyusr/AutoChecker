import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class IdenticalCatchBranchesRule extends AbstractJavaRulechainRule {

    public IdenticalCatchBranchesRule() {
        super(CatchClause.class, CatchClause.class);
    }

    @Override
    public Object visit(ASTCatchClause node, Object data) {
        Object result = super.visit(node, data);
        if (result instanceof List) {
            List<ASTCatchClause> catchClauses = (List<ASTCatchClause>) result;
            for (int i = 0; i < catchClauses.size(); i++) {
                for (int j = i + 1; j < catchClauses.size(); j++) {
                    if (isIdenticalCatchBranches(catchClauses.get(i), catchClauses.get(j))) {
                        addRuleViolation(node, "Identical catch branches use up vertical space and increase the complexity of code without adding functionality. It’s better style to collapse identical branches into a single multi-catch branch.");
                    }
                }
            }
        }
        return result;
    }

    private boolean isIdenticalCatchBranches(ASTCatchClause node1, ASTCatchClause node2) {
        if (node1 == null || node2 == null) {
            return false;
        }
        ASTCatchParameter catchParameter1 = node1.getCatchParameter();
        ASTCatchParameter catchParameter2 = node2.getCatchParameter();
        if (catchParameter1 == null || catchParameter2 == null) {
            return false;
        }
        if (!catchParameter1.isMulticatch() && !catchParameter2.isMulticatch()) {
            return false;
        }
        NodeStream<JavaNode> children1 = catchParameter1.children();
        NodeStream<JavaNode> children2 = catchParameter2.children();
        while (children1.hasNext() && children2.hasNext()) {
            JavaNode child1 = children1.next();
            JavaNode child2 = children2.next();
            if (child1 != null && child2 != null && child1.equals(child2)) {
                continue;
            } else {
                return false;
            }
        }
        return true;
    }

    @Override
    public Object visit(ASTCatchParameter node, Object data) {
        return super.visit(node, data);
    }
}

        List<ASTCatchParameter> catchParameters = new ArrayList<>();
        for (Object child : node.jjtGetChildren()) {
            if (child instanceof ASTCatchParameter) {
                catchParameters.add((ASTCatchParameter) child);
            }
        }
        if (catchParameters.size() > 1) {
            // ignore non-identical catch branches
            return super.visit(node, data);
        }

        List<ASTCatchClause> identicalCatchClauses = new ArrayList<>();

                        identicalCatchClauses.add(currentCatchClause);

                        seenCatchParameters.add(catchParameter);
