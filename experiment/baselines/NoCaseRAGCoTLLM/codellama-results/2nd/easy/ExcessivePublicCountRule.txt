package net.sourceforge.pmd.lang.java.rule.design;

import java.util.Set;
import java.util.HashSet;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ExcessivePublicCountRule extends AbstractJavaRulechainRule {

    @Override
    public Object visit(ASTCompilationUnit node, Object data) {
        int publicMethodCount = 0;
        int publicFieldCount = 0;
        Set<ASTMethodDeclaration> visitedMethods = new HashSet<>();
        Set<ASTFieldDeclaration> visitedFields = new HashSet<>();

        for (ASTClassOrInterfaceDeclaration classOrInterfaceDeclaration : node.findDescendants(ASTClassOrInterfaceDeclaration.class)) {
            if (visitedMethodsWithClass.contains(classOrInterfaceDeclaration)) {
                continue;
            }

            for (ASTMethodDeclaration methodDeclaration : classOrInterfaceDeclaration.findDescendants(ASTMethodDeclaration.class)) {
                if (methodDeclaration.isSyntacticallyPublic() && !visitedMethods.contains(methodDeclaration)) {
                    visitedMethods.add(methodDeclaration);
                    publicMethodCount++;
                }
            }

            for (ASTFieldDeclaration fieldDeclaration : classOrInterfaceDeclaration.findDescendants(ASTFieldDeclaration.class)) {
                if (fieldDeclaration.isSyntacticallyPublic() && !visitedFields.contains(fieldDeclaration)) {
                    visitedFields.add(fieldDeclaration);
                    publicFieldCount++;
                }
            }

            visitedMethodsWithClass.add(classOrInterfaceDeclaration);
        }

        if (publicMethodCount > 2 || publicFieldCount > 2) {
            addRuleViolation(node, "Classes with large numbers of public methods and attributes require disproportionate testing efforts since combinational side effects grow rapidly and increase risk.");
        }

        return super.visit(node, data);
    }

    private Set<ASTClassOrInterfaceDeclaration> visitedMethodsWithClass = new HashSet<>();

    @Override
    public Object visit(ASTMethodDeclaration node, Object data) {
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTFieldDeclaration node, Object data) {
        return super.visit(node, data);
    }
}

package net.sourceforge.pmd.lang.java.rule.design;


import java.util.HashSet;
import java.util.Set;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ExcessivePublicCountRule extends AbstractJavaRulechainRule {

    @Override
    public Object visit(ASTClassOrInterfaceDeclaration node, Object data) {
        if (!node.isSyntacticallyPublic()) {
            return data;
        }

        int methodCount = 0;
        int fieldCount = 0;
        Set<VariableSymbol> localVarsToTrack = new HashSet<>();

        for (ASTMethodDeclaration method : node.getChildNodesByType(ASTMethodDeclaration.class)) {
            if (method.isSyntacticallyPublic()) {
                methodCount++;
                localVarsToTrack.addAll(SymbolTableBuilder.buildForMethod(method).getVariables());
            }
        }

        for (ASTFieldDeclaration field : node.getChildNodesByType(ASTFieldDeclaration.class)) {
            if (field.isSyntacticallyPublic()) {
                fieldCount++;
                localVarsToTrack.add(SymbolTableBuilder.buildForField(field).getVariables().iterator().next());
            }
        }

        if (methodCount + fieldCount > 2) {
            addViolation(this, data, node);
        }

        return data;
    }
}

package net.sourceforge.pmd.lang.java.rule.design;

import java.util.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ExcessivePublicCountRule extends AbstractJavaRulechainRule {

    public ExcessivePublicCountRule() {
        super(ClassOrInterfaceDeclaration.class);
    }

    @Override
    public Object visit(ClassOrInterfaceDeclaration node, Object data) {
        int publicMethods = 0;
        int publicFields = 0;
        int totalMethods = 0;
        int totalFields = 0;

        totalMethods += node.getMethods().size();
        totalFields += node.getFields().size();

        for (MethodDeclaration method : node.getMethods()) {
            if (method.isSyntacticallyPublic()) {
                publicMethods++;
            }
        }

        for (FieldDeclaration field : node.getFields()) {
            if (field.isSyntacticallyPublic()) {
                publicFields++;
            }
        }

        int excessivePublicCount = publicMethods + publicFields - totalMethods - totalFields;
        if (excessivePublicCount > 2) {
            addViolation(this, data, node);
        }

        return super.visit(node, data);
    }
}

public class ExcessivePublicCountRule extends AbstractJavaRulechainRule {

    @Override
    public Object visit(ASTClassOrInterfaceDeclaration node, Object data) {
        int publicMethods = 0;
        int publicFields = 0;
        int totalMethods = 0;
        int totalFields = 0;

        InternalApiBridge.forceTypeResolutionPhase(node.getCompilationUnit(), node.getAST());

        for (ASTMethodDeclaration method : node.getMethods()) {
            if (method.isSyntacticallyPublic()) {
                publicMethods++;
            }
            totalMethods++;
        }

        for (ASTFieldDeclaration field : node.getFields()) {
            if (field.isSyntacticallyPublic()) {
                publicFields++;
            }
            totalFields++;
        }

        if (totalMethods > 2 || totalFields > 2) {
            addRuleViolation(this, node);
        }

        return super.visit(node, data);
    }

        return super.visit(node, data);

    }
