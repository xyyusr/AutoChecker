package net.sourceforge.pmd.lang.java.rule.codestyle;

import java.util.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class MethodNamingConventionsRule extends AbstractJavaRulechainRule {

    @Override
    public Object visit(ASTMethodDeclaration node, Object data) {
        String methodName = node.getName();
        if (isJUnitMethod(node)) {
            violateRule(node, "Method name does not match JUnit naming convention");
        } else {
            if (!matchesRegex(methodName)) {
                violateRule(node, "Method name does not match standard Java naming convention");
            }
        }
        return super.visit(node, data);
    }

    private boolean isJUnitMethod(ASTMethodDeclaration node) {
        return isJUnit4Method(node) || isJUnit5Method(node) || isJUnit3Method(node);
    }

    private boolean isJUnit4Method(ASTMethodDeclaration node) {
        return TestFrameworksUtil.isJUnit4Method(node);
    }

    private boolean isJUnit5Method(ASTMethodDeclaration node) {
        return TestFrameworksUtil.isJUnit5Method(node);
    }

    private boolean isJUnit3Method(ASTMethodDeclaration node) {
        return TestFrameworksUtil.isJUnit3Method(node);
    }

    private boolean matchesRegex(String methodName) {
        if (isNativeMethod(node)) {
            return methodName.matches("[a-z][a-zA-Z0-9]*");
        } else if (isStaticMethod(node)) {
            return methodName.matches("[a-z][a-zA-Z0-9]*");
        } else {
            return methodName.matches("[a-z][a-zA-Z0-9]*");
        }
    }

    private boolean isNativeMethod(ASTMethodDeclaration node) {
        return node.getParent() != null && node.getParent() instanceof ASTMethodDeclaration && node.getParent().getModifiers().isNative();
    }

    private boolean isStaticMethod(ASTMethodDeclaration node) {
        return node.getModifiers().isStatic();
    }

    private void violateRule(ASTMethodDeclaration node, String message) {
        RuleViolation violation = new RuleViolation(this, node, message);
        getRuleContext().addViolation(violation);
    }
}

package net.sourceforge.pmd.lang.java.rule.codestyle;


import java.util.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class MethodNamingConventionsRule extends AbstractJavaRulechainRule {

    public MethodNamingConventionsRule() {
        super(MethodDeclaration.class, ConstructorDeclaration.class);
    }

    @Override
    public Object visit(MethodDeclaration node, Object data) {
        String name = node.getName();
        String regex = "[a-z][a-zA-Z0-9]*";
        if (isJUnit4Method(node)) {
            regex = "[a-z][a-zA-Z0-9]*";
        } else if (isJUnit5Method(node)) {
            regex = "[a-z][a-zA-Z0-9]*";
        } else if (isJUnit3Method(node)) {
            regex = "test[A-Z0-9][a-zA-Z0-9]*";
        } else if (isNativeMethod(node)) {
            regex = "[a-z][a-zA-Z0-9]*";
        }
        if (!matchesRegex(name, regex)) {
            addViolation(data, node, "Method name does not match regex: " + regex);
        }
        return data;
    }

    @Override
    public Object visit(ConstructorDeclaration node, Object data) {
        String name = node.getName();
        String regex = "[a-z][a-zA-Z0-9]*";
        if (isNativeMethod(node)) {
            regex = "[a-z][a-zA-Z0-9]*";
        }
        if (!matchesRegex(name, regex)) {
            addViolation(data, node, "Method name does not match regex: " + regex);
        }
        return data;
    }

    private boolean matchesRegex(String name, String regex) {
        return name.matches(regex);
    }

    private boolean isJUnit4Method(MethodDeclaration node) {
        return isJUnit4Method(node) || node.getDeclaredAnnotations().stream()
                .anyMatch(TestFrameworksUtil::isJunit4TestAnnotation);
    }

    private boolean isJUnit5Method(MethodDeclaration node) {
        return TestFrameworksUtil.isJUnit5Method(node);
    }

    private boolean isJUnit3Method(MethodDeclaration node) {
        return TestFrameworksUtil.isJUnit3MethodSignature(node) || TestFrameworksUtil.isJUnit3Class(node.getParent());
    }

    private boolean isNativeMethod(MethodDeclaration node) {
        return node.getParent().isMethodDeclaration() && ((MethodDeclaration) node.getParent()).isNativeMethod();
    }
}

    }
