package net.sourceforge.pmd.lang.java.rule.errorprone;

import java.util.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class BrokenNullCheckRule extends AbstractJavaRulechainRule {

    @Override
    public Object visit(ASTMethodInvocation node, Object data) {
        if (JavaRuleUtil.isNullChecked(node)) {
            return super.visit(node, data);
        }
        Object result = null;
        if (node.getImage().equals("equals")) {
            List<ASTExpression> exprs = node.getExpressionList();
            if (exprs.size() > 1) {
                for (int i = 0; i < exprs.size() - 1; i++) {
                    ASTExpression expr = exprs.get(i);
                    ASTExpression nextExpr = exprs.get(i + 1);
                    if (JavaRuleUtil.isNullChecked(expr) && JavaRuleUtil.isNullChecked(nextExpr)) {
                        result = new Violation(this, node, "The null check is broken since it will throw a NullPointerException itself. It is likely that you used || instead of && or vice versa.");
                    }
                }
            }
        }
        return result;
    }

    @Override
    public Object visit(ASTBinaryExpression node, Object data) {
        if (node.getImage().equals("||") || node.getImage().equals("&&")) {
            if (JavaRuleUtil.isNullChecked(node.getLeftOperand()) && JavaRuleUtil.isNullChecked(node.getRightOperand())) {
                Object result = new Violation(this, node, "The null check is broken since it will throw a NullPointerException itself. It is likely that you used || instead of && or vice versa.");
                return result;
            }
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTConditionalExpression node, Object data) {
        if (JavaRuleUtil.isNullChecked(node.getExpression())) {
            return super.visit(node, data);
