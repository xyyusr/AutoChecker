package net.sourceforge.pmd.lang.java.rule.codestyle;

import java.util.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class UnnecessaryImportRule extends AbstractJavaRulechainRule {

    @Override
    public Object visit(ASTImportDeclaration node, Object data) {
        if (node.isStatic()) {
            ASTImportStaticDeclaration staticImport = (ASTImportStaticDeclaration) node.getFirstChild();
            String importedName = staticImport.getImportedSimpleName();
            if (importedName == null) {
                return super.visit(node, data);
            }
            // Check if importedName is already implicitly in scope
            if (importedName.startsWith("java.lang.")) {
                return super.visit(node, data);
            }
            // Check if importedName is already in the current package
            if (isInCurrentPackage(node, importedName)) {
                return super.visit(node, data);
            }
            // Check if importedName is duplicated in the import declaration
            if (isDuplicatedImport(node, importedName)) {
                return super.visit(node, data);
            }
            // Check if importedName is used anywhere in the code
            if (isUsed(importedName)) {
                return super.visit(node, data);
            }
        }
        return super.visit(node, data);
    }

    private boolean isInCurrentPackage(ASTImportDeclaration node, String importedName) {
        ASTPackageDeclaration packageDeclaration = (ASTPackageDeclaration) node.getAstInfo().getAstRoot().getFirstChild();
        String packageName = packageDeclaration.getPackageName();
        if (packageName.equals("java.lang") || packageName.equals("java")) {
            return true;
        }
        return false;
    }

    private boolean isDuplicatedImport(ASTImportDeclaration node, String importedName) {
        Iterable<ASTImportDeclaration> imports = node.getAstInfo().getAstRoot().getFirstChild().getFirstChild().getDescendants();
        for (ASTImportDeclaration importNode : imports) {
            if (importNode.getImportedSimpleName().equals(importedName)) {
                return true;
            }
        }
        return false;
    }

    private boolean isUsed(String importedName) {
        // This function is not implemented as it requires a full blown semantic analysis of the code.
        // For a simple check, we can just check if the importedName is used somewhere in the code.
        // However, this is a very basic check and might not cover all cases.
        NodeStream descendants = node.descendants();
        for (Node descendant : descendants) {
            if (descendant instanceof ASTClassOrInterfaceDeclaration) {
                ASTClassOrInterfaceDeclaration classOrInterface = (ASTClassOrInterfaceDeclaration) descendant;
                for (ASTMethodDeclaration method : classOrInterface.getMethods()) {
                    for (ASTExpression expression : method.getBody().getStatements()) {
                        if (expression instanceof ASTLiteral) {
                            ASTLiteral literal = (ASTLiteral) expression;
                            if (literal.getValue().equals(importedName)) {
                                return true;
                            }
                        }
                    }
                }
            }
        }
        return false;
    }
}

package net.sourceforge.pmd.lang.java.rule.codestyle;

import java.util.EnumSet;
import java.util.Set;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class UnnecessaryImportRule extends AbstractJavaRulechainRule {

    @Override
    public Object visit(ASTCompilationUnit node, Object data) {
        Set<String> usedNames = new HashSet<>();
        Set<ASTImportDeclaration> visitedImports = new HashSet<>();

        for (ASTImportDeclaration importDecl : node.findChildrenOfType(ASTImportDeclaration.class)) {
            if (importDecl.isStatic()) {
                String importedName = importDecl.getImportedSimpleName();
                if (importedName != null && !isImplicitlyImported(importedName)) {
                    usedNames.add(importedName);
                }
            } else {
                for (NameDeclaration nameDecl : importDecl.findChildrenOfType(NameDeclaration.class)) {
                    String name = nameDecl.getImage();
                    usedNames.add(name);
                }
            }
            visitedImports.add(importDecl);
        }

        Set<String> allNamesInCurrentPackage = getNamesInCurrentPackage(node);

        for (ASTImportDeclaration importDecl : node.findChildrenOfType(ASTImportDeclaration.class)) {
            if (!visitedImports.contains(importDecl)) {
                String importedSimpleName = importDecl.getImportedSimpleName();
                if (importedSimpleName != null && !usedNames.contains(importedSimpleName) && !allNamesInCurrentPackage.contains(importedSimpleName)) {
                    reportImport(importDecl);
                } else if (importDecl.isStatic() && importedSimpleName == null) {
                    reportStaticImport(importDecl);
                }
            }
        }

        return super.visit(node, data);
    }

    private boolean isImplicitlyImported(String name) {
        return name.startsWith("java.") || name.equals("java.lang.");
    }

    private Set<String> getNamesInCurrentPackage(ASTCompilationUnit node) {
        Set<String> allNames = new HashSet<>();
        for (NameDeclaration nameDecl : node.findChildrenOfType(NameDeclaration.class)) {
            allNames.add(nameDecl.getImage());
        }
        return allNames;
    }

    private void reportImport(ASTImportDeclaration importDecl) {
        reportIssue(importDecl.getReportLocation(), "Import of " + importDecl.getImportedSimpleName() + " is unnecessary.");
    }

    private void reportStaticImport(ASTImportDeclaration importDecl) {
        reportIssue(importDecl.getReportLocation(), "Static import of " + importDecl.getImportedSimpleName() + " is unnecessary.");
    }
}

package net.sourceforge.pmd.lang.java.rule.codestyle;

import java.util.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class UnnecessaryImportRule extends AbstractJavaRulechainRule {

    public UnnecessaryImportRule() {
        super(ASTCompilationUnit.class);
    }

    @Override
    public Object visit(ASTCompilationUnit node, Object data) {
        Set<ASTImportDeclaration> importDeclarations = new HashSet<>();
        Set<String> usedTypes = new HashSet<>();
        Set<String> definedTypes = new HashSet<>();

        // Collect used types
        collectUsedTypes(node, usedTypes);

        // Collect defined types
        collectDefinedTypes(node, definedTypes);

        // Check import declarations
        for (ASTImportDeclaration importDecl : node.findDescendantsOfType(ASTImportDeclaration.class, true)) {
            if (importDecl.isStatic()) {
                continue;
            }

            if (usedTypes.contains(importDecl.getImportedSimpleName())) {
                continue;
            }

            if (importDecl.equals(node.findImportDeclaration(importDecl.getImportedSimpleName()))) {
                continue;
            }

            if (definedTypes.contains(importDecl.getImportedSimpleName())) {
                continue;
            }

            if (isImplicitlyInScope(importDecl.getImportedSimpleName(), node)) {
                continue;
            }

            addViolation(this, importDecl.getReportLocation());
        }

        return data;
    }

    private void collectUsedTypes(Node node, Set<String> usedTypes) {
        if (node instanceof ASTCompilationUnit) {
            ASTCompilationUnit compilationUnit = (ASTCompilationUnit) node;
            for (ASTImportDeclaration importDecl : compilationUnit.findDescendantsOfType(ASTImportDeclaration.class, true)) {
                usedTypes.addAll(compilationUnit.getSymbolTable().getTypes(importDecl.getImportedSimpleName()));
            }
        }

        for (Node child : node.children()) {
            collectUsedTypes(child, usedTypes);
        }
    }

    private void collectDefinedTypes(Node node, Set<String> definedTypes) {
        if (node instanceof ASTCompilationUnit) {
            ASTCompilationUnit compilationUnit = (ASTCompilationUnit) node;
            for (ASTTypeDeclaration typeDecl : compilationUnit.findDescendantsOfType(ASTTypeDeclaration.class, true)) {
                for (ASTMemberDeclaration memberDecl : typeDecl.findDescendantsOfType(ASTMemberDeclaration.class, true)) {
                    definedTypes.add(memberDecl.getName());
                }
            }
        }

        for (Node child : node.children()) {
            collectDefinedTypes(child, definedTypes);

public class UnnecessaryImportRule extends AbstractJavaRulechainRule {

    @Override
    public Object visit(ASTImportDeclaration node, Object data) {
        if (node.isStatic() && node.getImportedSimpleName() != null) {
            return visitStaticImport(node, data);
        } else {
            return visitDynamicImport(node, data);
        }
    }

    private Object visitStaticImport(ASTImportDeclaration node, Object data) {
        if (isImplicitImport(node.getImportedSimpleName())) {
            addViolation(data, node);
            return data;
        }
        return super.visit(node, data);
    }

    private Object visitDynamicImport(ASTImportDeclaration node, Object data) {
        if (isImplicitImport(node.getImportedSimpleName())) {
            addViolation(data, node);
            return data;
        } else if (isDuplicatedImport(node, data)) {
            addViolation(data, node);
            return data;
        } else if (isUnusedImport(node, data)) {
            addViolation(data, node);
            return data;
        }
        return super.visit(node, data);
    }

    private boolean isImplicitImport(String importedSimpleName) {
        return importedSimpleName.startsWith("java.lang.") || importedSimpleName.startsWith(getCurrentPackageName());
    }

    private String getCurrentPackageName() {
        ASTPackageDeclaration packageDeclaration = (ASTPackageDeclaration) getCurrentNode().getParent().findFirst(ASTPackageDeclaration.class);
        if (packageDeclaration != null) {
            return packageDeclaration.getImage();
        }
        return "";
    }

    private boolean isDuplicatedImport(ASTImportDeclaration node, Object data) {
        NodeStream<ASTImportDeclaration> importDeclarations = node.getSymbolTable().getAllImportDeclarations();
        for (ASTImportDeclaration importDeclaration : importDeclarations) {
            if (importDeclaration != node && importDeclaration.getImportedSimpleName().equals(node.getImportedSimpleName())) {
                return true;
            }
        }
        return false;

    private boolean isUnusedImport(ASTImportDeclaration node, Object data) {
        NodeStream<ASTMemberDeclaration> memberDeclarations = node.descendantsOrSelf(ASTMemberDeclaration.class);
        for (ASTMemberDeclaration memberDeclaration : memberDeclarations) {
            if (memberDeclaration.getImage().startsWith(node.getImportedSimpleName())) {
                return false;
            }
        }
        return true;
    }

        return (Node) getCurrentRuleContext();

package net.sourceforge.pmd.lang.java.rule.codestyle;

import java.util.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class UnnecessaryImportRule extends AbstractJavaRulechainRule {

    @Override
    public Object visit(ASTImportDeclaration node, Object data) {
        if (node.isStatic()) {
            String importedSimpleName = node.getImportedSimpleName();
            if (importedSimpleName != null) {
                if (isMemberInJavaLang(importedSimpleName) || isMemberInCurrentPackage(importedSimpleName)) {
                    return false;
                }
                if (isMemberDuplicated(node, importedSimpleName)) {
                    return false;
                }
            }

            List<ASTReference> references = node.findChildrenOfType(ASTReference.class);
            for (ASTReference reference : references) {
                if (!isReferenceUnused(reference)) {
                    return false;
                }
            }
        }
        return super.visit(node, data);
    }

    private boolean isMemberInJavaLang(String member) {
        return member.startsWith("java.");
    }

    private boolean isMemberInCurrentPackage(String member) {
        try {
            Class<?> currentPackage = Class.forName(getCurrentPackageName());
            for (java.lang.reflect.Method method : currentPackage.getMethods()) {
                if (method.getName().equals(member)) {
                    return true;
                }
            }
            return false;
        } catch (Exception e) {
            return false;
        }
    }

    private boolean isMemberDuplicated(ASTImportDeclaration node, String member) {
        ASTImportDeclaration duplicate = findDuplicateImport(node);
        if (duplicate != null) {
            return !duplicate.getImportedSimpleName().equals(member);
        }
        return false;
    }

    private ASTImportDeclaration findDuplicateImport(ASTImportDeclaration node) {
        for (ASTImportDeclaration duplicate : node.getSymbolTable().getImportDeclarations()) {
            if (duplicate.isStatic() && duplicate.getImportedSimpleName().equals(node.getImportedSimpleName())) {
                return duplicate;
            }
        }
        return null;
    }

    private boolean isReferenceUnused(ASTReference reference) {
        return reference.findChildrenOfType(ASTIdentifier.class).isEmpty();
    }

    private String getCurrentPackageName() {
        return getCurrentCompilationUnit().getPackageDeclaration().getPackageName();
    }

    private ASTCompilationUnit getCurrentCompilationUnit() {
        return (ASTCompilationUnit) getCurrentNode().getParent().getParent();
    }
}
