package net.sourceforge.pmd.lang.java.rule.codestyle;

import java.util.*;
import java.lang.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class UnnecessaryImportRule extends AbstractJavaRulechainRule {

    private static final String[] STANDARDIMPORTS = new String[] {
        "java.lang.Math", "java.lang.String", "java.lang.System", "java.lang.Thread",
        "java.util.ArrayList", "java.util.Arrays", "java.util.Collections",
        "java.util.HashMap", "java.util.HashSet", "java.util.List",
        "java.util.Map", "java.util.Properties", "java.util.Set",
        "java.util.TreeMap", "java.util.TreeSet"
    };

    @Override
    public Object visit(ASTImportDeclaration node, Object data) {
        if (isStandardImport(node)) {
            return super.visit(node, data);
        }

        boolean used = isImportUsed(node, data);
        if (used) {
            return super.visit(node, data);
        }

        List<ASTImportDeclaration> duplicates = findDuplicates(node, data);
        if (!duplicates.isEmpty()) {
            return super.visit(node, data);
        }

        return super.visit(node, data);
    }

    private boolean isStandardImport(ASTImportDeclaration node) {
        for (String standardImport : STANDARDIMPORTS) {
            if (standardImport.equals(node.getImage())) {
                return true;
            }
        }
        return false;
    }

    private boolean isImportUsed(ASTImportDeclaration node, Object data) {
        ASTCompilationUnit unit = (ASTCompilationUnit) data;
        NodeStream<JavaNode> nodes = unit.getChildren();
        for (JavaNode n : nodes) {
            if (n instanceof ASTMethodDeclaration
                && node.getImage().equals(((ASTMethodDeclaration) n).getImage())) {
                return true;
            }
        }
        return false;
    }

    private List<ASTImportDeclaration> findDuplicates(ASTImportDeclaration node, Object data) {
        ASTCompilationUnit unit = (ASTCompilationUnit) data;
        List<ASTImportDeclaration> duplicates = new ArrayList<>();
        for (ASTImportDeclaration n : unit.getImports()) {
            if (n != node && n.getImage().equals(node.getImage())) {
                duplicates.add(n);
            }
        }
        return duplicates;
    }
}

package net.sourceforge.pmd.lang.java.rule.codestyle;

import java.util.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class UnnecessaryImportRule extends AbstractJavaRulechainRule {

    public UnnecessaryImportRule() {
        addRuleMessage("Import of " + ImportDeclaration.class.getSimpleName() + " is unnecessary");
    }

    @Override
    public Object visit(ImportDeclaration node, Object data) {
        boolean isStaticImport = node.isStatic();
        String importedSimpleName = node.getImportedSimpleName();

        if (importedSimpleName == null) {
            return super.visit(node, data);
        }

        String importedName = importedSimpleName;
        if (isStaticImport) {
            importedName = importedSimpleName + ".";
        }

        boolean importedInJavaLang = importedSimpleName.equals("java.lang") || importedSimpleName.startsWith("java.lang.");

        boolean importedInCurrentPackage = true;
        List<ClassOrInterfaceDeclaration> classes = node.findChildrenOfType(ClassOrInterfaceDeclaration.class);
        if (classes.size() > 0) {
            importedInCurrentPackage = false;
        }

        List<ImportDeclaration> importDeclarations = node.findChildrenOfType(ImportDeclaration.class);
        boolean duplicatedImport = false;
        for (ImportDeclaration importDeclaration : importDeclarations) {
            if (importDeclaration != node && importDeclaration.getImportedSimpleName().equals(importedSimpleName)) {
                duplicatedImport = true;
                break;
            }
        }

        if (importedInJavaLang || importedInCurrentPackage || duplicatedImport) {
            addRuleViolation(node.getReportLocation(), "Import of " + importedSimpleName + " is unnecessary");
        }

        return super.visit(node, data);
    }
}

package net.sourceforge.pmd.lang.java.rule.codestyle;

import java.util.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class UnnecessaryImportRule extends AbstractJavaRulechainRule {

    public UnnecessaryImportRule() {
        super(ASTImportDeclaration.class);
    }

    @Override
    public Object visit(ASTImportDeclaration node, Object data) {
        if (isUnusedImport(node)) {
            report(node.getReportLocation(), "Unused import");
        } else if (isDuplicateImport(node, node.getParent().getChildren())) {
            report(node.getReportLocation(), "Duplicate import");
        } else if (isImplicitImport(node)) {
            report(node.getReportLocation(), "Implicit import");
        }
        return super.visit(node, data);
    }

    private boolean isUnusedImport(ASTImportDeclaration node) {
        if (node.getSymbolTable().getTypes().contains(node.getImportedSimpleName())) {
            return false;
        }
        if (node.getParent() instanceof ASTCompilationUnit) {
            ASTCompilationUnit compilationUnit = (ASTCompilationUnit) node.getParent();
            if (compilationUnit.getImportedNames().contains(node.getImportedSimpleName())) {
                return false;
            }
        }
        return true;
    }

    private boolean isDuplicateImport(ASTImportDeclaration node, List<ASTImportDeclaration> imports) {
        for (ASTImportDeclaration imp : imports) {
            if (imp.equals(node) && imp != node) {
                return true;
            }
        }
        return false;
    }

    private boolean isImplicitImport(ASTImportDeclaration node) {
        return node.getImportedSimpleName().startsWith("java.lang.") || node.getImportedSimpleName().startsWith(node.getParent().getPackageName() + ".");
    }
}

package net.sourceforge.pmd.lang.java.rule.codestyle;

import java.util.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class UnnecessaryImportRule extends AbstractJavaRulechainRule {
    public UnnecessaryImportRule() {
        super(ASTImportDeclaration.class);
    }

    @Override
    public Object visit(ASTImportDeclaration node, Object data) {
        if (node.isStatic()) {
            String importedSimpleName = node.getImportedSimpleName();
            if (importedSimpleName == null) {
                return null;
            }
            if (importedSimpleName.equals("java.lang")) {
                return null;
            }
            if (isMemberInScope(node, importedSimpleName)) {
                return null;
            }
            if (isDuplicateImport(node, importedSimpleName)) {
                return null;
            }
            if (isImportedMemberUnused(node, importedSimpleName)) {
                addViolation(this, node.getReportLocation());
            }
        }
        return super.visit(node, data);
    }

    private boolean isMemberInScope(ASTImportDeclaration node, String importedSimpleName) {
        ASTCompilationUnit compilationUnit = node.getAstInfo().getRootNode();
        if (compilationUnit.getPackageDeclaration() != null) {
            return isMemberInScope(compilationUnit.getPackageDeclaration(), importedSimpleName);
        }
        return false;
    }

    private boolean isMemberInScope(ASTPackageDeclaration node, String importedSimpleName) {
        ASTCompilationUnit compilationUnit = node.getAstInfo().getRootNode();
        if (compilationUnit.getPackageDeclaration() != null) {
            return compilationUnit.getPackageDeclaration().getDeclaredAnnotations().stream()
                    .anyMatch(annotation -> isMemberInScope(annotation, importedSimpleName));
        }
        return false;
    }

    private boolean isMemberInScope(ASTAnnotation node, String importedSimpleName) {
        ASTCompilationUnit compilationUnit = node.getAstInfo().getRootNode();
        if (compilationUnit.getPackageDeclaration() != null) {
            return compilationUnit.getPackageDeclaration().getDeclaredAnnotations().stream()
                    .anyMatch(annotation -> isMemberInScope(annotation, importedSimpleName));
        }
        return false;
    }

    private boolean isMemberInScope(ASTImportDeclaration node, String importedSimpleName) {
        return false;
    }

    private boolean isDuplicateImport(ASTImportDeclaration node, String importedSimpleName) {
        ASTCompilationUnit compilationUnit = node.getAstInfo().getRootNode();
        ASTImportDeclaration duplicateImport = compilationUnit.findFirstDescendant(ASTImportDeclaration.class, true)
                .filter(importDeclaration -> importDeclaration.getImportedSimpleName().equals(importedSimpleName))
                .findFirst().orElse(null);
        return duplicateImport != null && !duplicateImport.equals(node);
    }

    private boolean isImportedMemberUnused(ASTImportDeclaration node, String importedSimpleName) {
        ASTCompilationUnit compilationUnit = node.getAstInfo().getRootNode();
        return compilationUnit.findDescendants(ASTCompilationUnit.class).stream()
                .filter(unit -> unit.findChildrenOfType(ASTVariableDeclarator.class).stream()
                        .anyMatch(variableDeclarator -> {
                            ASTName name = variableDeclarator.getName();
                            return name != null && name.getImage().equals(importedSimpleName);
                        })).findFirst().orElse(null) == null;
    }
}
