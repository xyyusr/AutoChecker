import net.sourceforge.pmd.lang.java.ast.internal. *;

    @Override

            ASTThrowsList list = (ASTThrowsList) node;
            ASTExpression exception = list.getExceptions().singleOrNull();
            if (exception != null) {
                if (exception instanceof ASTNullLiteral && !JavaRuleUtil.isNullChecked(exception)) {
                    addViolation(data, this, exception);
                }
            }
        }
        return super.visit(node, data);
    }
}

    public AvoidThrowingNullPointerExceptionRule() {
        super(MethodDeclaration.class);

    @Override
