package net.sourceforge.pmd.lang.java.rule.errorprone;

import java.util.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class NullAssignmentRule extends AbstractJavaRulechainRule {

    @Override
    public Object visit(ASTAssignmentNode node, Object data) {
        if (node.getOpToken().getImage().equals("=")) {
            ASTExpression left = node.getLeftOp();
            ASTExpression right = node.getRightOp();
            if (!isNullCheck(left, null)) {
                if (right instanceof ASTNullLiteral) {
                    addViolation(this, node, "Assigning a null to a variable is usually bad form.");
                }
            }
        }
        return super.visit(node, data);
    }

    private boolean isNullCheck(ASTExpression expression, JVariableSymbol variable) {
        if (expression instanceof ASTNullLiteral) {
            return true;
        }
        if (expression instanceof ASTMethodCallExpr) {
            ASTMethodCallExpr methodCall = (ASTMethodCallExpr) expression;
            if (methodCall.getMethodName().equals("NullPointerException")) {
                if (methodCall.getArguments().size() == 1 && methodCall.getArguments().get(0) instanceof ASTLiteral) {
                    return true;
                } else if (methodCall.getArguments().size() == 1 && methodCall.getArguments().get(0) instanceof ASTNameExpr) {
                    ASTNameExpr name = (ASTNameExpr) methodCall.getArguments().get(0);
                    if (name.getImage().equals("get") && name.getParent() instanceof ASTFieldAccessExpr) {
                        ASTFieldAccessExpr fieldAccess = (ASTFieldAccessExpr) name.getParent();
                        if (fieldAccess.getLeftOp() instanceof ASTNameExpr && ((ASTNameExpr) fieldAccess.getLeftOp()).getImage().equals("null")) {
                            return true;
                        }
                    }
                }
            }
        }
        return false;
    }

    @Override
    public Object visit(ASTVariableDeclarator node, Object data) {
        if (node.getInit() instanceof ASTNullLiteral) {
            addViolation(this, node, "Assigning a null to a variable is usually bad form.");
        }
        return super.visit(node, data);
    }
}
