public class InefficientEmptyStringCheckRule extends AbstractJavaRulechainRule {

    public InefficientEmptyStringCheckRule() {
        addRuleMessage("String.trim().length() == 0 and String.trim().isEmpty() are inefficient ways to check if a String is really blank.");
        addRulePriority(RulePriority.NORMAL);
    }

    @Override
    public Object visit(ASTMethodCall node, Object data) {
        if (node.getImage().contains("trim") && node.getImage().contains("length") && node.getImage().contains("==") && node.getImage().contains("0")) {
            addRuleViolation(node, "String.trim().length() == 0 is an inefficient way to check if a String is really blank.");
        }
        if (node.getImage().contains("trim") && node.getImage().contains("isEmpty") && node.getImage().contains("&&") && node.getImage().contains("true")) {
            addRuleViolation(node, "String.trim().isEmpty() is an inefficient way to check if a String is really blank.");
        }
        if (node.getImage().contains("trim") && node.getImage().contains("isEmpty")) {
            addRuleViolation(node, "String.trim().isEmpty() is an inefficient way to check if a String is really blank.");
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTMethodDeclaration node, Object data) {
        if (node.getBody() instanceof ASTBlockStmt) {
            ASTBlockStmt block = (ASTBlockStmt) node.getBody();
            for (ASTStatement stmt : block.getStatements()) {
                if (stmt instanceof ASTExpressionStmt) {
                    ASTExpressionStmt exprStmt = (ASTExpressionStmt) stmt;

                        ASTMethodCall methodCall = (ASTMethodCall) exprStmt.getExpression();
                        if (visit(methodCall, data) != null) {
                            break;
                        }
                    }
                }
            }
        }
        return super.visit(node, data);
    }
