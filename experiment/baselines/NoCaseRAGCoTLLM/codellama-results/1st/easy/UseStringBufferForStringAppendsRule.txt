package net.sourceforge.pmd.lang.java.rule.performance;


import java.util.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class UseStringBufferForStringAppendsRule extends AbstractJavaRulechainRule {

    @Override
    public Object visit(ASTAssignment node, Object data) {
        if (node.getOp() == ASTToken.EQUALS) {
            ASTExpression left = node.getLeftOp();
            if (left instanceof ASTCompoundAssignment && ((ASTCompoundAssignment) left).getOp() == ASTToken.PLUS_EQUALS) {
                if (visitCompoundAssignment((ASTCompoundAssignment) left, data)) {
                    addViolation(node, data);
                }
            }
        }
        return super.visit(node, data);
    }

    private boolean visitCompoundAssignment(ASTCompoundAssignment left, Object data) {
        ASTExpression right = left.getRightOp();
        if (JavaRuleUtil.isStringBuilderCtorOrAppend(right)) {
            List<ASTExpression> expressions = getExpressions(right);
            if (expressions != null && expressions.size() > 1) {
                return true;
            }
        } else if (JavaAstUtils.isStringConcatExpr(right)) {
            List<ASTExpression> expressions = getExpressions(right);
            if (expressions != null && expressions.size() > 1) {
                return true;
            }
        }
        return false;
    }

    private List<ASTExpression> getExpressions(ASTExpression expression) {
        List<ASTExpression> expressions = new ArrayList<>();
        if (expression instanceof ASTBinaryExpression && ((ASTBinaryExpression) expression).getOp() == ASTToken.PLUS) {
            expressions.add(((ASTBinaryExpression) expression).getLeftOp());
            expressions.add(((ASTBinaryExpression) expression).getRightOp());
        } else if (expression instanceof ASTArrayAccessExpression) {
            expressions = getExpressions(((ASTArrayAccessExpression) expression).getExpression());
        } else if (expression instanceof ASTMethodCallExpression) {
            if (expression.getTarget() instanceof ASTThisExpression) {
                expressions = getExpressions(expression.getArgumentList());
            } else if (expression.getTarget() instanceof ASTFieldAccessExpression) {
                expressions = getExpressions(((ASTFieldAccessExpression) expression.getTarget()).getExpression());
            }
        }
        return expressions;
    }

    @Override
    public Object visit(ASTMethodCallExpression node, Object data) {
        if (node.getTarget() instanceof ASTFieldAccessExpression) {
            ASTFieldAccessExpression fieldAccess = (ASTFieldAccessExpression) node.getTarget();
            ASTExpression base = fieldAccess.getExpression();
            if (base instanceof ASTThisExpression) {
                if (node.getImage().contains("append")) {
                    addViolation(node, data);
                }
            }
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTArrayAccessExpression node, Object data) {
        List<ASTExpression> expressions = getExpressions(node.getExpression());
        if (expressions != null && expressions.size() > 1) {
            addViolation(node, data);
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTMethodCallExpression node, Object data) {
        if (node.getImage().contains("append")) {
            List<ASTExpression> expressions = getExpressions(node.getArgumentList());
            if (expressions != null && expressions.size() > 1) {
                addViolation(node, data);
            }
        }
        return super.visit(node, data);
    }
}
