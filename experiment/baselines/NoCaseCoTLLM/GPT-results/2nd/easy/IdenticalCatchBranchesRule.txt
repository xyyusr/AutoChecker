package net.sourceforge.pmd.lang.java.rule.codestyle;


import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class IdenticalCatchBranchesRule extends AbstractJavaRulechainRule {

    private Map<String, ASTBlock> catchBlocks = new HashMap<>();

    public IdenticalCatchBranchesRule() {
        super(ASTCatchStatement.class);
    }

    @Override
    public Object visit(ASTCatchStatement node, Object data) {
        ASTBlock currentCatchBlock = node.getFirstDescendantOfType(ASTBlock.class);
        String currentExceptionType = node.getFirstDescendantOfType(ASTCatchStatement.class).getImage();

        for (Map.Entry<String, ASTBlock> entry : catchBlocks.entrySet()) {
            String storedExceptionType = entry.getKey();
            ASTBlock storedCatchBlock = entry.getValue();

            if (currentExceptionType.equals(storedExceptionType) && currentCatchBlock.equals(storedCatchBlock)) {
                addViolation(data, node);
                break;
            }
        }

        catchBlocks.put(currentExceptionType, currentCatchBlock);

        return super.visit(node, data);
    }
}

package net.sourceforge.pmd.lang.java.rule.codestyle;


import java.util.List;
import java.util.ArrayList;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class IdenticalCatchBranchesRule extends AbstractJavaRulechainRule {

    public IdenticalCatchBranchesRule() {
        super(ASTTryStatement.class);
    }

    @Override
    public Object visit(ASTTryStatement node, Object data) {
        List<String> catchBlockCodes = new ArrayList<>();

        for (int i = 0; i < node.getNumCatchStatements(); i++) {
            ASTCatchStatement catchStatement = node.getCatchStatement(i);
            ASTBlock block = (ASTBlock) catchStatement.getFirstDescendantOfType(ASTBlock.class);
            if (block != null) {
                String blockCode = block.getImage();
                catchBlockCodes.add(blockCode);
            }
        }

        if (hasDuplicates(catchBlockCodes)) {
            addViolation(data, node);
        }

        return super.visit(node, data);
    }

    private boolean hasDuplicates(List<String> list) {
        Set<String> set = new HashSet<>(list);
        return set.size() < list.size();
    }
}
