import java.util.List;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;

import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class UnnecessaryImportRule extends AbstractJavaRulechainRule {
    private List<String> imports = new ArrayList<>();
    private String packageName;

    public UnnecessaryImportRule() {
        super(ASTImportDeclaration.class);
    }

    @Override
    public Object visit(ASTCompilationUnit node, Object data) {
        packageName = node.getPackageDeclaration().getPackageNameImage();
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTImportDeclaration node, Object data) {
        String importName = node.getImportedName();

        if (imports.contains(importName)) {
            addViolation(data, node);
        } else {
            imports.add(importName);
            if (importName.startsWith("java.lang.") || importName.startsWith(packageName + ".")) {
                addViolation(data, node);
            }
        }

        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTTypeDeclaration node, Object data) {
        List<String> usedClasses = node.findDescendantsOfType(ASTClassOrInterfaceType.class)
                                       .stream()
                                       .map(ASTClassOrInterfaceType::getImage)
                                       .collect(Collectors.toList());

        for (String importName : imports) {
            if (!usedClasses.contains(importName)) {
                addViolation(data, node);
            }
        }

        return super.visit(node, data);
    }
}

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class UnnecessaryImportRule extends AbstractJavaRulechainRule {

    public UnnecessaryImportRule() {
        addRuleChainVisit(ASTImportDeclaration.class);
    }

    @Override
    public Object visit(ASTImportDeclaration node, Object data) {
        if (node.isImportOnDemand()) {
            return super.visit(node, data);
        }

        AbstractJavaNode typeNode = (AbstractJavaNode) node.jjtGetChild(0);
        String name = typeNode.getImage();

        Map<NameDeclaration, List<NameOccurrence>> declarations = node.getScope()
                .getDeclarations();

        boolean found = declarations.keySet().stream().anyMatch(decl -> decl.getName().equals(name));

        if (!found || node.isFromSamePackage() || node.isFromJavaLang() || node.isDuplicated()) {
            addViolation(data, node);
        }

        return super.visit(node, data);
    }
}

package net.sourceforge.pmd.lang.java.rule.codestyle;


import java.util.HashSet;
import java.util.Set;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class UnnecessaryImportRule extends AbstractJavaRulechainRule {

  private Set<String> importedTypes;
  private Set<String> usedTypes;

  public UnnecessaryImportRule() {
    super(ASTCompilationUnit.class);
  }

  @Override
  public Object visit(ASTCompilationUnit node, Object data) {
    importedTypes = new HashSet<>();
    usedTypes = new HashSet<>();

    super.visit(node, data);

    for (String importType : importedTypes) {
      if (!usedTypes.contains(importType)) {
        addViolation(data, node, importType);
      }
    }

    return data;
  }

  @Override
  public Object visit(ASTImportDeclaration node, Object data) {
    String importType = node.getImportedName();
    if (importedTypes.contains(importType) || importType.startsWith("java.lang.") || importType.startsWith(node.getPackageName() + ".")) {
      addViolation(data, node, importType);
    } else {
      importedTypes.add(importType);
    }
    return super.visit(node, data);
  }

  @Override
  public Object visit(ASTType node, Object data) {
    usedTypes.add(node.getTypeImage());
    return super.visit(node, data);
  }
}
