package net.sourceforge.pmd.lang.java.rule.bestpractices;

import java.util.Map;
import java.util.HashMap;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class AvoidDuplicateLiteralsRule extends AbstractJavaRulechainRule {
    private Map<String, Integer> literalsCount = new HashMap<>();

    public AvoidDuplicateLiteralsRule() {
        super(ASTLiteral.class);
    }

    @Override
    public Object visit(ASTLiteral node, Object data) {
        String literal = node.getImage();
        if (literal.length() < 3) {
            return super.visit(node, data);
        }

        literalsCount.put(literal, literalsCount.getOrDefault(literal, 0) + 1);

        return super.visit(node, data);
    }

    @Override
    public void end(final Object data) {
        for (Map.Entry<String, Integer> entry : literalsCount.entrySet()) {
            if (entry.getValue() >= 4) {
                addViolation(data, null, entry.getKey());
            }
        }
    }
}

import java.util.Map;

import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class AvoidDuplicateLiteralsRule extends AbstractJavaRulechainRule {
    private Map<String, Integer> stringLiterals;

    public AvoidDuplicateLiteralsRule() {
        super(ASTLiteral.class);
        stringLiterals = new HashMap<>();
    }

    @Override
    public Object visit(ASTLiteral node, Object data) {
        if (node.isStringLiteral() && node.getImage().length() > 2) {
            String image = node.getImage().substring(1, node.getImage().length() - 1);
            int count = stringLiterals.getOrDefault(image, 0) + 1;
            stringLiterals.put(image, count);
            if (count == 4) {
                addViolation(data, node);
            }
        }
        return super.visit(node, data);
    }

    @Override
    public void start(RuleContext ctx) {
        stringLiterals.clear();
        super.start(ctx);
    }

    @Override
    public void end(RuleContext ctx) {
        stringLiterals.clear();
        super.end(ctx);
    }
}

package net.sourceforge.pmd.lang.java.rule.errorprone;

import java.util.HashMap;
import java.util.Map;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class AvoidDuplicateLiteralsRule extends AbstractJavaRulechainRule {

    private Map<String, Integer> stringOccurrences;

    public AvoidDuplicateLiteralsRule() {
        super(ASTLiteral.class);
        stringOccurrences = new HashMap<>();
    }

    @Override
    public Object visit(ASTLiteral node, Object data) {
        if (node.isStringLiteral()) {
            String stringLiteral = node.getImage();
            if (stringLiteral.length() < 3) {
                return super.visit(node, data);
            }
            stringOccurrences.merge(stringLiteral, 1, Integer::sum);
            if (stringOccurrences.get(stringLiteral) >= 4) {
                addViolation(data, node);
            }
        }
        return super.visit(node, data);
    }
}
