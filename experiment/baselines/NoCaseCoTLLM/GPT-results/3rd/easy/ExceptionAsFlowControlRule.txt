package net.sourceforge.pmd.lang.java.rule.design;

import java.util.List;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ExceptionAsFlowControlRule extends AbstractJavaRulechainRule {

    public ExceptionAsFlowControlRule() {
        super(ASTTryStatement.class);
    }

    @Override
    public Object visit(ASTTryStatement node, Object data) {
        List<ASTThrowStatement> throwStatements = node.findDescendantsOfType(ASTThrowStatement.class);
        for (ASTThrowStatement throwStatement : throwStatements) {
            ASTClassOrInterfaceType exceptionType = throwStatement.getFirstDescendantOfType(ASTClassOrInterfaceType.class);
            if (exceptionType != null) {
                for (ASTCatchStatement catchStatement : node.getCatchStatements()) {
                    if (isExceptionCaught(catchStatement, exceptionType)) {
                        addViolation(data, throwStatement);
                        break;
                    }
                }
            }
        }
        return super.visit(node, data);
    }

    private boolean isExceptionCaught(ASTCatchStatement catchStatement, ASTClassOrInterfaceType exceptionType) {
        String caughtType = catchStatement.getFirstDescendantOfType(ASTClassOrInterfaceType.class).getImage();
        return exceptionType.getImage().equals(caughtType);
    }
}

                    ASTClassOrInterfaceType caughtExceptionType = catchStatement.getFirstDescendantOfType(ASTClassOrInterfaceType.class);
                    if (caughtExceptionType != null && caughtExceptionType.getImage().equals(thrownExceptionName)) {
                        addViolation(data, node);
                        break;
                    }
                }
            }
        }
        return super.visit(node, data);

                            addViolation(data, node);
                        }
                    }
                }
            }
