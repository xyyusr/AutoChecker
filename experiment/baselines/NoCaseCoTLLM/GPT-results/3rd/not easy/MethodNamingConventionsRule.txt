import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class MethodNamingConventionsRule extends AbstractJavaRulechainRule {

    private static final String INSTANCE_METHOD_NAME_REGEX = "[a-z][a-zA-Z0-9]*";
    private static final String STATIC_METHOD_NAME_REGEX = "[a-z][a-zA-Z0-9]*";
    private static final String NATIVE_METHOD_NAME_REGEX = "[a-z][a-zA-Z0-9]*";
    private static final String JUNIT_3_TEST_METHOD_NAME_REGEX = "test[A-Z0-9][a-zA-Z0-9]*";
    private static final String JUNIT_4_TEST_METHOD_NAME_REGEX = "[a-z][a-zA-Z0-9]*";
    private static final String JUNIT_5_TEST_METHOD_NAME_REGEX = "[a-z][a-zA-Z0-9]*";

    public MethodNamingConventionsRule() {
        super(ASTMethodDeclaration.class);
    }

    @Override
    public Object visit(ASTMethodDeclaration node, Object data) {
        String methodName = node.getFirstDescendantOfType(ASTMethodDeclarator.class).getImage();

        if (node.isStatic() && !methodName.matches(STATIC_METHOD_NAME_REGEX)) {
            addViolation(data, node);
        } else if (node.isNative() && !methodName.matches(NATIVE_METHOD_NAME_REGEX)) {
            addViolation(data, node);
        } else if (methodName.startsWith("test") && !methodName.matches(JUNIT_3_TEST_METHOD_NAME_REGEX)) {
            addViolation(data, node);
        } else if (methodName.startsWith("@Test") && !methodName.matches(JUNIT_4_TEST_METHOD_NAME_REGEX)) {
            addViolation(data, node);
        } else if (methodName.startsWith("@BeforeEach") && !methodName.matches(JUNIT_5_TEST_METHOD_NAME_REGEX)) {
            addViolation(data, node);
        } else if (!node.isStatic() && !node.isNative() && !methodName.matches(INSTANCE_METHOD_NAME_REGEX)) {
            addViolation(data, node);
        }

        return super.visit(node, data);
    }
}
