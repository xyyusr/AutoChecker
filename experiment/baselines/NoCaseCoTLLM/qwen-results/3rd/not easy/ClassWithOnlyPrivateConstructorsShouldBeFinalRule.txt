import java.util.List;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ClassWithOnlyPrivateConstructorsShouldBeFinalRule extends AbstractJavaRulechainRule {
    public ClassWithOnlyPrivateConstructorsShouldBeFinalRule() {
        super(ASTClassOrInterfaceDeclaration.class);
    }
    
    @Override
    public Object visit(ASTClassOrInterfaceDeclaration node, Object data) {
        if (!node.isInterface() && !node.isAbstract()) {
            boolean allConstructorsPrivate = true;
            List<ASTConstructorDeclaration> constructors = node.getDeclaredConstructors();
            if (constructors.isEmpty()) {
                allConstructorsPrivate = false;
            } else {
                for (ASTConstructorDeclaration constructor : constructors) {
                    if (!constructor.hasPrivateModifier()) {
                        allConstructorsPrivate = false;
                        break;
                    }
                }
            }
            if (allConstructorsPrivate && !node.isFinal()) {
                addViolation(data, node);
            }
        }
        return super.visit(node, data);
    }
}

package net.sourceforge.pmd.lang.java.rule.design;


import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ClassWithOnlyPrivateConstructorsShouldBeFinalRule extends AbstractJavaRulechainRule {
    public ClassWithOnlyPrivateConstructorsShouldBeFinalRule() {
        super(ASTClassOrInterfaceDeclaration.class);
    }

    @Override
    public Object visit(ASTClassOrInterfaceDeclaration node, Object data) {
        // Check if the node is a class (not an interface)
        if (node.isInterface()) {
            return data;
        }

        // Get all constructors of the class
        List<ASTConstructorDeclaration> constructors = node.getDeclarations(ASTConstructorDeclaration.class);

        // If there are no constructors, the class can be final, but we are not checking for that here
        if (constructors.isEmpty()) {
            return data;
        }

        // Check if all constructors are private
        boolean allPrivate = true;
        for (ASTConstructorDeclaration constructor : constructors) {
            ASTModifiers modifiers = constructor.getModifiers();
            if (!modifiers.isPrivate()) {
                allPrivate = false;
                break;
            }
        }

        // If all constructors are private and the class is not final, report it
        if (allPrivate && !node.getModifiers().isFinal()) {
            addViolation(data, node);
        }

        return data;
    }
}
