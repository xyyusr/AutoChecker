    public LiteralsFirstInComparisonsRule() {
        super(ASTMethodCall.class);

            List<ASTExpression> arguments = node.getArguments();

                ASTExpression arg2 = arguments.get(1);

                JType type2 = arg2.getType();
                if (!isStringLiteral(arg1) && isStringLiteral(arg2) && isStringType(type1)) {
                    addViolation(data, node);
                }
            }
        }
        return super.visit(node, data);
    }

               methodName.equals("compareTo") || methodName.equals("compareToIgnoreCase") ||

        return expression instanceof ASTLiteral && ((ASTLiteral) expression).getValue() instanceof String;
    }

        return type != null && typeSystem.isA(type, typeSystem.STRING_TYPE);
    }
