import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ExcessivePublicCountRule extends AbstractJavaRulechainRule {
    public ExcessivePublicCountRule() {
        super(ASTClassOrInterfaceDeclaration.class);
    }

    @Override
    public Object visit(ASTClassOrInterfaceDeclaration node, Object data) {
        int publicFieldCount = 0;
        int publicMethodCount = 0;

        NodeStream<ASTFieldDeclaration> fields = node.getFields();
        for (ASTFieldDeclaration field : fields) {
            if (field.isPublic()) {
                publicFieldCount++;
            }
        }

        NodeStream<ASTMethodDeclaration> methods = node.getDeclaredMethods();
        for (ASTMethodDeclaration method : methods) {
            if (method.isPublic()) {
                publicMethodCount++;
            }
        }

        int totalPublicMembers = publicFieldCount + publicMethodCount;
        if (totalPublicMembers > 2) {
            addViolation(data, node);
        }

        return super.visit(node, data);
    }
}

package net.sourceforge.pmd.lang.java.rule.design;


import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ExcessivePublicCountRule extends AbstractJavaRulechainRule {
    private static final int DEFAULT_THRESHOLD = 2;
    private static final IntegerProperty THRESHOLD_DESCRIPTOR = IntegerProperty.named("threshold")
            .defaultValue(DEFAULT_THRESHOLD)
            .description("Maximum number of public methods and attributes allowed")
            .build();

    public ExcessivePublicCountRule() {
        super(ASTClassOrInterfaceDeclaration.class);
        definePropertyDescriptor(THRESHOLD_DESCRIPTOR);
    }

    @Override
    public Object visit(ASTClassOrInterfaceDeclaration node, Object data) {
        int threshold = getProperty(THRESHOLD_DESCRIPTOR);
        int publicMethodCount = node.findDescendantsOfType(ASTMethodDeclaration.class)
                .stream()
                .filter(ASTMethodDeclaration::isPublic)
                .count();

        int publicFieldCount = node.findDescendantsOfType(ASTFieldDeclaration.class)
                .stream()
                .filter(ASTFieldDeclaration::isPublic)
                .count();

        int publicCount = (int) (publicMethodCount + publicFieldCount);

        if (publicCount > threshold) {
            addViolation(data, node);
        }

        return super.visit(node, data);
    }
}
