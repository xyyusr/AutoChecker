                    }
                }
            }
        }
        return null;

package net.sourceforge.pmd.lang.java.rule.errorprone;

import java.util.*;
import java.lang.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class NullAssignmentRule extends AbstractJavaRulechainRule {

    public NullAssignmentRule() {
        addRuleChainElement(new AssignmentExpression());
        addRuleChainElement(new MethodDeclaration());
        addRuleChainElement(new VariableDeclaratorId());
    }

    @Override
    public Object visit(AssignmentExpression node, Object data) {
        Object result = super.visit(node, data);
        if (result == null) {
            if (node.getRightOp() instanceof LiteralExpression) {
                LiteralExpression literal = (LiteralExpression) node.getRightOp();
                if (literal.getLiteral() == null) {
                    VariableDeclaratorId var = getVariableDeclarator(node.getLeftOp());
                    if (var != null && !isLocalVariable(var)) {
                        addViolation(data, node);
                    }
                }
            }
        }
        return result;
    }

    private VariableDeclaratorId getVariableDeclarator(JavaNode node) {
        NodeStream<VariableDeclarationFragment> stream = getMethodScope().getVariables();
        VariableDeclarationFragment fragment = null;
        for (VariableDeclarationFragment fragment1 : stream) {
            if (fragment1.getBodyDeclarations().contains(node)) {
                fragment = fragment1;
                break;
            }
        }
        if (fragment != null) {
            return fragment.getVariableDeclarator(node);
        }
        return null;
    }

    private boolean isLocalVariable(VariableDeclaratorId var) {
        return getMethodScope().getVariables().contains(var);
    }
}
