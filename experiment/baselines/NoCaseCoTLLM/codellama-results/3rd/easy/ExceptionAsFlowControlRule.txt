import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;

import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ExceptionAsFlowControlRule extends AbstractJavaRulechainRule {

    @Override
    public Object visit(TryStatement node, Object data) {
        if (node.getCatchClauses() != null && node.getCatchClauses().size() > 0) {
            for (CatchClause catchClause : node.getCatchClauses()) {
                if (catchClause.getException() != null) {
                    for (Statement statement : node.getTryBlock().getStatements()) {
                        if (statement instanceof ThrowStatement) {
                            ThrowStatement throwStatement = (ThrowStatement) statement;
                            if (throwStatement.getException() != null && throwStatement.getException().getType().getName().equals(catchClause.getException().getType().getName())) {
                                // Report a violation
                                addViolation(data, node);
                                return data;
                            }
                        }
                    }
                }
            }
        }
        return super.visit(node, data);
    }
}

package net.sourceforge.pmd.lang.java.rule.design;

import java.util.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ExceptionAsFlowControlRule extends AbstractJavaRulechainRule {

    public ExceptionAsFlowControlRule() {
        addRuleMatch(ASTTryStatement.class);
    }

    @Override
    public Object visit(ASTTryStatement node, Object data) {
        List<ASTCatch> catchNodes = node.getCatchClauses();
        for (ASTCatch catchNode : catchNodes) {
            Object result = super.visit(catchNode.getStatement(), data);
            if (result != null) {
                return result;
            }
            result = visit(catchNode, data);
            if (result != null) {
                return result;
            }
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTCatch catchNode, Object data) {
        ASTBlock statement = catchNode.getStatement();
        List<ASTThrow> throwNodes = statement.getNodesOfType(ASTThrow.class);
        for (ASTThrow throwNode : throwNodes) {
            addViolation(this, catchNode, "Exception thrown and caught in the same try statement.");
        }
        return super.visit(catchNode, data);
    }
}
