public class AvoidThrowingNullPointerExceptionRule extends AbstractJavaRulechainRule {
    public AvoidThrowingNullPointerExceptionRule() {
        addRuleChainElement(new ASTThrowExpression());
    }

    @Override
    public Object visit(ASTThrowExpression node, Object data) {
        if (node.getExpression() instanceof ASTNewExpression) {
            ASTNewExpression newExpression = (ASTNewExpression) node.getExpression();
            if (newExpression.getType() instanceof ASTClassOrInterfaceType) {
                ASTClassOrInterfaceType classType = (ASTClassOrInterfaceType) newExpression.getType();
                if (classType.getPrimaryPrefix() instanceof ASTName) {
                    ASTName name = (ASTName) classType.getPrimaryPrefix();
                    if (name.getImage().equals("NullPointerException") && !isAbstractClass(name)) {
                        addViolation(data, node);
                    }
                }
            }
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTCastExpression node, Object data) {
        if (node.getExpression() instanceof ASTThrowExpression) {
            ASTThrowExpression throwExpression = (ASTThrowExpression) node.getExpression();
            return visit(throwExpression, data);
        }
        return super.visit(node, data);
    }

    private boolean isAbstractClass(ASTName name) {
        Type type = TypeSystem.getTypeSystem().resolve(name.getImage());
        if (type instanceof ClassOrInterfaceDeclaration) {
            ClassOrInterfaceDeclaration classDeclaration = (ClassOrInterfaceDeclaration) type;
            return classDeclaration.isAbstract();
        }
        return false;
    }

import net.sourceforge.pmd.lang.ast.NodeStream;
public class AvoidThrowingNullPointerExceptionRule extends AbstractJavaRulechainRule {

    @Override
    public Object visit(MethodDeclaration node, Object data) {
        Object result = super.visit(node, data);
        if (node.getBody() != null) {
            return visitThrowStatement(node.getBody(), data);
        }
        return result;
    }

    @Override
    public Object visit(ThrowStatement node, Object data) {
        Object result = super.visit(node, data);
        if (node.getException() != null) {
            if (isNullPointerException(node.getException()) || isNullLiteral(node.getException())) {
                addViolation(data, this, node);
            }
        }
        return result;
    }

    private boolean isNullPointerException(JavaNode node) {
        if (node instanceof ClassOrInterfaceDeclaration) {
            ClassOrInterfaceDeclaration classNode = (ClassOrInterfaceDeclaration) node;
            if (classNode.getName().equals("NullPointerException")) {
                return true;
            }
        }
        return false;
    }

    private boolean isNullLiteral(JavaNode node) {
        if (node instanceof NullLiteral) {
            return true;
        }
        if (node instanceof PrimitiveExpression) {
            PrimitiveExpression primitiveExpression = (PrimitiveExpression) node;
            if (primitiveExpression.getOperator() == Operator.NULL) {
                return true;
            }
        }
        return false;
    }
}
