import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class StringInstantiationRule extends AbstractJavaRulechainRule {

    @Override
    public Object visit(ASTNewClassExpression node, Object data) {
        // Get the type of the class being instantiated
        ASTType type = (ASTType) node.getType();

        // Get the fully qualified name of the class
        String typeName = type.getImage();

        // Check if the class is String or its subclasses
        if (typeName.equals("java.lang.String") || isSubclassOf(typeName, "java.lang.String")) {
            // Report the instantiation as a rule violation
            addRuleViolation(node.getImage());
        }

        // Continue visiting other nodes
        return super.visit(node, data);
    }

    private boolean isSubclassOf(String typeName, String superClassType) {
        TypeResolver resolver = TypeResolver.getInstance();
        return resolver.isSubclassOf(typeName, superClassType);
    }
}
