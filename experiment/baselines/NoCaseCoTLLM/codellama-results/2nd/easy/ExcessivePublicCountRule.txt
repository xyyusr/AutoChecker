import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ExcessivePublicCountRule extends AbstractJavaRulechainRule {

    public ExcessivePublicCountRule() {
        super(ClassOrInterfaceDeclaration.class);
    }

    @Override
    public Object visit(ClassOrInterfaceDeclaration node, Object data) {

        // Count public methods
        for (MethodDeclaration method : node.getMethods()) {
            if (method.isPublic()) {
                publicCount++;

        // Count public attributes

            if (field.isPublic()) {
                publicCount++;
            }
        }

        // Apply the threshold
        if (publicCount > 2) {
            addViolation(data, this, node, "Class has too many public methods and attributes");
        }

        return super.visit(node, data);
    }
}

import java.util.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ExcessivePublicCountRule extends AbstractJavaRulechainRule {

    private static final int THRESHOLD = 2;
    private static final String RULE_NAME = "ExcessivePublicCountRule";

    public ExcessivePublicCountRule() {
        addVisitor(new ClassVisitor());
    }

    @Override
    public Object visit(Node node, Object data) {
        return super.visit(node, data);
    }

    private class ClassVisitor extends AbstractJavaRuleVisitor {

        @Override
        public Object visit(JavaClass node, Object data) {
            int publicMethodCount = 0;
            int publicFieldCount = 0;

            for (JavaBodyMethodDeclaration method : (List<JavaBodyMethodDeclaration>) node.getBody().getMethods()) {
                if (method.getModifiers().isPublic()) {
                    publicMethodCount++;
                }
            }

            SymbolTable symbolTable = node.getSymbolTable();
            for (SymbolReference symbolReference : symbolTable.getSymbolReferences()) {
                if (symbolReference.getSymbol() instanceof JavaFieldSymbol) {
                    JavaFieldSymbol fieldSymbol = (JavaFieldSymbol) symbolReference.getSymbol();
                    if (fieldSymbol.getModifiers().isPublic()) {
                        publicFieldCount++;
                    }
                }
            }

            if (publicMethodCount + publicFieldCount > THRESHOLD) {
                addViolation(data, node);
            }

            return super.visit(node, data);
        }
    }
}

import java.util.Map;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ExcessivePublicCountRule extends AbstractJavaRulechainRule {

    private static final int DEFAULT_THRESHOLD = 2;

    public ExcessivePublicCountRule() {
        addRuleChainElement(new MethodCallRuleChainElement());
    }

    @Override
    public Object visit(ASTClassOrInterfaceDeclaration node, Object data) {
        int publicMethodCount = 0;
        int publicAttributeCount = 0;

        SymbolTable symbolTable = SymbolTableBuilder.build(node);

        ClassSymbolTable classSymbolTable = (ClassSymbolTable) symbolTable.getClasses().get(node);

        if (classSymbolTable != null) {
            List<MethodSymbolTable> methods = classSymbolTable.getMethods();
            if (methods != null) {
                for (MethodSymbolTable method : methods) {
                    String name = method.getName();
                    if ((method.getAccess() & SymbolTable.ACCESS_PUBLIC) != 0) {
                        publicMethodCount++;
                    }
                }
            }

            List<FieldSymbolTable> fields = classSymbolTable.getFields();
            if (fields != null) {
                for (FieldSymbolTable field : fields) {
                    if ((field.getAccess() & SymbolTable.ACCESS_PUBLIC) != 0) {
                        publicAttributeCount++;
                    }
                }
            }
        }

        if (publicMethodCount > DEFAULT_THRESHOLD || publicAttributeCount > DEFAULT_THRESHOLD) {
            addViolation(this, node, "Class '" + node.getImage() + "' has excessive public methods (" + publicMethodCount + ") and/or attributes (" + publicAttributeCount + ").");
        }

        return super.visit(node, data);
    }
}
