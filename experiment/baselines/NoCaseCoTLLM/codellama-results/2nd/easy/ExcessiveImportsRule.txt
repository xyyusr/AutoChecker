package net.sourceforge.pmd.lang.java.rule.design;

import java.util.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ExcessiveImportsRule extends AbstractJavaRulechainRule {

    private static final int DEFAULT_THRESHOLD = 3;

    public ExcessiveImportsRule() {
        addVisitHandler(CompilationUnit.class);
    }

    @Override
    public Object visit(CompilationUnit node, Object data) {
        int importCount = countImports(node);
        if (importCount > DEFAULT_THRESHOLD) {
            reportViolation(node, importCount);
        }
        return super.visit(node, data);
    }

    private int countImports(CompilationUnit node) {
        return ((ImportDeclaration[]) node.findDescendants(ImportDeclaration.class)).length;
    }

    private void reportViolation(CompilationUnit node, int importCount) {
        Violation violation = new Violation(getName(), getDescription());
        violation.setLocation(node.getBeginLine(), node.getBeginColumn());
        node.addViolation(violation);
    }

    @Override
    public String getName() {
        return "Excessive Imports";
    }

    @Override
    public String getDescription() {
        return "A high number of imports can indicate a high degree of coupling within an object.";
    }
}
