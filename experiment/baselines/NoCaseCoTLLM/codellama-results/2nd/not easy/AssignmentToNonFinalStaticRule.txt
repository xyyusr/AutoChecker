package net.sourceforge.pmd.lang.java.rule.errorprone;

import java.util.*;
import java.lang.*;
import java.util.stream.Collectors;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class AssignmentToNonFinalStaticRule extends AbstractJavaRulechainRule {

    public AssignmentToNonFinalStaticRule() {
        super(ClassOrInterfaceDeclaration.class);
    }

    @Override
    public Object visit(ClassOrInterfaceDeclaration node, Object data) {
        List<String> nonFinalStaticFields = traverseNonFinalStaticFields(node);
        for (String field : nonFinalStaticFields) {
            checkAssignmentInConstructor(node, field);
        }
        return super.visit(node, data);
    }

    private List<String> traverseNonFinalStaticFields(ClassOrInterfaceDeclaration node) {
        List<String> nonFinalStaticFields = new ArrayList<>();
        for (FieldDeclaration field : node.getFields()) {
            if (field.isStatic() && !field.isFinal()) {
                nonFinalStaticFields.add(field.getImage());
            }
        }
        return nonFinalStaticFields;
    }

    private void checkAssignmentInConstructor(ClassOrInterfaceDeclaration node, String field) {
        for (MethodDeclaration method : node.getMethodDeclarations()) {
            if (method.isConstructor()) {
                for (AssignmentExpression assignment : method.getBody().getStatements().stream()
                        .filter(statement -> statement instanceof AssignmentExpression)
                        .map(statement -> (AssignmentExpression) statement)
                        .collect(Collectors.toList())) {
                    if (assignment.getLeftOperand().getImage().equals(field)) {
                        addRuleViolation(this, node, method, assignment);
                        break;
                    }
                }
            }
        }
    }

    private void addRuleViolation(Rule rule, ClassOrInterfaceDeclaration node, MethodDeclaration method, AssignmentExpression assignment) {
        addViolation(rule, node, rule.getDescription() + " " + method.getImage() + " at line " + method.getBeginLine());
    }

    @Override
    public Object visit(MethodDeclaration node, Object data) {
        return super.visit(node, data);
    }
}
