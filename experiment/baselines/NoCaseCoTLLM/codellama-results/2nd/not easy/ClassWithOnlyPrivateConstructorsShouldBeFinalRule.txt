    @Override
    public Object visit(ASTClassOrInterfaceDeclaration node, Object data) {

            addViolation(data, this, node);
        }
        return super.visit(node, data);
    }

    private boolean isSubclass(ASTClassOrInterfaceDeclaration node) {
        if (!(node instanceof ASTClassOrInterfaceDeclaration)) {
            return false;
        }
        return node.getParent() instanceof ASTClassOrInterfaceDeclaration;
    }

    private boolean areAllConstructorsPrivate(ASTClassOrInterfaceDeclaration node) {
        if (!(node instanceof ASTClassOrInterfaceDeclaration)) {
            return false;
        }
        ASTMethodDeclaration[] constructors = node.findDescendants(ASTMethodDeclaration.class);
        for (ASTMethodDeclaration constructor : constructors) {
            if (!constructor.getImage().startsWith("private")) {
                return false;
            }
        }
        return true;
    }

    private boolean haveSuperConstructorCall(ASTClassOrInterfaceDeclaration node) {
        if (!(node instanceof ASTClassOrInterfaceDeclaration)) {
            return false;
        }
        ASTMethodInvocation superConstructorCall = node.findDescendants(ASTMethodInvocation.class);
        if (superConstructorCall == null) {
            return false;
        }
        return superConstructorCall.getImage().startsWith("super");
    }

    private void checkConstructors(ASTClassOrInterfaceDeclaration node) {
        if (!(node instanceof ASTClassOrInterfaceDeclaration)) {
            return;
        }
        ASTMethodDeclaration[] constructors = node.findDescendants(ASTMethodDeclaration.class);
        for (ASTMethodDeclaration constructor : constructors) {
            if (!constructor.getImage().startsWith("private")) {
                return;
            }
            if (!haveSuperConstructorCall(node)) {
                return;
            }
        }
        addViolation(data, this, node);
    }

    @Override
    public Object visit(ASTMethodDeclaration node, Object data) {
        if (node instanceof ASTConstructorDeclaration) {
            ASTClassOrInterfaceDeclaration classNode = (ASTClassOrInterfaceDeclaration) node.getParent();
            checkConstructors(classNode);
        }
        return super.visit(node, data);
    }
}

package net.sourceforge.pmd.lang.java.rule.design;

import java.util.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ClassWithOnlyPrivateConstructorsShouldBeFinalRule extends AbstractJavaRulechainRule {

    @Override
    public Object visit(JClassOrInterfaceDeclaration node, Object data) {
        if (node.isFinal()) {
            return super.visit(node, data);
        }

        int privateConstructorCount = 0;
        int constructorCount = 0;

        for (Object member : node.getMembers()) {
            JMethodDeclaration method = (JMethodDeclaration) member;
            if (method.getName().equals(node.getName() + "()")) { // Check if it's a constructor
                constructorCount++;
                if (method.isPrivate()) {
                    privateConstructorCount++;
                }
            }
        }

        if (constructorCount == privateConstructorCount) {
            addViolation(this, data, node);
        }

        return super.visit(node, data);
    }
}

    @Override
    public Object visit(JavaClassOrInterfaceDeclaration node, Object data) {
        if (node.isClass()) {
            // Check if the class has a private constructor
            if (hasOnlyPrivateConstructors(node)) {
                // Report the class if it has only private constructors
                report(node);
            }
        }
        // Recursively visit child nodes
        return super.visit(node, data);
    }

    private boolean hasOnlyPrivateConstructors(JavaClassOrInterfaceDeclaration node) {
        // Get all constructors of the class
        List<JavaConstructorDeclaration> constructors = new ArrayList<>();
        for (JavaMethodDeclaration method : node.getMethods()) {
            if (method.isConstructor()) {
                constructors.add((JavaConstructorDeclaration) method);

        }

        // Check if all constructors are private
        for (JavaConstructorDeclaration constructor : constructors) {
            if (!constructor.isPrivate()) {
                return false;
            }
        }

public class ClassWithOnlyPrivateConstructorsShouldBeFinalRule extends AbstractJavaRulechainRule {

    public ClassWithOnlyPrivateConstructorsShouldBeFinalRule() {
        super(ClassOrInterfaceDeclaration.class);
    }

    @Override
    public Object visit(ClassOrInterfaceDeclaration n, Object data) {
        if (n.isClass()) {
            ClassDeclaration c = (ClassDeclaration) n;
            if (c.getConstructors().isEmpty()) {
                // Class has no constructors, skip
                return super.visit(n, data);
            }

            boolean allConstructorsPrivate = true;
            for (MethodDeclaration constructor : c.getConstructors()) {
                if (!constructor.isPrivate()) {
                    // Not all constructors are private, skip
                    allConstructorsPrivate = false;
                    break;
                }
            }

            if (allConstructorsPrivate) {
                // Report class as potential candidate for being made final
                addViolation(this, data, n);
            }
        }
        return super.visit(n, data);
    }
