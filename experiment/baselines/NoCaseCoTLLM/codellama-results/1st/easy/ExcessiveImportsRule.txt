public class ExcessiveImportsRule extends AbstractJavaRulechainRule {

    public ExcessiveImportsRule() {
        setLanguages(new String[] {"java"});
        addRuleChainElement(new CompilationUnitPropertyRuleChainElement());

    @Override
    public Object visit(CompilationUnit node, Object data) {
        int importCounter = 0;
        for (ImportDeclaration importDecl : node.findAll(ImportDeclaration.class)) {

        return super.visit(node, data);
    }
}

class ExcessiveImports extends RuleViolation {
    private CompilationUnit node;
    private int importCount;

    public ExcessiveImports(CompilationUnit node, int importCount) {
        super(node, "Excessive imports: " + importCount + " imports");
        this.node = node;
        this.importCount = importCount;
    }

    public CompilationUnit getNode() {
        return node;
    }

    public int getImportCount() {
        return importCount;
    }
}

package net.sourceforge.pmd.lang.java.rule.design;

import java.util.HashMap;
import java.util.regex.Matcher;
import java.util.regex.Pattern;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ExcessiveImportsRule extends AbstractJavaRulechainRule {

    public ExcessiveImportsRule() {
        super(ASTImportDeclaration.class);
    }

    @Override
    public Object apply(ASTImportDeclaration node, Object data) {
        String sourceCode = node.getImage();
        Pattern pattern = Pattern.compile("^import\\s+.*;$");
        Matcher matcher = pattern.matcher(sourceCode);
        int importCount = 0;
        while (matcher.find()) {
            importCount++;
        }
        
        if (importCount > 3) {
            addViolation(this, node, data, "There are too many imports in this file (" + importCount + ">");
        }
        
        return super.apply(node, data);
    }
}
