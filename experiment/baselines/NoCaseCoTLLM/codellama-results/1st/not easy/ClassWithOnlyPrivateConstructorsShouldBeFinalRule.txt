import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ClassWithOnlyPrivateConstructorsShouldBeFinalRule extends AbstractJavaRulechainRule {

    @Override
    public Object visit(ASTCompilationUnit node, Object data) {
        for (ASTClassOrInterfaceDeclaration classDecl : node.jjtGetChildren()) {
            if (classDecl instanceof ASTClassDeclaration) {
                ASTClassDeclaration classNode = (ASTClassDeclaration) classDecl;
                if (!hasNonPrivateConstructors(classNode)) {
                    if (!hasNonPrivateMethods(classNode)) {
                        addViolation(this, data, classNode);
                    }
                }
            }
        }
        return super.visit(node, data);
    }

    private boolean hasNonPrivateConstructors(ASTClassDeclaration classNode) {
        for (ASTMethodDeclaration method : classNode.getMethods()) {
            if (method instanceof ASTConstructorDeclaration) {
                ASTConstructorDeclaration constructor = (ASTConstructorDeclaration) method;
                if (constructor.getModifiers().isPublic()) {
                    return true;
                }
            }
        }
        return false;
    }

    private boolean hasNonPrivateMethods(ASTClassDeclaration classNode) {
        for (ASTMethodDeclaration method : classNode.getMethods()) {
            if (!(method instanceof ASTConstructorDeclaration) && !method.getModifiers().isStatic() && !method.getModifiers().isPrivate()) {
                return true;
            }
        }
        return false;
    }
}

import net.sourceforge.pmd.lang.ast.NodeStream;
public class ClassWithOnlyPrivateConstructorsShouldBeFinalRule extends AbstractJavaRulechainRule {

    public ClassWithOnlyPrivateConstructorsShouldBeFinalRule() {
        addRuleMatch(CategorizedMessage.java(Check, "class_with_only_private_constructors_should_be_final", "This class may be made final because it cannot be extended from outside its compilation unit anyway."));
    }

    @Override
    public Object visit(ClassOrInterfaceDeclaration node, Object data) {
        if (node.getModifiers().hasModifier(Modifier.FINAL)) {
            return data;
        }

        List<MethodDeclaration> constructors = new ArrayList<>();
        for (MethodDeclaration method : node.getBody().getMethods()) {
            if (method.getNameString().equals(node.getNameString()) && method.getParameters().isEmpty()) {
                constructors.add(method);
            }
        }

        if (constructors.isEmpty() || !constructors.stream().allMatch(m -> m.getModifiers().hasModifier(Modifier.PRIVATE))) {
            reportRuleViolation(CategorizedMessage.java(Check, "class_with_only_private_constructors_should_be_final", "This class may be made final because it cannot be extended from outside its compilation unit anyway."));
        }

        return super.visit(node, data);
    }
}

package net.sourceforge.pmd.lang.java.rule.design;


import java.util.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ClassWithOnlyPrivateConstructorsShouldBeFinalRule extends AbstractJavaRulechainRule {

    @Override
    public Object visit(CLASS_DEF node, Object data) {
        ClassScope classScope = SymbolTableManager.getInstance().getScope(node);
        Collection<MethodDeclaration> methods = classScope.getMethods().stream()
                .filter(method -> method.getIdentifier().equals(node.getName()))
                .collect(Collectors.toList());

        // If class has no methods, it may be final
        if (methods.isEmpty()) {
            return null;
        }

        Set<String> privateConstructors = new HashSet<>();
        methods.forEach(method -> {
            if (method.getModifiers().contains(Modifier.PRIVATE)) {
                privateConstructors.add(method.getIdentifier());
            }
        });

        if (methods.stream().allMatch(method -> privateConstructors.contains(method.getIdentifier()))) {
            addViolation(this, node.getImage());
        }

        return super.visit(node, data);
    }
}
