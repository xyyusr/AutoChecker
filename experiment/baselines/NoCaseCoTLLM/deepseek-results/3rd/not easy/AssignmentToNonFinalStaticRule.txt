            }
        }

    public Object visit(ASTConstructorDeclaration node, Object data) {
        ASTClassOrInterfaceDeclaration classNode = node.getFirstParentOfType(ASTClassOrInterfaceDeclaration.class);
        if (classNode != null) {
            List<ASTFieldDeclaration> staticFields = classNode.findChildrenOfType(ASTFieldDeclaration.class)
                    .filter(field -> field.hasModifiers(JModifier.STATIC) && !field.hasModifiers(JModifier.FINAL))

                            && ((ASTVariableAccess) assign.getLeftOperand()).getName().equals(fieldName))

            }
        }
        return super.visit(node, data);
    }

    public Object visit(ASTFieldDeclaration node, Object data) {
