package net.sourceforge.pmd.lang.java.rule.errorprone;

import java.util.*;
import java.lang.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class BrokenNullCheckRule extends AbstractJavaRulechainRule {
    public BrokenNullCheckRule() {
        super(ASTConditionalOrExpression.class, ASTConditionalAndExpression.class);
    }

    @Override
    public Object visit(ASTConditionalOrExpression node, Object data) {
        checkNullCheck(node, data);
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTConditionalAndExpression node, Object data) {
        checkNullCheck(node, data);
        return super.visit(node, data);
    }

    private void checkNullCheck(ASTExpression node, Object data) {
        if (node.getNumChildren() == 2) {
            ASTExpression left = (ASTExpression) node.getChild(0);
            ASTExpression right = (ASTExpression) node.getChild(1);

            if (isNullCheck(left, right) || isNullCheck(right, left)) {
                addViolation(data, node);
            }
        }
    }

    private boolean isNullCheck(ASTExpression expr1, ASTExpression expr2) {
        if (JavaAstUtils.isNullLiteral(expr1)) {
            return JavaRuleUtil.isNullChecked(expr2);
        }
        return false;
