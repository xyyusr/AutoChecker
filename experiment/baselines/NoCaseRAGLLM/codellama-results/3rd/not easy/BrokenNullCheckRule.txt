public class BrokenNullCheckRule extends AbstractJavaRulechainRule {

    public BrokenNullCheckRule() {
        super(MethodDeclaration.class);
    }

    @Override
    public Object visit(MethodDeclaration node, Object data) {
        List<ASTExpression> expressions = node.getBody().getExpressionList();
        for (ASTExpression expression : expressions) {
            if (JavaRuleUtil.isNullLiteral(expression)) {
                if (isLogicalBinaryOp(expression)) {
                    List<ASTExpression> operands = getOperands(expression);
                    for (ASTExpression operand : operands) {
                        if (JavaRuleUtil.isNullChecked(operand)) {
                            addViolation(data, this, operand);
                        }
                    }
                }
            }
        }
        return super.visit(node, data);
    }

    private boolean isLogicalBinaryOp(ASTExpression expression) {
        return expression instanceof ASTBinaryExpression && ((ASTBinaryExpression) expression).getOperator().isOperator(Operator.LOGICAL_OR) ||
               expression instanceof ASTBinaryExpression && ((ASTBinaryExpression) expression).getOperator().isOperator(Operator.LOGICAL_AND);
    }

    private List<ASTExpression> getOperands(ASTExpression expression) {
        if (expression instanceof ASTBinaryExpression) {
            return Arrays.asList(((ASTBinaryExpression) expression).getLeftOperand(), ((ASTBinaryExpression) expression).getRightOperand());
        }
        return Collections.emptyList();
    }
}
