import java.util.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class UnnecessaryImportRule extends AbstractJavaRulechainRule {

    public UnnecessaryImportRule() {
        super(ASTImportDeclaration.class, ASTClassOrInterfaceDeclaration.class, ASTMethodDeclaration.class, ASTFieldDeclaration.class, ASTCompilationUnit.class);
    }

    @Override
    public Object visit(ASTImportDeclaration node, Object data) {
        if (isUnusedImport(node, data)) {
            addViolation(data, node);
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTClassOrInterfaceDeclaration node, Object data) {
        if (isRedundantImport(node, node.getSymbolTable())) {
            addViolation(data, node);
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTMethodDeclaration node, Object data) {
        if (isRedundantImport(node, node.getSymbolTable())) {
            addViolation(data, node);
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTFieldDeclaration node, Object data) {
        if (isRedundantImport(node, node.getSymbolTable())) {
            addViolation(data, node);
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTCompilationUnit node, Object data) {
        for (ASTImportDeclaration importDeclaration : node.findDescendantsOfType(ASTImportDeclaration.class, false)) {
            if (isUnusedImport(importDeclaration, data) || isDuplicatedImport(importDeclaration, data)) {
                addViolation(data, importDeclaration);
            }
        }
        return super.visit(node, data);
    }

    private boolean isUnusedImport(ASTImportDeclaration importDeclaration, Object data) {
        JSymbolTable symbolTable = importDeclaration.getSymbolTable();
        String importedType = importDeclaration.getImportedSimpleName();
        if (importedType == null) {
            return false;
        }
        try {
            symbolTable.resolve(importedType);
        } catch (JSymbolTable.SymbolNotFoundException e) {
            return true;
        }
        return false;
    }

    private boolean isDuplicatedImport(ASTImportDeclaration importDeclaration, Object data) {
        List<ASTImportDeclaration> importDeclarations = getImportDeclarations(data);
        for (ASTImportDeclaration otherImportDeclaration : importDeclarations) {
            if (importDeclaration.equals(otherImportDeclaration) && !importDeclaration.isStatic() && otherImportDeclaration.isStatic()) {
                return true;
            }
        }
        return false;
    }

    private boolean isRedundantImport(ASTMethodDeclaration methodDeclaration, JSymbolTable symbolTable) {
        String importedType = methodDeclaration.getName();
        if (importedType == null) {
            return false;
        }
        Type type = symbolTable.resolve(importedType);
        if (type == null) {
            return false;
        }
        String typeQualifiedName = type.getQualifiedName();
        if (typeQualifiedName == null) {
            return false;
        }
        if (typeQualifiedName.startsWith("java.lang.") || typeQualifiedName.equals("")) {
            return true;
        }
        return false;
    }

    private boolean isRedundantImport(ASTFieldDeclaration fieldDeclaration, JSymbolTable symbolTable) {
        String importedType = fieldDeclaration.getType().getName();
        if (importedType == null) {
            return false;
        }
        Type type = symbolTable.resolve(importedType);
        if (type == null) {
            return false;
        }
        String typeQualifiedName = type.getQualifiedName();
        if (typeQualifiedName == null) {
            return false;
        }
        if (typeQualifiedName.startsWith("java.lang.") || typeQualifiedName.equals("")) {
            return true;
        }
        return false;
    }

    private List<ASTImportDeclaration> getImportDeclarations(Object data) {
        List<ASTImportDeclaration> importDeclarations = new ArrayList<>();
        NodeStream<ASTCompilationUnit> compilationUnits = dataDescendants(ASTCompilationUnit.class);
        for (ASTCompilationUnit compilationUnit : compilationUnits) {
            importDeclarations.addAll(compilationUnit.findDescendantsOfType(ASTImportDeclaration.class, true));
        }
        return importDeclarations;
    }
}

package net.sourceforge.pmd.lang.java.rule.codestyle;

import java.util.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class UnnecessaryImportRule extends AbstractJavaRulechainRule {

    public UnnecessaryImportRule() {
        super(ASTImportDeclaration.class);
    }

    @Override
    public Object visit(ASTImportDeclaration node, Object data) {
        if (node.isStatic()) {
            String importedSimpleName = node.getImportedSimpleName();
            if (importedSimpleName != null) {
                if (isImplicitlyDefined(node, importedSimpleName)) {
                    reportViolation(node, "Imported member '" + importedSimpleName + "' is implicitly in scope.");
                } else if (isDuplicateImport(node)) {
                    reportViolation(node, "Duplicate import of '" + importedSimpleName + "'.");
                } else if (isUnusedImport(node)) {
                    reportViolation(node, "Unused import '" + importedSimpleName + "'.");
                }
            }
        }
        return super.visit(node, data);
    }

    private boolean isImplicitlyDefined(ASTImportDeclaration node, String importedSimpleName) {
        ASTCompilationUnit compilationUnit = (ASTCompilationUnit) node.getParent().getParent();
        ASTPackageDeclaration packageDeclaration = compilationUnit.getPackage();
        if (packageDeclaration != null) {
            List<List<String>> importedTypes = packageDeclaration.getDeclaredAnnotations();
            for (List<String> type : importedTypes) {
                for (String name : type) {
                    if (name.equals(importedSimpleName)) {
                        return true;
                    }
                }
            }
        }
        return false;
    }

    private boolean isDuplicateImport(ASTImportDeclaration node) {
        ASTCompilationUnit compilationUnit = (ASTCompilationUnit) node.getParent().getParent();
        Set<String> importNames = new HashSet<>();
        for (ASTImportDeclaration importDeclaration : compilationUnit.getImports()) {
            if (importDeclaration.isStatic()) {
                String importedSimpleName = importDeclaration.getImportedSimpleName();
                if (importedSimpleName != null && importNames.contains(importedSimpleName)) {
                    return true;
                }
                importNames.add(importedSimpleName);
            }
        }
        return false;
    }

    private boolean isUnusedImport(ASTImportDeclaration node) {
        ASTCompilationUnit compilationUnit = (ASTCompilationUnit) node.getParent().getParent();
        for (ASTImportDeclaration importDeclaration : compilationUnit.getImports()) {
            if (importDeclaration.isStatic()) {
                String importedSimpleName = importDeclaration.getImportedSimpleName();
                if (importedSimpleName != null && !isUsed(importedSimpleName, compilationUnit)) {
                    return true;
                }
            }
        }
        return false;
    }

    private boolean isUsed(String importedSimpleName, ASTCompilationUnit compilationUnit) {
        for (ASTCompilationUnit.ImportEntry entry : compilationUnit.getImports()) {
            if (entry.getImportedSimpleName().equals(importedSimpleName)) {
                return true;
            }
        }
        return false;
    }

    private void reportViolation(ASTImportDeclaration node, String message) {
        addRuleViolation(node.getReportLocation(), message);
    }
}

    public UnnecessaryImportRule() {
        addRuleChainElement(new ASTImportDeclaration());
        addRuleChainElement(new ASTPackageDeclaration());
        addRuleChainElement(new ASTClassOrInterfaceDeclaration());
        addRuleChainElement(new ASTMethodDeclaration());
        addRuleChainElement(new ASTCompilationUnit());
    }

    @Override
    public Object visit(ASTImportDeclaration node, Object data) {
        if (node.isStatic() || isImportedMemberAlreadyImplicit(node)) {
            return addViolation(data, node);
        }
        return super.visit(node, data);
    }

    @Override

            if (!importDeclaration.isStatic()) {
                importNames.add(importDeclaration.getImportedSimpleName());

    @Override
    public Object visit(ASTImportDeclaration node, Object data) {
        if (isPackageImport(node) || isStaticImport(node)) {
            return super.visit(node, data);
        }

        Iterable<JavacToken> tokens = node.tokens();
        String importedSimpleName = node.getImportedSimpleName();

        boolean isImported = true;
        if (importedSimpleName == null || importedSimpleName.isEmpty()) {
            isImported = false;
        }

        if (!isImported) {
            return super.visit(node, data);
        }
