package net.sourceforge.pmd.lang.java.rule.design;

import java.util.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ExcessivePublicCountRule extends AbstractJavaRulechainRule {

    @Override
    public Object visit(ASTClassOrInterfaceDeclaration node, Object data) {
        int publicMethods = 0;
        int publicFields = 0;
        int publicConstructors = 0;
        int publicInnerClasses = 0;

        publicMethods += ASTMethodDeclaration.class.countDescendants(node);
        publicFields += ASTFieldDeclaration.class.countDescendants(node);
        publicConstructors += ASTConstructorDeclaration.class.countDescendants(node);
        publicInnerClasses += ASTInnerClassDeclaration.class.countDescendants(node);

        if (publicMethods > 2 || publicFields > 2 || publicConstructors > 2 || publicInnerClasses > 2) {
            addRuleViolation(data, node, "Too many public methods, fields, constructors, or inner classes: " + (publicMethods + publicFields + publicConstructors + publicInnerClasses));
        }

        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTInnerClassDeclaration node, Object data) {
        int publicMethods = 0;
        int publicFields = 0;

        publicMethods += ASTMethodDeclaration.class.countDescendants(node);
        publicFields += ASTFieldDeclaration.class.countDescendants(node);

        if (publicMethods > 2 || publicFields > 2) {
            addRuleViolation(data, node, "Too many public methods, fields: " + (publicMethods + publicFields));
        }

        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTInterfaceDeclaration node, Object data) {
        int publicMethods = 0;

        publicMethods += ASTMethodDeclaration.class.countDescendants(node);

        if (publicMethods > 2) {
            addRuleViolation(data, node, "Too many public methods: " + publicMethods);
        }

        return super.visit(node, data);
    }

    private void addRuleViolation(Object data, JavaNode node, String message) {
        addRuleViolation(data, node.getBeginLine(), node.getBeginColumn(), node.getEndLine(), node.getEndColumn(), message);
    }
}

    public Object visit(ASTClassOrInterfaceDeclaration node, Object data) {
        int publicMethods = 0;
        int publicFields = 0;
        int nonPublicMethods = 0;
        int nonPublicFields = 0;

        for (ASTMethodDeclaration method : node.getMethods()) {
            if (method.isSyntacticallyPublic()) {
                publicMethods++;
            } else {
                nonPublicMethods++;
            }
        }

        for (ASTFieldDeclaration field : node.getFields()) {
            if (field.isSyntacticallyPublic()) {
                publicFields++;

                nonPublicFields++;
