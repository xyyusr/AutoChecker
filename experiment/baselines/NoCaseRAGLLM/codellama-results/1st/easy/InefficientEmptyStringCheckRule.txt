        Object result = super.visit(node, data);
        NodeStream<JavaNode> statements = node.getBody().children(MethodDeclaration.BODY_STATEMENTS);
        for (JavaNode statement : statements) {
            if (statement instanceof ExpressionStatement) {
                Expression expression = ((ExpressionStatement) statement).getExpression();
                if (expression instanceof MethodCallExpression) {
                    MethodCall methodCall = (MethodCall) expression;
                    if (methodCall.getTarget() instanceof MethodName) {
                        String methodName = ((MethodName) methodCall.getTarget()).getImage();
                        if ("trim".equals(methodName) || "isEmpty".equals(methodName)) {
                            NodeStream<JavaNode> args = methodCall.getArguments();
                            if (args.size() == 1) {
                                ASTStringLiteral stringLiteral = (ASTStringLiteral) args.get(0);
                                if (stringLiteral.isEmpty()) {

                                        if (stringLiteral.isEmpty()) {
                                            addViolation(data, this, node);
                                        }
                                    }
                                }
                            }
                        }
                    }
                }
            } else if (statement instanceof Block) {
                result = super.visit(statement, data);
            }
        }
        return result;
    }

    @Override
    public Object visit(FieldDeclaration node, Object data) {

public class InefficientEmptyStringCheckRule extends AbstractJavaRulechainRule {
    public InefficientEmptyStringCheckRule() {
        super(ASTExpression.class, ASTMethodCall.class);
    }

    @Override
    public Object visit(ASTExpression node, Object data) {
        if (hasMethodCall(node, "trim") && hasMethodCall(node, "length") && hasMethodCall(node, "isEmpty")) {
            addViolation(data, node);
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTMethodCall node, Object data) {
        if (hasMethodCall(node, "length") && isCallOnTrim(node)) {
            addViolation(data, node);
        }
        if (hasMethodCall(node, "isEmpty") && isCallOnTrim(node)) {
            addViolation(data, node);
        }
        return super.visit(node, data);
    }

    private boolean hasMethodCall(ASTNode node, String methodName) {
        for (ASTNode methodCall : children(node, ASTMethodCall.class)) {
            if (methodCall.getImage().contains(methodName)) {
                return true;
            }
        }
        return false;
    }

    private boolean isCallOnTrim(ASTMethodCall node) {
        for (ASTNode argument : children(node, ASTArgument.class)) {
            if (argument.getImage().contains("trim")) {
                return true;
            }
        }
        return false;
    }

    private void addViolation(Object data, ASTNode node) {
        addRuleViolation(this, data, node.getImage());
    }
}
