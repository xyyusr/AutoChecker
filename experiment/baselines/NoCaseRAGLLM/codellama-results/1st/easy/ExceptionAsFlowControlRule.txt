        super(JavaTryStatement.class);

    public Object visit(JavaTryStatement n, Object data) {

        if (tryStatement != null && tryStatement.isFindBoundary()) {
            addViolation(data, n.getReportLocation(), "Exceptions thrown and caught in an enclosing try statement");

        return super.visit(n, data);

        NodeStream<ThrowStatement> descendants = node.descendants(ThrowStatement.class);
        for (ThrowStatement ts : descendants) {
            ts.acceptVisitor(this, data);
            throwStatements.add(ts);
