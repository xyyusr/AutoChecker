public class IdenticalCatchBranchesRule extends AbstractJavaRulechainRule {

    public IdenticalCatchBranchesRule() {
        super(CompoundStatement.class);
    }

    @Override
    public Object visit(CompoundStatement node, Object data) {
        NodeStream<JavaNode> catchClauses = node.jjtGetChild(0).children(ASTCatchClause.class);
        List<CatchClause> catchClausesList = new ArrayList<>();
        for (JavaNode child : catchClauses.asStream()) {
            catchClausesList.add((CatchClause) child);
        }
        catchClausesList.sort((a, b) -> {
            NodeStream<JavaNode> aDescendants = a.descendantsOrSelf();
            NodeStream<JavaNode> bDescendants = b.descendantsOrSelf();
            return aDescendants.asStream().findFirst().orElse(null).getClass().getName()
                    .compareTo(bDescendants.asStream().findFirst().orElse(null).getClass().getName());
        });
        Set<String> catchParamSet = new HashSet<>();
        for (int i = 0; i < catchClausesList.size() - 1; i++) {
            if (areCatchClausesIdentical(catchClausesList.get(i), catchClausesList.get(i + 1))) {
                throw new RuleViolation(catchClausesList.get(i), "Identical catch branches use up vertical space and increase the complexity of code without adding functionality.");
            } else {
                CatchParameter catchParam = catchClausesList.get(i).jjtGetChild(0).jjtGetChild(0);
                if (!catchParam.isMulticatch()) {
                    catchParamSet.add(catchParam.toString());
                }
            }
        }
        if (catchClausesList.size() > 1 && !catchParamSet.isEmpty()) {
            String catchParam = catchParamSet.iterator().next();
            throw new RuleViolation(catchClausesList.get(catchClausesList.size() - 1), "Multiple catch clauses with different single catch blocks can be simplified to a single catch clause with a multi-catch parameter '" + catchParam + "'.");
        }
        return super.visit(node, data);
    }

    private boolean areCatchClausesIdentical(CatchClause a, CatchClause b) {
        NodeStream<JavaNode> aDescendants = a.descendantsOrSelf();
        NodeStream<JavaNode> bDescendants = b.descendantsOrSelf();
        NodeStream<JavaNode> aChildren = a.children(ASTCatchParameter.class);
        NodeStream<JavaNode> bChildren = b.children(ASTCatchParameter.class);
        JavaNode aChild = aChildren.asStream().findFirst().orElse(null);
        JavaNode bChild = bChildren.asStream().findFirst().orElse(null);
        return aDescendants.equals(bDescendants) && aChild.getClass().equals(bChild.getClass());
    }

        super(JavaSwitchStatement.class);

    @Override
    public Object visit(JavaSwitchStatement node, Object data) {
        for (JavaSwitchCase node2 : node.getCases()) {
            if (node2.isDefaultCase()) {
                continue;
            }
            for (JavaCatchClause node3 : node2.getCatchClauses()) {
                if (isMulticatch(node3.getCatchParameter())) {

                            addRuleViolation(node, "Identical catch branches");
                        }

                }
            }
        }
        return super.visit(node, data);
    }

    private boolean isMulticatch(ASTCatchParameter node) {
        return node.isMulticatch();

import net.sourceforge.pmd.lang.java.ast.internal. *;

import net.sourceforge.pmd.lang.ast.NodeStream;
public class IdenticalCatchBranchesRule extends AbstractJavaRulechainRule {

    public IdenticalCatchBranchesRule() {
        super(JavaNode.class);
    }

    @Override
    public Object visit(CatchClause node, Object data) {
        if (node.isMulticatch()) {
            return null;
        }
        NodeStream<CatchClause> catchClauses = catchClauses(node);
        if (catchClauses == null) {
            return null;
        }
        Map<String, Set<CatchClause>> catchClausesMap = new HashMap<>();
        for (CatchClause catchClause : catchClauses) {
            ASTCatchParameter catchParameter = (ASTCatchParameter) catchClause.getFirstChild();
            String key = catchParameter.getExceptions().get(0).getImage();
            catchClausesMap.computeIfAbsent(key, k -> new HashSet<>()).add(catchClause);
        }
        for (Map.Entry<String, Set<CatchClause>> entry : catchClausesMap.entrySet()) {
            if (entry.getValue().size() > 1) {
                addViolation(data, node, "Identical catch branches should be collapsed into a single multi-catch branch");
            }
        }
        return null;
    }

    private NodeStream<CatchClause> catchClauses(CatchClause node) {

        if (catchClauses == null) {
            return null;

            if (catchClause.isMulticatch()) {
                continue;

            String key = catchParameter.getExceptions().get(0).getImage();

        for (ASTCatchClause catchClause : catchClauses) {
