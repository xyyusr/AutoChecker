import java.util.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class AvoidThrowingNullPointerExceptionRule extends AbstractJavaRulechainRule {

    public AvoidThrowingNullPointerExceptionRule() {
        super(JavaNode.class, ASTThrowStatement.class);
    }

    @Override
    public Object visit(JavaNode node, Object data) {
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTThrowStatement node, Object data) {
        Object result = super.visit(node, data);
        if (isThrowingNpe(node)) {
            addViolation(data, node);
        }
        return result;
    }

    private boolean isThrowingNpe(ASTThrowStatement node) {
        ASTExpression expression = node.getExpression();
        if (expression == null) {
            return false;
        }

        if (ASTNullLiteral.class.isInstance(expression)) {
            return true;
        }

        if (isThrowingNpe(expression)) {
            return true;
        }

        return false;
    }

    private boolean isThrowingNpe(ASTExpression expression) {
        if (expression == null) {
            return false;
        }

        if (ASTNullLiteral.class.isInstance(expression)) {
            return true;
        }

        for (ASTMethodInvocation methodInvocation : expression.getDescendants(ASTMethodInvocation.class)) {
            if (methodInvocation.getName().image.equals("NullPointerException") || methodInvocation.getName().image.equals(
                    "NullPointerException")) {
                return true;
            }
        }

        return false;
    }
}
