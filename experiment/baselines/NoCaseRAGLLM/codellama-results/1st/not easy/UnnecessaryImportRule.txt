import java.util.*;
import java.lang.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;

import net.sourceforge.pmd.lang.java.ast.internal. *;

import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class UnnecessaryImportRule extends AbstractJavaRulechainRule {

    public UnnecessaryImportRule() {
        super(ASTImportDeclaration.class, ASTPackageDeclaration.class);
    }

    @Override
    public Object visit(ASTImportDeclaration node, Object data) {
        if (isImplicitlyImported(node)) {
            addViolation(data, node);
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTPackageDeclaration node, Object data) {
        for (ASTImportDeclaration importDeclaration : node.findChildrenOfType(ASTImportDeclaration.class)) {
            if (isImplicitlyImported(importDeclaration)) {
                addViolation(data, importDeclaration);
            }
        }
        return super.visit(node, data);
    }

    private boolean isImplicitlyImported(ASTImportDeclaration importDeclaration) {
        String importedSimpleName = importDeclaration.getImportedSimpleName();
        if (importedSimpleName == null || importedSimpleName.equals("")) {
            return false;
        }

        if (isJavaLangImport(importDeclaration) || isThisPackageImport(importDeclaration)) {
            return true;
        }

        ASTCompilationUnitDeclaration compilationUnit = (ASTCompilationUnitDeclaration) findCompilationUnit(importDeclaration);

        if (compilationUnit == null) {
            return false;
        }

        Iterable<ASTImportDeclaration> imports = compilationUnit.findChildrenOfType(ASTImportDeclaration.class);

        for (ASTImportDeclaration declaration : imports) {
            if (declaration.equals(importDeclaration)) {
                continue;
            }

            if (declaration.getImportedSimpleName().equals(importedSimpleName)) {
                return true;
            }
        }

        return false;
    }

    private boolean isJavaLangImport(ASTImportDeclaration importDeclaration) {
        ASTCompilationUnitDeclaration compilationUnit = (ASTCompilationUnitDeclaration) findCompilationUnit(importDeclaration);
        if (compilationUnit == null) {
            return false;
        }

        for (ASTImportDeclaration declaration : compilationUnit.findChildrenOfType(ASTImportDeclaration.class)) {
            if (declaration.getImportedSimpleName().equals("java.lang")) {
                return true;
            }
        }

        return false;
    }

    private boolean isThisPackageImport(ASTImportDeclaration importDeclaration) {
        ASTCompilationUnitDeclaration compilationUnit = (ASTCompilationUnitDeclaration) findCompilationUnit(importDeclaration);
        if (compilationUnit == null) {
            return false;
        }

        return compilationUnit.getPackageDeclaration().getImportedSimpleName().equals(importDeclaration.getImportedSimpleName());
    }

    private ASTCompilationUnitDeclaration findCompilationUnit(ASTImportDeclaration importDeclaration) {
        NodeStream<JavaNode> descendants = importDeclaration.descendantsOrSelf();
        while (true) {
            JavaNode node = descendants.next();
            if (node instanceof ASTCompilationUnitDeclaration) {
                return (ASTCompilationUnitDeclaration) node;
            }
            if (node == null) {
                break;
            }
        }
        return null;
    }
}

package net.sourceforge.pmd.lang.java.rule.codestyle;

import java.util.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class UnnecessaryImportRule extends AbstractJavaRulechainRule {

    public UnnecessaryImportRule() {
        super(JavaCompilationUnit.class);
    }

    @Override
    public Object visit(JavaCompilationUnit n, Object data) {
        Set<String> importedNames = new HashSet<>();
        Set<String> usedNames = new HashSet<>();
        Set<String> declaredNames = new HashSet<>();

        for (ASTImportDeclaration importDeclaration : n.findDescendantsOfType(ASTImportDeclaration.class, true)) {
            String importedSimpleName = importDeclaration.getImportedSimpleName();
            if (importedSimpleName != null) {
                importedNames.add(importedSimpleName);
            }

            for (ASTClassOrInterfaceDeclaration classDeclaration : importDeclaration.findDescendantsOfType(ASTClassOrInterfaceDeclaration.class, true)) {
                declaredNames.addAll(classDeclaration.getQualifiedName().split("\\."));
            }

            for (ASTMethodOrConstructorDeclaration methodDeclaration : importDeclaration.findDescendantsOfType(ASTMethodOrConstructorDeclaration.class, true)) {
                declaredNames.add(methodDeclaration.getName().getImage());
            }
        }

        for (ASTClassOrInterfaceDeclaration classDeclaration : n.findDescendantsOfType(ASTClassOrInterfaceDeclaration.class)) {
            usedNames.addAll(classDeclaration.getQualifiedName().split("\\."));
        }

        for (ASTMethodOrConstructorDeclaration methodDeclaration : n.findDescendantsOfType(ASTMethodOrConstructorDeclaration.class)) {
            usedNames.add(methodDeclaration.getName().getImage());
        }

        for (ASTImportDeclaration importDeclaration : n.findDescendantsOfType(ASTImportDeclaration.class)) {
            String importedSimpleName = importDeclaration.getImportedSimpleName();
            if (importedSimpleName != null) {
                if (!usedNames.contains(importedSimpleName) && !declaredNames.contains(importedSimpleName)) {
                    if (importedNames.contains(importedSimpleName)) {
                        // Report the import declaration
                        addViolation(this, importDeclaration.getReportLocation(), "Import of " + importedSimpleName + " is unnecessary.");
                    }
                }
            }
        }

        for (ASTImportDeclaration importDeclaration : n.findDescendantsOfType(ASTImportDeclaration.class)) {
            String importedSimpleName = importDeclaration.getImportedSimpleName();
            if (importedSimpleName != null) {
                String importedName = importDeclaration.getQualifiedName();
                if (importedName.equals("java.lang." + importedSimpleName) || importedName.equals(importedSimpleName)) {
                    if (importedNames.contains(importedSimpleName)) {
                        // Report the import declaration
                        addViolation(this, importDeclaration.getReportLocation(), "Import of " + importedSimpleName + " is unnecessary.");
                    }
                }
            }
        }

        return super.visit(n, data);
    }

package net.sourceforge.pmd.lang.java.rule.codestyle;

import java.util.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class UnnecessaryImportRule extends AbstractJavaRulechainRule {

    @Override
    public Object visit(ASTImportDeclaration node, Object data) {
        if (isUnusedImport(node)) {
            addViolation(data, node);
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTImportStaticDeclaration node, Object data) {
        if (isUnusedImport(node)) {
            addViolation(data, node);
        }
        return super.visit(node, data);
    }

    private boolean isUnusedImport(ASTImportDeclaration node) {
        String importedSimpleName = node.getImportedSimpleName();
        if (importedSimpleName == null) {
            return false;
        }

        SymbolTable symbolTable = node.getSymbolTable();
        if (symbolTable.findSymbol(importedSimpleName) == null) {
            return true;
        }

        if (isImplicitlyInScope(node, importedSimpleName)) {
            return true;
        }

        NodeStream<ASTCompilationUnit> compilationUnits = node.getParent().descendantsOrSelf(ASTCompilationUnit.class);
        boolean isImportedInAnotherCompilationUnit = false;
        for (ASTCompilationUnit compilationUnit : compilationUnits) {
            NodeStream<ASTImportDeclaration> importDeclarations = compilationUnit.descendants(ASTImportDeclaration.class);
            for (ASTImportDeclaration importDeclaration : importDeclarations) {
                if (importDeclaration.equals(node)) {
                    continue;
                }
                if (isImportedInSameCompilationUnit(importDeclaration, importedSimpleName)) {
                    isImportedInAnotherCompilationUnit = true;
                    break;
                }
            }
            if (isImportedInAnotherCompilationUnit) {
                break;
            }
        }

        return !isImportedInAnotherCompilationUnit;
    }

    private boolean isImplicitlyInScope(ASTImportDeclaration node, String importedSimpleName) {
        if (importedSimpleName.startsWith("java.")) {
            return true;
        }

        ASTPackageDeclaration packageDeclaration = node.getParent().getParent();
        if (packageDeclaration == null) {
            return false;
        }

        String packageName = packageDeclaration.getPackageName();
        if (packageName.equals("") || packageName.equals("java") || packageName.equals("javax")) {
            return true;
        }

        ASTClassOrInterfaceDeclaration classOrInterfaceDeclaration = packageDeclaration.getFirstChild(ASTClassOrInterfaceDeclaration.class);
        if (classOrInterfaceDeclaration == null) {
            return false;
        }

        Type type = classOrInterfaceDeclaration.getType();
        if (type == null) {
            return false;
        }

        return type.getName().equals(importedSimpleName);
    }

    private boolean isImportedInSameCompilationUnit(ASTImportDeclaration node, String importedSimpleName) {
