package net.sourceforge.pmd.lang.java.rule.codestyle;

import java.util.*;
import java.lang.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class MethodNamingConventionsRule extends AbstractJavaRulechainRule {

    public MethodNamingConventionsRule() {
        super(MethodDeclaration.class, MethodDeclaration.class);
    }

    @Override
    public Object visit(MethodDeclaration node, Object data) {
        String methodName = node.getName();
        String methodType = node.isStatic() ? "static" : "instance";
        boolean isNativeMethod = node.isNative();
        boolean isJUnit3Method = TestFrameworksUtil.isJUnit3Method(node);
        boolean isJUnit4Method = TestFrameworksUtil.isJUnit4Method(node);
        boolean isJUnit5Method = TestFrameworksUtil.isJUnit5Method(node);

        if (isNativeMethod) {
            if (!methodName.matches("[a-z][a-zA-Z0-9]*")) {
                throw new RuleViolation(node, "Native method names must start with a lowercase letter followed by word characters");
            }
        } else if (isJUnit3Method) {
            if (!methodName.matches("test[A-Z0-9][a-zA-Z0-9]*")) {
                throw new RuleViolation(node, "JUnit 3 test method names must start with 'test' followed by a uppercase letter or digit followed by word characters");
            }
        } else if (isJUnit4Method || isJUnit5Method) {
            if (!methodName.matches("[a-z][a-zA-Z0-9]*")) {
                throw new RuleViolation(node, "JUnit 4 and 5 test method names must start with a lowercase letter followed by word characters");
            }
        } else {
            if (!methodName.matches("[a-z][a-zA-Z0-9]*")) {
                throw new RuleViolation(node, "Method names must start with a lowercase letter followed by word characters");
            }
        }

        return super.visit(node, data);
    }
}

package net.sourceforge.pmd.lang.java.rule.codestyle;

import java.util.*;
import java.lang.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class MethodNamingConventionsRule extends AbstractJavaRulechainRule {
    public MethodNamingConventionsRule() {
        super(ASTMethodDeclaration.class, ASTMethodReference.class);
    }

    @Override
    public Object visit(ASTMethodDeclaration node, Object data) {
        String methodName = node.getName();
        if (isJUnitMethod(node)) {
            if (!isJUnit5Method(node)) {
                if (!isJUnit3MethodSignature(node)) {
                    if (!isJUnit4Method(node)) {
                        reportRegexViolation(node, methodName);
                    }
                }
            }
        } else if (node.isStatic()) {
            reportRegexViolation(node, methodName);
        } else {
            reportRegexViolation(node, methodName);
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTMethodReference node, Object data) {
        String methodName = node.getMethodName();
        if (isNativeMethod(node)) {
            reportRegexViolation(node, methodName);
        }
        return super.visit(node, data);
    }

    private void reportRegexViolation(Node node, String methodName) {
        if (!matchRegex(methodName, "[a-z][a-zA-Z0-9]*")) {
            addRuleViolation(node, "Method " + methodName + " does not match the naming convention.");
        }
    }

    private boolean matchRegex(String input, String regex) {
        try {
            java.util.regex.Pattern pattern = java.util.regex.Pattern.compile(regex);
            java.util.regex.Matcher matcher = pattern.matcher(input);
            return matcher.matches();
        } catch (java.util.regex.PatternSyntaxException e) {
            return false;
        }
    }

    private boolean isJUnitMethod(ASTMethodDeclaration node) {
        return TestFrameworksUtil.isJUnitMethod(node) || TestFrameworksUtil.isJUnit3Method(node) || TestFrameworksUtil.isJUnit4Method(node);
    }

    private boolean isNativeMethod(ASTMethodReference node) {
        return TestFrameworksUtil.isNativeMethod(node);
    }
}

import net.sourceforge.pmd.lang.ast.NodeStream;
public class MethodNamingConventionsRule extends AbstractJavaRulechainRule {

    public MethodNamingConventionsRule() {
        addRuleChainVisitor(this);
    }

    @Override
    public Object visit(ASTMethodDeclaration node, Object data) {
        String methodName = node.getName();
        boolean isStatic = node.isStatic();
        boolean isNative = node.isNative();
        boolean isJUnitTest = isJUnitMethod(node);

        if (isStatic) {
            if (!node.getName().matches("[a-z][a-zA-Z0-9]*")) {
                addViolation(data, node, "Static method name does not match camel case convention");
            }
        } else if (isNative) {
            if (!node.getName().matches("[a-z][a-zA-Z0-9]*")) {
                addViolation(data, node, "Native method name does not match camel case convention");
            }
        } else if (isJUnitTest) {
            if (isJUnit5Method(node)) {
                if (!node.getName().matches("[a-z][a-zA-Z0-9]*")) {
                    addViolation(data, node, "JUnit 5 test method name does not match camel case convention");
                }
            } else if (isJUnit4Method(node)) {
                if (!node.getName().matches("[a-z][a-zA-Z0-9]*")) {
                    addViolation(data, node, "JUnit 4 test method name does not match camel case convention");
                }
            } else if (isJUnit3Method(node)) {
                if (!node.getName().matches("test[A-Z0-9][a-zA-Z0-9]*")) {
                    addViolation(data, node, "JUnit 3 test method name does not match camel case convention");
                }
            }
        } else {
            if (!node.getName().matches("[a-z][a-zA-Z0-9]*")) {
                addViolation(data, node, "Method name does not match camel case convention");
            }
        }

        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTClassOrInterfaceDeclaration node, Object data) {
        if (isJUnit3Class(node)) {

            }
        }

        return super.visit(node, data);
    }
