package net.sourceforge.pmd.lang.java.rule.codestyle;

import java.util.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class EmptyControlStatementRule extends AbstractJavaRulechainRule {

    public EmptyControlStatementRule() {
        addRuleMessage("Empty control statement body");
        addRuleChainVisitorClass(EmptyControlStatementRule.class);
    }

    @Override
    public Object visit(ASTTryStatement node, Object data) {
        if (isBodyEmpty(node.getBody())) {
            addRuleViolation(node, "Try statement body is empty");
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTFinallyStatement node, Object data) {
        if (isBodyEmpty(node.getBody())) {
            addRuleViolation(node, "Finally statement body is empty");
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTSwitchStatement node, Object data) {
        if (isBodyEmpty(node.getStatements())) {
            addRuleViolation(node, "Switch statement body is empty");
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTSynchronizedStatement node, Object data) {
        if (isBodyEmpty(node.getBody())) {
            addRuleViolation(node, "Synchronized statement body is empty");
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTIfStatement node, Object data) {
        if (isBodyEmpty(node.getThenStatement())) {
            addRuleViolation(node, "If statement body is empty");
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTForStatement node, Object data) {
        if (isBodyEmpty(node.getBody())) {
            addRuleViolation(node, "For statement body is empty");
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTWhileStatement node, Object data) {
        if (isBodyEmpty(node.getBody())) {
            addRuleViolation(node, "While statement body is empty");
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTDoStatement node, Object data) {
        if (isBodyEmpty(node.getBody())) {
            addRuleViolation(node, "Do statement body is empty");
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTEmptyStatement node, Object data) {
        if (isBodyEmpty(node)) {
            addRuleViolation(node, "Empty statement");
        }
        return super.visit(node, data);
    }

    private boolean isBodyEmpty(ASTStatement statement) {
        if (statement == null) {
            return false;
        }
        if (statement instanceof ASTEmptyStatement) {
            return true;
        }
        return statement.getChildren().isEmpty();
    }
}

package net.sourceforge.pmd.lang.java.rule.codestyle;

import java.util.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class EmptyControlStatementRule extends AbstractJavaRulechainRule {

    public EmptyControlStatementRule() {
        addRuleMessage("Empty control statement body.");
    }

    @Override
    public Object visit(ASTTryStatement node, Object data) {
        if (isBodyEmpty(node.getBody())) {
            addRuleViolation(node.getReportLocation(), "Empty body in try statement.");
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTFinallyStatement node, Object data) {
        if (isBodyEmpty(node.getBody())) {
            addRuleViolation(node.getReportLocation(), "Empty body in finally statement.");
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTSwitchStatement node, Object data) {
        if (isBodyEmpty(node.getStatements())) {
            addRuleViolation(node.getReportLocation(), "Empty switch statement body.");
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTSynchronizedStatement node, Object data) {
        if (isBodyEmpty(node.getBody())) {
            addRuleViolation(node.getReportLocation(), "Empty body in synchronized statement.");
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTIfStatement node, Object data) {
        if (node.getElse() == null && isBodyEmpty(node.getThen())) {
            addRuleViolation(node.getReportLocation(), "Empty body in if statement.");
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTWhileStatement node, Object data) {
        if (isBodyEmpty(node.getBody())) {
            addRuleViolation(node.getReportLocation(), "Empty body in while statement.");
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTForStatement node, Object data) {
        if (isBodyEmpty(node.getBody())) {
            addRuleViolation(node.getReportLocation(), "Empty body in for statement.");
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTDoStatement node, Object data) {
        if (isBodyEmpty(node.getBody())) {
            addRuleViolation(node.getReportLocation(), "Empty body in do statement.");
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTBlock node, Object data) {
        if (isBlockEmpty(node)) {
            addRuleViolation(node.getReportLocation(), "Empty block statement.");
        }
        return super.visit(node, data);
    }

    private boolean isBodyEmpty(Node body) {
        if (body instanceof ASTEmptyStatement) {
            return true;
        }
        return (body == null || body.getChildren().isEmpty()) && !isBlockEmpty(body);
    }

    private boolean isBlockEmpty(Node block) {
        return block == null || block.getChildren().isEmpty();
    }
}
