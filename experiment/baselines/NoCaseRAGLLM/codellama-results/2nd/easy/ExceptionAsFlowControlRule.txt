import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ExceptionAsFlowControlRule extends AbstractJavaRulechainRule {

    @Override
    public Object visit(ASTTryStatement node, Object data) {
        if (node.isFinallyBlock()) {
            Object[] violations = new Object[node.getCatchClauses().size()];
            int i = 0;
            for (ASTCatchClause catchClause : node.getCatchClauses()) {
                if (catchClause.getEnclosingType() == null && !isInEnclosingType(node)) {
                    violations[i++] = new RuleViolation(node, "Exception thrown and caught in an enclosing try statement", getRulePriority());
                }
            }
            if (i > 0) {
                return violations;
            }
        }
        return super.visit(node, data);
    }

    private boolean isInEnclosingType(ASTTryStatement node) {
        if (node.getEnclosingType() != null) {
            return true;
        }
        for (ASTNode ancestor : node.ancestors()) {
            if (ancestor instanceof ASTEnclosingStatement || ancestor instanceof ASTBlock) {
                return true;
            }
        }
        return false;
    }

    @Override
    public Object visit(ASTThrowStatement node, Object data) {
        if (node.getEnclosingType() == null && !isInEnclosingType(node)) {
            return new RuleViolation(node, "Exception thrown in an enclosing try statement", getRulePriority());
        }
        return super.visit(node, data);
    }

    private boolean isInEnclosingType(ASTThrowStatement node) {
        if (node.getEnclosingType() != null) {
            return true;
        }
        for (ASTNode ancestor : node.ancestors()) {
            if (ancestor instanceof ASTEnclosingStatement || ancestor instanceof ASTBlock) {
                return true;
            }
        }
        return false;
    }
}

package net.sourceforge.pmd.lang.java.rule.design;
import java.util.*;
import java.lang.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class ExceptionAsFlowControlRule extends AbstractJavaRulechainRule {
    public ExceptionAsFlowControlRule() {
        addRuleChainVisitor(this);
    }

    @Override
    public Object visit(ASTTryStatement node, Object data) {
        if (node.isFindBoundary()) {
            return data;
        }
        if (node.getCatchClauses().isEmpty()) {
            return data;
        }
        if (node.getEnclosingType() != null) {
            return data;
        }
        return findThrowStatements(node);
    }

    private Object findThrowStatements(ASTTryStatement node) {
        NodeStream<ASTThrowStatement> descendants = node.descendants(ASTThrowStatement.class);
        for (ASTThrowStatement throwStatement : descendants) {
            if (throwStatement.isFindBoundary()) {
                ReportLocation reportLocation = node.getReportLocation();
                StringBuilder message = new StringBuilder();
                message.append("Exception thrown and caught in an enclosing try statement.");
                message.append(" Consider adding necessary validation or using an alternate control structure.");
                addRuleViolation(node, reportLocation, message.toString());
                return data;
            }
        }
        return data;
    }
}
