    public InefficientEmptyStringCheckRule() {
        super(ASTEqualityExpression.class);
    }

    @Override
    public Object visit(ASTEqualityExpression node, Object data) {
        if (node.getImage().equals("==")) {
            NodeStream<ASTMethodCall> methodCalls = node.children(ASTExpression.class).descendants(ASTMethodCall.class);
            if (methodCalls.count() == 2) {
                ASTMethodCall methodCall1 = methodCalls.get(0);
                ASTMethodCall methodCall2 = methodCalls.get(1);

                if (isTrimLengthZeroCheck(methodCall1, methodCall2) || isTrimLengthZeroCheck(methodCall2, methodCall1)) {
                    addViolation(data, node);
                } else if (isTrimIsEmptyCheck(methodCall1, methodCall2) || isTrimIsEmptyCheck(methodCall2, methodCall1)) {
                    addViolation(data, node);
                }
            }
        }
        return super.visit(node, data);
    }

    private boolean isTrimLengthZeroCheck(ASTMethodCall methodCall1, ASTMethodCall methodCall2) {
        return isTrimCall(methodCall1) && methodCall2.hasImageEqualTo("0");
    }

    private boolean isTrimIsEmptyCheck(ASTMethodCall methodCall1, ASTMethodCall methodCall2) {
        return isTrimCall(methodCall1) && methodCall2.hasImageEqualTo("true");
    }

    private boolean isTrimCall(ASTMethodCall methodCall) {
        if (methodCall.getMethodName().equals("trim")) {
            ASTPrimaryExpression primaryExpression = methodCall.getFirstParentOfType(ASTPrimaryExpression.class);
            if (primaryExpression != null) {
                ASTPrimaryPrefix primaryPrefix = primaryExpression.getFirstChildOfType(ASTPrimaryPrefix.class);
                if (primaryPrefix != null && primaryPrefix.getFirstChildOfType(ASTPrimarySuffix.class) != null) {
                    ASTPrimarySuffix suffix = primaryPrefix.getFirstChildOfType(ASTPrimarySuffix.class);
                    return suffix.hasImageEqualTo(".length()") || suffix.hasImageEqualTo(".isEmpty()");
                }
            }
        }
        return false;
    }
