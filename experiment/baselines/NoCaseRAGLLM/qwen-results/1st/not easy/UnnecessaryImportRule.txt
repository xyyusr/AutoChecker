import java.util.HashSet;
import java.util.List;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class UnnecessaryImportRule extends AbstractJavaRulechainRule {
    public UnnecessaryImportRule() {
        super(ASTCompilationUnit.class);
    }

    @Override
    public Object visit(ASTCompilationUnit node, Object data) {
        Set<String> usedImports = new HashSet<>();
        Set<String> javaLangImplicitImports = new HashSet<>(Arrays.asList("Object", "String", "Boolean", "Character", "Byte", "Short", "Integer", "Long", "Float", "Double", "Void", "Class", "Package", "Thread", "Runnable"));
        String currentPackageName = node.getPackageNameDeclaration().map(ASTPackageDeclaration::getImage).orElse("");

        node.descendants(ASTName.class).forEach(astName -> {
            JType type = astName.getType();
            if (type != null) {
                JTypeSymbol typeSymbol = type.getSymbol();
                if (typeSymbol != null) {
                    String typeName = typeSymbol.getQualifiedName();
                    if (!typeName.startsWith(currentPackageName) && !typeName.startsWith("java.lang")) {
                        usedImports.add(typeName);
                    }
                }
            }
        });

        node.descendants(ASTImportDeclaration.class).forEach(importDeclaration -> {
            if (!importDeclaration.isStatic()) {
                String importName = importDeclaration.getImportedSimpleName();
                String qualifiedName = importDeclaration.getImportedType().getQualifiedName();
                if (importName == null || usedImports.contains(qualifiedName) || javaLangImplicitImports.contains(importName)) {
                    addViolation(data, importDeclaration);
                }
            }
        });

        return super.visit(node, data);
    }
}

package net.sourceforge.pmd.lang.java.rule.codestyle;


import java.util.HashSet;
import java.util.List;
import java.util.Set;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class UnnecessaryImportRule extends AbstractJavaRulechainRule {
    public UnnecessaryImportRule() {
        super(ASTImportDeclaration.class, ASTCompilationUnit.class);
    }

    @Override
    public Object visit(ASTCompilationUnit node, Object data) {
        Set<String> usedImports = new HashSet<>();
        String currentPackageName = getCurrentPackageName(node);

        node.descendants(ASTPrimaryExpression.class).forEach(expr -> {
            expr.descendants(ASTPrimaryPrefix.class).forEach(prefix -> {
                prefix.descendants(ASTClassOrInterfaceType.class).forEach(type -> {
                    type.children(ASTName.class).forEach(name -> {
                        usedImports.add(name.getImage());
                    });
                });
            });
        });

        node.descendants(ASTVariableDeclaratorId.class).forEach(var -> {
            var.ancestors(ASTClassOrInterfaceType.class).forEach(type -> {
                type.children(ASTName.class).forEach(name -> {
                    usedImports.add(name.getImage());
                });
            });
        });

        node.descendants(ASTLiteral.class).forEach(literal -> {
            if (literal.getType() == TypeSystem.STRING_TYPE) {
                usedImports.add(literal.getImage());
            }
        });

        node.descendants(ASTMethodReference.class).forEach(methodRef -> {
            methodRef.descendants(ASTClassOrInterfaceType.class).forEach(type -> {
                type.children(ASTName.class).forEach(name -> {
                    usedImports.add(name.getImage());
                });
            });
        });

        node.descendants(ASTImportDeclaration.class).forEach(importDecl -> {
            String importName = importDecl.getImportedSimpleName();
            if (!importDecl.isStatic() && !importDecl.getImportedName().toString().startsWith("java.lang")) {
                if (importName == null) {
                    return; // on-demand import, cannot determine simplicity
                }
                if (importName.equals(currentPackageName)) {
                    addViolation(importDecl, data);
                } else if (!usedImports.contains(importName)) {
                    addViolation(importDecl, data);
                }
            }
        });

        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTImportDeclaration node, Object data) {
        return data;
    }

    private String getCurrentPackageName(ASTCompilationUnit node) {
        ASTPackageDeclaration packageDecl = node.getFirstChildOfType(ASTPackageDeclaration.class);
        return packageDecl != null ? packageDecl.getName().getImage() : "";
    }
}

package net.sourceforge.pmd.lang.java.rule.codestyle;

import java.util.HashSet;
import java.util.List;
import java.util.Set;
import java.util.stream.Collectors;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class UnnecessaryImportRule extends AbstractJavaRulechainRule {

    public UnnecessaryImportRule() {
        super(ASTCompilationUnit.class);
    }

    @Override
    public Object visit(ASTCompilationUnit node, Object data) {
        Set<String> usedImports = new HashSet<>();
        Set<String> implicitImports = new HashSet<>();

        implicitImports.add("java.lang");
        ASTPackageDeclaration packageDeclaration = node.getFirstChildOfType(ASTPackageDeclaration.class);
        if (packageDeclaration != null) {
            implicitImports.add(packageDeclaration.getPackageName());
        }

        node.descendants(ASTClassOrInterfaceDeclaration.class).forEach(c -> {
            c.descendants(ASTName.class).forEach(name -> {
                JClassSymbol classSymbol = name.getType().getSymbol();
                if (classSymbol != null) {
                    usedImports.add(classSymbol.getPackageName());
                }
            });
        });

        node.descendants(ASTFieldDeclaration.class).forEach(f -> {
            f.descendants(ASTName.class).forEach(name -> {
                JFieldSymbol fieldSymbol = name.getReferencedField();
                if (fieldSymbol != null) {
                    usedImports.add(fieldSymbol.getEnclosingClass().getPackageName());
                }
            });
        });

        node.descendants(ASTMethodDeclaration.class).forEach(m -> {
            m.descendants(ASTName.class).forEach(name -> {
                JMethodSymbol methodSymbol = name.getReferencedMethod();
                if (methodSymbol != null) {
                    usedImports.add(methodSymbol.getEnclosingClass().getPackageName());
                }
            });
        });

        node.descendants(ASTConstructorDeclaration.class).forEach(c -> {
            c.descendants(ASTName.class).forEach(name -> {
                JMethodSymbol constructorSymbol = name.getReferencedConstructor();
                if (constructorSymbol != null) {
                    usedImports.add(constructorSymbol.getEnclosingClass().getPackageName());
                }
            });
        });

        List<ASTImportDeclaration> importDeclarations = node.findChildrenOfType(ASTImportDeclaration.class);
        List<String> imports = importDeclarations.stream()
                .map(ASTImportDeclaration::getImportedPackageName)
                .collect(Collectors.toList());

        for (ASTImportDeclaration importDeclaration : importDeclarations) {
            String importedPackageName = importDeclaration.getImportedPackageName();
            String importedSimpleName = importDeclaration.getImportedSimpleName();
            if (!importDeclaration.isStatic() && (usedImports.contains(importedPackageName) || implicitImports.contains(importedPackageName))) {
                addViolation(data, importDeclaration);
            } else if (importDeclaration.isStatic() && (usedImports.contains(importedPackageName) || implicitImports.contains(importedPackageName) || (importedSimpleName != null && !imports.contains(importedPackageName + "." + importedSimpleName)))) {
                addViolation(data, importDeclaration);
            }
        }

        for (int i = 0; i < importDeclarations.size(); i++) {
            ASTImportDeclaration importDeclaration = importDeclarations.get(i);
            for (int j = i + 1; j < importDeclarations.size(); j++) {
                ASTImportDeclaration nextImportDeclaration = importDeclarations.get(j);
                if (importDeclaration.equals(nextImportDeclaration)) {
                    addViolation(data, nextImportDeclaration);
                }
            }
        }

        return super.visit(node, data);
    }
}

import net.sourceforge.pmd.lang.ast.NodeStream;
public class UnnecessaryImportRule extends AbstractJavaRulechainRule {
    public UnnecessaryImportRule() {
        super(ASTCompilationUnit.class, ASTImportDeclaration.class);
    }

    private Set<String> usedImports = new HashSet<>();
    private String currentPackageName;

    @Override
    public Object visit(ASTCompilationUnit node, Object data) {
        usedImports.clear();
        currentPackageName = node.getPackageDeclaration() != null ? node.getPackageDeclaration().getPackageName() : null;
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTImportDeclaration node, Object data) {
        if (node.isStatic()) {
            return data;
        }
        String importedName = node.getImportedName();
        if (importedName.startsWith("java.lang")) {
            return data;
        }
        if (importedName.equals(currentPackageName)) {
            return data;
        }
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTReferenceType node, Object data) {
        trackUsage(node);
        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTPrimaryPrefix node, Object data) {
        trackUsage(node);
        return super.visit(node, data);
    }

    private void trackUsage(JavaNode node) {
        ASTName nameNode = node.getFirstChildOfType(ASTName.class);
        if (nameNode != null) {
            JType type = nameNode.getType();
            if (type instanceof JClassType) {
                JClassType classType = (JClassType) type;
                JClassSymbol classSymbol = classType.getSymbol();
                if (classSymbol != null) {
                    usedImports.add(classSymbol.getQualifiedName());
                }
            }
        }
    }

    @Override
    public void end(ASTCompilationUnit node, Object data) {
        NodeStream<ASTImportDeclaration> imports = node.descendants(ASTImportDeclaration.class);
        imports.filter(importDeclaration -> !importDeclaration.isStatic())
               .filter(importDeclaration -> !importDeclaration.getImportedName().startsWith("java.lang"))
               .filter(importDeclaration -> !importDeclaration.getImportedName().equals(currentPackageName))
               .filter(importDeclaration -> !usedImports.contains(importDeclaration.getImportedName()))
               .forEach(importDeclaration -> addViolation(data, importDeclaration));
    }

package net.sourceforge.pmd.lang.java.rule.codestyle;

import java.util.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class UnnecessaryImportRule extends AbstractJavaRulechainRule {
    public UnnecessaryImportRule() {
        super(ASTCompilationUnit.class, ASTImportDeclaration.class);
    }

    @Override
    public Object visit(ASTCompilationUnit node, Object data) {
        Set<String> usedImports = new HashSet<>();
        Set<String> currentPackageClasses = new HashSet<>();

        ASTPackageDeclaration packageDeclaration = node.getFirstChildOfType(ASTPackageDeclaration.class);
        if (packageDeclaration != null) {
            String packageName = packageDeclaration.getPackageName();
            currentPackageClasses = getClassesInPackage(packageName);
        }

        node.descendants(ASTName.class).forEach(nameNode -> {
            JType type = nameNode.getType();
            if (type != null && type instanceof JClassType) {
                JClassSymbol symbol = ((JClassType) type).getSymbol();
                if (symbol != null) {
                    usedImports.add(symbol.getQualifiedName());
                }
            }
        });

        node.descendants(ASTImportDeclaration.class).forEach(importNode -> {
            if (!importNode.isStatic() && !importNode.isFindBoundary()) {
                String importedName = importNode.getImportedSimpleName();
                String fullName = importNode.getImportedName();
                if (importedName == null || usedImports.contains(fullName) || currentPackageClasses.contains(fullName) || fullName.startsWith("java.lang")) {
                    addViolation(data, importNode);
                }
            }
        });

        return data;
    }

    @Override
    public Object visit(ASTImportDeclaration node, Object data) {
        return data;
    }

    private Set<String> getClassesInPackage(String packageName) {
        Set<String> classes = new HashSet<>();
        JSymbolTable symbolTable = getCurrentScope().getSymbolTable();
        symbolTable.visibleClasses().forEach(classSymbol -> {
            if (classSymbol instanceof JDefinedClassSymbol) {
                String classPackageName = ((JDefinedClassSymbol) classSymbol).getPackageName();
                if (packageName.equals(classPackageName)) {
                    classes.add(classSymbol.getQualifiedName());
                }
            }
        });
        return classes;
    }
}
