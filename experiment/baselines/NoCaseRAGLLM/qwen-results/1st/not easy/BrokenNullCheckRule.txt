            JType leftType = node.getLeftOperand().getType();
            JType rightType = node.getRightOperand().getType();
            TypeSystem typeSystem = node.getTypeSystem();

            if (typeSystem.isA(typeSystem.OBJECT, leftType) || typeSystem.isA(typeSystem.OBJECT, rightType)) {
                boolean isLeftNullCheck = JavaRuleUtil.isNullChecked(node.getLeftOperand());
                boolean isRightNullCheck = JavaRuleUtil.isNullChecked(node.getRightOperand());

                if (isLeftNullCheck && isRightNullCheck) {
                    if (node.getOperator() == ASTBinaryExpression.Operator.OR) {
                        checkBrokenNullCheck(node.getLeftOperand(), node.getRightOperand(), data);
                    }
                } else if (isLeftNullCheck || isRightNullCheck) {
                    ASTExpression nullCheckedExpr = isLeftNullCheck ? node.getLeftOperand() : node.getRightOperand();
                    ASTExpression otherExpr = isLeftNullCheck ? node.getRightOperand() : node.getLeftOperand();

                    if (isNullComparison(otherExpr)) {
                        checkBrokenNullCheck(nullCheckedExpr, otherExpr, data);
                    }
                }
            }
        }
        return super.visit(node, data);
    }

    private void checkBrokenNullCheck(ASTExpression nullCheckedExpr, ASTExpression otherExpr, Object data) {
        ASTExpression firstOperand = getFirstOperand(otherExpr);

        if (firstOperand != null && secondOperand != null) {
            if (firstOperand.equals(nullCheckedExpr) && secondOperand instanceof ASTNullLiteral) {
                addViolation(data, nullCheckedExpr);
            }

    private boolean isNullComparison(ASTExpression expr) {

            return binaryExpr.getOperator() == ASTBinaryExpression.Operator.EQUAL || binaryExpr.getOperator() == ASTBinaryExpression.Operator.NOT_EQUAL;

        return false;

    private ASTExpression getFirstOperand(ASTExpression expr) {

        }

    }

package net.sourceforge.pmd.lang.java.rule.errorprone;


import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class BrokenNullCheckRule extends AbstractJavaRulechainRule {
    public BrokenNullCheckRule() {
        super(ASTBinaryExpression.class);
    }

    @Override
    public Object visit(ASTBinaryExpression node, Object data) {
        if (isBrokenNullCheck(node)) {
            addViolation(data, node);
        }
        return super.visit(node, data);
    }

    private boolean isBrokenNullCheck(ASTBinaryExpression node) {
        ASTExpression leftOperand = node.getLeftOperand();
        ASTExpression rightOperand = node.getRightOperand();
        JType leftType = leftOperand.getType();
        JType rightType = rightOperand.getType();
        TypeSystem typeSystem = leftOperand.getTypeSystem();

        boolean isLeftNullLiteral = JavaRuleUtil.isNullLiteral(leftOperand);
        boolean isRightNullLiteral = JavaRuleUtil.isNullLiteral(rightOperand);

        if (typeSystem.isA(leftType, typeSystem.OBJECT) && isRightNullLiteral) {
            return isBrokenNullCheck(node, leftOperand);
        } else if (typeSystem.isA(rightType, typeSystem.OBJECT) && isLeftNullLiteral) {
            return isBrokenNullCheck(node, rightOperand);
        }
        return false;
    }

    private boolean isBrokenNullCheck(ASTBinaryExpression node, ASTExpression nullCheckedOperand) {
        ASTExpression otherOperand = node.getLeftOperand().equals(nullCheckedOperand) ? node.getRightOperand() : node.getLeftOperand();
        NodeStream<ASTBinaryExpression> parentNullChecks = node.ancestors(ASTBinaryExpression.class)
                .filter(n -> JavaRuleUtil.isNullLiteral(n.getLeftOperand()) || JavaRuleUtil.isNullLiteral(n.getRightOperand()));

        return parentNullChecks.anyMatch(n -> n.getOperator() != node.getOperator() && JavaRuleUtil.isNullLiteral(otherOperand));
    }
}
