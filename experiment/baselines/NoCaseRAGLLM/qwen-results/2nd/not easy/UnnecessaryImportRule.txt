package net.sourceforge.pmd.lang.java.rule.codestyle;

import java.util.HashSet;
import java.util.List;
import java.util.Set;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class UnnecessaryImportRule extends AbstractJavaRulechainRule {
    public UnnecessaryImportRule() {
        super(ASTCompilationUnit.class, ASTImportDeclaration.class);
    }

    @Override
    public Object visit(ASTCompilationUnit node, Object data) {
        JSymbolTable symbolTable = node.getSymbolTable();
        JPackageSymbol currentPackage = symbolTable.getCurrentPackage();
        Set<String> usedImports = new HashSet<>();
        Set<String> implicitImports = new HashSet<>(Arrays.asList("Object", "Exception", "String", "Integer", "Boolean", "Double", "Float", "Long", "Short", "Byte", "Character", "Void", "Thread", "Runnable", "System", "Class", "Package", "Math", "StrictMath"));
        implicitImports.addAll(currentPackage.getTopLevelClasses().stream().map(JClassSymbol::getSimpleName).toList());

        List<ASTTypeDeclaration> typeDeclarations = node.findDescendantsOfType(ASTTypeDeclaration.class, false);
        for (ASTTypeDeclaration typeDeclaration : typeDeclarations) {
            typeDeclaration.descendants(ASTName.class).forEach(name -> usedImports.add(name.getImage()));
        }

        List<ASTImportDeclaration> importDeclarations = node.findDescendantsOfType(ASTImportDeclaration.class, false);
        for (ASTImportDeclaration importDeclaration : importDeclarations) {
            String importName = importDeclaration.getImportedSimpleName();
            boolean isOnDemand = importDeclaration.getImage().endsWith(".*");
            if (isOnDemand) {
                JTypeSystem typeSystem = importDeclaration.getTypeSystem();
                JType type = typeSystem.findTypeInPackage(importDeclaration.getImage().replace(".*", ""), currentPackage);
                if (type instanceof JClassType) {
                    JClassType classType = (JClassType) type;
                    classType.getDeclaredFields().stream().map(field -> field.getSimpleName()).forEach(implicitImports::add);
                    classType.getDeclaredMethods().stream().map(method -> method.getSimpleName()).forEach(implicitImports::add);
                }
            } else if (importName != null && (!usedImports.contains(importName) || implicitImports.contains(importName))) {
                addViolation(data, importDeclaration);
            }
        }

        return super.visit(node, data);
    }

    @Override
    public Object visit(ASTImportDeclaration node, Object data) {
        return super.visit(node, data);
    }
}

package net.sourceforge.pmd.lang.java.rule.codestyle;

import java.util.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class UnnecessaryImportRule extends AbstractJavaRulechainRule {
    public UnnecessaryImportRule() {
        super(ASTCompilationUnit.class);
    }

    @Override
    public Object visit(ASTCompilationUnit node, Object data) {
        Set<String> usedImports = new HashSet<>();
        Set<String> currentPackageTypes = new HashSet<>();

        ASTPackageDeclaration packageDeclaration = node.getFirstChildOfType(ASTPackageDeclaration.class);
        String currentPackageName = packageDeclaration != null ? packageDeclaration.getPackageName() : "";

        node.descendantsOrSelf(ASTImportDeclaration.class).forEach(importDeclaration -> {
            if (importDeclaration.isStatic()) {
                return;
            }
            if (importDeclaration.getImportedSimpleName() == null) {
                return;
            }
            String fullyQualifiedImport = importDeclaration.getImportedType().getQualifiedName();
            if (fullyQualifiedImport.startsWith("java.lang")) {
                return;
            }
            if (fullyQualifiedImport.startsWith(currentPackageName)) {
                currentPackageTypes.add(importDeclaration.getImportedSimpleName());
                return;
            }
        });

        node.descendants(ASTPrimaryExpression.class).forEach(primaryExpression -> {
            primaryExpression.getFirstChildOfType(ASTPrimaryPrefix.class).findFirstDescendantOfType(ASTPrimarySuffix.class).ifPresent(primarySuffix -> {
                String name = primarySuffix.getImage();
                if (!Character.isUpperCase(name.charAt(0))) {
                    return;
                }
                usedImports.add(name);
            });
        });

        node.descendantsOrSelf(ASTImportDeclaration.class).forEach(importDeclaration -> {
            if (importDeclaration.isStatic()) {
                return;
            }
            String simpleName = importDeclaration.getImportedSimpleName();
            if (simpleName == null) {
                return;
            }
            String fullyQualifiedImport = importDeclaration.getImportedType().getQualifiedName();
            if (!usedImports.contains(simpleName) || currentPackageTypes.contains(simpleName)) {
                addViolation(data, importDeclaration);
            }
        });

        return data;
    }
}

import java.util.*;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class UnnecessaryImportRule extends AbstractJavaRulechainRule {
    public UnnecessaryImportRule() {
        super(ASTCompilationUnit.class);
    }

    @Override
    public Object visit(ASTCompilationUnit node, Object data) {
        Set<String> usedImports = new HashSet<>();
        Set<String> implicitImports = new HashSet<>(Arrays.asList("java.lang.*", node.getPackageName()));
        JSymbolTable symbolTable = node.getSymbolTable();

        node.descendants(ASTName.class)
            .forEach(name -> {
                JType type = symbolTable.resolveGenericType(name.getQualifier(), name.getImage());
                if (type != null) {
                    usedImports.add(type.getPackageName() + "." + type.getSimpleName());
                }
            });

        node.descendants(ASTImportDeclaration.class)
            .filter(importDecl -> !importDecl.isStatic())
            .forEach(importDecl -> {
                String importName = importDecl.getImportedName();
                if (importName.endsWith(".*")) {
                    if (implicitImports.contains(importName)) {
                        addViolation(data, importDecl);
                    }
                } else {
                    if (!usedImports.contains(importName) || implicitImports.contains(importName)) {
                        addViolation(data, importDecl);
                    }
                }
            });

        node.descendants(ASTImportDeclaration.class)
            .filter(importDecl -> importDecl.isStatic())
            .forEach(importDecl -> {
                String importName = importDecl.getImportedName();
                String simpleName = importDecl.getImportedSimpleName();
                if (!usedImports.contains(importName + "." + simpleName) || implicitImports.contains(importName)) {
                    addViolation(data, importDecl);
                }
            });

        return super.visit(node, data);
    }
}

            }
        }

        NodeStream<ASTImportDeclaration> importDeclarations = node.descendants(ASTImportDeclaration.class);
        for (ASTImportDeclaration importDeclaration : importDeclarations) {
            if (importDeclaration.isStatic() || importDeclaration.isFindBoundary()) {
                continue;
            }
            String importedName = importDeclaration.getImportedSimpleName();
            if (importedName == null) {
                continue;
            }
