                JavaNode right = expression.getChild(1);

                if (isTrimLengthCheck(left, right) || isTrimLengthCheck(right, left)) {
                    addViolation(data, node);
                } else if (isTrimIsEmptyCheck(left, right) || isTrimIsEmptyCheck(right, left)) {
                    addViolation(data, node);
                }
            }

        return super.visit(node, data);
    }

    private boolean isTrimLengthCheck(JavaNode potentialTrimCall, JavaNode potentialZero) {
        return potentialTrimCall instanceof ASTMethodCall && potentialZero instanceof ASTLiteral &&
                ((ASTLiteral) potentialZero).getValue().equals(0) &&
                isTrimCall((ASTMethodCall) potentialTrimCall);
    }

    private boolean isTrimIsEmptyCheck(JavaNode potentialTrimCall, JavaNode potentialEmptyMethodCall) {
        return potentialTrimCall instanceof ASTMethodCall && potentialEmptyMethodCall instanceof ASTMethodCall &&
                isTrimCall((ASTMethodCall) potentialTrimCall) &&
                ((ASTMethodCall) potentialEmptyMethodCall).getMethodSignature().getName().equals("isEmpty");

package net.sourceforge.pmd.lang.java.rule.performance;

import java.util.List;

import net.sourceforge.pmd.lang.java.rule.AbstractJavaRulechainRule;
import net.sourceforge.pmd.lang.java.ast. *;
import net.sourceforge.pmd.lang.java.ast.internal. *;
import net.sourceforge.pmd.lang.java.types. *;
import net.sourceforge.pmd.lang.java.symbols. *;
import net.sourceforge.pmd.lang.ast.NodeStream;
public class InefficientEmptyStringCheckRule extends AbstractJavaRulechainRule {
    public InefficientEmptyStringCheckRule() {
        super(ASTMethodCall.class);
    }

    @Override
    public Object visit(ASTMethodCall node, Object data) {
        if (isInefficientEmptyCheck(node)) {
            addViolation(data, node);
        }
        return super.visit(node, data);
    }

    private boolean isInefficientEmptyCheck(ASTMethodCall node) {
        String methodName = node.getName();
        if (!"trim".equals(methodName)) {
            return false;
        }

        ASTPrimaryExpression primaryExpression = node.getFirstParentOfType(ASTPrimaryExpression.class);
        if (primaryExpression == null) {
            return false;
        }

        ASTPrimarySuffix suffix = node.getFirstChildOfType(ASTPrimarySuffix.class);
        if (suffix == null || suffix.getImage().equals(".length()")) {
            return checkLengthMethod(primaryExpression);
        } else if (suffix.getImage().equals(".isEmpty()")) {
            return checkIsEmptyMethod(primaryExpression);
        }

        return false;
    }

    private boolean checkLengthMethod(ASTPrimaryExpression primaryExpression) {
        NodeStream<ASTMethodCall> methodCalls = primaryExpression.descendants(ASTMethodCall.class);
        List<ASTMethodCall> lengthCalls = methodCalls.filter(call -> "length".equals(call.getName())).toList();
        List<ASTMethodCall> trimCalls = primaryExpression.descendants(ASTMethodCall.class).filter(call -> "trim".equals(call.getName())).toList();

        return lengthCalls.size() == 1 && trimCalls.size() == 1;
    }

    private boolean checkIsEmptyMethod(ASTPrimaryExpression primaryExpression) {
        NodeStream<ASTMethodCall> methodCalls = primaryExpression.descendants(ASTMethodCall.class);
        List<ASTMethodCall> isEmptyCalls = methodCalls.filter(call -> "isEmpty".equals(call.getName())).toList();
        List<ASTMethodCall> trimCalls = primaryExpression.descendants(ASTMethodCall.class).filter(call -> "trim".equals(call.getName())).toList();

        return isEmptyCalls.size() == 1 && trimCalls.size() == 1;
    }
}

import net.sourceforge.pmd.lang.java.types. *;

import net.sourceforge.pmd.lang.ast.NodeStream;
public class InefficientEmptyStringCheckRule extends AbstractJavaRulechainRule {
    public InefficientEmptyStringCheckRule() {
        super(ASTEqualityExpression.class);
    }

    @Override
    public Object visit(ASTEqualityExpression node, Object data) {
        if (node.getImage().equals("==")) {
            NodeStream<ASTMethodCall> methodCalls = node.children(ASTPrimaryExpression.class)
                    .map(JavaNode::getFirstChild)
                    .filter(ASTPrimaryPrefix.class::isInstance)
                    .map(ASTPrimaryPrefix.class::cast)
                    .flatMap(ASTPrimaryPrefix::descendants)
                    .filter(ASTMethodCall.class::isInstance)
                    .map(ASTMethodCall.class::cast);

            boolean isTrimLengthCheck = methodCalls.anyMatch(this::isTrimLengthCall);
            boolean isTrimEmptyCheck = methodCalls.anyMatch(this::isTrimEmptyCall);

            if (isTrimLengthCheck || isTrimEmptyCheck) {
                ASTLiteral literal = node.children(ASTLiteral.class).singleOrNull();
                if (literal != null && literal.getValue() instanceof Integer && (Integer) literal.getValue() == 0) {
                    addViolation(data, node);
                }
            }
        }
        return super.visit(node, data);
    }

    private boolean isTrimLengthCall(ASTMethodCall methodCall) {
        return methodCall.getImage().equals("length") && isTrimCall(methodCall);
    }

    private boolean isTrimEmptyCall(ASTMethodCall methodCall) {
        return methodCall.getImage().equals("isEmpty") && isTrimCall(methodCall);
    }

    private boolean isTrimCall(ASTMethodCall methodCall) {
        return methodCall.getParent() instanceof ASTPrimaryPrefix
                && ((ASTPrimaryPrefix) methodCall.getParent()).getFirstChild() instanceof ASTPrimaryExpression
                && ((ASTPrimaryExpression) ((ASTPrimaryPrefix) methodCall.getParent()).getFirstChild()).children(ASTPrimarySuffix.class)
                .map(ASTPrimarySuffix::getImage)
                .anyMatch(image -> image.equals("trim"));
    }
