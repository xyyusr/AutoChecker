public class BrokenNullCheckRule extends AbstractJavaRulechainRule {
    public BrokenNullCheckRule() {
        super(ASTConditionalExpression.class);
    }

    @Override
    public Object visit(ASTConditionalExpression node, Object data) {
        NodeStream<ASTExpression> expressions = node.children(ASTExpression.class);
        if (expressions.count() >= 2) {
            ASTExpression firstExpr = expressions.get(0);
            ASTExpression secondExpr = expressions.get(1);
            if (isPotentialNullCheck(firstExpr) && isNullCheckThrow(secondExpr)) {
                addViolation(data, node);
            } else if (isPotentialNullCheck(secondExpr) && isNullCheckThrow(firstExpr)) {
                addViolation(data, node);
            }
        }
        return super.visit(node, data);
    }

    private boolean isPotentialNullCheck(ASTExpression expression) {
        return JavaRuleUtil.isNullChecked(expression) || JavaAstUtils.isNullLiteral(expression);
    }

    private boolean isNullCheckThrow(ASTExpression expression) {
        if (expression instanceof ASTPrimaryExpression) {
            ASTPrimaryExpression primaryExpr = (ASTPrimaryExpression) expression;
            Optional<ASTPrimaryPrefix> prefix = primaryExpr.getFirstChildOfType(ASTPrimaryPrefix.class);
            Optional<ASTPrimarySuffix> suffix = primaryExpr.getFirstChildOfType(ASTPrimarySuffix.class);
            if (prefix.isPresent() && suffix.isPresent() && suffix.get().isMethodCall()) {
                JMethodSymbol methodSymbol = suffix.get().getEnclosingMethodSymbol();
                if (methodSymbol != null && methodSymbol.getName().equals("equals")) {
                    List<ASTExpression> args = suffix.get().getArguments().children(ASTExpression.class).toList();
                    if (args.size() == 1 && JavaAstUtils.isNullLiteral(args.get(0))) {
                        JVariableSymbol variableSymbol = prefix.get().getVariableSymbol();
                        if (variableSymbol != null) {
                            JType type = variableSymbol.getType();
                            return type != null && type.isReferenceType();
                        }
                    }
                }
            }
        }
        return false;
    }

    public BrokenNullCheckRule() {
        super(ASTBinaryExpression.class);
    }

    @Override
    public Object visit(ASTBinaryExpression node, Object data) {
        if (node.getOperator() == ASTBinaryExpression.Operator.OR || node.getOperator() == ASTBinaryExpression.Operator.AND) {
            ASTExpression leftOperand = node.getLeftOperand();
            ASTExpression rightOperand = node.getRightOperand();

            boolean isLeftOperandNullChecked = JavaRuleUtil.isNullChecked(leftOperand);
            boolean isRightOperandNullChecked = JavaRuleUtil.isNullChecked(rightOperand);

            if (isLeftOperandNullChecked && isRightOperandNullChecked) {
                boolean leftContainsNullLiteral = leftOperand.descendantsOrSelf(ASTNullLiteral.class).first() != null;
                boolean rightContainsNullLiteral = rightOperand.descendantsOrSelf(ASTNullLiteral.class).first() != null;

                if (node.getOperator() == ASTBinaryExpression.Operator.OR && leftContainsNullLiteral && rightContainsNullLiteral) {
                    addViolation(data, node);
                } else if (node.getOperator() == ASTBinaryExpression.Operator.AND && (leftContainsNullLiteral || rightContainsNullLiteral)) {
                    addViolation(data, node);
                }
            }
        }
        return super.visit(node, data);
    }
}
