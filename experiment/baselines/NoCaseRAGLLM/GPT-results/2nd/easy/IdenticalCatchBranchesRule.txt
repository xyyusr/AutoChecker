public class IdenticalCatchBranchesRule extends AbstractJavaRulechainRule {
    public IdenticalCatchBranchesRule() {
        super(ASTCatchStatement.class);
    }
    
    @Override
    public Object visit(ASTCatchStatement node, Object data) {
        List<ASTCatchClause> catchClauses = node.findDescendantsOfType(ASTCatchClause.class);
        for (int i = 0; i < catchClauses.size() - 1; i++) {
            ASTCatchClause currentCatchClause = catchClauses.get(i);
            ASTCatchClause nextCatchClause = catchClauses.get(i + 1);
            Node currentCatchBody = currentCatchClause.getBlock();
            Node nextCatchBody = nextCatchClause.getBlock();
            if (currentCatchBody.jjtGetNumChildren() != nextCatchBody.jjtGetNumChildren()) {
                continue;
            }
            boolean areIdentical = true;
            for (int j = 0; j < currentCatchBody.jjtGetNumChildren(); j++) {
                String currentCatchBodyChild = currentCatchBody.jjtGetChild(j).toString();
                String nextCatchBodyChild = nextCatchBody.jjtGetChild(j).toString();
                if (!currentCatchBodyChild.equals(nextCatchBodyChild)) {
                    areIdentical = false;
                    break;
                }
            }
            if (areIdentical) {
                addViolation(data, currentCatchClause);
                addViolation(data, nextCatchClause);
            }
        }
        return super.visit(node, data);
    }
}

                addViolation(data, catchClause);

                catchBlocks.put(blockCode, catchClause);
            }
        }

    }
